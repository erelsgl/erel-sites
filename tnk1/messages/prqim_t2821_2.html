סוס מוכן ליום מלחמה וליהוה התשועה

קוד: סוס מוכן ליום מלחמה וליהוה התשועה בתנ"ך

סוג: בסיס

מאת: מורה לתורה משה דין

אל: DAIAN1 @ 013.NET

ספר משלי פרק כא (לא) סוּס מוּכָן לְיוֹם מִלְחָמָה וְלַיהֹוָה הַתְּשׁוּעָה:
לבני ישראל בדורנו יש סוס מוכן לצהול ליום מלחמה
מה שחסר לבני ישראל לכבוש הארץ זה רק תשועת יהוה
תשועת יהוה משיגים רק אם בוטחים באלהים ללכת בדרכיו
אחוזת יהוה נתנה ליהושע ובני ישראל בתשועת יהוה
ספר יהושע פרק כד כתוב:
(יא) וַתַּעַבְרוּ אֶת הַיַּרְדֵּן וַתָּבֹאוּ אֶל יְרִיחוֹ וַיִּלָּחֲמוּ בָכֶם בַּעֲלֵי יְרִיחוֹ הָאֱמֹרִי וְהַפְּרִזִּי וְהַכְּנַעֲנִי וְהַחִתִּי וְהַגִּרְגָּשִׁי הַחִוִּי וְהַיְבוּסִי וָאֶתֵּן אוֹתָם בְּיֶדְכֶם: [חומות העיר נפלו תחתם משופרות כהני יהוה שופרות היובלים המזכירות את קדוש שנת היובל לעשות צדק חברתי]
(יב) וָאֶשְׁלַח לִפְנֵיכֶם אֶת הַצִּרְעָה וַתְּגָרֶשׁ אוֹתָם מִפְּנֵיכֶם שְׁנֵי מַלְכֵי הָאֱמֹרִי לֹא בְחַרְבְּךָ וְלֹא בְקַשְׁתֶּךָ:
(יג) וָאֶתֵּן לָכֶם אֶרֶץ אֲשֶׁר לֹא יָגַעְתָּ בָּהּ וְעָרִים אֲשֶׁר לֹא בְנִיתֶם וַתֵּשְׁבוּ בָּהֶם כְּרָמִים וְזֵיתִים אֲשֶׁר לֹא נְטַעְתֶּם אַתֶּם אֹכְלִים:
ספר שמואל א פרק יז כתוב:
(מה) וַיֹּאמֶר דָּוִד אֶל הַפְּלִשְׁתִּי אַתָּה בָּא אֵלַי בְּחֶרֶב וּבַחֲנִית וּבְכִידוֹן וְאָנֹכִי בָא אֵלֶיךָ בְּשֵׁם יְהֹוָה צְבָאוֹת אֱלֹהֵי מַעַרְכוֹת יִשְׂרָאֵל אֲשֶׁר חֵרַפְתָּ:
(מו) הַיּוֹם הַזֶּה יְסַגֶּרְךָ יְהֹוָה בְּיָדִי וְהִכִּיתִךָ וַהֲסִרֹתִי אֶת רֹאשְׁךָ מֵעָלֶיךָ וְנָתַתִּי פֶּגֶר מַחֲנֵה פְלִשְׁתִּים הַיּוֹם הַזֶּה לְעוֹף הַשָּׁמַיִם וּלְחַיַּת הָאָרֶץ וְיֵדְעוּ כָּל הָאָרֶץ כִּי יֵשׁ אֱלֹהִים לְיִשְׂרָאֵל:
(מז) וְיֵדְעוּ כָּל הַקָּהָל הַזֶּה כִּי לֹא בְּחֶרֶב וּבַחֲנִית יְהוֹשִׁיעַ יְהֹוָה כִּי לַיהֹוָה הַמִּלְחָמָה וְנָתַן אֶתְכֶם בְּיָדֵנוּ:
ספר תהילות פרק לג כתוב:
(טז) אֵין הַמֶּלֶךְ נוֹשָׁע בְּרָב חָיִל גִּבּוֹר לֹא יִנָּצֵל בְּרָב כֹּחַ:
(יז) שֶׁקֶר הַסּוּס לִתְשׁוּעָה וּבְרֹב חֵילוֹ לֹא יְמַלֵּט:
(יח) הִנֵּה עֵין יְהֹוָה אֶל יְרֵאָיו לַמְיַחֲלִים לְחַסְדּוֹ:
(יט) לְהַצִּיל מִמָּוֶת נַפְשָׁם וּלְחַיּוֹתָם בָּרָעָב:
(כ) נַפְשֵׁנוּ חִכְּתָה לַיהֹוָה עֶזְרֵנוּ וּמָגִנֵּנוּ הוּא:
(כא) כִּי בוֹ יִשְׂמַח לִבֵּנוּ כִּי בְשֵׁם קָדְשׁוֹ בָטָחְנוּ:
(כב) יְהִי חַסְדְּךָ יְהֹוָה עָלֵינוּ כַּאֲשֶׁר יִחַלְנוּ לָךְ:
ספר שמות פרק יד כתוב:

(יד) יְהֹוָה יִלָּחֵם לָכֶם וְאַתֶּם תַּחֲרִשׁוּן: על כן יש להחריש בזמן המלחמה וגם המודיעין יודע זאת

ובספר דברי הימים א פרק ה כתוב:

(כ) וַיֵּעָזְרוּ עֲלֵיהֶם וַיִּנָּתְנוּ בְיָדָם הַהַגְרִיאִים וְכֹל שֶׁעִמָּהֶם  כִּי לֵאלֹהִים זָעֲקוּ בַּמִּלְחָמָה וְנַעְתּוֹר לָהֶם כִּי בָטְחוּ בוֹ:
על כן במלחמה יש לזעוק לאלהים ולבטוח בו למען יעתר לנו

בספר דברי הימים פרק יז כתוב:
(ט) וַיְלַמְּדוּ בִּיהוּדָה וְעִמָּהֶם סֵפֶר תּוֹרַת יְהֹוָה וַיָּסֹבּוּ בְּכָל עָרֵי יְהוּדָה וַיְלַמְּדוּ בָּעָם:
(י) וַיְהִי פַּחַד יְהֹוָה עַל כָּל מַמְלְכוֹת הָאֲרָצוֹת אֲשֶׁר סְבִיבוֹת יְהוּדָה וְלֹא נִלְחֲמוּ עִם יְהוֹשָׁפָט:
ו בספר בראשית פרק לה כתוב:
(א) וַיֹּאמֶר אֱלֹהִים אֶל יַעֲקֹב קוּם עֲלֵה בֵית אֵל וְשֶׁב שָׁם וַעֲשֵׂה שָׁם מִזְבֵּחַ לָאֵל הַנִּרְאֶה אֵלֶיךָ בְּבָרְחֲךָ מִפְּנֵי עֵשָׂו אָחִיךָ:
(ב) וַיֹּאמֶר יַעֲקֹב אֶל בֵּיתוֹ וְאֶל כָּל אֲשֶׁר עִמּוֹ הָסִרוּ אֶת אֱלֹהֵי הַנֵּכָר אֲשֶׁר בְּתֹכֲכֶם וְהִטַּהֲרוּ וְהַחֲלִיפוּ שִׂמְלֹתֵיכֶם:
(ג) וְנָקוּמָה וְנַעֲלֶה בֵּית אֵל וְאֶעֱשֶׂה שָּׁם מִזְבֵּחַ לָאֵל הָעֹנֶה אֹתִי בְּיוֹם צָרָתִי וַיְהִי עִמָּדִי בַּדֶּרֶךְ אֲשֶׁר הָלָכְתִּי:
(ד) וַיִּתְּנוּ אֶל יַעֲקֹב אֵת כָּל אֱלֹהֵי הַנֵּכָר אֲשֶׁר בְּיָדָם וְאֶת הַנְּזָמִים אֲשֶׁר בְּאָזְנֵיהֶם וַיִּטְמֹן אֹתָם יַעֲקֹב תַּחַת הָאֵלָה אֲשֶׁר עִם שְׁכֶם:
(ה) וַיִּסָּעוּ וַיְהִי חִתַּת אֱלֹהִים עַל הֶעָרִים אֲשֶׁר סְבִיבוֹתֵיהֶם וְלֹא רָדְפוּ אַחֲרֵי בְּנֵי יַעֲקֹב:
ואתה אמור נא אל כל בית יַעֲקֹב אשר יש עמהם אֱלֹהֵי הַנֵּכָר כסף וזהב וטמון אותם מתחת לאדמה לאמור תן את הכסף והזהב לעניים המרודים אשר חייהם אינם חיים על פני האדמה והם למעשה כשורשים מתחת לאדמה ויונקים את מסורת אלהי אבותינו הקדושים כזרע מתחת לאדמה אוֹר זָרֻעַ לַצַּדִּיק וּלְיִשְׁרֵי לֵב שִׂמְחָה כאשר אמר דוד משיח אלהים  וְאָנֹכִי תוֹלַעַת וְלֹא אִישׁ חֶרְפַּת אָדָם וּבְזוּי עָם: וכאשר אמר אברהם אבינו הצדיק ירא האלהים וַיַּעַן אַבְרָהָם וַיֹּאמַר הִנֵּה נָא הוֹאַלְתִּי לְדַבֵּר אֶל אֲדֹנָי וְאָנֹכִי עָפָר וָאֵפֶר: 
לאמור כי הצדיקים כולם אנשי אדמה ותן להם את הכסף והזהב וַיְהִי חִתַּת אֱלֹהִים על כל אויבי ישראל ושונאי יעקב ולא ירדפו אַחֲרֵי בְּנֵי יַעֲקֹב כי מלאך האלהים לָנוּ
