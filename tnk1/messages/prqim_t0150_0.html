מערת המכפלה = אחוזת-קבר למתים בשלום

קוד: מערת המכפלה = אחוזת-קבר למתים בשלום בתנ"ך

סוג: מאמר

מאת: רמי ניר

אל:

בְּשָׁלוֹם... בני אברהם ובני יצחק ובני יעקב...
בִּמְעָרַת שְׂדֵה הַמַּכְפֵּלָה שָׁמָּה קָבְרוּ אֶת-אַבְרָהָם וְאֵת שָׂרָה אִשְׁתּוֹ שָׁמָּה קָבְרוּ אֶת-יִצְחָק וְאֵת רִבְקָה אִשְׁתּוֹ וְשָׁמָּה קָבַר יַעֲקֹב אֶת-לֵאָה... וְשָׁמָּה קָבְרוּ אוֹתוֹ (את יעקב החנוט) בָנָיו יוֹסֵף וְאֶחָיו...

כלומר... בִּמְעָרַת הַמַּכְפֵּלָה קָבְרוּ אֶת הַמֵּתִים בְּשָׁלוֹם...
על-כן... "מְעָרַת הַמַּכְפֵּלָה" = "אֲחֻזַּת-קָבֶר לְחַיֵּי שָׁלוֹם"...

ומתועד - כדלקמן:

1.
הבטחתו של אלהים לְאַבְרָם (בראשית טו' 15) "וְאַתָּה תָּבוֹא אֶל-אֲבוֹתֶיךָ בְּשָׁלוֹם תִּקָּבֵר בְּשֵׂיבָה טוֹבָה"...

2.
אברהם ביקש לקנות ואכן קנה את מערת המכפלה והשדה להיות אחוזת-קבר לבני משפחתו ואחר-כך (במערת המכפלה) קבר את שרה אשתו "בְּשָׁלוֹם":

וַיְדַבֵּר אִתָּם לֵאמוֹר אִם-יֵשׁ אֶת-נַפְשְׁכֶם לִקְבּוֹר אֶת-מֵתִי מִלְּפָנַי--שְׁמָעוּנִי וּפִגְעוּ-לִי בְּעֶפְרוֹן בֶּן-צוֹחַר וְיִתֶּן-לִי אֶת-מְעָרַת הַמַּכְפֵּלָה אֲשֶׁר-לוֹ אֲשֶׁר בִּקְצֵה שָׂדֵהוּ בְּכֶסֶף מָלֵא יִתְּנֶנָּה לִּי בְּתוֹכְכֶם--לַאֲחֻזַּת-קָבֶר...
(בראשית כג' 8 -9)

וַיִּשְׁמַע אַבְרָהָם אֶל-עֶפְרוֹן וַיִּשְׁקוֹל אַבְרָהָם לְעֶפְרוֹן אֶת-הַכֶּסֶף אֲשֶׁר דִּבֶּר בְּאָזְנֵי בְנֵי-חֵת--אַרְבַּע מֵאוֹת שֶׁקֶל כֶּסֶף עוֹבֵר לַסּוֹחֵר וַיָּקָם שְׂדֵה עֶפְרוֹן אֲשֶׁר בַּמַּכְפֵּלָה אֲשֶׁר לִפְנֵי מַמְרֵא הַשָּׂדֶה וְהַמְּעָרָה אֲשֶׁר-בּוֹ וְכָל-הָעֵץ אֲשֶׁר בַּשָּׂדֶה אֲשֶׁר בְּכָל-גְּבֻלוֹ סָבִיב לְאַבְרָהָם לְמִקְנָה לְעֵינֵי בְנֵי-חֵת בְּכוֹל בָּאֵי שַׁעַר-עִירוֹ וְאַחֲרֵי-כֵן קָבַר אַבְרָהָם אֶת-שָׂרָה אִשְׁתּוֹ אֶל-מְעָרַת שְׂדֵה הַמַּכְפֵּלָה עַל-פְּנֵי מַמְרֵא--הִוא חֶבְרוֹן בְּאֶרֶץ כְּנָעַן וַיָּקָם הַשָּׂדֶה וְהַמְּעָרָה אֲשֶׁר-בּוֹ לְאַבְרָהָם--לַאֲחֻזַּת-קָבֶר מֵאֵת בְּנֵי-חֵת...
(בראשית כג' 16 - 20)

3.
יצחק וישמעאל קברו את אברהם במערת המכפלה "בְּשָׁלוֹם":

וַיִּגְוַע וַיָּמָת אַבְרָהָם בְּשֵׂיבָה טוֹבָה זָקֵן וְשָׂבֵעַ וַיֵּאָסֶף אֶל-עַמָּיו וַיִּקְבְּרוּ אוֹתוֹ יִצְחָק וְיִשְׁמָעֵאל בָּנָיו אֶל-מְעָרַת הַמַּכְפֵּלָה אֶל-שְׂדֵה עֶפְרוֹן בֶּן-צוֹחַר הַחִתִּי אֲשֶׁר עַל-פְּנֵי מַמְרֵא הַשָּׂדֶה אֲשֶׁר-קָנָה אַבְרָהָם מֵאֵת בְּנֵי-חֵת--שָׁמָּה קֻבַּר אַבְרָהָם וְשָׂרָה אִשְׁתּוֹ...
(בראשית כה' 8 - 10)

4.
יעקב ועשו (האחים התאומים) קברו את יצחק במערת המכפלה "בְּשָׁלוֹם":

וַיִּהְיוּ יְמֵי יִצְחָק--מְאַת שָׁנָה וּשְׁמוֹנִים שָׁנָה וַיִּגְוַע יִצְחָק וַיָּמָת וַיֵּאָסֶף אֶל-עַמָּיו זָקֵן וּשְׂבַע יָמִים וַיִּקְבְּרוּ אוֹתוֹ עֵשָׂו וְיַעֲקֹב בָּנָיו...
(בראשית לה' 28 - 29)

5.
יעקב ציווה את בניו כי "בְּשָׁלוֹם" יקברו אותו במערת המכפלה:

וַיְצַו אוֹתָם וַיּוֹאמֶר אֲלֵהֶם אֲנִי נֶאֱסָף אֶל-עַמִּי--קִבְרוּ אוֹתִי אֶל-אֲבוֹתָי אֶל-הַמְּעָרָה--אֲשֶׁר בִּשְׂדֵה עֶפְרוֹן הַחִתִּי בַּמְּעָרָה אֲשֶׁר בִּשְׂדֵה הַמַּכְפֵּלָה אֲשֶׁר עַל-פְּנֵי-מַמְרֵא--בְּאֶרֶץ כְּנָעַן אֲשֶׁר קָנָה אַבְרָהָם אֶת-הַשָּׂדֶה מֵאֵת עֶפְרוֹן הַחִתִּי--לַאֲחֻזַּת-קָבֶר שָׁמָּה קָבְרוּ אֶת-אַבְרָהָם וְאֵת שָׂרָה אִשְׁתּוֹ שָׁמָּה קָבְרוּ אֶת-יִצְחָק וְאֵת רִבְקָה אִשְׁתּוֹ וְשָׁמָּה קָבַרְתִּי אֶת-לֵאָה מִקְנֵה הַשָּׂדֶה וְהַמְּעָרָה אֲשֶׁר-בּוֹ מֵאֵת בְּנֵי-חֵת וַיְכַל יַעֲקֹב לְצַוּוֹת אֶת-בָּנָיו וַיֶּאֱסוֹף רַגְלָיו אֶל-הַמִּטָּה וַיִּגְוַע וַיֵּאָסֶף אֶל-עַמָּיו...
(בראשית מט' 29 - 33)

6.
יוסף ציווה על חניטת יעקב:

וַיְצַו יוֹסֵף אֶת-עֲבָדָיו אֶת-הָרוֹפְאִים לַחֲנוֹט אֶת-אָבִיו וַיַּחַנְטוּ הָרוֹפְאִים אֶת-יִשְׂרָאֵל וַיִּמְלְאוּ-לוֹ אַרְבָּעִים יוֹם כִּי כֵּן יִמְלְאוּ יְמֵי הַחֲנֻטִים וַיִּבְכּוּ אוֹתוֹ מִצְרַיִם שִׁבְעִים יוֹם...
(בראשית נ' 2 - 3)

7.
יוסף ואחיו קברו את יעקב (החנוט) במערת המכפלה "בְּשָׁלוֹם":

וַיַּעַל יוֹסֵף לִקְבּוֹר אֶת-אָבִיו וַיַּעֲלוּ אִתּוֹ כָּל-עַבְדֵי פַרְעֹה זִקְנֵי בֵיתוֹ וְכוֹל זִקְנֵי אֶרֶץ-מִצְרָיִם וְכוֹל בֵּית יוֹסֵף וְאֶחָיו וּבֵית אָבִיו רַק טַפָּם וְצוֹאנָם וּבְקָרָם--עָזְבוּ בְּאֶרֶץ גּוֹשֶׁן וַיַּעַל עִמּוֹ גַּם-רֶכֶב גַּם-פָּרָשִׁים וַיְהִי הַמַּחֲנֶה כָּבֵד מְאוֹד וַיָּבוֹאוּ עַד-גּוֹרֶן הָאָטָד אֲשֶׁר בְּעֵבֶר הַיַּרְדֵּן וַיִּסְפְּדוּ-שָׁם מִסְפֵּד גָּדוֹל וְכָבֵד מְאוֹד וַיַּעַשׂ לְאָבִיו אֵבֶל שִׁבְעַת יָמִים וַיַּרְא יוֹשֵׁב הָאָרֶץ הַכְּנַעֲנִי אֶת-הָאֵבֶל בְּגוֹרֶן הָאָטָד וַיּוֹאמְרוּ אֵבֶל-כָּבֵד זֶה לְמִצְרָיִם עַל-כֵּן קָרָא שְׁמָהּ אָבֵל מִצְרַיִם אֲשֶׁר בְּעֵבֶר הַיַּרְדֵּן וַיַּעֲשׂוּ בָנָיו לוֹ--כֵּן כַּאֲשֶׁר צִוָּם וַיִּשְׂאוּ אוֹתוֹ בָנָיו אַרְצָה כְּנַעַן וַיִּקְבְּרוּ אוֹתוֹ בִּמְעָרַת שְׂדֵה הַמַּכְפֵּלָה אֲשֶׁר קָנָה אַבְרָהָם אֶת-הַשָּׂדֶה לַאֲחֻזַּת-קֶבֶר מֵאֵת עֶפְרוֹן הַחִתִּי--עַל-פְּנֵי מַמְרֵא...
(בראשית נ' 7 - 13)

***

ללא ספק יוסף ציווה על עבדיו המצרים לחפור ליעקב אביו חדר קבורה סודי ומיוחד במערת המכפלה ובו הניחו לנצח נצחים את ארון האבן ובתוכו המומיה החנוטה של יעקב...

יוסף ככל הנראה (מתועד בתעודות ההיסטוריה המצרית) באותו מסע גם ציווה על עבדיו המצרים לחפור בחלקת השדה אשר קנה יעקב מאת בני חמור בשכם "בְּחֶלְקַת הַשָּׂדֶה אֲשֶׁר קָנָה יַעֲקֹב מֵאֵת בְּנֵי-חֲמוֹר אֲבִי-שְׁכֶם בְּמֵאָה קְשִׂיטָה וַיִּהְיוּ לִבְנֵי-יוֹסֵף לְנַחֲלָה"... את קברו הסודי העתידי (שם קבר יהושע באותו קבר מצרי סודי את המומיה של יוסף) על צלע הר גריזים... כל זאת, כאשר באותו מסע פרעה סנווסרת ה-3 כבש את שכם...
