והגר קראה את שם המלאך: "אתה אל ראי"

מאת: רמי ניר

נכתב ב: 23:24:58  22.11.2010, כתוספת/תגובה ל: בראשית טז

כדאי לתת את הדעת על תיעוד החפיפה המתמקד במלאך אלהים, המדבר אל הגר...

א)
בפרק (בראשית טז') מתועד כי הגר ראתה את פני המלאך משתקפים בראי "עֵין הַמַּיִם" ואמנם (טז' 13):
וַתִּקְרָא שֵׁם-יְהוָה הַדּוֹבֵר אֵלֶיהָ "אַתָּה אֵל רֳאִי"...

ב)
והתיעוד (באותו פסוק) חופף:
"כִּי אָמְרָה הֲגַם הֲלוֹם רָאִיתִי--אַחֲרֵי רוֹאִי"...

כלומר, החפיפה:
אותו מלאך (פעמיים) השקיף על הגר "מִן-הַשָּׁמַיִם", כמתועד (בראשית כא' 17) "וַיִּשְׁמַע אֱלוֹהִים אֶת-קוֹל הַנַּעַר וַיִּקְרָא מַלְאַךְ אֱלוֹהִים אֶל-הָגָר מִן-הַשָּׁמַיִם וַיּוֹאמֶר לָהּ מַה-לָּךְ הָגָר אַל-תִּירְאִי כִּי-שָׁמַע אֱלוֹהִים אֶל-קוֹל הַנַּעַר בַּאֲשֶׁר הוּא-שָׁם"...


להלן התיעוד (בראשית טז' 7 - 14):

וַיִּמְצָאָהּ מַלְאַךְ יְהוָה עַל-עֵין הַמַּיִם--בַּמִּדְבָּר
עַל-הָעַיִן בְּדֶרֶךְ שׁוּר
וַיּוֹאמַר הָגָר שִׁפְחַת שָׂרַי אֵי-מִזֶּה בָאת--וְאָנָה תֵלֵכִי
וַתּוֹאמֶר--מִפְּנֵי שָׂרַי גְּבִרְתִּי אָנוֹכִי בּוֹרַחַת
וַיּוֹאמֶר לָהּ מַלְאַךְ יְהוָה שׁוּבִי אֶל-גְּבִרְתֵּךְ
וְהִתְעַנִּי תַּחַת יָדֶיהָ...

וַיּוֹאמֶר לָהּ מַלְאַךְ יְהוָה הַרְבָּה אַרְבֶּה אֶת-זַרְעֵךְ
וְלֹא יִסָּפֵר, מֵרוֹב
וַיּוֹאמֶר לָהּ מַלְאַךְ יְהוָה הִנָּךְ הָרָה וְיוֹלַדְתְּ בֵּן וְקָרָאת שְׁמוֹ יִשְׁמָעֵאל
כִּי-שָׁמַע יְהוָה אֶל-עָנְיֵךְ
וְהוּא יִהְיֶה פֶּרֶא אָדָם--יָדוֹ בַכּוֹל וְיַד כּוֹל בּוֹ
וְעַל-פְּנֵי כָל-אֶחָיו יִשְׁכּוֹן...

וַתִּקְרָא שֵׁם-יְהוָה הַדּוֹבֵר אֵלֶיהָ אַתָּה אֵל רֳאִי
כִּי אָמְרָה הֲגַם הֲלוֹם רָאִיתִי--אַחֲרֵי רוֹאִי
עַל-כֵּן קָרָא לַבְּאֵר בְּאֵר לַחַי רוֹאִי
הִנֵּה בֵין-קָדֵשׁ וּבֵין בָּרֶד...

***

אולם... לא מתועד האם מלאך זה שעיניינו בנצח לפקח על "יִשְׁמָעֵאל" הוא אחד משלשת המלאכים (בראשית יח' 2):
"וַיִּשָּׂא עֵינָיו וַיַּרְא וְהִנֵּה שְׁלוֹשָׁה אֲנָשִׁים נִצָּבִים עָלָיו"...
