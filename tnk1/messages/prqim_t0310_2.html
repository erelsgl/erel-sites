(ו) וַיֹּאמֶר אָנֹכִי אֱלֹהֵי אָבִיךָ אֱלֹהֵי אַבְרָהָם אֱלֹהֵי יִצְחָק וֵאלֹהֵי יַעֲקֹב וַיַּסְתֵּר משֶׁה פָּנָיו כִּי יָרֵא מֵהַבִּיט אֶל הָאֱלֹהִים
כי אבותינו הקדושים כולם אמנם עבדו את יהוה האחד והיחיד אך כל אחד וסיפורו האישי להתמודד עם צרות ומה שהאלהים מנסה אותך בחיים וכל צדיק הלך על פי דרכו ותכונת האלהים [האופיינית] המיוחדת לו על כן כתוב בספר משלי פרק כב (ו) חֲנֹךְ לַנַּעַר עַל פִּי דַרְכּוֹ גַּם כִּי יַזְקִין לֹא יָסוּר מִמֶּנָּה:
לכל אחד מאבותינו היה נתיב אלהים מיוחד משלו אשר הלך בו וכל אחד על פי דרכו 
וככתוב בספר משלי פרק ג (יז) דְּרָכֶיהָ דַרְכֵי נֹעַם וְכָל נְתִיבוֹתֶיהָ שָׁלוֹם:
על כן ניתן לעבוד את יהוה גם דרך אלהי אברהם וגם דרך אלהי יצחק וגם דרך אלהי יעקב

ועל כן ניתן לעבוד את יהוה גם דרך אלהי השמים וגם דרך אלהי הארץ הוא אל שדי

דברי דוד איש האלהים בספר תהילות פרק נא כתוב : 
(יז) אֲדֹנָי שְׂפָתַי תִּפְתָּח וּפִי יַגִּיד תְּהִלָּתֶךָ
(יח) כִּי לֹא תַחְפֹּץ זֶבַח וְאֶתֵּנָה עוֹלָה לֹא תִרְצֶה
(יט) זִבְחֵי אֱלֹהִים רוּחַ נִשְׁבָּרָה לֵב נִשְׁבָּר וְנִדְכֶּה אֱלֹהִים לֹא תִבְזֶה  ואלה זבחי אלהי השמים. וכאשר...
(כ) הֵיטִיבָה בִרְצוֹנְךָ אֶת צִיּוֹן תִּבְנֶה חוֹמוֹת יְרוּשָׁלָםִ
(כא) אָז תַּחְפֹּץ זִבְחֵי צֶדֶק עוֹלָה וְכָלִיל אָז יַעֲלוּ עַל מִזְבַּחֲךָ פָרִים :

כי כאשר רוּחַ נִשְׁבָּרָה לֵב נִשְׁבָּר וְנִדְכֶּה זה כבר זִבְחֵי אֱלֹהִים וזבח אלהים הוא טהור ושלם ואין לאכול עליהם בנוסף בשר
כי לא יהיה זבח אלהי הארץ על זבח אלהי השמים
ספר מלאכי פרק ב כתוב :
(יג) וְזֹאת שֵׁנִית תַּעֲשׂוּ כַּסּוֹת דִּמְעָה אֶת מִזְבַּח יְהֹוָה בְּכִי וַאֲנָקָה מֵאֵין עוֹד פְּנוֹת אֶל הַמִּנְחָה וְלָקַחַת רָצוֹן מִיֶּדְכֶם
לאמור כי אין לכַּסּוֹת דִּמְעָה אֶת מִזְבַּח יְהֹוָה
על כן אהרן לא אכל את שעיר החטאת ממִזְבַּח יְהֹוָה כי ידע כי לא יהיה לרָצוֹן כי את הזבח יש לזבוח לרָצוֹן בשמחה
ספר ויקרא פרק י כתוב :

(טז) וְאֵת שְׂעִיר הַחַטָּאת דָּרשׁ דָּרַשׁ משֶׁה וְהִנֵּה שׂרָף וַיִּקְצֹף עַל אֶלְעָזָר וְעַל אִיתָמָר בְּנֵי אַהֲרֹן הַנּוֹתָרִם לֵאמֹר
(יז) מַדּוּעַ לֹא אֲכַלְתֶּם אֶת הַחַטָּאת בִּמְקוֹם הַקֹּדֶשׁ כִּי קֹדֶשׁ קָדָשִׁים הִוא וְאֹתָהּ נָתַן לָכֶם לָשֵׂאת אֶת עֲוֹן הָעֵדָה לְכַפֵּר עֲלֵיהֶם לִפְנֵי יְהֹוָה
(יח) הֵן לֹא הוּבָא אֶת דָּמָהּ אֶל הַקֹּדֶשׁ פְּנִימָה אָכוֹל תֹּאכְלוּ אֹתָהּ בַּקֹּדֶשׁ כַּאֲשֶׁר צִוֵּיתִי
(יט) וַיְדַבֵּר אַהֲרֹן אֶל משֶׁה הֵן הַיּוֹם הִקְרִיבוּ אֶת חַטָּאתָם וְאֶת עֹלָתָם לִפְנֵי יְהֹוָה וַתִּקְרֶאנָה אֹתִי כָּאֵלֶּה וְאָכַלְתִּי חַטָּאת הַיּוֹם הַיִּיטַב בְּעֵינֵי יְהֹוָה :
(כ) וַיִּשְׁמַע משֶׁה וַיִּיטַב בְּעֵינָיו  כי משה איש האלהים כתב בתורתו  "לִרְצֹנְכֶם תִּזְבָּחוּ" 
ןבספר ויקרא פרק כב כתוב :
(יז) וַיְדַבֵּר יְהֹוָה אֶל משֶׁה לֵּאמֹר
(יח) דַּבֵּר אֶל אַהֲרֹן וְאֶל בָּנָיו וְאֶל כָּל בְּנֵי יִשְׂרָאֵל וְאָמַרְתָּ אֲלֵהֶם אִישׁ אִישׁ מִבֵּית יִשְׂרָאֵל וּמִן הַגֵּר בְּיִשְׂרָאֵל אֲשֶׁר יַקְרִיב קָרְבָּנוֹ לְכָל נִדְרֵיהֶם וּלְכָל נִדְבוֹתָם אֲשֶׁר יַקְרִיבוּ לַיהֹוָה לְעֹלָה
(יט) לִרְצֹנְכֶם תָּמִים זָכָר בַּבָּקָר בַּכְּשָׂבִים וּבָעִזִּים
ולרצון בשמחה כי אם לא ברצון ושמחה לעבוד את יהוה ואיך ישע לך
אז לכן מזהיר משה איש האלהים בתורתו בספר דברים פרק כח ככתוב: 

(מה) וּבָאוּ עָלֶיךָ כָּל הַקְּלָלוֹת הָאֵלֶּה וּרְדָפוּךָ וְהִשִּׂיגוּךָ עַד הִשָּׁמְדָךְ כִּי לֹא שָׁמַעְתָּ בְּקוֹל יְהֹוָה אֱלֹהֶיךָ לִשְׁמֹר מִצְוֹתָיו וְחֻקֹּתָיו אֲשֶׁר צִוָּךְ
(מו) וְהָיוּ בְךָ לְאוֹת וּלְמוֹפֵת וּבְזַרְעֲךָ עַד עוֹלָם
(מז) תַּחַת אֲשֶׁר לֹא עָבַדְתָּ אֶת יְהֹוָה אֱלֹהֶיךָ בְּשִׂמְחָה וּבְטוּב לֵבָב מֵרֹב כֹּל
וכי זֶבַח וּמִנְחָה נחשבים לעבודת יהוה אשר יש לעבוד בשמחה
ובספר ישעיהו פרק יט כתוב:
(כא) וְנוֹדַע יְהֹוָה לְמִצְרַיִם וְיָדְעוּ מִצְרַיִם אֶת יְהֹוָה בַּיּוֹם הַהוּא וְעָבְדוּ זֶבַח וּמִנְחָה וְנָדְרוּ נֵדֶר לַיהֹוָה וְשִׁלֵּמוּ
ואם אין שמחה לאדם לא יוכל במזבח האדמה לעבוד את יהוה דרך אלהי הארץ ואל [ואסור] לו לאכול מבשר הזבח כי אם לא בשמחה אז לא לרצון יהיה
אז מה יעשה אם לא יעבוד את יהוה דרך אלהי הארץ
לכן רחום וחנון יהוה ונתן לו האלהים דרך אלהי השמים בכי ודמעה זִבְחֵי אֱלֹהִים רוּחַ נִשְׁבָּרָה לֵב נִשְׁבָּר וְנִדְכֶּה אֱלֹהִים לֹא תִבְזֶה
אך אין לכסות דמעה של בכי מזבח אלהי השמים את מזבח האדמה כי זבח לאלהי הארץ יש לעבוד בשמחה ולא עת לרוּחַ נִשְׁבָּרָה לֵב נִשְׁבָּר וְנִדְכֶּה

ספר שמואל פרק א כתוב:
(ז) וְכֵן יַעֲשֶׂה שָׁנָה בְשָׁנָה מִדֵּי עֲלֹתָהּ בְּבֵית יְהֹוָה כֵּן תַּכְעִסֶנָּ ה וַתִּבְכֶּה וְלֹא תֹאכַל :
ספר שמואל א פרק כ (לד) וַיָּקָם יְהוֹנָתָן מֵעִם הַשֻּׁלְחָן בָּחֳרִי אָף וְלֹא אָכַל בְּיוֹם הַחֹדֶשׁ הַשֵּׁנִי לֶחֶם כִּי נֶעְצַב אֶל דָּוִד כִּי הִכְלִמוֹ אָבִיו
ספר ישעיהו פרק נו כתוב:
(ז) וַהֲבִיאוֹתִים אֶל הַר קָדְשִׁי וְשִׂמַּחְתִּים בְּבֵית תְּפִלָּתִי עוֹלֹתֵיהֶם וְזִבְחֵיהֶם לְרָצוֹן עַל מִזְבְּחִי כִּי בֵיתִי בֵּית תְּפִלָּה יִקָּרֵא לְכָל הָעַמִּים
ובספר קהלת פרק ט כתוב:
(ז) לֵךְ אֱכֹל בְּשִׂמְחָה לַחְמֶךָ וּשְׁתֵה בְלֶב טוֹב יֵינֶךָ כִּי כְבָר רָצָה הָאֱלֹהִים אֶת מַעֲשֶׂיךָ:

וגם אני כאיש אלהים רוצה שתאכל בשמחה ותזבח לרצון ואל תחשבו להחליף את הזבח לאלהי הארץ בזבח לאלהי השמים
כי זִבְחֵי אֱלֹהִים רוּחַ נִשְׁבָּרָה לֵב נִשְׁבָּר וְנִדְכֶּה אֱלֹהִים לֹא תִבְזֶה ואין אדם יכול [לתזמן זאת] להחליף את זבחי אלהי השמים בזבחי אלהי הארץ הכתובים בתורה אשר להם מועדים [קבועים] וכי לזבחי אלהי השמים יש עתים אשר אלהים לבדו יודע אותם והוא לבדו בוחר את הזמנים מהשמים לאמור כי דמעה של בכי מעומק הלב זה פרץ רגשות מרוח האלהים ובלתי נשלט לאדם ואין אדם שליט ברוח לכלוא את הרוח
