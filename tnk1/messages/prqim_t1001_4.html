דברי אלהים חֻקּוֹת עוֹלָם תָּמִיד
(א) חֲזוֹן יְשַׁעְיָהוּ בֶן אָמוֹץ אֲשֶׁר חָזָה עַל יְהוּדָה וִירוּשָׁלָם בִּימֵי עֻזִּיָּהוּ יוֹתָם אָחָז יְחִזְקִיָּהוּ מַלְכֵי יְהוּדָה:
(ב) שִׁמְעוּ שָׁמַיִם וְהַאֲזִינִי אֶרֶץ כִּי יְהֹוָה דִּבֵּר בָּנִים גִּדַּלְתִּי וְרוֹמַמְתִּי וְהֵם פָּשְׁעוּ בִי:
(ג) יָדַע שׁוֹר קֹנֵהוּ וַחֲמוֹר אֵבוּס בְּעָלָיו יִשְׂרָאֵל לֹא יָדַע עַמִּי לֹא הִתְבּוֹנָן:
(ד) הוֹי גּוֹי חֹטֵא עַם כֶּבֶד עָוֹן זֶרַע מְרֵעִים בָּנִים מַשְׁחִיתִים עָזְבוּ אֶת יְהֹוָה נִאֲצוּ אֶת קְדוֹשׁ יִשְׂרָאֵל נָזֹרוּ אָחוֹר:
(ה) עַל מֶה תֻכּוּ עוֹד תּוֹסִיפוּ סָרָה כָּל רֹאשׁ לָחֳלִי וְכָל לֵבָב דַּוָּי:
(ו) מִכַּף רֶגֶל וְעַד רֹאשׁ אֵין בּוֹ מְתֹם פֶּצַע וְחַבּוּרָה וּמַכָּה טְרִיָּה לֹא זֹרוּ וְלֹא חֻבָּשׁוּ וְלֹא רֻכְּכָה בַּשָּׁמֶן:
(ז) אַרְצְכֶם שְׁמָמָה עָרֵיכֶם שְׂרֻפוֹת אֵשׁ אַדְמַתְכֶם לְנֶגְדְּכֶם זָרִים אֹכְלִים אֹתָהּ וּשְׁמָמָה כְּמַהְפֵּכַת זָרִים:
כִּי עֲדַת חָנֵף גַּלְמוּד וְאֵשׁ אָכְלָה אָהֳלֵי שֹׁחַד:
כִּי אֵשׁ קָדְחָה בְאַפִּי וַתִּיקַד עַד שְׁאוֹל תַּחְתִּית וַתֹּאכַל אֶרֶץ וִיבֻלָהּ וַתְּלַהֵט מוֹסְדֵי הָרִים:
לָכֵן כֶּאֱכֹל קַשׁ לְשׁוֹן אֵשׁ וַחֲשַׁשׁ לֶהָבָה יִרְפֶּה שָׁרְשָׁם כַּמָּק יִהְיֶה וּפִרְחָם כָּאָבָק יַּעֲלֶה כִּי מָאֲסוּ אֵת תּוֹרַת יְהֹוָה צְבָאוֹת וְאֵת אִמְרַת קְדוֹשׁ יִשְׂרָאֵל נִאֵצוּ:
וַיִּשְׁכַּח יִשְׂרָאֵל אֶת עֹשֵׂהוּ וַיִּבֶן הֵיכָלוֹת וִיהוּדָה הִרְבָּה עָרִים בְּצֻרוֹת וְשִׁלַּחְתִּי אֵשׁ בְּעָרָיו וְאָכְלָה אַרְמְנֹתֶיהָ:
וּפָקַדְתִּי עֲלֵיכֶם כִּפְרִי מַעַלְלֵיכֶם נְאֻם יְהֹוָה וְהִצַּתִּי אֵשׁ בְּיַעְרָהּ וְאָכְלָה כָּל סְבִיבֶיהָ:
כִּלָּה יְהֹוָה אֶת חֲמָתוֹ שָׁפַךְ חֲרוֹן אַפּוֹ וַיַּצֶּת אֵשׁ בְּצִיּוֹן וַתֹּאכַל יְסוֹדֹתֶיהָ:
וְאִם לֹא תִשְׁמְעוּ אֵלַי לְקַדֵּשׁ אֶת יוֹם הַשַּׁבָּת וּלְבִלְתִּי שְׂאֵת מַשָּׂא וּבֹא בְּשַׁעֲרֵי יְרוּשָׁלַם בְּיוֹם הַשַּׁבָּת וְהִצַּתִּי אֵשׁ בִּשְׁעָרֶיהָ וְאָכְלָה אַרְמְנוֹת יְרוּשָׁלַם וְלֹא תִכְבֶּה:
