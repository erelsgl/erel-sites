מיהו "אור לגויים"?

ישנם 4 שירי עבד ה':

ישעיהו פרק מב

א  הֵן עַבְדִּי אֶתְמָךְ-בּוֹ, בְּחִירִי רָצְתָה נַפְשִׁי; נָתַתִּי רוּחִי עָלָיו, מִשְׁפָּט לַגּוֹיִם יוֹצִיא.   ב  לֹא יִצְעַק, וְלֹא יִשָּׂא; וְלֹא-יַשְׁמִיעַ בַּחוּץ, קוֹלוֹ.   ג  קָנֶה רָצוּץ לֹא יִשְׁבּוֹר, וּפִשְׁתָּה כֵהָה לֹא יְכַבֶּנָּה; לֶאֱמֶת, יוֹצִיא מִשְׁפָּט.   ד  לֹא יִכְהֶה וְלֹא יָרוּץ, עַד-יָשִׂים בָּאָרֶץ מִשְׁפָּט; וּלְתוֹרָתוֹ, אִיִּים יְיַחֵלוּ.  {פ}

ה  כֹּה-אָמַר הָאֵל יְהוָה, בּוֹרֵא הַשָּׁמַיִם וְנוֹטֵיהֶם, רֹקַע הָאָרֶץ, וְצֶאֱצָאֶיהָ; נֹתֵן נְשָׁמָה לָעָם עָלֶיהָ, וְרוּחַ לַהֹלְכִים בָּהּ.   ו  אֲנִי יְהוָה קְרָאתִיךָ בְצֶדֶק, וְאַחְזֵק בְּיָדֶךָ; וְאֶצָּרְךָ, וְאֶתֶּנְךָ לִבְרִית עָם--לְאוֹר גּוֹיִם.   ז לִפְקֹחַ, עֵינַיִם עִוְרוֹת; לְהוֹצִיא מִמַּסְגֵּר אַסִּיר, מִבֵּית כֶּלֶא יֹשְׁבֵי חֹשֶׁךְ.

ישעיהו פרק מט

א  שִׁמְעוּ אִיִּים אֵלַי, וְהַקְשִׁיבוּ לְאֻמִּים מֵרָחוֹק; יְהוָה מִבֶּטֶן קְרָאָנִי, מִמְּעֵי אִמִּי הִזְכִּיר שְׁמִי.   ב  וַיָּשֶׂם פִּי כְּחֶרֶב חַדָּה, בְּצֵל יָדוֹ הֶחְבִּיאָנִי; וַיְשִׂימֵנִי לְחֵץ בָּרוּר, בְּאַשְׁפָּתוֹ הִסְתִּירָנִי.   ג  וַיֹּאמֶר לִי, עַבְדִּי-אָתָּה--יִשְׂרָאֵל, אֲשֶׁר-בְּךָ אֶתְפָּאָר.   ד  וַאֲנִי אָמַרְתִּי לְרִיק יָגַעְתִּי, לְתֹהוּ וְהֶבֶל כֹּחִי כִלֵּיתִי; אָכֵן מִשְׁפָּטִי אֶת-יְהוָה, וּפְעֻלָּתִי אֶת-אֱלֹהָי.  {ס}

ה  וְעַתָּה אָמַר יְהוָה, יוֹצְרִי מִבֶּטֶן לְעֶבֶד לוֹ, לְשׁוֹבֵב יַעֲקֹב אֵלָיו, וְיִשְׂרָאֵל לא (לוֹ) יֵאָסֵף; וְאֶכָּבֵד בְּעֵינֵי יְהוָה, וֵאלֹהַי הָיָה עֻזִּי.   ו  וַיֹּאמֶר, נָקֵל מִהְיוֹתְךָ לִי עֶבֶד, לְהָקִים אֶת-שִׁבְטֵי יַעֲקֹב, ונצירי (וּנְצוּרֵי) יִשְׂרָאֵל לְהָשִׁיב; וּנְתַתִּיךָ לְאוֹר גּוֹיִם, לִהְיוֹת יְשׁוּעָתִי עַד-קְצֵה הָאָרֶץ.  {ס}

ז  כֹּה אָמַר-יְהוָה גֹּאֵל יִשְׂרָאֵל קְדוֹשׁוֹ, לִבְזֹה-נֶפֶשׁ לִמְתָעֵב גּוֹי לְעֶבֶד מֹשְׁלִים, מְלָכִים יִרְאוּ וָקָמוּ, שָׂרִים וְיִשְׁתַּחֲווּ--לְמַעַן יְהוָה אֲשֶׁר נֶאֱמָן, קְדֹשׁ יִשְׂרָאֵל וַיִּבְחָרֶךָּ.  {ס}

ח  כֹּה אָמַר יְהוָה, בְּעֵת רָצוֹן עֲנִיתִיךָ, וּבְיוֹם יְשׁוּעָה, עֲזַרְתִּיךָ; וְאֶצָּרְךָ, וְאֶתֶּנְךָ לִבְרִית עָם, לְהָקִים אֶרֶץ, לְהַנְחִיל נְחָלוֹת שֹׁמֵמוֹת.   ט  לֵאמֹר לַאֲסוּרִים צֵאוּ, לַאֲשֶׁר בַּחֹשֶׁךְ הִגָּלוּ; עַל-דְּרָכִים יִרְעוּ, וּבְכָל-שְׁפָיִים מַרְעִיתָם.

 

ישעיהו פרק נ

ד  אֲדֹנָי יְהוִה, נָתַן לִי לְשׁוֹן לִמּוּדִים, לָדַעַת לָעוּת אֶת-יָעֵף, דָּבָר; יָעִיר בַּבֹּקֶר בַּבֹּקֶר, יָעִיר לִי אֹזֶן, לִשְׁמֹעַ, כַּלִּמּוּדִים.   ה  אֲדֹנָי יְהוִה פָּתַח-לִי אֹזֶן, וְאָנֹכִי לֹא מָרִיתִי:  אָחוֹר, לֹא נְסוּגֹתִי.   ו  גֵּוִי נָתַתִּי לְמַכִּים, וּלְחָיַי לְמֹרְטִים; פָּנַי לֹא הִסְתַּרְתִּי, מִכְּלִמּוֹת וָרֹק.   ז  וַאדֹנָי יְהוִה יַעֲזָר-לִי, עַל-כֵּן לֹא נִכְלָמְתִּי; עַל-כֵּן שַׂמְתִּי פָנַי, כַּחַלָּמִישׁ, וָאֵדַע, כִּי-לֹא אֵבוֹשׁ.   ח  קָרוֹב, מַצְדִּיקִי--מִי-יָרִיב אִתִּי, נַעַמְדָה יָּחַד; מִי-בַעַל מִשְׁפָּטִי, יִגַּשׁ אֵלָי.   ט  הֵן אֲדֹנָי יְהוִה יַעֲזָר-לִי, מִי-הוּא יַרְשִׁיעֵנִי; הֵן כֻּלָּם כַּבֶּגֶד יִבְלוּ, עָשׁ יֹאכְלֵם.  {ס}

 

שיר רביעי: ישעיה נב 13-15, נג.

 

ישנה מחלוקת עתיקה האם עבד ה' המתואר בפרקים אלה הוא ישראל (כדעת רבים מהמפרשים היהודים) או המשיח (לדעת מפרשים יהודים אחרים וכן הנוצרים). לכאורה נראה ברור הזיהוי עם ישראל לפי המקור השני: ג וַיֹּאמֶר לִי, עַבְדִּי-אָתָּה--יִשְׂרָאֵל, אֲשֶׁר-בְּךָ אֶתְפָּאָר .   אולם המצב מורכב יותר:

במקור הראשון עבד ה' מוציא את ישראל מהגלות:

ו אֲנִי יְהוָה קְרָאתִיךָ בְצֶדֶק, וְאַחְזֵק בְּיָדֶךָ; וְאֶצָּרְךָ, וְאֶתֶּנְךָ לִבְרִית עָם--לְאוֹר גּוֹיִם .   זלִפְקֹחַ, עֵינַיִם עִוְרוֹת; לְהוֹצִיא מִמַּסְגֵּר אַסִּיר, מִבֵּית כֶּלֶא יֹשְׁבֵי חֹשֶׁךְ .

וכן גם במקור השני, שם ישנה הפרדה ברורה בין העבד ובין ישראל:

ו וַיֹּאמֶר, נָקֵל מִהְיוֹתְךָ לִי עֶבֶד, לְהָקִים אֶת-שִׁבְטֵי יַעֲקֹב, ונצירי (וּנְצוּרֵי) יִשְׂרָאֵל לְהָשִׁיב; וּנְתַתִּיךָ לְאוֹר גּוֹיִם, לִהְיוֹת יְשׁוּעָתִי עַד-קְצֵה הָאָרֶץ.  {ס }
