עַל כֵּן קָרָא שְׁמוֹ

אָבֵל מִצְרַיִם - אֵבֶל כָּבֵד זֶה לְמִצְרָיִם (בר' נ 11)
אֶבֶן הָעָזֶר - עַד הֵנָּה עֲזָרָנוּ יְיָ (שמ"א ז 12)
אֵל בֵּית אֵל - כִּי שָׁם נִגְלוּ אֵלָיו הָאֱלֹהִים בְּבָרְחוֹ מִפְּנֵי אָחִיו (בר' לה 7)
אֵל רֳאִי - הֲגַם הֲלֹם רָאִיתִי אַחֲרֵי רֹאִי (בר' טז 13), בְּאֵר לַחַי רֹאִי (בר' טז 14)
בְּאֵר שֶבַע - כִּי שָׁם נִשְׁבְּעוּ שְׁנֵיהֶם (בר' כא 31; שֶׁבַע כִּבְשׂת הַצֹּאן – כו 33)
בָּבֶל - כִּי שָׁם בָּלַל יְיָ שְׂפַת כָּל הָאָרֶץ (בר' יא 9)
בֵּית אֵל - אֵין זֶה כִּי אִם בֵּית אֱלֹהִים וְזֶה שַׁעַר הַשָּׁמָיִם (בר' ח 17)
בֹּכִים - וַיִּשְׂאוּ הָעָם אֶת קוֹלָם וַיִּבְכּוּ (שופ' ב 5-4)
בָּמָה - מָה הַבָּמָה אֲשֶׁר אַתֶּם הַבָּאִים שָׁם (יחז' כ 29)
בַּעַל פְּרָצִים - פָּרַץ יְיָ אֶת אֹיְבַי לְפָנַי כְּפֶרֶץ מָיִם (שמ"ב ה 20; דה"א יד 11)
גִּלְגָּל - הַיּוֹם גַּלּוֹתִי אֶת חֶרְפַּת מִצְרַיִם מֵעֲלֵיכֶם (יהו' ה 9)
גַּלְעֵד - הַגַּל הַזֶּה עֵד בֵּינִי וּבֵינְךָ הַיּוֹם (בר' לא 48)
דָּן (עיר) - בְּשֵׁם דָּן אֲבִיהֶם אֲשֶׁר יוּלַּד לְיִשְׂרָאֵל (שופ' יח 29)
חָרְמָה - וַיַּחֲרֵם אֶתְהֶם וְאֶת עָרֵיהֶם (במד' כא 3), וַיַּחֲרִימוּ אוֹתָהּ (שופ' א 17)
יְיָ יִרְאֶה - (הַר יְיָ יֵרָאֶה – בר' כב 14) אֱלֹהִים יִרְאֶה לּוֹ הַשֶּׂה לְעֹלָה (בר' כב 8)
יְיָ נִסִּי - (כִּי יָד עַל כֵּס יָהּ מִלְחָמָה לַיְיָ בַּעֲמָלֵק מִדֹּר דֹּר – שמ' יז 16-15)
מַחֲנָיִם - מַחֲנֵה אֱלֹהִים זֶה (בר' לב 3)
מַסָּה וּמְרִיבָה - עַל רִיב בְּנֵי יִשְׂרָאֵל וְעַל נַסֹּתָם אֶת יְיָ לֵאמֹר הֲיֵשׁ יְיָ בְּקִרְבֵּנוּ אִם אָיִן (שמ' יז 7)
הַמִּצְפָּה - יִצֶף יְיָ בֵּינִי וּבֵינֶךָ כִּי נִסָּתֵר אִישׁ מֵרֵעֵהוּ (בר' לא 49)
מָרָה - וְלֹא יָכְלוּ לִשְׁתֹּת מַיִם (מִמָּרָה) כִּי מָרִים הֵם (שמ' טו 23)
נֹבַח - וַיִּקְרָא לָה נֹבַח בִּשְׁמוֹ (במד' לב 42)
נַחַל אֶשְׁכּוֹל - עַל אֹדוֹת הָאֶשְׁכּוֹל אֲשֶׁר כָּרְתוּ מִשָּׁם בְּנֵי יִשְׂרָאֵל (במד' יג 24)
סֻכּוֹת - וּלְמִקְנֵהוּ עָשָׂה סֻכֹּת (בר' לג 17)
סֶלַע הַמַּחְלְקוֹת -? וַיֵּלֶךְ שָׁאוּל מִצַּד הָהָר מִזֶּה וְדָוִד וַאֲנָשָׁיו מִצַּד הָהָר מִזֶּה (שמ"א כג 28-26)
עֵין הַקּוֹרֵא - וַיִּקְרָא אֶל יְיָ (שופ' טו 19-18)
עִיר דָּוִיד - וַיֵּשֶׁב דָּוִיד בַּמְצָד (דה"א יא 6)
עֵמֶק בְּרָכָה - כִּי שָׁם בֵּרְכוּ אֶת יְיָ (דה"ב כ 26)
עֵמֶק עָכוֹר - מֶה עֲכַרְתָּנוּ יַעְכֳּרְךָ יְיָ בַּיּוֹם הַזֶּה (יהו' ז 25-24)
עֵשֶׂק - כִּי הִתְעַשְּׂקוּ עִמּוֹ (בר' כו 20)
פְּנִיאֵל (פְּנוּאֵל) - כִּי רָאִיתִי אֱלֹהִים פָּנִים אֶל פָּנִים וַתִּנָּצֵל נַפְשִׁי (בר' לב 31)
פֶּרֶץ עֻזָּה - עַל אֲשֶׁר פָּרַץ יְיָ פֶּרֶץ בְּעֻזָּה (שמ"ב ו 8)
פֶּרֶץ - מַה פָּרַצְתָּ עָלֶיךָ פָּרֶץ (בר' לח 29)
צוֹעַר - הָעִיר הַזֹּאת קְרֹבָה לָנוּס שָׁמָּה וְהִוא מִצְעָר הֲלֹא מִצְעָר הִוא (בר' יט 20)
קִבְרוֹת הַתַּאֲוָה - כִּי שָׁם קָבְרוּ אֶת הָעָם הַמִּתְאַוִּים (במד' יא 34)
רְחֹבוֹת - כִּי עַתָּה הִרְחִיב יְיָ לָנוּ וּפָרִינוּ בָאָרֶץ (בר' כו 22)
רָמַת לֶחִי - בִּלְחִי הַחֲמוֹר חֲמוֹר חֲמֹרָתָיִם בִּלְחִי הַחֲמוֹר הִכֵּיתִי אֶלֶף אִישׁ (שופ' טו 17-16)
שִׂטְנָה (וַיָּרִיבוּ גַּם עָלֶיהָ – בר' כו 21)
תַּבְעֵרָה - כִּי בָעֲרָה בָם אֵשׁ יְיָ (במד' יא 3)
