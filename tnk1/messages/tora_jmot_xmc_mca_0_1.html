(לט) וַיֹּאפוּ אֶת הַבָּצֵק אֲשֶׁר הוֹצִיאוּ מִמִּצְרַיִם עֻגֹת מַצּוֹת כִּי לֹא חָמֵץ כִּי גֹרְשׁוּ מִמִּצְרַיִם וְלֹא יָכְלוּ לְהִתְמַהְמֵהַּ וְגַם צֵדָה לֹא עָשׂוּ לָהֶם:

(מ) וּמוֹשַׁב בְּנֵי יִשְׂרָאֵל אֲשֶׁר יָשְׁבוּ בְּמִצְרָיִם שְׁלשִׁים שָׁנָה וְאַרְבַּע מֵאוֹת שָׁנָה:

(מא) וַיְהִי מִקֵּץ שְׁלשִׁים שָׁנָה וְאַרְבַּע מֵאוֹת שָׁנָה וַיְהִי בְּעֶצֶם הַיּוֹם הַזֶּה יָצְאוּ כָּל צִבְאוֹת יְהֹוָה מֵאֶרֶץ מִצְרָיִם:

(מב) לֵיל שִׁמֻּרִים הוּא לַיהֹוָה לְהוֹצִיאָם מֵאֶרֶץ מִצְרָיִם הוּא הַלַּיְלָה הַזֶּה לַיהֹוָה שִׁמֻּרִים לְכָל בְּנֵי יִשְׂרָאֵל לְדֹרֹתָם:
