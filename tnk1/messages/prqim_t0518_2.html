ספר עמוס פרק ג  (ז) כִּי לֹא יַעֲשֶׂה אֲדֹנָי יֱהֹוִה דָּבָר כִּי אִם גָּלָה סוֹדוֹ אֶל עֲבָדָיו הַנְּבִיאִים :
 שימו לב כי במקרא הקודש כתוב קודם כל עבדיו של האלהים ורק אחר כך נביאים:
1.  עבדיו
2.  נביאים  
לאמור כי קודם כל עליך להיות עבד יהוה ורק אחר כך נביא 
בלשון המקרא מבדילים בין נביא אמת לנביא שקר
 אוב וידעוני זה בשפה החילונית תקשור ומתקשרים שמאניזם ביאורגונומי ותטא הילינג וכל נביאי השקר למיניהם כי הם מתנבאים אך הם לא עבדי יהוה

ובספר ויקרא כתוב:
(לא) אַל תִּפְנוּ אֶל הָאֹבֹת וְאֶל הַיִּדְּעֹנִים אַל תְּבַקְשׁוּ לְטָמְאָה בָהֶם אֲנִי יְהֹוָה אֱלֹהֵיכֶם:
(ו) וְהַנֶּפֶשׁ אֲשֶׁר תִּפְנֶה אֶל הָאֹבֹת וְאֶל הַיִּדְּעֹנִים לִזְנֹת אַחֲרֵיהֶם וְנָתַתִּי אֶת פָּנַי בַּנֶּפֶשׁ הַהִוא וְהִכְרַתִּי אֹתוֹ מִקֶּרֶב עַמּוֹ:
(כז) וְאִישׁ אוֹ אִשָּׁה כִּי יִהְיֶה בָהֶם אוֹב אוֹ יִדְּעֹנִי מוֹת יוּמָתוּ בָּאֶבֶן יִרְגְּמוּ אֹתָם דְּמֵיהֶם בָּם:
ספר דברים פרק יח כתוב:
(י) לֹא יִמָּצֵא בְךָ מַעֲבִיר בְּנוֹ וּבִתּוֹ בָּאֵשׁ קֹסֵם קְסָמִים מְעוֹנֵן וּמְנַחֵשׁ וּמְכַשֵּׁף:
(יא) וְחֹבֵר חָבֶר וְשֹׁאֵל אוֹב וְיִדְּעֹנִי וְדֹרֵשׁ אֶל הַמֵּתִים:
(יב) כִּי תוֹעֲבַת יְהֹוָה כָּל עֹשֵׂה אֵלֶּה וּבִגְלַל הַתּוֹעֵבֹת הָאֵלֶּה יְהֹוָה אֱלֹהֶיךָ מוֹרִישׁ אוֹתָם מִפָּנֶיךָ:
(יג) תָּמִים תִּהְיֶה עִם יְהֹוָה אֱלֹהֶיךָ:
(יד) כִּי הַגּוֹיִם הָאֵלֶּה אֲשֶׁר אַתָּה יוֹרֵשׁ אוֹתָם אֶל מְעֹנֲנִים וְאֶל קֹסְמִים יִשְׁמָעוּ וְאַתָּה לֹא כֵן נָתַן לְךָ יְהֹוָה אֱלֹהֶיךָ:
(טו) נָבִיא מִקִּרְבְּךָ מֵאַחֶיךָ כָּמֹנִי יָקִים לְךָ יְהֹוָה אֱלֹהֶיךָ אֵלָיו תִּשְׁמָעוּן:
ספר מלכים ב פרק כא
(א) בֶּן שְׁתֵּים עֶשְׂרֵה שָׁנָה מְנַשֶּׁה בְמָלְכוֹ וַחֲמִשִּׁים וְחָמֵשׁ שָׁנָה מָלַךְ בִּירוּשָׁלָם וְשֵׁם אִמּוֹ חֶפְצִי בָהּ:
(ב) וַיַּעַשׂ הָרַע בְּעֵינֵי יְהֹוָה כְּתוֹעֲבֹת הַגּוֹיִם אֲשֶׁר הוֹרִישׁ יְהֹוָה מִפְּנֵי בְּנֵי יִשְׂרָאֵל:
(ג) וַיָּשָׁב וַיִּבֶן אֶת הַבָּמוֹת אֲשֶׁר אִבַּד חִזְקִיָּהוּ אָבִיו וַיָּקֶם מִזְבְּחֹת לַבַּעַל וַיַּעַשׂ אֲשֵׁרָה כַּאֲשֶׁר עָשָׂה אַחְאָב מֶלֶךְ יִשְׂרָאֵל וַיִּשְׁתַּחוּ לְכָל צְבָא הַשָּׁמַיִם וַיַּעֲבֹד אֹתָם:
(ד) וּבָנָה מִזְבְּחֹת בְּבֵית יְהֹוָה אֲשֶׁר אָמַר יְהֹוָה בִּירוּשָׁלַם אָשִׂים אֶת שְׁמִי:
(ה) וַיִּבֶן מִזְבְּחוֹת לְכָל צְבָא הַשָּׁמָיִם בִּשְׁתֵּי חַצְרוֹת בֵּית יְהֹוָה:
(ו) וְהֶעֱבִיר אֶת בְּנוֹ בָּאֵשׁ וְעוֹנֵן וְנִחֵשׁ וְעָשָׂה אוֹב וְיִדְּעֹנִים הִרְבָּה לַעֲשֹוֹת הָרַע בְּעֵינֵי יְהֹוָה לְהַכְעִיס:

ובספר שמואל א פרק כח כתוב:
(ג) וּשְׁמוּאֵל מֵת וַיִּסְפְּדוּ לוֹ כָּל יִשְׂרָאֵל וַיִּקְבְּרֻהוּ בָרָמָה וּבְעִירוֹ וְשָׁאוּל הֵסִיר הָאֹבוֹת וְאֶת הַיִּדְּעֹנִים מֵהָאָרֶץ
(ט) וַתֹּאמֶר הָאִשָּׁה אֵלָיו הִנֵּה אַתָּה יָדַעְתָּ אֵת אֲשֶׁר עָשָׂה שָׁאוּל אֲשֶׁר הִכְרִית אֶת הָאֹבוֹת וְאֶת הַיִּדְּעֹנִי מִן הָאָרֶץ וְלָמָה אַתָּה מִתְנַקֵּשׁ בְּנַפְשִׁי לַהֲמִיתֵנִי:

ובספר מלכים ב פרק כג כתוב על המלך יאשיהו הצדיק:
(כד) וְגַם אֶת הָאֹבוֹת וְאֶת הַיִּדְּעֹנִים וְאֶת הַתְּרָפִים וְאֶת הַגִּלֻּלִים וְאֵת כָּל הַשִּׁקֻּצִים אֲשֶׁר נִרְאוּ בְּאֶרֶץ יְהוּדָה וּבִירוּשָׁלַם בִּעֵר יֹאשִׁיָּהוּ לְמַעַן הָקִים אֶת דִּבְרֵי הַתּוֹרָה הַכְּתֻבִים עַל הַסֵּפֶר אֲשֶׁר מָצָא חִלְקִיָּהוּ הַכֹּהֵן בֵּית יְהֹוָה:

ובספר הנביא ישעיהו פרק ח כתוב:
(יט) וְכִי יֹאמְרוּ אֲלֵיכֶם דִּרְשׁוּ אֶל הָאֹבוֹת וְאֶל הַיִּדְּעֹנִים הַמְצַפְצְפִים וְהַמַּהְגִּים הֲלוֹא עַם אֶל אֱלֹהָיו יִדְרֹשׁ בְּעַד הַחַיִּים אֶל הַמֵּתִים:
וכי אלהים הוא אל חי ויש לדרוש אותו לבדו בדמות נביא האלהים ומי שדורש אל המתים אז האלהים שלו מת
וכי אבותינו חשבו תמיד להביט ביה בארץ החיים
וכך אמר בצער ובכאב הנביא ישעיהו בפרק לח ככתוב:

(יא) אָמַרְתִּי לֹא אֶרְאֶה יָהּ יָהּ בְּאֶרֶץ הַחַיִּים לֹא אַבִּיט אָדָם עוֹד עִם יוֹשְׁבֵי חָדֶל:

ויש אמנם חכמה למתים כי כאשר בספר איוב פרק כח כתוב:


(יב) וְהַחָכְמָה מֵאַיִן תִּמָּצֵא וְאֵי זֶה מְקוֹם בִּינָה:

(יג) לֹא יָדַע אֱנוֹשׁ עֶרְכָּהּ וְלֹא תִמָּצֵא בְּאֶרֶץ הַחַיִּים:

ואין לדרוש במתים גם אם חכמה להם ומי אשר דורש את חכמת המתים לא ימצא בארץ החיים
הלא בפרי עץ הדעת גם חכמה רבה הפוקחת עיניים אך אין לאכול מעץ הדעת כי כל רצונו של האלהים למען אשר תתהלך תמים לפניו ותמצא חן בעיניו ובספר קהלת פרק ט כתוב: (יא) שַׁבְתִּי וְרָאֹה תַחַת הַשֶּׁמֶשׁ כִּי לֹא לַקַּלִּים הַמֵּרוֹץ וְלֹא לַגִּבּוֹרִים הַמִּלְחָמָה וְגַם לֹא לַחֲכָמִים לֶחֶם וְגַם לֹא לַנְּבֹנִים עֹשֶׁר וְגַם לֹא לַיֹּדְעִים חֵן כִּי עֵת וָפֶגַע יִקְרֶה אֶת כֻּלָּם: לאמור כי אם כאשר חלילה אוכלים וחוטאים מפרי עץ הדעת אז מאבדים את החן בעיני הקדוש ברוך הוא וכל רצונו של הבורא יתברך הוא למען נמצא חן בעיניו ויש לעבוד את האלהים באמונה מבלי חטא עץ הדעת אז תהיה בנו דעת אלהים קדושים

הדעת היא הנחשת על כן הנחש אשר ערום הוא מכל חית השדה דעת בו אך לא לנו לנחש
