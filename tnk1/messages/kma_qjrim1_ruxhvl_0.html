וַיֵּצֵא הָרוּחַ וַיַּעֲמֹד לִפְנֵי יְהוָה וַיֹּאמֶר אֲנִי אֲפַתֶּנּוּ וַיֹּאמֶר יְהוָה אֵלָיו בַּמָּה
וַיֹּאמֶר אֵצֵא וְהָיִיתִי רוּחַ שֶׁקֶר בְּפִי כָּל-נְבִיאָיו
וַיֹּאמֶר תְּפַתֶּה וְגַם-תּוּכָל--צֵא וַעֲשֵׂה-כֵן...
וְעַתָּה הִנֵּה נָתַן יְהוָה רוּחַ שֶׁקֶר בְּפִי כָּל-נְבִיאֶיךָ אֵלֶּה
וַיהוָה דִּבֶּר עָלֶיךָ רָעָה...

וַיִּגַּשׁ צִדְקִיָּהוּ בֶן-כְּנַעֲנָה וַיַּכֶּה אֶת-מִיכָיְהוּ עַל-הַלֶּחִי
וַיֹּאמֶר אֵי-זֶה עָבַר רוּחַ-יְהוָה מֵאִתִּי לְדַבֵּר אוֹתָךְ
וַיֹּאמֶר מִיכָיְהוּ הִנְּךָ רֹאֶה בַּיּוֹם הַהוּא אֲשֶׁר תָּבֹא חֶדֶר בְּחֶדֶר לְהֵחָבֵה...

וַיֹּאמֶר מֶלֶךְ יִשְׂרָאֵל קַח אֶת-מִיכָיְהוּ וַהֲשִׁיבֵהוּ אֶל-אָמֹן שַׂר-הָעִיר וְאֶל-יוֹאָשׁ בֶּן-הַמֶּלֶךְ
וְאָמַרְתָּ כֹּה אָמַר הַמֶּלֶךְ שִׂימוּ אֶת-זֶה בֵּית הַכֶּלֶא
וְהַאֲכִלֻהוּ לֶחֶם לַחַץ וּמַיִם לַחַץ עַד בֹּאִי בְשָׁלוֹם...

וַיֹּאמֶר מִיכָיְהוּ--אִם-שׁוֹב תָּשׁוּב בְּשָׁלוֹם לֹא-דִבֶּר יְהוָה בִּי
וַיֹּאמֶר שִׁמְעוּ עַמִּים כֻּלָּם...
(מלכים א' כב' 28-21)
