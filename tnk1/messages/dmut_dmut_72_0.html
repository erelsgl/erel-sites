תולדותיה וחשיבותה של השפה העברית

קוד: תולדותיה וחשיבותה של השפה העברית בתנ"ך

סוג: מאמר

מאת: רמי ניר

אל:

כָּל-בְּנֵי-עֵבֶר ו/או אֵלֶּה תוֹלְדוֹת הָעִבְרִית

נֹחַ אִישׁ צַדִּיק תָּמִים הָיָה בְּדֹרֹתָיו (בראשית ו' 9)

שְׁלֹשִׁים וּשְׁמֹנֶה שָׁנָה--עַד-תֹּם כָּל-הַדּוֹר (דברים ב' 14)

"וַיִּהְיוּ כָּל-יְמֵי-נֹחַ תְּשַׁע מֵאוֹת שָׁנָה וַחֲמִשִּׁים שָׁנָה" [= 25 דּוֹרוֹת] (בראשית ט' 29)

"וַיּוֹלֶד נֹחַ אֶת-שֵׁם" (בראשית ה' 32)

"וּלְשֵׁם יֻלַּד גַּם-הוּא אֲבִי כָּל-בְּנֵי-עֵבֶר אֲחִי יֶפֶת הַגָּדוֹל... בְּנֵי שֵׁם עֵילָם וְאַשּׁוּר וְאַרְפַּכְשַׁד וְלוּד וַאֲרָם... וְאַרְפַּכְשַׁד יָלַד אֶת-שָׁלַח וְשֶׁלַח יָלַד אֶת-עֵבֶר... וּלְעֵבֶר יֻלַּד שְׁנֵי בָנִים: שֵׁם הָאֶחָד פֶּלֶג כִּי בְיָמָיו נִפְלְגָה הָאָרֶץ וְשֵׁם אָחִיו יָקְטָן" (בראשית י 21-25)

"וַיְהִי כָל-הָאָרֶץ שָׂפָה אֶחָת וּדְבָרִים אֲחָדִים... וַיֹּאמְרוּ הָבָה נִבְנֶה-לָּנוּ עִיר וּמִגְדָּל וְרֹאשׁוֹ בַשָּׁמַיִם וְנַעֲשֶׂה-לָּנוּ שֵׁם פֶּן-נָפוּץ עַל-פְּנֵי כָל-הָאָרֶץ... וַיֵּרֶד יְהוָה לִרְאֹת אֶת-הָעִיר וְאֶת-הַמִּגְדָּל אֲשֶׁר בָּנוּ בְּנֵי הָאָדָם וַיֹּאמֶר יְהוָה הֵן עַם אֶחָד וְשָׂפָה אַחַת לְכֻלָּם וְזֶה הַחִלָּם (מפענחים) לַעֲשׂוֹת וְעַתָּה לֹא-יִבָּצֵר מֵהֶם כֹּל אֲשֶׁר יָזְמוּ לַעֲשׂוֹת... הָבָה נֵרְדָה וְנָבְלָה שָׁם שְׂפָתָם אֲשֶׁר לֹא יִשְׁמְעוּ אִישׁ שְׂפַת רֵעֵהוּ וַיָּפֶץ יְהוָה אֹתָם מִשָּׁם עַל-פְּנֵי כָל-הָאָרֶץ וַיַּחְדְּלוּ לִבְנֹת הָעִיר עַל-כֵּן קָרָא שְׁמָהּ בָּבֶל כִּי-שָׁם בָּלַל יְהוָה שְׂפַת כָּל-הָאָרֶץ וּמִשָּׁם הֱפִיצָם יְהוָה עַל-פְּנֵי כָּל-הָאָרֶץ..." = [1758 שנה לבריאת אדם].  (בראשית יא' 1 - 9)

יחסים של 22/7 בְּמַעְגְּלֵי-צֶדֶק.

כלומר, בשנת -2408 לפנה"ס, "וַיּוֹלֶד עֵבֶר אֶת-פָּלֶג" = התחיל עֵבֶר לייסד את השפה הָעִבְרִית.


1. הָעִבְרִית = [22 אוֹתִיּוֹת] = שְׂפַת אֱלוֹהֵי צֶדֶק – כמתועד: "יַנְחֵנִי בְמַעְגְּלֵי-צֶדֶק לְמַעַן שְׁמוֹ" (תהלים כג' 3), "וַיֵּרָא יְהוָה אֶל-אַבְרָם וַיֹּאמֶר לְזַרְעֲךָ אֶתֵּן אֶת-הָאָרֶץ הַזֹּאת" (בראשית יב' 7), "וַיּוֹצֵא אֹתוֹ הַחוּצָה וַיֹּאמֶר הַבֶּט-נָא הַשָּׁמַיְמָה וּסְפֹר הַכּוֹכָבִים--אִם-תּוּכַל לִסְפֹּר אֹתָם וַיֹּאמֶר לוֹ כֹּה יִהְיֶה זַרְעֶךָ וְהֶאֱמִן בַּיהוָה וַיַּחְשְׁבֶהָ לּוֹ צְדָקָה" (בראשית טו' 5 - 6)

2. הָעִבְרִית היא תעודת הזהות של הָעִבְרִים – כמתועד: "וַיָּבֹא הַפָּלִיט וַיַּגֵּד לְאַבְרָם הָעִבְרִי וְהוּא שֹׁכֵן בְּאֵלֹנֵי מַמְרֵא" (בראשית יד' 13), "וַתִּקְרָא לְאַנְשֵׁי בֵיתָהּ וַתֹּאמֶר לָהֶם לֵאמֹר רְאוּ הֵבִיא לָנוּ אִישׁ עִבְרִי לְצַחֶק בָּנוּ בָּא אֵלַי לִשְׁכַּב עִמִּי וָאֶקְרָא בְּקוֹל גָּדוֹל וַיְהִי כְשָׁמְעוֹ כִּי-הֲרִימֹתִי קוֹלִי וָאֶקְרָא וַיַּעֲזֹב בִּגְדוֹ אֶצְלִי וַיָּנָס וַיֵּצֵא הַחוּצָה וַתַּנַּח בִּגְדוֹ אֶצְלָהּ עַד-בּוֹא אֲדֹנָיו אֶל-בֵּיתוֹ וַתְּדַבֵּר אֵלָיו כַּדְּבָרִים הָאֵלֶּה לֵאמֹר: בָּא-אֵלַי הָעֶבֶד הָעִבְרִי..."(בראשית לט' 14 - 17), "כִּי-גֻנֹּב גֻּנַּבְתִּי מֵאֶרֶץ הָעִבְרִים..." (בראשית מ' 15) "וְשָׁם אִתָּנוּ נַעַר עִבְרִי עֶבֶד לְשַׂר הַטַּבָּחִים..." (בראשית מא' 12), "וַיָּשִׂימוּ לוֹ לְבַדּוֹ וְלָהֶם לְבַדָּם וְלַמִּצְרִים הָאֹכְלִים אִתּוֹ לְבַדָּםכִּי לֹא יוּכְלוּן הַמִּצְרִים לֶאֱכֹל אֶת-הָעִבְרִים לֶחֶם..." (בראשית מג' 32), "וַיֹּאמֶר מֶלֶךְ מִצְרַיִם לַמְיַלְּדֹת הָעִבְרִיֹּת..." (שמות א' 15), "וַיֵּצֵא בַּיּוֹם הַשֵּׁנִי וְהִנֵּה שְׁנֵי-אֲנָשִׁים עִבְרִים נִצִּים..." (שמות ב' 13),

3. הָעִבְרִית הִיא שְׂפַת אֱלוֹהֵי אֱמֶת וְעוֹשֶׂה חֶסֶד – כמתועד: "וַיֹּאמְרוּ אֱלֹהֵי הָעִבְרִים נִקְרָא עָלֵינוּ..." (שמות ה' 3), "וַיְדַבֵּר אֱלֹהִים אֵת כָּל-הַדְּבָרִים הָאֵלֶּה לֵאמֹר אָנֹכִי יְהוָה אֱלֹהֶיךָ אֲשֶׁר הוֹצֵאתִיךָ מֵאֶרֶץ מִצְרַיִם מִבֵּית עֲבָדִים לֹא-יִהְיֶה לְךָ אֱלֹהִים אֲחֵרִים עַל-פָּנָי לֹא-תַעֲשֶׂה לְךָ פֶסֶל וְכָל-תְּמוּנָה אֲשֶׁר בַּשָּׁמַיִם מִמַּעַל וַאֲשֶׁר בָּאָרֶץ מִתָּחַת וַאֲשֶׁר בַּמַּיִם מִתַּחַת לָאָרֶץ לֹא-תִשְׁתַּחֲוֶה לָהֶם וְלֹא תָעָבְדֵם כִּי אָנֹכִי יְהוָה אֱלֹהֶיךָ אֵל קַנָּא פֹּקֵד עֲוֹן אָבֹת עַל-בָּנִים עַל-שִׁלֵּשִׁים וְעַל-רִבֵּעִים לְשֹׂנְאָי וְעֹשֶׂה חֶסֶד לַאֲלָפִים לְאֹהֲבַי וּלְשֹׁמְרֵי מִצְוֹתָי..." (שמות כ' 1 - 5)

4. הָעִבְרִית הִיא שְׂפַת מֹשֶׁה בֵּאֵר אֶת-הַתּוֹרָה – כמתועד: "כִּי תִקְנֶה עֶבֶד עִבְרִי שֵׁשׁ שָׁנִים יַעֲבֹד וּבַשְּׁבִעִת--יֵצֵא לַחָפְשִׁי חִנָּם..." (שמות כא' 2)

5. הָעִבְרִית הִיא שְׂפַת נֵצַח יִשְׂרָאֵל = שְׂפַת הָאֱמֶת הַהִיסְטוֹרִית – כמתועד: "וַיֹּאמְרוּ מֶה קוֹל הַתְּרוּעָה הַגְּדוֹלָה הַזֹּאת בְּמַחֲנֵה הָעִבְרִים וַיֵּדְעוּ--כִּי אֲרוֹן יְהוָה בָּא אֶל-הַמַּחֲנֶה..." (שמואל א' ד' 6) "וַיַּךְ יוֹנָתָן אֵת נְצִיב פְּלִשְׁתִּים אֲשֶׁר בְּגֶבַע וַיִּשְׁמְעוּ פְּלִשְׁתִּים וְשָׁאוּל תָּקַע בַּשּׁוֹפָר בְּכָל-הָאָרֶץ לֵאמֹר יִשְׁמְעוּ הָעִבְרִים..." (שמואל א' יג' 3), "וְחָרָשׁ לֹא יִמָּצֵא בְּכֹל אֶרֶץ יִשְׂרָאֵל כִּי-אמר (אָמְרוּ) פְלִשְׁתִּים--פֶּן יַעֲשׂוּ הָעִבְרִים חֶרֶב אוֹ חֲנִית..." (שמואל א' יג' 19).

6. הָעִבְרִית הִיא שְׂפַת הַנְּבִיאִים – כמתועד: "וַיֹּאמֶר אֲלֵיהֶם עִבְרִי אָנֹכִי וְאֶת-יְהוָה אֱלֹהֵי הַשָּׁמַיִם אֲנִי יָרֵא..." (יונה א' 9), "לְשַׁלַּח אִישׁ אֶת-עַבְדּוֹ וְאִישׁ אֶת-שִׁפְחָתוֹ  הָעִבְרִי וְהָעִבְרִיָּה--חָפְשִׁים לְבִלְתִּי עֲבָד-בָּם בִּיהוּדִי אָחִיהוּ אִישׁ..." (ירמיה לד' 9)
