רוח החיה באופנים

ספר יחזקאל פרק א כתוב:
(כ) עַל אֲשֶׁר יִהְיֶה שָּׁם הָרוּחַ לָלֶכֶת יֵלֵכוּ שָׁמָּה הָרוּחַ לָלֶכֶת וְהָאוֹפַנִּים יִנָּשְׂאוּ לְעֻמָּתָם כִּי רוּחַ הַחַיָּה בָּאוֹפַנִּים:
(כא) בְּלֶכְתָּם יֵלֵכוּ וּבְעָמְדָם יַעֲמֹדוּ וּבְהִנָּשְׂאָם מֵעַל הָאָרֶץ יִנָּשְׂאוּ הָאוֹפַנִּים לְעֻמָּתָם כִּי רוּחַ הַחַיָּה בָּאוֹפַנִּים:

ספר בראשית פרק א כתוב:
(כו) וַיֹּאמֶר אֱלֹהִים נַעֲשֶׂה אָדָם בְּצַלְמֵנוּ כִּדְמוּתֵנוּ וְיִרְדּוּ בִדְגַת הַיָּם וּבְעוֹף הַשָּׁמַיִם וּבַבְּהֵמָה וּבְכָל הָאָרֶץ וּבְכָל הָרֶמֶשׂ הָרֹמֵשׂ עַל הָאָרֶץ:
כי אדם הראשון בצלם אלהים כדמות אלהים והאדם נפש חיה הוא ורוח החיה באופנים גם לאדם וגם לחית השדה כי האופנים הם בצלם כדמות החיה וכל נפש חיה עם האופנים שלה כי כל יצור חי עם האופן שלו ואם יהיה יצור חי נעלה יותר אז בהתאם כך גם האופנים שלו
אך לאדם אופנים גבוהים ונעלים יותר מכל החיות לכן אלהים המשיל את האדם על כל נפש חיה ככתוב בספר תהילות פרק ח:
(ה) מָה אֱנוֹשׁ כִּי תִזְכְּרֶנּוּ וּבֶן אָדָם כִּי תִפְקְדֶנּוּ:
(ו) וַתְּחַסְּרֵהוּ מְּעַט מֵאֱלֹהִים וְכָבוֹד וְהָדָר תְּעַטְּרֵהוּ:
(ז) תַּמְשִׁילֵהוּ בְּמַעֲשֵׂי יָדֶיךָ כֹּל שַׁתָּה תַחַת רַגְלָיו:
(ח) צֹנֶה וַאֲלָפִים כֻּלָּם וְגַם בַּהֲמוֹת שָׂדָי:
(ט) צִפּוֹר שָׁמַיִם וּדְגֵי הַיָּם עֹבֵר אָרְחוֹת יַמִּים:
(י) יְהֹוָה אֲדֹנֵינוּ מָה אַדִּיר שִׁמְךָ בְּכָל הָאָרֶץ:

אך אם אדם משחית את דמותו נפשו אז צלמו נשחת גם הוא ואלהים ישיב עליו אופן לאמור את צלם הדמות המתאימה לאדם הנחות יותר שהתנהג כאחת החיות
ועל פי ספר משלי פרק כ כתוב: (כו) מְזָרֶה רְשָׁעִים מֶלֶךְ חָכָם וַיָּשֶׁב עֲלֵיהֶם אוֹפָן: לאמור השיב עליהם את צלם הדמות המאימה לרשעים ואם נבאר למי שאינו מבין שפת מקרא אז הכוונה שהגנים התורשתיים של הDNA של הרשעים יהיה נחות יותר ואז החיה תגבר על האדם הנחות הזה והאדם יפחד מכל חיה רעה הנמצאת בבני האדם הרשעים וזאת במקום תחת שכתוב בספר בראשית לנח ובניו:(ב) וּמוֹרַאֲכֶם וְחִתְּכֶם יִהְיֶה עַל כָּל חַיַּת הָאָרֶץ וְעַל כָּל עוֹף הַשָּׁמָיִם בְּכֹל אֲשֶׁר תִּרְמֹשׂ הָאֲדָמָה וּבְכָל דְּגֵי הַיָּם בְּיֶדְכֶם נִתָּנוּ:

לכן אם ההיפך מכך אז כל החיות העופות והדגים הנזכרות בפסוק הזה יאכלו את האדם הרע במקום שהאדם יאכל אותם לכן בתורת משה בספר דברים פרק כח נכתב קללה לרשע שבאדם:(כו) וְהָיְתָה נִבְלָתְךָ לְמַאֲכָל לְכָל עוֹף הַשָּׁמַיִם וּלְבֶהֱמַת הָאָרֶץ וְאֵין מַחֲרִיד:
כאשר הקדוש ברוך הוא כחכמת מלך המלכים משיב אופן על כל חי אז הוא עושה חסד עמו כי מתאים לכל נפש חי את אופן צלם לדמות כי אם נפש הכלב בצלם הכלב אז הכלב חיה מקסימה ואכן אהובה על האדם אך אם אדם ירד לדרגת נפש הכלב אז האדם רע מאד לכן מקללים אותו יא-כלב אך אדגיש שוב כי כשמושב אופן לחיות אז כמובן שהחיות שכולנו רואים הם מקסימות ואין כמוני האוהב את בעלי החיים
אַךְ בְּצֶלֶם יִתְהַלֶּךְ אִישׁ לֹא יִכּוֹן אָדָם בְּרֶשַׁע לכן במות האדם ביום במשפט ביום הדין ישיב יהוה במשפט צדק לכל נפש חי את האופן שלו ועם האופן הזה הוא יורד לעולם ומתחיל מבחן חדש לחיים חדשים והתהליך הזה נעשה כל יום יוֹם לְיוֹם יַבִּיעַ אֹמֶר וְלַיְלָה לְּלַיְלָה יְחַוֶּה דָּעַת
מי יתן ויהוה אלהים ישיב לנו את משיח-יהוה כאדם בצלם וכדמות אלהים ורוח נפש החיה באדם באופני הקודש
