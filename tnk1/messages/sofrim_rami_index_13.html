לחכות לאלהים כי "שֵׁם-יְהוָה" - כי "מָשִׁיחַ" לא יבוא

"וְהָיוּ לִי אָמַר יְהוָה צְבָאוֹת"
"לַיּוֹם אֲשֶׁר אֲנִי עֹשֶׂה סְגֻלָּה"
"וְחָמַלְתִּי עֲלֵיהֶם"
"כַּאֲשֶׁר יַחְמֹל אִישׁ עַל-בְּנוֹ הָעֹבֵד אֹתוֹ"...
(מלאכי ג' 17)

חֶמְלָה אלהית זו מכוונת גם כלפי "משיחיסטים" המבססים את אמונתם על חזון "הִנֵּה מַלְכֵּךְ יָבוֹא לָךְ":

"גִּילִי מְאֹד בַּת-צִיּוֹן"
"הָרִיעִי בַּת יְרוּשָׁלִַם"
"הִנֵּה מַלְכֵּךְ יָבוֹא לָךְ"
"צַדִּיק וְנוֹשָׁע הוּא"
"עָנִי וְרֹכֵב עַל-חֲמוֹר"
"וְעַל-עַיִר בֶּן-אֲתֹנוֹת"

"וְהִכְרַתִּי-רֶכֶב מֵאֶפְרַיִם"
"וְסוּס מִירוּשָׁלִַם"
"וְנִכְרְתָה קֶשֶׁת מִלְחָמָה"
"וְדִבֶּר שָׁלוֹם לַגּוֹיִם"
"וּמָשְׁלוֹ מִיָּם עַד-יָם"
"וּמִנָּהָר עַד-אַפְסֵי-אָרֶץ"...
(זכריה ט' 10-9)

חזון זה "הִנֵּה מַלְכֵּךְ יָבוֹא לָךְ"...קדם לחזון "יִכָּרֵת מָשִׁיחַ" אשר תיעד דניאל:

"שָׁבֻעִים שִׁבְעִים נֶחְתַּךְ עַל-עַמְּךָ וְעַל-עִיר קָדְשֶׁךָ"
"לְכַלֵּא הַפֶּשַׁע וּלְהָתֵם חַטָּאת וּלְכַפֵּר עָוֹן"
"וּלְהָבִיא צֶדֶק עֹלָמִים"

"וְלַחְתֹּם חָזוֹן וְנָבִיא"
"וְלִמְשֹׁחַ קֹדֶשׁ קָדָשִׁים"

"וְתֵדַע וְתַשְׂכֵּל מִן-מֹצָא דָבָר"
"לְהָשִׁיב וְלִבְנוֹת יְרוּשָׁלִַם"
"עַד-מָשִׁיחַ נָגִיד--שָׁבֻעִים שִׁבְעָה"
"וְשָׁבֻעִים שִׁשִּׁים וּשְׁנַיִם"
"תָּשׁוּב וְנִבְנְתָה רְחוֹב וְחָרוּץ"
"וּבְצוֹק הָעִתִּים"

"וְאַחֲרֵי הַשָּׁבֻעִים שִׁשִּׁים וּשְׁנַיִם"
"יִכָּרֵת מָשִׁיחַ וְאֵין לוֹ"
"וְהָעִיר וְהַקֹּדֶשׁ יַשְׁחִית עַם נָגִיד הַבָּא"
"וְקִצּוֹ בַשֶּׁטֶף"
"וְעַד קֵץ מִלְחָמָה נֶחֱרֶצֶת"
"שֹׁמֵמוֹת"...
(דניאל ט' 26-24)

חזון זה (דניאל ט' 26-24) מבוסס על נבואת ירמיה:

"הַעֶצֶב נִבְזֶה נָפוּץ הָאִישׁ הַזֶּה כָּנְיָהוּ"
"אִם-כְּלִי אֵין חֵפֶץ בּוֹ"
"מַדּוּעַ הוּטְלוּ הוּא וְזַרְעוֹ"
"וְהֻשְׁלְכוּ עַל-הָאָרֶץ אֲשֶׁר לֹא-יָדָעוּ"

"אֶרֶץ אֶרֶץ אָרֶץ"
"שִׁמְעִי דְּבַר-יְהוָה"
"כֹּה אָמַר יְהוָה"
"כִּתְבוּ אֶת-הָאִישׁ הַזֶּה עֲרִירִי"
"גֶּבֶר לֹא-יִצְלַח בְּיָמָיו"
"כִּי לֹא יִצְלַח מִזַּרְעוֹ אִישׁ יֹשֵׁב עַל-כִּסֵּא דָוִד וּמֹשֵׁל עוֹד בִּיהוּדָה"...
(ירמיה כב' 30-28)

ואמנם, אחרי חזון "יִכָּרֵת מָשִׁיחַ וְאֵין לוֹ"...נבנה המקדש השני אשר היה בית חסר כבוד...כאשר "לוחות הברית" נשארו בקבר כספת "אבן השתייה"...באותם ימים התחזקה בעם היהודי אמונת ההבל ב"תחיית המתים"...המאמינים חיברו אל התקיימות אמונת הבל זו...אמונת הבל על בואו של "מָשִׁיחַ" בשר ודם...אולם, כן יהיה..."וְדָוִד עַבְדִּי נָשִׂיא לָהֶם לְעוֹלָם"...(יחזקאל לז' 25)


ואלה הן נבואות האמת אשר תועדו - לחכות לאלהים...כי "שֵׁם-יְהוָה" - ישעיה:

1. "בֵּית יַעֲקֹב--לְכוּ וְנֵלְכָה בְּאוֹר יְהוָה"...
(ב' 5)

2. "כִּי יוֹם לַיהוָה צְבָאוֹת עַל כָּל-גֵּאֶה--וָרָם וְעַל כָּל-נִשָּׂא וְשָׁפֵל"...
(ב' 12)

3. "וְשַׁח גַּבְהוּת הָאָדָם"
"וְשָׁפֵל רוּם אֲנָשִׁים"
"וְנִשְׂגַּב יְהוָה לְבַדּוֹ"
"בַּיּוֹם הַהוּא"
"וְהָאֱלִילִים כָּלִיל יַחֲלֹף"...
(ב' 18-17)

4. "וְהָיָה בַּיּוֹם הַהוּא"
"לֹא-יוֹסִיף עוֹד שְׁאָר יִשְׂרָאֵל וּפְלֵיטַת בֵּית-יַעֲקֹב לְהִשָּׁעֵן עַל-מַכֵּהוּ"
"וְנִשְׁעַן עַל-יְהוָה קְדוֹשׁ יִשְׂרָאֵל--בֶּאֱמֶת"...
(י' 20)

5. "וְאָמַר בַּיּוֹם הַהוּא"
"הִנֵּה אֱלֹהֵינוּ זֶה קִוִּינוּ לוֹ וְיוֹשִׁיעֵנוּ"
"זֶה יְהוָה קִוִּינוּ לוֹ"
"נָגִילָה וְנִשְׂמְחָה בִּישׁוּעָתוֹ"...
(כה' 9)

6. "הִנֵּה שֵׁם-יְהוָה"
"בָּא מִמֶּרְחָק"
"בֹּעֵר אַפּוֹ"
"וְכֹבֶד מַשָּׂאָה"
"שְׂפָתָיו מָלְאוּ זַעַם"
"וּלְשׁוֹנוֹ כְּאֵשׁ אֹכָלֶת"...
(ל' 27)

7. "קוֹל אֹמֵר קְרָא"
"וְאָמַר מָה אֶקְרָא"
"כָּל-הַבָּשָׂר חָצִיר"
"וְכָל-חַסְדּוֹ כְּצִיץ הַשָּׂדֶה"

"יָבֵשׁ חָצִיר נָבֵל צִיץ"
"כִּי רוּחַ יְהוָה נָשְׁבָה בּוֹ"
"אָכֵן חָצִיר הָעָם"

יָבֵשׁ חָצִיר נָבֵל צִיץ"
"וּדְבַר-אֱלֹהֵינוּ יָקוּם לְעוֹלָם"...
(מ' 8-6)

8. "וְאֶל-מִי תְדַמְּיוּנִי וְאֶשְׁוֶה"
"יֹאמַר קָדוֹשׁ"
"שְׂאוּ-מָרוֹם עֵינֵיכֶם"
"וּרְאוּ מִי-בָרָא אֵלֶּה"
"הַמּוֹצִיא בְמִסְפָּר צְבָאָם"
"לְכֻלָּם בְּשֵׁם יִקְרָא"
"מֵרֹב אוֹנִים"
"וְאַמִּיץ כֹּחַ"
"אִישׁ לֹא נֶעְדָּר"...
(מ' 26-25)

9. "קוּמִי אוֹרִי כִּי בָא אוֹרֵךְ"
"וּכְבוֹד יְהוָה עָלַיִךְ זָרָח"...
(ס' 1)

10. "כִּי-הִנֵּה יְהוָה בָּאֵשׁ יָבוֹא"
"וְכַסּוּפָה מַרְכְּבֹתָיו"
"לְהָשִׁיב בְּחֵמָה אַפּוֹ"
"וְגַעֲרָתוֹ בְּלַהֲבֵי-אֵשׁ"
"כִּי בָאֵשׁ יְהוָה נִשְׁפָּט"
"וּבְחַרְבּוֹ אֶת-כָּל-בָּשָׂר"
"וְרַבּוּ חַלְלֵי יְהוָה"...
(סו' 16-15)

ירמיה:

11. "אִם-תָּשׁוּב יִשְׂרָאֵל נְאֻם-יְהוָה אֵלַי תָּשׁוּב"...
(ד' 1)

12. "כִּי אִם-בְּזֹאת יִתְהַלֵּל הַמִּתְהַלֵּל"
"הַשְׂכֵּל וְיָדֹעַ אוֹתִי"
"כִּי אֲנִי יְהוָה"
"עֹשֶׂה חֶסֶד מִשְׁפָּט וּצְדָקָה בָּאָרֶץ"
"כִּי-בְאֵלֶּה חָפַצְתִּי"
"נְאֻם-יְהוָה"...
(ט' 23)

13. "מֵאֵין כָּמוֹךָ יְהוָה"
"גָּדוֹל אַתָּה"
"וְגָדוֹל שִׁמְךָ בִּגְבוּרָה"...
(י' 6)

14. "כֹּה אָמַר יְהוָה"
"אָרוּר הַגֶּבֶר אֲשֶׁר יִבְטַח בָּאָדָם"
"וְשָׂם בָּשָׂר זְרֹעוֹ"
"וּמִן-יְהוָה יָסוּר לִבּוֹ"

"וְהָיָה כְּעַרְעָר בָּעֲרָבָה"
"וְלֹא יִרְאֶה כִּי-יָבוֹא טוֹב"
"וְשָׁכַן חֲרֵרִים בַּמִּדְבָּר"
"אֶרֶץ מְלֵחָה וְלֹא תֵשֵׁב"...
(יז' 6-5)

15. "בָּרוּךְ הַגֶּבֶר אֲשֶׁר יִבְטַח בַּיהוָה"
"וְהָיָה יְהוָה מִבְטַחוֹ"
"וְהָיָה כְּעֵץ שָׁתוּל עַל-מַיִם"
"וְעַל-יוּבַל יְשַׁלַּח שָׁרָשָׁיו"
"וְלֹא יִרָא כִּי-יָבֹא חֹם"
"וְהָיָה עָלֵהוּ רַעֲנָן"
"וּבִשְׁנַת בַּצֹּרֶת לֹא יִדְאָג"
"וְלֹא יָמִישׁ מֵעֲשׂוֹת פֶּרִי"...
(יז' 8-7)

16. "לֹא יָשׁוּב אַף-יְהוָה"
"עַד-עֲשֹׂתוֹ וְעַד-הֲקִימוֹ מְזִמּוֹת לִבּוֹ"
"בְּאַחֲרִית הַיָּמִים תִּתְבּוֹנְנוּ בָהּ בִּינָה"...
(כג' 20)

17. "וִהְיִיתֶם לִי לְעָם"
"וְאָנֹכִי אֶהְיֶה לָכֶם לֵאלֹהִים"...
(ל' 22)

18. "וְלֹא יְלַמְּדוּ עוֹד אִישׁ אֶת-רֵעֵהוּ"
"וְאִישׁ אֶת-אָחִיו לֵאמֹר"
"דְּעוּ אֶת-יְהוָה"

"כִּי-כוּלָּם יֵדְעוּ אוֹתִי"
"לְמִקְּטַנָּם וְעַד-גְּדוֹלָם"
"נְאֻם-יְהוָה"
"כִּי אֶסְלַח לַעֲוֹנָם"
"וּלְחַטָּאתָם לֹא אֶזְכָּר-עוֹד"...
(לא' 33)

19. "הִנְנִי מַעֲלֶה-לָּהּ אֲרֻכָה וּמַרְפֵּא"
"וּרְפָאתִים"
"וְגִלֵּיתִי לָהֶם עֲתֶרֶת שָׁלוֹם וֶאֱמֶת"...
(לג' 6)

20. "הוֹצִיא יְהוָה אֶת-צִדְקֹתֵינוּ"
"בֹּאוּ וּנְסַפְּרָה בְצִיּוֹן אֶת-מַעֲשֵׂה יְהוָה אֱלֹהֵינוּ"...
(נא' 10)

יחזקאל:

21. "לְמַעַן בְּחֻקֹּתַי יֵלֵכוּ"
"וְאֶת-מִשְׁפָּטַי יִשְׁמְרוּ"
"וְעָשׂוּ אֹתָם"
"וְהָיוּ-לִי לְעָם"

(יא' 20)

22. "וַהֲשִׁבֹתִי אֶת-יָדִי"
"וָאַעַשׂ לְמַעַן שְׁמִי"
"לְבִלְתִּי הֵחֵל לְעֵינֵי הַגּוֹיִם"

23. "וִידַעְתֶּם כִּי-אֲנִי יְהוָה"

"אֶל-הָאָרֶץ אֲשֶׁר נָשָׂאתִי אֶת-יָדִי לָתֵת אוֹתָהּ לַאֲבוֹתֵיכֶם"
"וּזְכַרְתֶּם-שָׁם אֶת-דַּרְכֵיכֶם וְאֵת כָּל-עֲלִילוֹתֵיכֶם אֲשֶׁר נִטְמֵאתֶם בָּם"

"וּנְקֹטֹתֶם בִּפְנֵיכֶם"
"בְּכָל-רָעוֹתֵיכֶם אֲשֶׁר עֲשִׂיתֶם"
"וִידַעְתֶּם כִּי-אֲנִי יְהוָה"
"בַּעֲשׂוֹתִי אִתְּכֶם לְמַעַן שְׁמִי"

"לֹא כְדַרְכֵיכֶם הָרָעִים"
"וְכַעֲלִילוֹתֵיכֶם הַנִּשְׁחָתוֹת"
"בֵּית יִשְׂרָאֵל"
"נְאֻם אֲדֹנָי יְהוִה"...
(כ' 44-42)

24. "וְהִרְבֵּיתִי עֲלֵיכֶם אָדָם וּבְהֵמָה"
"וְרָבוּ וּפָרוּ"
"וְהוֹשַׁבְתִּי אֶתְכֶם כְּקַדְמוֹתֵיכֶם"
"וְהֵיטִבֹתִי מֵרִאשֹׁתֵיכֶם"
"וִידַעְתֶּם כִּי-אֲנִי יְהוָה"...
(לו' 11)

25. "לָכֵן אֱמֹר לְבֵית-יִשְׂרָאֵל"
"כֹּה אָמַר אֲדֹנָי יְהוִה"
"לֹא לְמַעַנְכֶם אֲנִי עֹשֶׂה בֵּית יִשְׂרָאֵל"
" כִּי אִם-לְשֵׁם-קָדְשִׁי"
"אֲשֶׁר חִלַּלְתֶּם בַּגּוֹיִם אֲשֶׁר-בָּאתֶם שָׁם"

"וְקִדַּשְׁתִּי אֶת-שְׁמִי הַגָּדוֹל"
"הַמְחֻלָּל בַּגּוֹיִם אֲשֶׁר חִלַּלְתֶּם בְּתוֹכָם"
"וְיָדְעוּ הַגּוֹיִם כִּי-אֲנִי יְהוָה"
"נְאֻם אֲדֹנָי יְהוִה"
"בְּהִקָּדְשִׁי בָכֶם לְעֵינֵיהֶם"...
(לו' 23-22)

..."הַנָּשִׂיא"...

26. "וְעַבְדִּי דָוִד מֶלֶךְ עֲלֵיהֶם"
"וְרוֹעֶה אֶחָד יִהְיֶה לְכֻלָּם"
"וּבְמִשְׁפָּטַי יֵלֵכוּ"
"וְחֻקּוֹתַי יִשְׁמְרוּ וְעָשׂוּ אוֹתָם"
"וְיָשְׁבוּ עַל-הָאָרֶץ אֲשֶׁר נָתַתִּי לְעַבְדִּי לְיַעֲקֹב"
"אֲשֶׁר יָשְׁבוּ-בָהּ אֲבוֹתֵיכֶם"
"וְיָשְׁבוּ עָלֶיהָ הֵמָּה וּבְנֵיהֶם וּבְנֵי בְנֵיהֶם עַד-עוֹלָם"
"וְדָוִד עַבְדִּי נָשִׂיא לָהֶם לְעוֹלָם"...
(לז' 25-24)

27. "וְכָרַתִּי לָהֶם בְּרִית שָׁלוֹם"
"בְּרִית עוֹלָם יִהְיֶה אוֹתָם"
"וּנְתַתִּים וְהִרְבֵּיתִי אוֹתָם"
"וְנָתַתִּי אֶת-מִקְדָּשִׁי בְּתוֹכָם לְעוֹלָם"

"וְהָיָה מִשְׁכָּנִי עֲלֵיהֶם"
"וְהָיִיתִי לָהֶם לֵאלֹהִים"
"וְהֵמָּה יִהְיוּ-לִי לְעָם"

"וְיָדְעוּ הַגּוֹיִם"
"כִּי אֲנִי יְהוָה"
"מְקַדֵּשׁ אֶת-יִשְׂרָאֵל"
"בִּהְיוֹת מִקְדָּשִׁי בְּתוֹכָם לְעוֹלָם"...
(לז' 28-26)

28. "וְאֶת-שֵׁם קָדְשִׁי"
"אוֹדִיעַ בְּתוֹךְ עַמִּי יִשְׂרָאֵל"
"וְלֹא-אַחֵל אֶת-שֵׁם-קָדְשִׁי עוֹד"
"וְיָדְעוּ הַגּוֹיִם כִּי-אֲנִי יְהוָה קָדוֹשׁ בְּיִשְׂרָאֵל"
"הִנֵּה בָאָה וְנִהְיָתָה"
"נְאֻם אֲדֹנָי יְהוִה"
"הוּא הַיּוֹם אֲשֶׁר דִּבַּרְתִּי"...
(לט' 8-7)

29. "וְנָתַתִּי אֶת-כְּבוֹדִי בַּגּוֹיִם"
"וְרָאוּ כָל-הַגּוֹיִם אֶת-מִשְׁפָּטִי אֲשֶׁר עָשִׂיתִי"
"וְאֶת-יָדִי אֲשֶׁר-שַׂמְתִּי בָהֶם"
"וְיָדְעוּ בֵּית יִשְׂרָאֵל"
"כִּי אֲנִי יְהוָה אֱלֹהֵיהֶם"
"מִן-הַיּוֹם הַהוּא וָהָלְאָה"...
(לט' 22-21)


30. "וְעַל-הַנָּשִׂיא יִהְיֶה הָעוֹלוֹת וְהַמִּנְחָה וְהַנֵּסֶךְ בַּחַגִּים וּבֶחֳדָשִׁים וּבַשַּׁבָּתוֹת בְּכָל-מוֹעֲדֵי בֵּית יִשְׂרָאֵל"...
(מה' 17)

31. "וּבְבוֹא הַנָּשִׂיא--דֶּרֶךְ אוּלָם הַשַּׁעַר יָבוֹא וּבְדַרְכּוֹ יֵצֵא"...
(מו' 8)

32. "וְלֹא-יִקַּח הַנָּשִׂיא מִנַּחֲלַת הָעָם לְהוֹנֹתָם מֵאֲחֻזָּתָם"...
(מח' 18)

33. "וּמֵאֲחֻזַּת הַלְוִיִּם וּמֵאֲחֻזַּת הָעִיר בְּתוֹךְ אֲשֶׁר לַנָּשִׂיא יִהְיֶה"
"בֵּין גְּבוּל יְהוּדָה וּבֵין גְּבוּל בִּנְיָמִן"
"לַנָּשִׂיא יִהְיֶה"...
(מח' 22)
