גלוח פאת הזקן לא משחית את פאת הזקן

מאת: מורה לתורה משה דין

אל: DAIAN1 @ 013.NET

נכתב ב: 18:05:38  16.02.2011, כתוספת/תגובה ל: ויקרא יט

בספר ויקרא פרק יט כתוב (כז) לֹא תַקִּפוּ פְּאַת רֹאשְׁכֶם וְלֹא תַשְׁחִית אֵת פְּאַת זְקָנֶךָ:
ובספר ויקרא יג (מא) וְאִם מִפְּאַת פָּנָיו יִמָּרֵט רֹאשׁוֹ גִּבֵּחַ הוּא טָהוֹר הוּא:
לאמור כי הגלוח מביא לטהרה
ובספר במדבר פרק ו כתוב על הנזיר וְגִלַּח רֹאשׁוֹ בְּיוֹם טָהֳרָתוֹ בַּיּוֹם הַשְּׁבִיעִי יְגַלְּחֶנּוּ
גם הלוים התגלחו וְהֶעֱבִירוּ תַעַר עַל כָּל בְּשָׂרָם וְכִבְּסוּ בִגְדֵיהֶם וְהִטֶּהָרוּ וגם אבשלום בן-דוד התגלח
גם יוסף הצדיק התגלח ככתוב בספר בראשית פרק מא (יד) וַיִּשְׁלַח פַּרְעֹה וַיִּקְרָא אֶת יוֹסֵף וַיְרִיצֻהוּ מִן הַבּוֹר וַיְגַלַּח וַיְחַלֵּף שִׂמְלֹתָיו וַיָּבֹא אֶל פַּרְעֹה:
כל בני ישראל התגלחו לכן עם ישראל יכול להתגלח מלבד הכהנים אשר כתוב להם במפורש לא להתגלח
תחשבו הלא התורה נתנה לכל בני ישראל אז מדוע לא די לכתוב את הפסוק "לא תשחית את פאת זקנך"
ומדוע יש לכתוב בנוסף לכהנים בספר ויקרא כא (ה) לֹא יִקְרְחֻה }יִקְרְחוּ{ קָרְחָה בְּרֹאשָׁם וּפְאַת זְקָנָם לֹא יְגַלֵּחוּ
ובכן יש הבדל בין השחתת הזקן לבין גלוח
בגלוח פאת הזקן אין השחתה כי פאת הזקן ממשיכה לצמוח
אך אם יסירו את השערות בלייזר לאמור יעקרו אותן מן השורש אז הזקן לא יוכל להמשיך לצמוח לכן זו השחתה של הזקן
השחתת פאת הזקן אסורה לכל בני ישראל
וגלוח פאת הזקן אסור רק לכהנים בני אהרן
