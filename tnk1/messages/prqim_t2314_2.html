ונסתם (בימינו) כאשר נסתם מפני הרעש בימי עזיה מלך-יהודה

מאת: רמי ניר

נכתב ב: 04:53:30  05.02.2010, כתוספת/תגובה ל: זכריה יד

תעודת "נצח ישראל" = "האמת ההיסטורית" מוכיחה ומודיעה על התגשמות חזון - כדלקמן:

דִּבְרֵי עָמוֹס אֲשֶׁר-הָיָה בַנּוֹקְדִים מִתְּקוֹעַ
אֲשֶׁר חָזָה עַל-יִשְׂרָאֵל בִּימֵי עֻזִּיָּה מֶלֶךְ-יְהוּדָה
וּבִימֵי יָרָבְעָם בֶּן-יוֹאָשׁ מֶלֶךְ יִשְׂרָאֵל
שְׁנָתַיִם לִפְנֵי הָרָעַשׁ...
(עמוס א' 1)

להבדיל מרעש בימינו אם בהאיטי או בתאילנד או באינדונזיה או בקליפורניה עליהם לא מתריע נביא (לא חודש ולא שנה לפני בואם)... את הרעש בִּימֵי עֻזִּיָּה מֶלֶךְ-יְהוּדָה וּבִימֵי יָרָבְעָם בֶּן-יוֹאָשׁ מֶלֶךְ יִשְׂרָאֵל חָזָה עָמוֹס "שְׁנָתַיִם לִפְנֵי הָרָעַשׁ"...

כלומר, "דִּבְרֵי עָמוֹס"... כי יש לצפות "שְׁנָתַיִם" לרעש זה (אשר התגשם בשנה 14 לעוזיה) עם מלאת 700 שנה ליציאת מצרים (עמוס ז' 8) "הִנְנִי שָׂם אֲנָךְ בְּקֶרֶב עַמִּי יִשְׂרָאֵל--לֹא-אוֹסִיף עוֹד עֲבוֹר לוֹ"... מייחדים את האירוע של תחילת חורבן ממלכת ישראל כתיעוד "אמת היסטורית" ואכן התגשמות "הָרָעַשׁ" באה בזמן אמת... ואמנם, הרעש גם מלווה היה בצונאמי גדול:
(עמוס ט' 6) "הַקּוֹרֵא לְמֵי-הַיָּם וַיִּשְׁפְּכֵם עַל-פְּנֵי הָאָרֶץ--יְהוָה שְׁמוֹ"...

ואכן, התגשם חזון חורבן ממלכת ישראל:

וַיְהִי בַּשָּׁנָה הָרְבִיעִית לַמֶּלֶךְ חִזְקִיָּהוּ
הִיא הַשָּׁנָה הַשְּׁבִיעִית לְהוֹשֵׁעַ בֶּן-אֵלָה מֶלֶךְ יִשְׂרָאֵל
עָלָה שַׁלְמַנְאֶסֶר מֶלֶךְ-אַשּׁוּר עַל-שׁוֹמְרוֹן--וַיָּצַר עָלֶיהָ
וַיִּלְכְּדֻהָ מִקְצֵה שָׁלֹשׁ שָׁנִים
בִּשְׁנַת-שֵׁשׁ לְחִזְקִיָּה הִיא שְׁנַת-תֵּשַׁע לְהוֹשֵׁעַ מֶלֶךְ יִשְׂרָאֵל נִלְכְּדָה שׁוֹמְרוֹן
וַיֶּגֶל מֶלֶךְ-אַשּׁוּר אֶת-יִשְׂרָאֵל אַשּׁוּרָה וַיַּנְחֵם בַּחְלַח וּבְחָבוֹר נְהַר גּוֹזָן--וְעָרֵי מָדָי
עַל אֲשֶׁר לֹא-שָׁמְעוּ בְּקוֹל יְהוָה אֱלוֹהֵיהֶם
וַיַּעַבְרוּ אֶת-בְּרִיתוֹ אֵת כָּל-אֲשֶׁר צִוָּה מֹשֶׁה עֶבֶד יְהוָה
וְלֹא שָׁמְעוּ וְלֹא עָשׂוּ...
(מלכים ב' יח' 9 - 12)


כך, יש לראות "אמת היסטורית" בחפיפה המתועדת בספר זכריה (יד') שייעודה להוכיח עוצמה של "רַעַשׁ גָּדוֹל" הצפוי לסיים את המלחמה האחרונה, אשר תתגשם בימינו, "עַל אַדְמַת יִשְׂרָאֵל":

(זכריה יד' 5) "וְנַסְתֶּם כַּאֲשֶׁר נַסְתֶּם מִפְּנֵי הָרַעַשׁ בִּימֵי עֻזִּיָּה מֶלֶךְ-יְהוּדָה וּבָא יְהוָה אֱלוֹהַי"...


ואכן (יחזקאל לח' 18 - 20) מחזק את עיקרון "האמת ההיסטורית" הייחודית אשר תביא (בימינו) "רַעַשׁ גָּדוֹל" כאות להתגשמות סיום המלחמה האחרונה:

וְהָיָה בַּיּוֹם הַהוּא
בְּיוֹם בּוֹא גוֹג עַל-אַדְמַת יִשְׂרָאֵל--נְאֻם אֲדוֹנָי יְהוִה
תַּעֲלֶה חֲמָתִי בְּאַפִּי וּבְקִנְאָתִי בְאֵשׁ-עֶבְרָתִי דִּבַּרְתִּי:
אִם-לֹא בַּיּוֹם הַהוּא יִהְיֶה רַעַשׁ גָּדוֹל עַל אַדְמַת יִשְׂרָאֵל...

וְרָעֲשׁוּ מִפָּנַי דְּגֵי הַיָּם וְעוֹף הַשָּׁמַיִם וְחַיַּת הַשָּׂדֶה
וְכָל-הָרֶמֶשׂ הָרוֹמֵשׂ עַל-הָאֲדָמָה וְכוֹל הָאָדָם אֲשֶׁר עַל-פְּנֵי הָאֲדָמָה
וְנֶהֶרְסוּ הֶהָרִים
וְנָפְלוּ הַמַּדְרֵגוֹת
וְכָל-חוֹמָה לָאָרֶץ תִּפּוֹל"...


או אז, (דניאל יב' 7) יתגשם (בימינו) החזון שהבטיח "הָאִישׁ לְבוּשׁ הַבַּדִּים":

וָאֶשְׁמַע אֶת-הָאִישׁ לְבוּשׁ הַבַּדִּים אֲשֶׁר מִמַּעַל לְמֵימֵי הַיְאוֹר
וַיָּרֶם יְמִינוֹ וּשְׂמוֹאלוֹ אֶל-הַשָּׁמַיִם וַיִּשָּׁבַע בְּחֵי הָעוֹלָם
כִּי לְמוֹעֵד מוֹעֲדִים וָחֵצִי וּכְכַלּוֹת נַפֵּץ יַד-עַם-קוֹדֶשׁ--תִּכְלֶינָה כָל-אֵלֶּה...

כלומר, יתגשם (בימינו) החזון... תנופץ "אבן השתייה" ויוצאו מקבר "כספת האבן"... ארון הברית ובו "לוחות הברית"...
