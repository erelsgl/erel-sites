אראל שלום
"נגע" מעבר להיותו בגוף יכול להיות גם ברוח
וגם מעבר להיותו פוגע ומזיק יכול גם "הנגע" להיטיב
ובספר דניאל פרק י כתוב:
ח) וַאֲנִי נִשְׁאַרְתִּי לְבַדִּי וָאֶרְאֶה אֶת הַמַּרְאָה הַגְּדֹלָה הַזֹּאת וְלֹא נִשְׁאַר בִּי כֹּח וְהוֹדִי נֶהְפַּךְ עָלַי לְמַשְׁחִית וְלֹא עָצַרְתִּי כֹּחַ:
(ט) וָאֶשְׁמַע אֶת קוֹל דְּבָרָיו וּכְשָׁמְעִי אֶת קוֹל דְּבָרָיו וַאֲנִי הָיִיתִי נִרְדָּם עַל פָּנַי וּפָנַי אָרְצָה:
(י) וְהִנֵּה יָד נָגְעָה בִּי וַתְּנִיעֵנִי עַל בִּרְכַּי וְכַפּוֹת יָדָי:
(יח) וַיֹּסֶף וַיִּגַּע בִּי כְּמַרְאֵה אָדָם וַיְחַזְּקֵנִי:

ובספר מלכים א פרק יט כתוב:
(ה) וַיִּשְׁכַּב וַיִּישַׁן תַּחַת רֹתֶם אֶחָד וְהִנֵּה זֶה מַלְאָךְ נֹגֵעַ בּוֹ וַיֹּאמֶר לוֹ קוּם אֱכוֹל:
(ו) וַיַּבֵּט וְהִנֵּה מְרַאֲשֹׁתָיו עֻגַת רְצָפִים וְצַפַּחַת מָיִם וַיֹּאכַל וַיֵּשְׁתְּ וַיָּשָׁב וַיִּשְׁכָּב:
(ז) וַיָּשָׁב מַלְאַךְ יְהֹוָה שֵׁנִית וַיִּגַּע בּוֹ וַיֹּאמֶר קוּם אֱכֹל כִּי רַב מִמְּךָ הַדָּרֶךְ:
(ח) וַיָּקָם וַיֹּאכַל וַיִּשְׁתֶּה וַיֵּלֶךְ בְּכֹחַ הָאֲכִילָה הַהִיא אַרְבָּעִים יוֹם וְאַרְבָּעִים לַיְלָה עַד הַר הָאֱלֹהִים חֹרֵב:

לאמור כי לגעת "הנגע" יכול להיות גם ברוח ולטובה או לרעה וגם בגוף לטובה או לרעה
וגם לא בגוף כי "נגע" זה מהמילה "לגעת" וכי יכולים שני דברים לגעת אחד בשני כי "נגע" לגעת זה קרבה עד אפס מקום לאמור "להשיג"
למשל בספרי עזרא ונחמיה כתוב:
(עב) וַיֵּשְׁבוּ הַכֹּהֲנִים וְהַלְוִיִּם וְהַשּׁוֹעֲרִים וְהַמְשֹׁרְרִים וּמִן הָעָם וְהַנְּתִינִים וְכָל יִשְׂרָאֵל בְּעָרֵיהֶם וַיִּגַּע הַחֹדֶשׁ הַשְּׁבִיעִי וּבְנֵי יִשְׂרָאֵל בְּעָרֵיהֶם:
לאמור כי החודש קרב עד אפס מקום ללא רווח עד אשר "הדביק" ולכן "הגיע" "נגע"
ולכן אומרים "תזהר אל תדביק אותי" לאדם חולה נגוע
אך יש הבדל בין המילה "לגעת" ובין המילה "לדבוק"
כי "לדבוק" מהמילה "דבק" זה לגעת מבלי להרפות לאמור "נגע" [אם זה במובן רע] שלא מרפה שלא מרפא ולכן כולם מפחדים "להדבק"
או "נגע" [אם זה במובן טוב] כאשר נגע אלהים ללבך ואתה דבק בו כי בספר שמואל א פרק י כתוב:
(כו) וְגַם שָׁאוּל הָלַךְ לְבֵיתוֹ גִּבְעָתָה וַיֵּלְכוּ עִמּוֹ הַחַיִל אֲשֶׁר נָגַע אֱלֹהִים בְּלִבָּם:
ובספר בראשית פרק 31 כתוב:
כג) וַיִּקַּח אֶת אֶחָיו עִמּוֹ וַיִּרְדֹּף אַחֲרָיו דֶּרֶךְ שִׁבְעַת יָמִים וַיַּדְבֵּק אֹתוֹ בְּהַר הַגִּלְעָד:
(כד) וַיָּבֹא אֱלֹהִים אֶל לָבָן הָאֲרַמִּי בַּחֲלֹם הַלָּיְלָה וַיֹּאמֶר לוֹ הִשָּׁמֶר לְךָ פֶּן תְּדַבֵּר עִם יַעֲקֹב מִטּוֹב עַד רָע:
(כה) וַיַּשֵּׂג לָבָן אֶת יַעֲקֹב וְיַעֲקֹב תָּקַע אֶת אָהֳלוֹ בָּהָר וְלָבָן תָּקַע אֶת אֶחָיו בְּהַר הַגִּלְעָד:
במקרה זה לבן רדף אחרי יעקב "הגיע" אליו ו"הדביק" את יעקב לאמור "נגע ולא חלף" לאמור מבלי להרפות ועל כן כתוב:(כה) וַיַּשֵּׂג לָבָן אֶת יַעֲקֹב וְיַעֲקֹב תָּקַע אֶת אָהֳלוֹ בָּהָר וְלָבָן תָּקַע אֶת אֶחָיו בְּהַר הַגִּלְעָד:
כי אדם שנגע והגיע למשהו השיג אותו ואם יתמיד "ידבק" בו
