(יג) וְלֹא מֵעֵבֶר לַיָּם הִוא לֵאמֹר מִי יַעֲבָר לָנוּ אֶל עֵבֶר הַיָּם וְיִקָּחֶהָ לָּנוּ וְיַשְׁמִעֵנוּ אֹתָהּ וְנַעֲשֶׂנָּה:

(ל) הִנֵּה יָמִים בָּאִים נְאֻם יְהֹוָה וְכָרַתִּי אֶת בֵּית יִשְׂרָאֵל וְאֶת בֵּית יְהוּדָה בְּרִית חֲדָשָׁה:
(לא) לֹא כַבְּרִית אֲשֶׁר כָּרַתִּי אֶת אֲבוֹתָם בְּיוֹם הֶחֱזִיקִי בְיָדָם לְהוֹצִיאָם מֵאֶרֶץ מִצְרָיִם אֲשֶׁר הֵמָּה הֵפֵרוּ אֶת בְּרִיתִי וְאָנֹכִי בָּעַלְתִּי בָם נְאֻם יְהֹוָה:
(לב) כִּי זֹאת הַבְּרִית אֲשֶׁר אֶכְרֹת אֶת בֵּית יִשְׂרָאֵל אַחֲרֵי הַיָּמִים הָהֵם נְאֻם יְהֹוָה נָתַתִּי אֶת תּוֹרָתִי בְּקִרְבָּם וְעַל לִבָּם אֶכְתֲּבֶנָּה וְהָיִיתִי לָהֶם לֵאלֹהִים וְהֵמָּה יִהְיוּ לִי לְעָם:

מורה לתורה משה דין משיב:
ובכן ברית-יהוה אמנם כתובה על לוחות-הברית אך ברית יהוה תורת משה תהיה בקרב בית עם ישראל ועל לבם יכתבנה אלהים כי נהיה לו לעם
אך אם אתה רמי ניר ממאן להקים את דברי תורת משה עד שתגיע לקבר "אבן השתייה" אשר אתה בודה מלבך אז איש לא ימנע את הכסילות הזאת ממך
ולידיעתך בית האלהים האחרון כבר היה

(יא) וָאֹמַר עַד מָתַי אֲדֹנָי וַיֹּאמֶר עַד אֲשֶׁר אִם שָׁאוּ עָרִים מֵאֵין יוֹשֵׁב וּבָתִּים מֵאֵין אָדָם וְהָאֲדָמָה תִּשָּׁאֶה שְׁמָמָה
ובספר ישעיהו פרק ד כתוב:
(ג) וְהָיָה הַנִּשְׁאָר בְּצִיּוֹן וְהַנּוֹתָר בִּירוּשָׁלִַם קָדוֹשׁ יֵאָמֶר לוֹ כָּל הַכָּתוּב לַחַיִּים בִּירוּשָׁלִָם:
(ד) אִם רָחַץ אֲדֹנָי אֵת צֹאַת בְּנוֹת צִיּוֹן וְאֶת דְּמֵי יְרוּשָׁלִַם יָדִיחַ מִקִּרְבָּהּ בְּרוּחַ מִשְׁפָּט וּבְרוּחַ בָּעֵר:
(ה) וּבָרָא יְהֹוָה עַל כָּל מְכוֹן הַר צִיּוֹן וְעַל מִקְרָאֶהָ עָנָן יוֹמָם וְעָשָׁן וְנֹגַהּ אֵשׁ לֶהָבָה לָיְלָה כִּי עַל כָּל כָּבוֹד חֻפָּה:
(ו) וְסֻכָּה תִּהְיֶה לְצֵל יוֹמָם מֵחֹרֶב וּלְמַחְסֶה וּלְמִסְתּוֹר מִזֶּרֶם וּמִמָּטָר
כי לא יהיה בית אבן ואָוֶן כִּי נָבָל נְבָלָה יְדַבֵּר וְלִבּוֹ יַעֲשֶׂה אָוֶן וגם תוחלתו היא לאבן [השתייה]
