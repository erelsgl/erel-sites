ספר בראשית פרק כז כתוב :
ויאמר אליו יצחק אביו גשה-נא ושקה-לי בני. 27 ויגש וישק-לו וירח את-ריח בגדיו ויברכהו ויאמר ראה ריח בני כריח שדה אשר ברכו יהוה. 28 ויתן-לך האלהים מטל השמים ומשמני הארץ ורב דגן ותירש. 1 29 יעבדוך עמים וישתחו לך לאמים הוה גביר לאחיך וישתחוו לך בני אמך ארריך ארור ומברכיך ברוך.

שימו לב כי כבר על פי הריח של הבגדים יצחק אבינו מברך את יעקב
הבגדים על פי התורה זה הגוף עצמו
ויגש וישק-לו וירח את-ריח בגדיו ויברכהו ויאמר ראה ריח בני כריח שדה אשר ברכו יהוה.
וזה מזכיר את בגדי הקודש של הכהן עם אבני החושן כִּי עִם אַבְנֵי הַשָּׂדֶה בְרִיתֶךָ וְחַיַּת הַשָּׂדֶה הָשְׁלְמָה לָךְ
כל בני האדם אוהבים אדם שיש לו ריח-גוף טוב לכן עלינו להטהר במעשינו כי כשאנחנו טהורים אז הטומאה המתבטאת בריח-גוף רע סרה מעלינו
בתורת משה איש האלהים בספר דברים פרק כא כתוב :

(טו) כִּי תִהְיֶיןָ לְאִישׁ שְׁתֵּי נָשִׁים הָאַחַת אֲהוּבָה וְהָאַחַת שְׂנוּאָה וְיָלְדוּ לוֹ בָנִים הָאֲהוּבָה וְהַשְּׂנוּאָה וְהָיָה הַבֵּן הַבְּכֹר לַשְּׂנִיאָה:
(טז) וְהָיָה בְּיוֹם הַנְחִילוֹ אֶת בָּנָיו אֵת אֲשֶׁר יִהְיֶה לוֹ לֹא יוּכַל לְבַכֵּר אֶת בֶּן הָאֲהוּבָה עַל פְּנֵי בֶן הַשְּׂנוּאָה הַבְּכֹר:
(יז) כִּי אֶת הַבְּכֹר בֶּן הַשְּׂנוּאָה יַכִּיר לָתֶת לוֹ פִּי שְׁנַיִם בְּכֹל אֲשֶׁר יִמָּצֵא לוֹ כִּי הוּא רֵאשִׁית אֹנוֹ לוֹ מִשְׁפַּט הַבְּכֹרָה:

אבל כידוע יעקב אבינו לא נתן הבכורה לבני לאה השנואה הגם כי ראובן חלל את יצועי אביו ובזאת אבדה בכורתו
הלא יכלה הבכורה לעבור לשמעון הבא בתור או אחריו ללוי או אחריו ליהודה אבל הבכורה ליוסף בן רחל האהובה

ספר דברי הימים א פרק ה כתוב :
(א) וּבְנֵי רְאוּבֵן בְּכוֹר יִשְׂרָאֵל כִּי הוּא הַבְּכוֹר וּבְחַלְּלוֹ יְצוּעֵי אָבִיו נִתְּנָה בְּכֹרָתוֹ לִבְנֵי יוֹסֵף בֶּן יִשְׂרָאֵל וְלֹא לְהִתְיַחֵשׂ לַבְּכֹרָה:
(ב) כִּי יְהוּדָה גָּבַר בְּאֶחָיו וּלְנָגִיד מִמֶּנּוּ וְהַבְּכֹרָה לְיוֹסֵף :

כי רחל לקחה את ריח דודאי ראובן הבכור והעבירה את ריח דודאי הבכורה ליוסף [בנה הנולד לה ראשון מאז השפעת הדודאים]

בספר שיר השירים פרק ז כתוב:
(יד) הַדּוּדָאִים נָתְנוּ רֵיחַ וְעַל פְּתָחֵינוּ כָּל מְגָדִים חֲדָשִׁים גַּם יְשָׁנִים דּוֹדִי צָפַנְתִּי לָךְ:
ספר בראשית פרק ל כתוב:
(יד) וַיֵּלֶךְ רְאוּבֵן בִּימֵי קְצִיר חִטִּים וַיִּמְצָא דוּדָאִים בַּשָּׂדֶה וַיָּבֵא אֹתָם אֶל לֵאָה אִמּוֹ וַתֹּאמֶר רָחֵל אֶל לֵאָה תְּנִי נָא לִי מִדּוּדָאֵי בְּנֵךְ :
(טו) וַתֹּאמֶר לָהּ הַמְעַט קַחְתֵּךְ אֶת אִישִׁי וְלָקַחַת גַּם אֶת דּוּדָאֵי בְּנִי וַתֹּאמֶר רָחֵל לָכֵן יִשְׁכַּב עִמָּךְ הַלַּיְלָה תַּחַת דּוּדָאֵי בְנֵךְ:
(טז) וַיָּבֹא יַעֲקֹב מִן הַשָּׂדֶה בָּעֶרֶב וַתֵּצֵא לֵאָה לִקְרָאתוֹ וַתֹּאמֶר אֵלַי תָּבוֹא כִּי שָׂכֹר שְׂכַרְתִּיךָ בְּדוּדָאֵי בְּנִי וַיִּשְׁכַּב עִמָּהּ בַּלַּיְלָה הוּא:
(יז) וַיִּשְׁמַע אֱלֹהִים אֶל לֵאָה וַתַּהַר וַתֵּלֶד לְיַעֲקֹב בֵּן חֲמִישִׁי:
(יח) וַתֹּאמֶר לֵאָה נָתַן אֱלֹהִים שְׂכָרִי אֲשֶׁר נָתַתִּי שִׁפְחָתִי לְאִישִׁי וַתִּקְרָא שְׁמוֹ יִשָּׂשׂכָר:
(יט) וַתַּהַר עוֹד לֵאָה וַתֵּלֶד בֵּן שִׁשִּׁי לְיַעֲקֹב:
(כ) וַתֹּאמֶר לֵאָה זְבָדַנִי אֱלֹהִים אֹתִי זֶבֶד טוֹב הַפַּעַם יִזְבְּלֵנִי אִישִׁי כִּי יָלַדְתִּי לוֹ שִׁשָּׁה בָנִים וַתִּקְרָא אֶת שְׁמוֹ זְבֻלוּן:
(כא) וְאַחַר יָלְדָה בַּת וַתִּקְרָא אֶת שְׁמָהּ דִּינָה:
(כב) וַיִּזְכֹּר אֱלֹהִים אֶת רָחֵל וַיִּשְׁמַע אֵלֶיהָ אֱלֹהִים וַיִּפְתַּח אֶת רַחְמָהּ:
(כג) וַתַּהַר וַתֵּלֶד בֵּן וַתֹּאמֶר אָסַף אֱלֹהִים אֶת חֶרְפָּתִי:
(כד) וַתִּקְרָא אֶת שְׁמוֹ יוֹסֵף לֵאמֹר יֹסֵף יְהֹוָה לִי בֵּן אַחֵר:

ספר תהילות פרק מה כתוב:
(א) לַמְנַצֵּחַ עַל שֹׁשַׁנִּים לִבְנֵי קֹרַח מַשְׂכִּיל שִׁיר יְדִידֹת
(ב) רָחַשׁ לִבִּי דָּבָר טוֹב אֹמֵר אָנִי מַעֲשַׂי לְמֶלֶךְ לְשׁוֹנִי עֵט סוֹפֵר מָהִיר
(ג) יָפְיָפִיתָ מִבְּנֵי אָדָם הוּצַק חֵן בְּשִׂפְתוֹתֶיךָ עַל כֵּן בֵּרַכְךָ אֱלֹהִים לְעוֹלָם
(ד) חֲגוֹר חַרְבְּךָ עַל יָרֵךְ גִּבּוֹר הוֹדְךָ וַהֲדָרֶךָ
(ה) וַהֲדָרְךָ צְלַח רְכַב עַל דְּבַר אֱמֶת וְעַנְוָה צֶדֶק וְתוֹרְךָ נוֹרָאוֹת יְמִינֶךָ
(ו) חִצֶּיךָ שְׁנוּנִים עַמִּים תַּחְתֶּיךָ יִפְּלוּ בְּלֵב אוֹיְבֵי הַמֶּלֶךְ
(ז) כִּסְאֲךָ אֱלֹהִים עוֹלָם וָעֶד שֵׁבֶט מִישֹׁר שֵׁבֶט מַלְכוּתֶךָ
(ח) אָהַבְתָּ צֶּדֶק וַתִּשְׂנָא רֶשַׁע עַל כֵּן מְשָׁחֲךָ אֱלֹהִים אֱלֹהֶיךָ שֶׁמֶן שָׂשׂוֹן מֵחֲבֵרֶךָ
(ט) מֹר וַאֲהָלוֹת קְצִיעוֹת כָּל בִּגְדֹתֶיךָ מִן הֵיכְלֵי שֵׁן מִנִּי שִׂמְּחוּךָ
(י) בְּנוֹת מְלָכִים בִּיקְּרוֹתֶיךָ נִצְּבָה שֵׁגַל לִימִינְךָ בְּכֶתֶם אוֹפִיר
(יא) שִׁמְעִי בַת וּרְאִי וְהַטִּי אָזְנֵךְ וְשִׁכְחִי עַמֵּךְ וּבֵית אָבִיךְ
(יב) וְיִתְאָו הַמֶּלֶךְ יָפְיֵךְ כִּי הוּא אֲדֹנַיִךְ וְהִשְׁתַּחֲוִי לוֹ
(יג) וּבַת צֹר בְּמִנְחָה פָּנַיִךְ יְחַלּוּ עֲשִׁירֵי עָם
(יד) כָּל כְּבוּדָּה בַת מֶלֶךְ פְּנִימָה מִמִּשְׁבְּצוֹת זָהָב לְבוּשָׁהּ
(טו) לִרְקָמוֹת תּוּבַל לַמֶּלֶךְ בְּתוּלוֹת אַחֲרֶיהָ רֵעוֹתֶיהָ מוּבָאוֹת לָךְ
(טז) תּוּבַלְנָה בִּשְׂמָחֹת וָגִיל תְּבֹאֶינָה בְּהֵיכַל מֶלֶךְ
(יז) תַּחַת אֲבֹתֶיךָ יִהְיוּ בָנֶיךָ תְּשִׁיתֵמוֹ לְשָׂרִים בְּכָל הָאָרֶץ
(יח) אַזְכִּירָה שִׁמְךָ בְּכָל דֹּר וָדֹר עַל כֵּן עַמִּים יְהוֹדוּךָ לְעֹלָם וָעֶד:

  בת מלך קדוש בת אלהי ישראל  וּבַת צֹר יקרא לה כי ריח בגדיה [גופה] כְּרֵיחַ לְבָנוֹן:

ספר שיר השירים פרק ד כתוב:
(ח) אִתִּי מִלְּבָנוֹן כַּלָּה אִתִּי מִלְּבָנוֹן תָּבוֹאִי תָּשׁוּרִי מֵרֹאשׁ אֲמָנָה מֵרֹאשׁ שְׂנִיר וְחֶרְמוֹן מִמְּעֹנוֹת אֲרָיוֹת מֵהַרְרֵי נְמֵרִים
(ט) לִבַּבְתִּנִי אֲחֹתִי כַלָּה לִבַּבְתִּינִי בְּאַחַד {בְּאַחַת} מֵעֵינַיִךְ בְּאַחַד עֲנָק מִצַּוְּרֹנָיִךְ
(י) מַה יָּפוּ דֹדַיִךְ אֲחֹתִי כַלָּה מַה טֹּבוּ דֹדַיִךְ מִיַּיִן וְרֵיחַ שְׁמָנַיִךְ מִכָּל בְּשָׂמִים
(יא) נֹפֶת תִּטֹּפְנָה שִׂפְתוֹתַיִךְ כַּלָּה דְּבַשׁ וְחָלָב תַּחַת לְשׁוֹנֵךְ וְרֵיחַ שַׂלְמֹתַיִךְ כְּרֵיחַ לְבָנוֹן :
