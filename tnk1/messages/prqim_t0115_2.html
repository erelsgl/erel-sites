שכר פרי הבטן על פי האב העובד את האלהים
אם האב עבד בלבב שלם את האלהים גם שכרו יהיה בהתאם ושכר פרי בטנו הלא הוא בנו יהיה גבור בארץ
ספר תהילים פרק קיב

(א) הַלְלוּיָהּ אַשְׁרֵי אִישׁ יָרֵא אֶת יְהֹוָה בְּמִצְוֹתָיו חָפֵץ מְאֹד:

(ב) גִּבּוֹר בָּאָרֶץ יִהְיֶה זַרְעוֹ דּוֹר יְשָׁרִים יְבֹרָךְ:


ספר בראשית פרק טו (ב) וַיֹּאמֶר אַבְרָם אֲדֹנָי יֱהוִֹה מַה תִּתֶּן לִי וְאָנֹכִי הוֹלֵךְ עֲרִירִי וּבֶן מֶשֶׁק בֵּיתִי הוּא דַּמֶּשֶׂק אֱלִיעֶזֶר:
(ג) וַיֹּאמֶר אַבְרָם הֵן לִי לֹא נָתַתָּה זָרַע וְהִנֵּה בֶן בֵּיתִי יוֹרֵשׁ אֹתִי:
אברם אמר ליהוה מַה תִּתֶּן לִי הֵן לִי לֹא נָתַתָּה זָרַע
כי ידע אברם כי שכר הצדיק בארץ יְשֻׁלָּם ככתוב בספר תהילות פרק קכז (ג) הִנֵּה נַחֲלַת יְהֹוָה בָּנִים שָׂכָר פְּרִי הַבָּטֶן:

ספר משלי פרק יא (לא) הֵן צַדִּיק בָּאָרֶץ יְשֻׁלָּם אַף כִּי רָשָׁע וְחוֹטֵא:
