ראשי>סופרים בני דורנו>רמי ניר>

כדאי לדעת ולעשות את יום השבת

קוד: כדאי לדעת ולעשות את יום השבת בתנ"ך

סוג: מאמר

מאת: רמי ניר

אל:

לַעֲשׂוֹת אֶת-יוֹם הַשַּׁבָּת

כדאי לדעת:

הַשַּׁבָּת... בֵּינִי וּבֵין בְּנֵי יִשְׂרָאֵל--אוֹת הִוא לְעֹלָם
(שמות לא' 17)

וְגַם אֶת-שַׁבְּתוֹתַי נָתַתִּי לָהֶם לִהְיוֹת לְאוֹת בֵּינִי וּבֵינֵיהֶם
לָדַעַת כִּי אֲנִי יְהוָה מְקַדְּשָׁם
(יחזקאל כ' 12)

*******

כדאי לדעת עד מתי הַיּוֹם הַשְּׁבִיעִי עדיין לא נקרא יוֹם הַשַּׁבָּת.

יש עֶרֶב, ויש בּוֹקֶר, ואין מְאוֹרוֹת בִּרְקִיעַ הַשָּׁמַיִם, שלושה ימים לפני היום הרביעי לבריאה;

כלומר, כל יום בריאה אלהי הוא נצחי, באין סוף של זמן, ואינו נמדד להיות 24 שעות.

*******

ומתועד כדלקמן:

בְּרֵאשִׁית בָּרָא אֱלֹהִים אֵת הַשָּׁמַיִם
וְאֵת הָאָרֶץ...
(בראשית א' 1)

וְהָאָרֶץ הָיְתָה תֹהוּ וָבֹהוּ
וְחֹשֶׁךְ עַל-פְּנֵי תְהוֹם
וְרוּחַ אֱלֹהִים מְרַחֶפֶת עַל-פְּנֵי הַמָּיִם
(בראשית א' 2)

וַיֹּאמֶר אֱלֹהִים יְהִי אוֹר וַיְהִי-אוֹר
(בראשית א' 3)

וַיַּרְא אֱלֹהִים אֶת-הָאוֹר כִּי-טוֹב
וַיַּבְדֵּל אֱלֹהִים בֵּין הָאוֹר וּבֵין הַחֹשֶׁךְ
(בראשית א' 4)

וַיִּקְרָא אֱלֹהִים לָאוֹר יוֹם
וְלַחֹשֶׁךְ קָרָא לָיְלָה
וַיְהִי-עֶרֶב וַיְהִי-בֹקֶר יוֹם אֶחָד...
(בראשית א' 5)

וַיְהִי-עֶרֶב וַיְהִי-בֹקֶר יוֹם שֵׁנִי
(בראשית א' 8)

וַיְהִי-עֶרֶב וַיְהִי-בֹקֶר יוֹם שְׁלִישִׁי
(בראשית א' 13)
וַיֹּאמֶר אֱלֹהִים
יְהִי מְאֹרֹת בִּרְקִיעַ הַשָּׁמַיִם
לְהַבְדִּיל בֵּין הַיּוֹם וּבֵין הַלָּיְלָה
וְהָיוּ לְאֹתֹת וּלְמוֹעֲדִים וּלְיָמִים וְשָׁנִים....
(בראשית א' 14)

וַיְהִי-עֶרֶב וַיְהִי-בֹקֶר יוֹם רְבִיעִי
(בראשית א' 19)

וַיְהִי-עֶרֶב וַיְהִי-בֹקֶר יוֹם חֲמִישִׁי
(בראשית א' 23)

וַיְהִי-עֶרֶב וַיְהִי-בֹקֶר יוֹם הַשִּׁשִּׁי
(בראשית א' 31)

וַיְכֻלּוּ הַשָּׁמַיִם וְהָאָרֶץ וְכָל-צְבָאָם
(בראשית ב' 1)

וַיְכַל אֱלֹהִים בַּיּוֹם הַשְּׁבִיעִי מְלַאכְתּוֹ אֲשֶׁר עָשָׂה
וַיִּשְׁבֹּת בַּיּוֹם הַשְּׁבִיעִי מִכָּל-מְלַאכְתּוֹ אֲשֶׁר עָשָׂה
(בראשית ב' 2)

וַיְבָרֶךְ אֱלֹהִים אֶת-יוֹם הַשְּׁבִיעִי וַיְקַדֵּשׁ אֹתוֹ
כִּי בוֹ שָׁבַת מִכָּל-מְלַאכְתּוֹ אֲשֶׁר-בָּרָא אֱלֹהִים לַעֲשׂוֹת...
(בראשית ב' 3)


*******

וְהַיּוֹם הַשְּׁבִיעִי עדיין לא נקרא "יוֹם הַשַּׁבָּת".

*******


וְנֹחַ בֶּן-שֵׁשׁ מֵאוֹת שָׁנָה
וְהַמַּבּוּל הָיָה מַיִם עַל-הָאָרֶץ
(בראשית ז' 6)

וַיְהִי לְשִׁבְעַת הַיָּמִים
(בראשית ז' 10)

בַּחֹדֶשׁ הַשֵּׁנִי
בְּשִׁבְעָה-עָשָׂר יוֹם לַחֹדֶשׁ
בַּיּוֹם הַזֶּה נִבְקְעוּ כָּל-מַעְיְנֹת תְּהוֹם רַבָּה וַאֲרֻבֹּת הַשָּׁמַיִם נִפְתָּחוּ
(בראשית ז' 11)

וַיִּגְבְּרוּ הַמַּיִם עַל-הָאָרֶץ חֲמִשִּׁים וּמְאַת יוֹם
(בראשית ז' 24)

וַיַּחְסְרוּ הַמַּיִם--מִקְצֵה חֲמִשִּׁים וּמְאַת יוֹם
(בראשית ח' 3)

וַתָּנַח הַתֵּבָה בַּחֹדֶשׁ הַשְּׁבִיעִי בְּשִׁבְעָה-עָשָׂר יוֹם לַחֹדֶשׁ עַל הָרֵי אֲרָרָט
(בראשית ח' 4)

וְהַמַּיִם הָיוּ הָלוֹךְ וְחָסוֹר עַד הַחֹדֶשׁ הָעֲשִׂירִי
בָּעֲשִׂירִי בְּאֶחָד לַחֹדֶשׁ נִרְאוּ רָאשֵׁי הֶהָרִים
(בראשית ח' 5)

וַיְהִי מִקֵּץ אַרְבָּעִים יוֹם וַיִּפְתַּח נֹחַ אֶת-חַלּוֹן הַתֵּבָה אֲשֶׁר עָשָׂה
(בראשית ח' 6)

וַיְשַׁלַּח אֶת-הָעֹרֵב וַיֵּצֵא יָצוֹא וָשׁוֹב עַד-יְבֹשֶׁת הַמַּיִם מֵעַל הָאָרֶץ
(בראשית ח' 7)

וַיָּחֶל עוֹד שִׁבְעַת יָמִים אֲחֵרִים
וַיֹּסֶף שַׁלַּח אֶת-הַיּוֹנָה מִן-הַתֵּבָה
(בראשית ח' 10)

וַתָּבֹא אֵלָיו הַיּוֹנָה לְעֵת עֶרֶב
וְהִנֵּה עֲלֵה-זַיִת טָרָף בְּפִיהָ
וַיֵּדַע נֹחַ כִּי-קַלּוּ הַמַּיִם מֵעַל הָאָרֶץ
(בראשית ח' 11)

וַיִּיָּחֶל עוֹד שִׁבְעַת יָמִים אֲחֵרִים
וַיְשַׁלַּח אֶת-הַיּוֹנָה וְלֹא-יָסְפָה שׁוּב-אֵלָיו עוֹד
(בראשית ח' 12)

וַיְהִי בְּאַחַת וְשֵׁשׁ-מֵאוֹת שָׁנָה... (כלומר ביום הולדתו של נֹחַ)
בָּרִאשׁוֹן בְּאֶחָד לַחֹדֶשׁ חָרְבוּ הַמַּיִם מֵעַל הָאָרֶץ
(בראשית ח' 13)

וּבַחֹדֶשׁ הַשֵּׁנִי בְּשִׁבְעָה וְעֶשְׂרִים יוֹם לַחֹדֶשׁ
יָבְשָׁה הָאָרֶץ
(בראשית ח' 14)

*******

לוח המועדים של נוח קיים, וְהַיּוֹם הַשְּׁבִיעִי עדיין לא נקרא "יוֹם הַשַּׁבָּת".

*******


וַיִּסְעוּ, מֵאֵילִם וַיָּבֹאוּ כָּל-עֲדַת בְּנֵי-יִשְׂרָאֵל אֶל-מִדְבַּר-סִין אֲשֶׁר בֵּין-אֵילִם וּבֵין סִינָי
בַּחֲמִשָּׁה עָשָׂר יוֹם לַחֹדֶשׁ הַשֵּׁנִי לְצֵאתָם מֵאֶרֶץ מִצְרָיִם
(שמות טז' 1)

*******

(וְהַיּוֹם הַשְּׁבִיעִי), בַּחֲמִשָּׁה עָשָׂר יוֹם לַחֹדֶשׁ הַשֵּׁנִי לְצֵאתָם מֵאֶרֶץ מִצְרָיִם, עדיין לא נקרא "יוֹם הַשַּׁבָּת".

*******

וַיֹּאמֶר יְהוָה אֶל-מֹשֶׁה
הִנְנִי מַמְטִיר לָכֶם לֶחֶם מִן-הַשָּׁמָיִם
וְיָצָא הָעָם וְלָקְטוּ דְּבַר-יוֹם בְּיוֹמוֹ לְמַעַן אֲנַסֶּנּוּ הֲיֵלֵךְ בְּתוֹרָתִי אִם-לֹא
(שמות טז' 4)

וְהָיָה בַּיּוֹם הַשִּׁשִּׁי
וְהֵכִינוּ אֵת אֲשֶׁר-יָבִיאוּ
וְהָיָה מִשְׁנֶה עַל אֲשֶׁר-יִלְקְטוּ יוֹם יוֹם
(שמות טז' 5)

וַיְהִי בַּיּוֹם הַשִּׁשִּׁי
לָקְטוּ לֶחֶם מִשְׁנֶה--שְׁנֵי הָעֹמֶר לָאֶחָד
(שמות טז' 22)

וַיֹּאמֶר אֲלֵהֶם הוּא אֲשֶׁר דִּבֶּר יְהוָה--שַׁבָּתוֹן שַׁבַּת-קֹדֶשׁ לַיהוָה מָחָר
(שמות טז' 23)

וַיֹּאמֶר מֹשֶׁה אִכְלֻהוּ הַיּוֹם כִּי-שַׁבָּת הַיּוֹם לַיהוָה
הַיּוֹם לֹא תִמְצָאֻהוּ בַּשָּׂדֶה
(שמות טז' 25)

וַיְהִי בַּיּוֹם הַשְּׁבִיעִי
יָצְאוּ מִן-הָעָם לִלְקֹט וְלֹא מָצָאוּ
(שמות טז' 27)

רְאוּ כִּי-יְהוָה נָתַן לָכֶם הַשַּׁבָּת
עַל-כֵּן הוּא נֹתֵן לָכֶם בַּיּוֹם הַשִּׁשִּׁי לֶחֶם יוֹמָיִם
שְׁבוּ אִישׁ תַּחְתָּיו אַל-יֵצֵא אִישׁ מִמְּקֹמוֹ--בַּיּוֹם הַשְּׁבִיעִי
(שמות טז' 29)

וַיִּשְׁבְּתוּ הָעָם בַּיּוֹם הַשְּׁבִעִי
(שמות טז' 30)

*******

כדאי לדעת: רַק בַּיּוֹם הַ-38 לְצֵאתָם מֵאֶרֶץ מִצְרָיִם, הַיּוֹם הַשְּׁבִיעִי נקרא "יוֹם הַשַּׁבָּת"ץ

*******

כדאי לדעת על הַשַּׁבָּת:


שֵׁשֶׁת יָמִים יֵעָשֶׂה מְלָאכָה
וּבַיּוֹם הַשְּׁבִיעִי שַׁבַּת שַׁבָּתוֹן קֹדֶשׁ לַיהוָה
כָּל-הָעֹשֶׂה מְלָאכָה בְּיוֹם הַשַּׁבָּת מוֹת יוּמָת
(שמות לא' 15)

וְשָׁמְרוּ בְנֵי-יִשְׂרָאֵל אֶת-הַשַּׁבָּת
לַעֲשׂוֹת אֶת-הַשַּׁבָּת לְדֹרֹתָם בְּרִית עוֹלָם
(שמות לא' 16)

בֵּינִי וּבֵין בְּנֵי יִשְׂרָאֵל--אוֹת הִוא לְעֹלָם
כִּי-שֵׁשֶׁת יָמִים עָשָׂה יְהוָה אֶת-הַשָּׁמַיִם וְאֶת-הָאָרֶץ
וּבַיּוֹם הַשְּׁבִיעִי שָׁבַת וַיִּנָּפַשׁ...
(שמות לא' 17)


*******

לַעֲשׂוֹת אֶת-הַשַּׁבָּת, מתועד כדלקמן:


א) זָכוֹר:

זָכוֹר אֶת-יוֹם הַשַּׁבָּת לְקַדְּשׁוֹ
(שמות כ' 7)

שֵׁשֶׁת יָמִים תַּעֲבֹד וְעָשִׂיתָ כָּל-מְלַאכְתֶּךָ
(שמות כ' 8)

וְיוֹם הַשְּׁבִיעִי--שַׁבָּת לַיהוָה אֱלֹהֶיךָ
לֹא-תַעֲשֶׂה כָל-מְלָאכָה אַתָּה וּבִנְךָ וּבִתֶּךָ עַבְדְּךָ וַאֲמָתְךָ וּבְהֶמְתֶּךָ וְגֵרְךָ אֲשֶׁר בִּשְׁעָרֶיךָ
(שמות כ' 9)

כִּי שֵׁשֶׁת-יָמִים עָשָׂה יְהוָה אֶת-הַשָּׁמַיִם וְאֶת-הָאָרֶץ
אֶת-הַיָּם וְאֶת-כָּל-אֲשֶׁר-בָּם
וַיָּנַח בַּיּוֹם הַשְּׁבִיעִי
עַל-כֵּן בֵּרַךְ יְהוָה אֶת-יוֹם הַשַּׁבָּת--וַיְקַדְּשֵׁהוּ...
(שמות כ' 10)

*******


ב) שָׁמוֹר לַעֲשׂוֹת:

שָׁמוֹר אֶת-יוֹם הַשַּׁבָּת לְקַדְּשׁוֹ כַּאֲשֶׁר צִוְּךָ יְהוָה אֱלֹהֶיךָ
(דברים ה' 11)

לְמַעַן יָנוּחַ עַבְדְּךָ וַאֲמָתְךָ כָּמוֹךָ
(דברים ה' 13)

וְזָכַרְתָּ כִּי עֶבֶד הָיִיתָ בְּאֶרֶץ מִצְרַיִם
וַיֹּצִאֲךָ יְהוָה אֱלֹהֶיךָ מִשָּׁם בְּיָד חֲזָקָה וּבִזְרֹעַ נְטוּיָה
עַל-כֵּן צִוְּךָ יְהוָה אֱלֹהֶיךָ
לַעֲשׂוֹת אֶת-יוֹם הַשַּׁבָּת...
(דברים ה' 14)
