(ב) בַּיָּמִים הָהֵם אֲנִי דָנִיֵּאל הָיִיתִי מִתְאַבֵּל שְׁלשָׁה שָׁבֻעִים יָמִים:

(ג) לֶחֶם חֲמֻדוֹת לֹא אָכַלְתִּי וּבָשָׂר וָיַיִן לֹא בָא אֶל פִּי וְסוֹךְ [שמן-תורק] לֹא סָכְתִּי עַד מְלֹאת שְׁלֹשֶׁת שָׁבֻעִים יָמִים: [ לא כתוב שלא התרחץ במים ]

בראשית לז (לד) וַיִּקְרַע יַעֲקֹב שִׂמְלֹתָיו וַיָּשֶׂם שַׂק בְּמָתְנָיו וַיִּתְאַבֵּל עַל בְּנוֹ יָמִים רַבִּים:

דברי הימים א פרק ז (כב) וַיִּתְאַבֵּל אֶפְרַיִם אֲבִיהֶם יָמִים רַבִּים וַיָּבֹאוּ אֶחָיו לְנַחֲמוֹ:

[ולא יתכן מבלי להתרחץ ימים רבים כי על פי מגילת אסתר פרק א פסוק ד כתוב: יָמִים רַבִּים שְׁמוֹנִים וּמְאַת יוֹם:]

ספר שמואל ב פרק יג (לז) וְאַבְשָׁלוֹם בָּרַח וַיֵּלֶךְ אֶל תַּלְמַי בֶּן עַמִּיחוּר {עַמִּיהוּד} מֶלֶךְ גְּשׁוּר וַיִּתְאַבֵּל עַל בְּנוֹ כָּל הַיָּמִים: ולא יתכן בלי רחצה

ספר נחמיה פרק א (ד) וַיְהִי כְּשָׁמְעִי אֶת הַדְּבָרִים הָאֵלֶּה יָשַׁבְתִּי וָאֶבְכֶּה וָאֶתְאַבְּלָה יָמִים [לא כתוב ימים רבים] וָאֱהִי צָם וּמִתְפַּלֵּל לִפְנֵי אֱלֹהֵי הַשָּׁמָיִם: ניתן להתאבל בצום [מבלי לאכל ולשתות]ובתפילה ולא מוזכר שאין להתרחץ באבל
בספר קהלת פרק ז כתוב:
(א) טוֹב שֵׁם מִשֶּׁמֶן טוֹב וְיוֹם הַמָּוֶת מִיּוֹם הִוָּלְדוֹ: לכן האבלים באבל הזוכרים את יום המות לא יסכו שמן טוב

(ב) טוֹב לָלֶכֶת אֶל בֵּית אֵבֶל מִלֶּכֶת אֶל בֵּית מִשְׁתֶּה בַּאֲשֶׁר הוּא סוֹף כָּל הָאָדָם וְהַחַי יִתֵּן אֶל לִבּוֹ: לכן האבלים לא ישתו כל משרת ענבים ולא תירוש ולא יין המשמח לבב אנוש כי כעת באבל

(ג) טוֹב כַּעַס מִשְּׂחוֹק כִּי בְרֹעַ פָּנִים יִיטַב לֵב: לכן האבלים כדניאל לא יאכלו בשר אשר על פי ספר דניאל כתוב בפרק א' כי הבשר [פת-בג] מיטיב את הפנים ומונע את זעף האדם
(ד) לֵב חֲכָמִים בְּבֵית אֵבֶל וְלֵב כְּסִילִים בְּבֵית שִׂמְחָה:
לכן האבלים כדניאל לא יאכלו לחם חמודות שהוא לחם שמן כי כתוב בספר תהילות פרק קד (טו) וְיַיִן יְשַׂמַּח לְבַב אֱנוֹשׁ ---לְהַצְהִיל פָּנִים מִשָּׁמֶן וְלֶחֶם לְבַב אֱנוֹשׁ יִסְעָד:
אז בסופו של דבר האבלות היא למעשה במניעת אוכל טוב ומשקה טוב אך כן מתרחצים באבל כי לא ינתק עניין הטהרה אשר מרחיקה אותנו מכל טומאה וגם לא נדרש לקרוע את הבגדים כי על פי ספר הנביא יואל פרק ב ככתוב:
(יב) וְגַם עַתָּה נְאֻם יְהֹוָה שֻׁבוּ עָדַי בְּכָל לְבַבְכֶם וּבְצוֹם וּבִבְכִי וּבְמִסְפֵּד:

(יג) וְקִרְעוּ לְבַבְכֶם וְאַל בִּגְדֵיכֶם  וְשׁוּבוּ אֶל יְהֹוָה אֱלֹהֵיכֶם כִּי חַנּוּן וְרַחוּם הוּא אֶרֶךְ אַפַּיִם וְרַב חֶסֶד וְנִחָם עַל הָרָעָה:
