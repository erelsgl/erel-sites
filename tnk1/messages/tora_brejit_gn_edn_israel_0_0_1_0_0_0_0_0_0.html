ירמיהו הנביא נתן את הספר לשריה בן-נריה בן-מחסיה וזאת למען יקרא את כל דברי האלהים שבספר מפי ירמיהו אל יושבי בבל אֶל הַמָּקֹום הַזֶּה לְהַכְרִיתֹו לְבִלְתִּי הֱיֹות בּוֹ יוֹשֵׁב לְמֵאָדָם וְעַד בְּהֵמָה כִּי שִׁמְמֹות עוֹלָם תִּהְיֶה: ועל כן באמת המקום הזה נהפך למדבר [סוריה] שלא ישב בו איש וכמובן גם לא זרע יעקב אשר הובטח לו ארץ זבת חלב ודבש כאשר בספר דברים פרק ח כתוב:
(ו) וְשָׁמַרְתָּ אֶת מִצְוֹת יְהֹוָה אֱלֹהֶיךָ לָלֶכֶת בִּדְרָכָיו וּלְיִרְאָה אֹתוֹ:
(ז) כִּי יְהֹוָה אֱלֹהֶיךָ מְבִיאֲךָ אֶל אֶרֶץ טוֹבָה אֶרֶץ נַחֲלֵי מָיִם עֲיָנֹת וּתְהֹמֹת יֹצְאִים בַּבִּקְעָה וּבָהָר:
(ח) אֶרֶץ חִטָּה וּשְׂעֹרָה וְגֶפֶן וּתְאֵנָה וְרִמּוֹן אֶרֶץ זֵית שֶׁמֶן וּדְבָשׁ:
(ט) אֶרֶץ אֲשֶׁר לֹא בְמִסְכֵּנֻת תֹּאכַל בָּהּ לֶחֶם לֹא תֶחְסַר כֹּל בָּהּ אֶרֶץ אֲשֶׁר אֲבָנֶיהָ בַרְזֶל וּמֵהֲרָרֶיהָ תַּחְצֹב נְחשֶׁת:
אך אתה אראל סגל וכל הולך בדרך רבני חזל וְהַמַּטִּים עַקַלְקַלּוֹתָם יוֹלִיכֵם יְהֹוָה אֶת פֹּעֲלֵי הָאָוֶן לַעֲזָאזֵל הַמִּדְבָּרָה שָׁלוֹם עַל יִשְׂרָאֵל ורק אם תשמור אֶת מִצְוֹת יְהֹוָה אֱלֹהֶיךָ לָלֶכֶת בִּדְרָכָיו וּלְיִרְאָה אֹתוֹ: כאשר מורה לך המורה לתורה עבד יהוה מזרע אהרן הכהן הלוי אז יטב לך בארץ הקודש

וגם הצוי מפי ירמיהו איש האלהים הוא להשמיע את דברי הספר ליושבי המקום בממלכת בבל ורק אחרי כן לקשור אבן ולהשליך לנהר פרת לנהר הירדן בשובו לישראל ויש להקים את דברי האלהים ככתוב בתורה מבלי לסלף להוסיף לשוא או לגרוע מתורת יהוה כאשר לא השכילו לעשות רבני חזל וההולכים בדרכם אז קרא עוד פעם ואולי תבין

(ס) וַיִּכְתֹּב יִרְמְיָהוּ אֵת כָּל הָרָעָה אֲשֶׁר תָּבֹוא אֶל בָּבֶל אֶל סֵפֶר אֶחָד אֵת כָּל הַדְּבָרִים הָאֵלֶּה הַכְּתֻבִים אֶל בָּבֶל:
(סא) וַיֹּאמֶר יִרְמְיָהוּ אֶל שְׂרָיָה כְבֹאֲךָ בָבֶל וְרָאִיתָ וְקָרָאתָ אֵת כָּל הַדְּבָרִים הָאֵלֶּה:
(סב) וְאָמַרְתָּ יְהֹוָה אַתָּה דִבַּרְתָּ אֶל הַמָּקֹום הַזֶּה לְהַכְרִיתֹו לְבִלְתִּי הֱיֹות בּוֹ יוֹשֵׁב לְמֵאָדָם וְעַד בְּהֵמָה כִּי שִׁמְמֹות עוֹלָם תִּהְיֶה:
(סג) וְהָיָה כְּכַלֹּתְךָ לִקְרֹא אֶת הַסֵּפֶר הַזֶּה תִּקְשֹׁר עָלָיו אֶבֶן וְהִשְׁלַכְתֹּו אֶל תֹּוךְ פְּרָת:
(סד) וְאָמַרְתָּ כָּכָה תִּשְׁקַע בָּבֶל וְלֹא תָקוּם מִפְּנֵי הָרָעָה אֲשֶׁר אָנֹכִי מֵבִיא עָלֶיהָ וְיָעֵפוּ עַד הֵנָּה דִּבְרֵי יִרְמְיָהוּ:
