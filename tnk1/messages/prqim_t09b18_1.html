וּבְאַרְבַּע עֶשְׂרֵה שָׁנָה לַמֶּלֶךְ חִזְקִיָּה עָלָה סַנְחֵרִיב מֶלֶךְ-אַשּׁוּר עַל כָּל-עָרֵי יְהוּדָה הַבְּצֻרוֹת--וַיִּתְפְּשֵׂם...
(מלכים ב' יח' 13)

וַיָּבוֹאוּ עַבְדֵי הַמֶּלֶךְ חִזְקִיָּהוּ--אֶל-יְשַׁעְיָהוּ
וַיּוֹאמֶר לָהֶם יְשַׁעְיָהוּ
כֹּה תוֹאמְרוּן אֶל-אֲדוֹנֵיכֶם
כֹּה אָמַר יְהוָה אַל-תִּירָא מִפְּנֵי הַדְּבָרִים אֲשֶׁר שָׁמַעְתָּ אֲשֶׁר גִּדְּפוּ נַעֲרֵי מֶלֶךְ-אַשּׁוּר אוֹתִי
הִנְנִי נוֹתֵן בּוֹ רוּחַ וְשָׁמַע שְׁמוּעָה וְשָׁב לְאַרְצוֹ וְהִפַּלְתִּיו בַּחֶרֶב בְּאַרְצוֹ...
(מלכים יט' 5 - 7)

לָכֵן כֹּה-אָמַר יְהוָה אֶל-מֶלֶךְ אַשּׁוּר לֹא יָבוֹא אֶל-הָעִיר הַזּוֹאת וְלֹא-יוֹרֶה שָׁם חֵץ
וְלֹא-יְקַדְּמֶנָּה מָגֵן וְלֹא-יִשְׁפּוֹךְ עָלֶיהָ סוֹלְלָה
בַּדֶּרֶךְ אֲשֶׁר-יָבוֹא בָּהּ יָשׁוּב וְאֶל-הָעִיר הַזּוֹאת לֹא יָבוֹא נְאֻם-יְהוָה
וְגַנּוֹתִי אֶל-הָעִיר הַזּוֹאת לְהוֹשִׁיעָהּ--לְמַעֲנִי וּלְמַעַן דָּוִד עַבְדִּי...

וַיְהִי בַּלַּיְלָה הַהוּא וַיֵּצֵא מַלְאַךְ יְהוָה וַיַּךְ בְּמַחֲנֵה אַשּׁוּר מֵאָה שְׁמוֹנִים וַחֲמִשָּׁה אָלֶף וַיַּשְׁכִּימוּ בַבּוֹקֶר וְהִנֵּה כֻלָּם פְּגָרִים מֵתִים
וַיִּסַּע וַיֵּלֶךְ וַיָּשָׁב סַנְחֵרִיב מֶלֶךְ-אַשּׁוּר וַיֵּשֶׁב בְּנִינְוֵה
וַיְהִי הוּא מִשְׁתַּחֲוֶה בֵּית נִסְרוֹךְ אֱלוֹהָיו וְאַדְרַמֶּלֶךְ וְשַׂרְאֶצֶר בָּנָיו הִכֻּהוּ בַחֶרֶב וְהֵמָּה נִמְלְטוּ אֶרֶץ אֲרָרָט...

ד)
בַּיָּמִים הָהֵם חָלָה חִזְקִיָּהוּ לָמוּת
וַיָּבוֹא אֵלָיו יְשַׁעְיָהוּ בֶן-אָמוֹץ הַנָּבִיא וַיּוֹאמֶר אֵלָיו כֹּה-אָמַר יְהוָה צַו לְבֵיתֶךָ--כִּי מֵת אַתָּה וְלֹא תִחְיֶה
וַיַּסֵּב אֶת-פָּנָיו אֶל-הַקִּיר וַיִּתְפַּלֵּל--אֶל-יְהוָה לֵאמוֹר
אָנָּה יְהוָה זְכָר-נָא אֵת אֲשֶׁר הִתְהַלַּכְתִּי לְפָנֶיךָ בֶּאֱמֶת וּבְלֵבָב שָׁלֵם
וְהַטּוֹב בְּעֵינֶיךָ עָשִׂיתִי וַיֵּבְךְּ חִזְקִיָּהוּ בְּכִי גָדוֹל
וַיְהִי יְשַׁעְיָהוּ--לֹא יָצָא העיר (חָצֵר) הַתִּיכוֹנָה
וּדְבַר-יְהוָה--הָיָה אֵלָיו לֵאמוֹר שׁוּב וְאָמַרְתָּ אֶל-חִזְקִיָּהוּ נְגִיד-עַמִּי
כֹּה-אָמַר יְהוָה אֱלוֹהֵי דָּוִד אָבִיךָ שָׁמַעְתִּי אֶת-תְּפִלָּתֶךָ רָאִיתִי אֶת-דִּמְעָתֶךָ
הִנְנִי רוֹפֶא לָךְ--בַּיּוֹם הַשְּׁלִישִׁי תַּעֲלֶה בֵּית יְהוָה וְהוֹסַפְתִּי עַל-יָמֶיךָ חֲמֵשׁ עֶשְׂרֵה שָׁנָה
וּמִכַּף מֶלֶךְ-אַשּׁוּר אַצִּילְךָ וְאֵת הָעִיר הַזּוֹאת וְגַנּוֹתִי עַל-הָעִיר הַזּוֹאת--לְמַעֲנִי וּלְמַעַן דָּוִד עַבְדִּי
וַיּוֹאמֶר יְשַׁעְיָהוּ קְחוּ דְּבֶלֶת תְּאֵנִים וַיִּקְחוּ וַיָּשִׂימוּ עַל-הַשְּׁחִין וַיֶּחִי
וַיּוֹאמֶר חִזְקִיָּהוּ אֶל-יְשַׁעְיָהוּ מָה אוֹת כִּי-יִרְפָּא יְהוָה לִי וְעָלִיתִי בַּיּוֹם הַשְּׁלִישִׁי בֵּית יְהוָה
וַיּוֹאמֶר יְשַׁעְיָהוּ זֶה-לְּךָ הָאוֹת מֵאֵת יְהוָה כִּי יַעֲשֶׂה יְהוָה אֶת-הַדָּבָר אֲשֶׁר דִּבֵּר
הָלַךְ הַצֵּל עֶשֶׂר מַעֲלוֹת אִם-יָשׁוּב עֶשֶׂר מַעֲלוֹת
וַיּוֹאמֶר יְחִזְקִיָּהוּ נָקֵל לַצֵּל לִנְטוֹת עֶשֶׂר מַעֲלוֹת לֹא כִי יָשׁוּב הַצֵּל אֲחוֹרַנִּית עֶשֶׂר מַעֲלוֹת
וַיִּקְרָא יְשַׁעְיָהוּ הַנָּבִיא אֶל-יְהוָה וַיָּשֶׁב אֶת-הַצֵּל בַּמַּעֲלוֹת אֲשֶׁר יָרְדָה בְּמַעֲלוֹת אָחָז אֲחוֹרַנִּית--עֶשֶׂר מַעֲלוֹת...
(מלכים ב' כ' 1 - 11)

האם אז כתת חזקיה את נחש הנחושת נחושתן?
האם אז הוכח כי אבד כוחו של נחש הנחושת להחיות ולרפא חולים?
"כִּי עַד-הַיָּמִים הָהֵמָּה הָיוּ בְנֵי-יִשְׂרָאֵל מְקַטְּרִים לוֹ וַיִּקְרָא-לוֹ נְחֻשְׁתָּן"...
הלא אז... "וַיּוֹאמֶר יְשַׁעְיָהוּ קְחוּ דְּבֶלֶת תְּאֵנִים וַיִּקְחוּ וַיָּשִׂימוּ עַל-הַשְּׁחִין וַיֶּחִי"...

ואמנם "הַמּוֹפֵת" אשר ראו כל עמי הארצות איך השמש הלכה אחורנית עשר מעלות נראה גם בבבל - ומתועד:
וְכֵן בִּמְלִיצֵי שָׂרֵי בָבֶל הַמְשַׁלְּחִים עָלָיו לִדְרוֹשׁ הַמּוֹפֵת אֲשֶׁר הָיָה בָאָרֶץ...
(דבה"י ב' לב' 31)

ואכן... ס"ה חיי חזקיה:
14 שנה + 15 שנה = "וְעֶשְׂרִים וָתֵשַׁע שָׁנָה מָלַךְ בִּירוּשָׁלִָם"...

והלא משה צֻוָּה לעשות את נחש הנחושת - ומתועד:

וַיְשַׁלַּח יְהוָה בָּעָם אֵת הַנְּחָשִׁים הַשְּׂרָפִים
וַיְנַשְּׁכוּ אֶת-הָעָם וַיָּמָת עַם-רָב מִיִּשְׂרָאֵל
וַיָּבוֹא הָעָם אֶל-מֹשֶׁה וַיּוֹאמְרוּ
חָטָאנוּ כִּי-דִבַּרְנוּ בַיהוָה וָבָךְ--הִתְפַּלֵּל אֶל-יְהוָה וְיָסֵר מֵעָלֵינוּ אֶת-הַנָּחָשׁ...
וַיִּתְפַּלֵּל מֹשֶׁה בְּעַד הָעָם
וַיּוֹאמֶר יְהוָה אֶל-מֹשֶׁה עֲשֵׂה לְךָ שָׂרָף וְשִׂים אוֹתוֹ עַל-נֵס
וְהָיָה כָּל-הַנָּשׁוּךְ וְרָאָה אוֹתוֹ וָחָי...
וַיַּעַשׂ מֹשֶׁה נְחַשׁ נְחוֹשֶׁת וַיְשִׂמֵהוּ עַל-הַנֵּס
וְהָיָה אִם-נָשַׁךְ הַנָּחָשׁ אֶת-אִישׁ--וְהִבִּיט אֶל-נְחַשׁ הַנְּחוֹשֶׁת וָחָי...
(במדבר כא' 6 - 9)
