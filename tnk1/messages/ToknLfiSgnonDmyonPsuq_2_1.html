שחיטת הרבנים אינה כשרה
כי בשחיטה כשרה באמת יש לשפך את הדם על מזבח אלהים לבדו
לאמור מזבח אשר עשה אותו איש אלהים כנביא כהן משיח
כאשר עקד אברהם את האיל במזבח
אך אם כל איש מעם הארץ שעשה קורס שוחטים ישחט ארצה ויטבח בכל מקום שיראה אז זה נקרא לאכול על הדם וזה מה שקורה היום בישראל

ספר שמואל א פרק יד

לב) וַיַּעַשׂ {וַיַּעַט} הָעָם אֶל ַשָּׁלָל {הַשָּׁלָל} וַיִּקְחוּ צֹאן וּבָקָר וּבְנֵי בָקָר וַיִּשְׁחֲטוּ אָרְצָה וַיֹּאכַל הָעָם עַל הַדָּם:

(לג) וַיַּגִּידוּ לְשָׁאוּל לֵאמֹר הִנֵּה הָעָם חֹטִאים לַיהֹוָה לֶאֱכֹל עַל הַדָּם וַיֹּאמֶר בְּגַדְתֶּם גֹּלּוּ אֵלַי הַיּוֹם אֶבֶן גְּדוֹלָה:

(לד) וַיֹּאמֶר שָׁאוּל פֻּצוּ בָעָם וַאֲמַרְתֶּם לָהֶם הַגִּישׁוּ אֵלַי אִישׁ שׁוֹרוֹ וְאִישׁ שְׂיֵהוּ וּשְׁחַטְתֶּם בָּזֶה וַאֲכַלְתֶּם וְלֹא תֶחֶטְאוּ לַיהֹוָה לֶאֱכֹל אֶל הַדָּם וַיַּגִּשׁוּ כָל הָעָם אִישׁ שׁוֹרוֹ בְיָדוֹ הַלַּיְלָה וַיִּשְׁחֲטוּ שָׁם:

(לה) וַיִּבֶן שָׁאוּל מִזְבֵּחַ לַיהֹוָה אֹתוֹ הֵחֵל לִבְנוֹת מִזְבֵּחַ לַיהֹוָה:
