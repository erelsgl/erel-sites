לא נתנה ארץ ביד רשע כי אם להשמידו בהכרת רשעים תראה

מאת: מורה לתורה משה דין

אל: DAIAN1 @ 013.NET

נכתב ב: 10:24:39  22.12.2011, כתוספת/תגובה ל: איוב ט

ספר איוב פרק ט (כד) אֶרֶץ נִתְּנָה בְיַד רָשָׁע פְּנֵי שֹׁפְטֶיהָ יְכַסֶּה אִם לֹא אֵפוֹא מִי הוּא
אך לא נתנה ארץ ביד רשע כי אם להשמידו על כן זה עֹשֶׁר שָׁמוּר לִבְעָלָיו  לְרָעָתוֹ:
בספר קהלת פרק ה כתוב:
(יב) יֵשׁ רָעָה חוֹלָה רָאִיתִי תַּחַת הַשָּׁמֶשׁ עֹשֶׁר שָׁמוּר לִבְעָלָיו  לְרָעָתוֹ:
(יג) וְאָבַד הָעֹשֶׁר הַהוּא בְּעִנְיַן רָע וְהוֹלִיד בֵּן וְאֵין בְּיָדוֹ מְאוּמָה
לכן בספר תהילות פרק מט כתוב: 
(יז) אַל תִּירָא כִּי יַעֲשִׁר אִישׁ כִּי יִרְבֶּה כְּבוֹד  בֵּיתוֹ
(יח) כִּי לֹא בְמוֹתוֹ יִקַּח הַכֹּל לֹא יֵרֵד אַחֲרָיו כְּבוֹדוֹ:
ספר שמואל א פרק כה כתוב:
(לז) וַיְהִי בַבֹּקֶר בְּצֵאת הַיַּיִן מִנָּבָל וַתַּגֶּד לוֹ אִשְׁתּוֹ אֶת הַדְּבָרִים הָאֵלֶּה וַיָּמָת לִבּוֹ בְּקִרְבּוֹ וְהוּא הָיָה לְאָבֶן
היין כחמרא טבא [סמל החומריות והעושר]
וכאשר יצא היין מנבל הכרמלי כך יצא כל טוב העושר מהרשעים בעיני אלהים כי לפני אלהים רק מעשיו ודברי תורתו יעמדו
הגם אם נתנה ארץ ביד רשע ורע מאד לארץ ובכל זאת ישמור אלהים את הארץ גם אם בשבי הרשע עד שתצא מידי הרשע כאשר יצא היין מנבל הכרמלי  
כי הצדיק בארץ הוא כיין המשמח את האלהים ואם הצדיק נמצא בשבי הרשע שאינו משמח את האלהים כי אין אלהים בקרבו אז נגזר על שניהם להפרד כאשר יצא היין הצדיק מנבל הכרמלי כי כאשר יאסף הצדיק משבי הרשע ואז הרשע לא יוכל לחיות עוד מטובו של הצדיק "שהתאבד" בתוך הרשע ונתן את הרשע קברו ולא תנתן ארץ [צדקת הצדיק=חמרא טבא] ביד רשע כי פנה אל רהבים ושטי כזב אך הצדיק אשר שם יהוה מבטחו הוא כארץ חמרא טבא והענבים לענוים והיין לנסך ליהוה צבאות לבדו
ובספר ישעיהו פרק נג  (ט) וַיִּתֵּן אֶת רְשָׁעִים קִבְרוֹ וְאֶת עָשִׁיר בְּמֹתָיו עַל לֹא חָמָס עָשָׂה וְלֹא מִרְמָה בְּפִיו:

אֶרֶץ נִתְּנָה בְיַד רָשָׁע פְּנֵי שֹׁפְטֶיהָ יְכַסֶּה עד אשר תדעו אֵפוֹא מִי הוּא: לאמור תראו אותו ביראת אלהים
ואמר אלהים בספר שמות פרק לג ככתוב: 
(כ) וַיֹּאמֶר לֹא תוּכַל לִרְאֹת אֶת פָּנָי כִּי לֹא יִרְאַנִי הָאָדָם וָחָי: לכן במות האדם ידע אֵפוֹא יראה מִי הוּא והארץ לא תנתן עוד ביד רשע כי לאלהים
ובספר תהילות פרק לז (לד) קַוֵּה אֶל יְהֹוָה וּשְׁמֹר דַּרְכּוֹ וִירוֹמִמְךָ לָרֶשֶׁת אָרֶץ בְּהִכָּרֵת רְשָׁעִים  תִּרְאֶה:
בהכרת רשעים תראה ובמות הרשע ידעו כולם כי הרשע אבד וְהָאָרֶץ לְעוֹלָם עֹמָדֶת
ספר קהלת פרק א (ד) דּוֹר הֹלֵךְ וְדוֹר בָּא וְהָאָרֶץ לְעוֹלָם עֹמָדֶת:
לאמור כי כך אלהים סולה מלשון סולת וזו מערכת סינון לרוחות בני האדם
אשר רק צַדִּיקִים יִירְשׁוּ אָרֶץ וְיִשְׁכְּנוּ לָעַד עָלֶיהָ: כי רוח אלהים המה והארץ לאלהים
הַרְנִינוּ גוֹיִם עַמּוֹ כִּי דַם עֲבָדָיו יִקּוֹם וְנָקָם יָשִׁיב לְצָרָיו וְכִפֶּר אַדְמָתוֹ עַמּוֹ
