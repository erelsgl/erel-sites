על לוחות הברית לב תורת משה עץ החיים מגנים הכרובים ולהט החרב המתהפכת אשר בידי כהני יהוה מזרע אהרן הלוי לשמור את דברי הברית בתורת משה אשר היא עץ החיים למחזיקים בה ותומכיה מאושר
כי על לוחות הברית שומרים ברוח הקודש ולא בחומרי בנייה אבן ואָוֶן
רק שוטה שותה מ"אבן השתיה" שרמי ניר בודה מִלִּבּוֹ
וכי שריה נקרא "כהן הראש" כי הוא הכהן הראשי לאמור ראש הכהנים הלא הוא הכהן הגדול מאחיו כאשר כתוב:
בספר דברי הימים ב פרק יט לאמור:
(יא) וְהִנֵּה אֲמַרְיָהוּ כֹהֵן הָרֹאשׁ עֲלֵיכֶם לְכֹל דְּבַר יְהֹוָה וּזְבַדְיָהוּ בֶן יִשְׁמָעֵאל הַנָּגִיד לְבֵית יְהוּדָה לְכֹל דְּבַר הַמֶּלֶךְ וְשֹׁטְרִים הַלְוִיִּם לִפְנֵיכֶם חִזְקוּ וַעֲשׂוּ וִיהִי יְהֹוָה עִם הַטּוֹב:
וגם בספר דברי הימים ב פרק כד כתוב:
יא) וַיְהִי בְּעֵת יָבִיא אֶת הָאָרוֹן אֶל פְּקֻדַּת הַמֶּלֶךְ בְּיַד הַלְוִיִּם וְכִרְאוֹתָם כִּי רַב הַכֶּסֶף וּבָא סוֹפֵר הַמֶּלֶךְ וּפְקִיד כֹּהֵן הָרֹאשׁ וִיעָרוּ אֶת הָאָרוֹן וְיִשָּׂאֻהוּ וִישִׁיבֻהוּ אֶל מְקֹמוֹ כֹּה עָשׂוּ לְיוֹם בְּיוֹם וַיַּאַסְפוּ כֶסֶף לָרֹב:
וגם בספר דברי הימים ב פרק כו כתוב:
כ) וַיִּפֶן אֵלָיו עֲזַרְיָהוּ כֹהֵן הָרֹאשׁ וְכָל הַכֹּהֲנִים וְהִנֵּה הוּא מְצֹרָע בְּמִצְחוֹ וַיַּבְהִלוּהוּ מִשָּׁם וְגַם הוּא נִדְחַף לָצֵאת כִּי נִגְּעוֹ יְהֹוָה:
וגם בספר עזרא כתוב:
ספר עזרא פרק ז
(א) וְאַחַר הַדְּבָרִים הָאֵלֶּה בְּמַלְכוּת אַרְתַּחְשַׁסְתְּא מֶלֶךְ פָּרָס עֶזְרָא בֶּן שְׂרָיָה בֶּן עֲזַרְיָה בֶּן חִלְקִיָּה:
(ב) בֶּן שַׁלּוּם בֶּן צָדוֹק בֶּן אֲחִיטוּב:
(ג) בֶּן אֲמַרְיָה בֶן עֲזַרְיָה בֶּן מְרָיוֹת:
(ד) בֶּן זְרַחְיָה בֶן עֻזִּי בֶּן בֻּקִּי:
(ה) בֶּן אֲבִישׁוּעַ בֶּן פִּינְחָס בֶּן אֶלְעָזָר בֶּן אַהֲרֹן הַכֹּהֵן הָרֹאשׁ:

(ו) הוּא עֶזְרָא עָלָה מִבָּבֶל וְהוּא סֹפֵר מָהִיר בְּתוֹרַת מֹשֶׁה אֲשֶׁר נָתַן יְהֹוָה אֱלֹהֵי יִשְׂרָאֵל וַיִּתֶּן לוֹ הַמֶּלֶךְ כְּיַד יְהֹוָה אֱלֹהָיו עָלָיו כֹּל בַּקָּשָׁתוֹ:

לא אבן ולא אסטרואיד קרעו את ים-סוף
כי היד הגדולה של יהוה אלהים ברוח הקודש כאשר בספר שמות כתוב:
כא) וַיֵּט משֶׁה אֶת יָדוֹ עַל הַיָּם וַיּוֹלֶךְ יְהֹוָה אֶת הַיָּם בְּרוּחַ קָדִים עַזָּה כָּל הַלַּיְלָה וַיָּשֶׂם אֶת הַיָּם לֶחָרָבָה וַיִּבָּקְעוּ הַמָּיִם:
(ח) וּבְרוּחַ אַפֶּיךָ נֶעֶרְמוּ מַיִם נִצְּבוּ כְמוֹ נֵד נֹזְלִים קָפְאוּ תְהֹמֹת בְּלֶב יָם:
ראו כולם כי לֹא בְחַיִל וְלֹא בְכֹחַ כִּי אִם בְּרוּחִי אָמַר יְהֹוָה צְבָאוֹת:
