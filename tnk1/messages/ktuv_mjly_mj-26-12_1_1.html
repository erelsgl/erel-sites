שקל הקודש ערכו כעשרה גרם לפי זהב טהור [24 קראט] כאלף שקל כסף של היום
מחצית השקל הלא היא "בקע" = כחמישה גרם לפי זהב טהור [24 קראט] כחמש מאות שקל כסף של יום
וערך הככר הוא כאלף שקל בשקל הקודש לאמור 10 ק"ג לפי זהב טהור [24 קראט] בשווי מיליון שקל כסף של היום
בספר שמות פרק לח כתוב:(כט) וּנְחשֶׁת הַתְּנוּפָה שִׁבְעִים כִּכָּר וְאַלְפַּיִם וְאַרְבַּע מֵאוֹת שָׁקֶל
לאמור שבעים אלף שקל ואלפיים וארבע מאות שקל בשקל הקודש [לפי עשרה גרם שוויו]
וכי לא כותבים שבעים ושניים אלף
כי במקרא יכתב:שבעים אלף ואלפיים:
לשם השוואה בספר במדבר פרק 31 כתוב:
(מג) וַתְּהִי מֶחֱצַת הָעֵדָה מִן הַצֹּאן שְׁלשׁ מֵאוֹת אֶלֶף וּשְׁלשִׁים אֶלֶף שִׁבְעַת אֲלָפִים וַחֲמֵשׁ מֵאוֹת
על כן ככר היא אלף שקל בשקל הקודש
ככר היא לא שלשת אלפים שקל וכי מאין המצאתם את זה הלא זה לא כתוב במקרא.
אך בספר שמואל ב פרק יב פסוק 30 כתוב:
(ל) וַיִּקַּח אֶת עֲטֶרֶת מַלְכָּם מֵעַל רֹאשׁוֹ וּמִשְׁקָלָהּ כִּכַּר זָהָב וְאֶבֶן יְקָרָה וַתְּהִי עַל רֹאשׁ דָּוִד וּשְׁלַל הָעִיר הוֹצִיא הַרְבֵּה מְאֹד
ככר זהב כאשר כתבתי לכם ערכה הוא כאלף שקל בשקל הקודש לאמור כעשרה ק"ג בלבד אשר נשא דוד המלך את הכתר על ראשו
אך לפי דבריכם אתם המצאת שהככר היא שלשת אלפים שקל
ולכן אני שואל אתכם האם נראה לכם שדוד המלך נשא על ראשו כתר של 30 ק"ג
האם יצאתם מדעתכם?
ברצוני להגיד כי היתי לוחם מתנדב ביחידה מובחרת לפני כעשרים שנה ובזמנו היה טיל טאו אמריקני [פק"ל עורב] במשקל עשרות ק"ג אשר היה כבד מאד לחיילים הלוחמים הצעירים
ונא להבין כי דוד המלך היה איש שפל קומה ולא גבוה כאחיו אשר כתוב:
(ז) וַיֹּאמֶר יְהֹוָה אֶל שְׁמוּאֵל אַל תַּבֵּט אֶל מַרְאֵהוּ וְאֶל גְּבֹהַּ קוֹמָתוֹ כִּי מְאַסְתִּיהוּ כִּי לֹא אֲשֶׁר יִרְאֶה הָאָדָם כִּי הָאָדָם יִרְאֶה לַעֵינַיִם וַיהֹוָה יִרְאֶה לַלֵּבָב:
(ח) וַיִּקְרָא יִשַׁי אֶל אֲבִינָדָב וַיַּעֲבִרֵהוּ לִפְנֵי שְׁמוּאֵל וַיֹּאמֶר גַּם בָּזֶה לֹא בָחַר יְהֹוָה:
(ט) וַיַּעֲבֵר יִשַׁי שַׁמָּה וַיֹּאמֶר גַּם בָּזֶה לֹא בָחַר יְהֹוָה:
(י) וַיַּעֲבֵר יִשַׁי שִׁבְעַת בָּנָיו לִפְנֵי שְׁמוּאֵל וַיֹּאמֶר שְׁמוּאֵל אֶל יִשַׁי לֹא בָחַר יְהֹוָה בָּאֵלֶּה:
(יא) וַיֹּאמֶר שְׁמוּאֵל אֶל יִשַׁי הֲתַמּוּ הַנְּעָרִים וַיֹּאמֶר עוֹד שָׁאַר הַקָּטָן וְהִנֵּה רֹעֶה בַּצֹּאן וַיֹּאמֶר שְׁמוּאֵל אֶל יִשַׁי שִׁלְחָה וְקָחֶנּוּ כִּי לֹא נָסֹב עַד בֹּאוֹ פֹה:
(יב) וַיִּשְׁלַח וַיְבִיאֵהוּ וְהוּא אַדְמוֹנִי עִם יְפֵה עֵינַיִם וְטוֹב רֹאִי וַיֹּאמֶר יְהֹוָה קוּם מְשָׁחֵהוּ כִּי זֶה הוּא:
