צריך לזכור כי דניאל היה שליט בבל בימי דריוש... כל זאת מאז מינה נבוכדנצר את דניאל לשליט בבל...
כמו-כן צריך לזכור כי בשנה 1 לדריוש דניאל ערך את חשבון השנים לנבואת ירמיה (כה' 11)...
כלומר... ללא ספק דניאל היה מעורב בהחלטות מלכי פרס לתת אישור לבניין המקדש... גם בשנה 1 לכורש וגם בשנה 2 לדריוש...

לכן... עזרא הסופר מתעד כי עבודות לבניין המקדש בטלו עד שנת 2 לדריוש:

בֵּאדַיִן בְּטֵלַת עֲבִידַת בֵּית-אֱלָהָא דִּי בִּירוּשְׁלֶם
וַהֲוָת בָּטְלָא עַד שְׁנַת תַּרְתֵּין לְמַלְכוּת דָּרְיָוֶשׁ מֶלֶךְ-פָּרָס.
(עזרא ד' 24)

עזרא הסופר גם מתעד כי בשנת 2 לדריוש... חגי וזכריה התנבאו לעידוד בניין המקדש:

וְהִתְנַבִּי חַגַּי נְבִיָּאה וּזְכַרְיָה בַר-עִדּוֹא נְבִיַּאיָּא
עַל-יְהוּדָיֵא דִּי בִיהוּד וּבִירוּשְׁלֶם--בְּשֻׁם אֱלָהּ יִשְׂרָאֵל עֲלֵיהוֹן.
בֵּאדַיִן קָמוּ זְרֻבָּבֶל בַּר-שְׁאַלְתִּיאֵל וְיֵשׁוּעַ בַּר-יוֹצָדָק
וְשָׁרִיו לְמִבְנֵא בֵּית אֱלָהָא דִּי בִירוּשְׁלֶם
וְעִמְּהוֹן נְבִיַּאיָּא דִי-אֱלָהָא מְסָעֲדִין לְהוֹן...
(עזרא ה' 1 - 2)

עזרא הסופר גם מתעד כי דריוש בדק בארכיון הפרסי ומצא את הכרזת כורש המתירה את בניין המקדש... והמתירה את החזרת כלי המקדש שלקח נבוכדנצר...
אולם צורת המקדש שהתיר כורש לבנות... הייתה בית בסגנון פרסי... בו ישאו תפילות לשלומו של כורש:

בֵּאדַיִן דָּרְיָוֶשׁ מַלְכָּא שָׂם טְעֵם
וּבַקַּרוּ בְּבֵית סִפְרַיָּא דִּי גִנְזַיָּא מְהַחֲתִין תַּמָּה--בְּבָבֶל
וְהִשְׁתְּכַח בְּאַחְמְתָא בְּבִירְתָא דִּי בְּמָדַי מְדִינְתָּא
מְגִלָּה חֲדָה וְכֵן-כְּתִיב בְּגַוַּהּ דִּכְרוֹנָה...

בִּשְׁנַת חֲדָה לְכוֹרֶשׁ מַלְכָּא
כּוֹרֶשׁ מַלְכָּא שָׂם טְעֵם--בֵּית-אֱלָהָא בִירוּשְׁלֶם בַּיְתָא יִתְבְּנֵא
אֲתַר דִּי-דָבְחִין דִּבְחִין וְאֻשּׁוֹהִי מְסוֹבְלִין
רוּמֵהּ אַמִּין שִׁתִּין פְּתָיֵהּ אַמִּין שִׁתִּין
נִדְבָּכִין דִּי-אֶבֶן גְּלָל תְּלָתָא
וְנִדְבָּךְ דִּי-אָע חֲדַת
וְנִפְקְתָא--מִן-בֵּית מַלְכָּא תִּתְיְהִב
וְאַף מָאנֵי בֵית-אֱלָהָא דִּי דַהֲבָה וְכַסְפָּא דִּי נְבוּכַדְנֶצַּר
הַנְפֵּק מִן-הֵיכְלָא דִי-בִירוּשְׁלֶם
וְהֵיבֵל לְבָבֶל--יַהֲתִיבוּן
וִיהָךְ לְהֵיכְלָא דִי-בִירוּשְׁלֶם לְאַתְרֵהּ
וְתַחֵת בְּבֵית אֱלָהָא...
(עזרא ו' 1 - 5)

עזרא הסופר גם מתעד כי המקדש נבנה עד שנת 6 לדריוש:

וְשֵׁיצִיא בַּיְתָה דְנָה
עַד יוֹם תְּלָתָה לִירַח אֲדָר
דִּי-הִיא שְׁנַת-שֵׁת לְמַלְכוּת דָּרְיָוֶשׁ מַלְכָּא...
(עזרא ו' 15)
