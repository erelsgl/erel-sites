רמי ניר כתב כך:
תיעוד הפסוק צ"ל:
"וַאֲחִיָּה אֲחִי אִיכָבוֹד בֶּן-פִּינְחָס בֶּן-עֵלִי בֶן-אֲחִטוּב כֹּהֵן יְהוָה בְּשִׁלוֹ--נֹשֵׂא אֵפוֹד":
(שמואל א' יד' 3)

כנראה מעתיק שיבש את הפסוק במהופך:
"וַאֲחִיָּה בֶן-אֲחִטוּב אֲחִי אִיכָבוֹד בֶּן-פִּינְחָס בֶּן-עֵלִי כֹּהֵן יְהוָה בְּשִׁלוֹ--נֹשֵׂא אֵפוֹד":
(שמואל א' יד' 3)


מורה לתורה משה דין משיב:
הלא אחיה הוא אחימלך [כהן נב] בן-אחיטוב בן-פנחס בן-עלי הכהן הגדול שהיה בשילו
ובספר שמואל א פרק כא כתוב:
(ה) וַיַּעַן הַכֹּהֵן אֶת דָּוִד וַיֹּאמֶר אֵין לֶחֶם חֹל אֶל תַּחַת יָדִי כִּי אִם לֶחֶם קֹדֶשׁ יֵשׁ אִם נִשְׁמְרוּ הַנְּעָרִים אַךְ מֵאִשָּׁה:
(ו) וַיַּעַן דָּוִד אֶת הַכֹּהֵן וַיֹּאמֶר לוֹ כִּי אִם אִשָּׁה עֲצֻרָה לָנוּ כִּתְמוֹל שִׁלְשֹׁם בְּצֵאתִי וַיִּהְיוּ כְלֵי הַנְּעָרִים קֹדֶשׁ וְהוּא דֶּרֶךְ חֹל וְאַף כִּי הַיּוֹם יִקְדַּשׁ בַּכֶּלִי:
(ז) וַיִּתֶּן לוֹ הַכֹּהֵן קֹדֶשׁ כִּי לֹא הָיָה שָׁם לֶחֶם כִּי אִם לֶחֶם הַפָּנִים הַמּוּסָרִים מִלִּפְנֵי יְהֹוָה לָשׂוּם לֶחֶם חֹם בְּיוֹם הִלָּקְחוֹ:
אם הית כהן מורה לתורה כמוני אז הית מבין כי לֶחֶם קֹדֶשׁ לֶחֶם הַפָּנִים הַמּוּסָרִים מִלִּפְנֵי יְהֹוָה נמצא אצל הכהן הגדול


וגם כה אמר לעלי איש האלהים: (לג) וְאִישׁ לֹא אַכְרִית לְךָ מֵעִם מִזְבְּחִי לְכַלּוֹת אֶת עֵינֶיךָ וְלַאֲדִיב אֶת נַפְשֶׁךָ וְכָל מַרְבִּית בֵּיתְךָ יָמוּתוּ אֲנָשִׁים: על כן נרצחו כהני נבה מבית עלי הכהן הגדול וכי גם דוד משיח אלהים הלך לכהן הגדול בימים ההם היה אחימלך בן-אחיטוב אשר נתן לדוד את חרב גולית הפלישתי
(לד) וְזֶה לְּךָ הָאוֹת אֲשֶׁר יָבֹא אֶל שְׁנֵי בָנֶיךָ אֶל חָפְנִי וּפִינְחָס בְּיוֹם אֶחָד יָמוּתוּ שְׁנֵיהֶם:
(לה) וַהֲקִימֹתִי לִי כֹּהֵן נֶאֱמָן כַּאֲשֶׁר בִּלְבָבִי וּבְנַפְשִׁי יַעֲשֶׂה וּבָנִיתִי לוֹ בַּיִת נֶאֱמָן וְהִתְהַלֵּךְ לִפְנֵי מְשִׁיחִי כָּל הַיָּמִים:
הכהן הנאמן הוא הכהן הגדול צדוק בן-אחיטוב מבית אלעזר בן-אהרן ולא מבית-עלי אשר מבית איתמר בן-אהרן
וראה כי בספר דברי הימים א פרק כד כתוב:
(ג) וַיֶּחָלְקֵם דָּוִיד
וְצָדוֹק מִן בְּנֵי אֶלְעָזָר
וַאֲחִימֶלֶךְ מִן בְּנֵי אִיתָמָר
כי אֲחִימֶלֶךְ בן-אביתר בן-אחימלך בן-אחיטוב בן-פנחס בן-עלי מזרע איתמר הוא איתמר בן-אהרן הכהן הלוי
ועל כן
עלי הכהן מבני איתמר
צדוק הכהן מבני אלעזר

אז איך אתה מעז להצדיק את עצמך לסלף את דברי הקודש של המקרא כי אתה אומר למעשה שסופרי הנביאים הצדיקים כתבו שלא ברוח הקודש ונפלה שגגה בידם וכי אתה מחשיב עצמך לאלהים עליהם אז ישפוט אותך אלהים ויבער אותך מן הארץ

וגם כתבת כך:
בתיעוד שושלת הכהנים הגדולים...שמו של עֵלִי אינו כלול ברשימה המתועדת...
כלומר, הלוי (מר משה דין) אינו מכבד את התוכן המתועד בתנ"ך כי בדא מליבו שם של "כהן גדול"..."עֵלִי" שלא היה אלא "כהן" בלבד:

מורה לתורה משה דין משיב:
לא אני בודא את דברי כי אם אתה אשר אומר "שושלת הכהנים הגדולים" זאת אינך מבין כי זו אמנם שושלת כהנים של בית אלעזר בן אהרן הכהן אך כאשר בזמן שמשכן יהוה היה בשילו בימי השופטים אז אחיטוב בן-אמריה בן-מריות הכהן כל שלשת אלה אמנם היו כהנים מבית אלעזר בן-אהרן אך לא היו כהנים גדולים בימים ההם כי הכהונה הגדולה היתה בידי בני איתמר בן-אהרן הכהן וכי שושלת הכהנים הגדולים מבית אלעזר התחדשה רק מאז צדוק הכהן בימי שלמה המלך אשר נמשחו בשמן משחת הקודש של נתן הנביא

ובכל מקרה יש הבדל של שנים רבות בין ימי צדוק ובין ימי עלי וכי לא מדובר על אותו דור הלא בספר שמואל א פרק ד כתוב:
טו) וְעֵלִי בֶּן תִּשְׁעִים וּשְׁמֹנֶה שָׁנָה וְעֵינָיו קָמָה וְלֹא יָכוֹל לִרְאוֹת:
(טז) וַיֹּאמֶר הָאִישׁ אֶל עֵלִי אָנֹכִי הַבָּא מִן הַמַּעֲרָכָה וַאֲנִי מִן הַמַּעֲרָכָה נַסְתִּי הַיּוֹם וַיֹּאמֶר מֶה הָיָה הַדָּבָר בְּנִי:
(יז) וַיַּעַן הַמְבַשֵּׂר וַיֹּאמֶר נָס יִשְׂרָאֵל לִפְנֵי פְלִשְׁתִּים וְגַם מַגֵּפָה גְדוֹלָה הָיְתָה בָעָם וְגַם שְׁנֵי בָנֶיךָ מֵתוּ חָפְנִי וּפִינְחָס וַאֲרוֹן הָאֱלֹהִים נִלְקָחָה:
(יח) וַיְהִי כְּהַזְכִּירוֹ אֶת אֲרוֹן הָאֱלֹהִים וַיִּפֹּל מֵעַל הַכִּסֵּא אֲחֹרַנִּית בְּעַד יַד הַשַּׁעַר וַתִּשָּׁבֵר מַפְרַקְתּוֹ וַיָּמֹת כִּי זָקֵן הָאִישׁ וְכָבֵד וְהוּא שָׁפַט אֶת יִשְׂרָאֵל אַרְבָּעִים שָׁנָה:

אם הית כהן מורה לתורה כמוני אז הית יודע כי על ארון האלהים עם לוּחֹת הָאֲבָנִים לוּחֹת הַבְּרִית שומר הכהן הגדול ולזאת אתה לא יכול להתכחש וגם שים לב שכתוב: וְעֵלִי בֶּן תִּשְׁעִים וּשְׁמֹנֶה שָׁנָה כאשר הוא מת בטרם מלך שאול וכי צדוק הכהן שרת בימי שלמה המלך
אז אם תאמר שצדוק ועלי הם אחים כי הם שניהם מזרע בית אהרן הכהן הלוי אשר הוא אב לכל הכהנים אז אסלח לך כי אתה לא כהן מורה לתורה
אך אם תעשה בזדון ולא תתקן גם אחרי שבארתי לך אז אשלח בך רוח רעה
כי אני ידעתי את התעודה "עדות לדרך המשיח" אשר אנכי כתבתי

כִּי שִׂפְתֵי כֹהֵן יִשְׁמְרוּ דַעַת וְתוֹרָה יְבַקְשׁוּ מִפִּיהוּ כִּי מַלְאַךְ יְהֹוָה צְבָאוֹת הוּא
