תעודת "נצח ישראל" = "האמת ההיסטורית" מוכיחה - כדלקמן:

1.
(מלכים ב' יד' 1 - 2) "בִּשְׁנַת שְׁתַּיִם לְיוֹאָשׁ בֶּן-יוֹאָחָז מֶלֶךְ יִשְׂרָאֵל מָלַךְ אֲמַצְיָהוּ בֶן-יוֹאָשׁ מֶלֶךְ יְהוּדָה בֶּן-עֶשְׂרִים וְחָמֵשׁ שָׁנָה הָיָה בְמָלְכוֹ וְעֶשְׂרִים וָתֵשַׁע שָׁנָה מָלַךְ בִּירוּשָׁלִָם"...

כלומר, "האמת ההיסטורית" = "אֲמַצְיָהוּ בֶן-יוֹאָשׁ מָלַךְ 29 שָׁנָה בִּירוּשָׁלִָם"...

2.
(מלכים ב' יד 17) "וַיְחִי אֲמַצְיָהוּ בֶן-יוֹאָשׁ מֶלֶךְ יְהוּדָה אַחֲרֵי מוֹת יְהוֹאָשׁ בֶּן-יְהוֹאָחָז מֶלֶךְ יִשְׂרָאֵל--חֲמֵשׁ עֶשְׂרֵה שָׁנָה"...

כלומר, "האמת ההיסטורית" = "בִּשְׁנַת 14 לַאֲמַצְיָהוּ בֶן-יוֹאָשׁ מֶלֶךְ יְהוּדָה" = "מוֹת יְהוֹאָשׁ בֶּן-יְהוֹאָחָז מֶלֶךְ יִשְׂרָאֵל"...

ואמנם, "האמת ההיסטורית" = "אֲמַצְיָהוּ בֶן-יוֹאָשׁ מֶלֶךְ יְהוּדָה" = "וַיְחִי אַחֲרֵי מוֹת יְהוֹאָשׁ בֶּן-יְהוֹאָחָז מֶלֶךְ יִשְׂרָאֵל--חֲמֵשׁ עֶשְׂרֵה שָׁנָה"...

3.
(מלכים ב' יד' 23) "בִּשְׁנַת חֲמֵשׁ-עֶשְׂרֵה שָׁנָה לַאֲמַצְיָהוּ בֶן-יוֹאָשׁ מֶלֶךְ יְהוּדָה--מָלַךְ יָרָבְעָם בֶּן-יוֹאָשׁ מֶלֶךְ-יִשְׂרָאֵל בְּשׁוֹמְרוֹן אַרְבָּעִים וְאַחַת שָׁנָה"...

כלומר, "האמת ההיסטורית" = "29 שנה - 15 שנה" = "בִּשְׁנַת 14 לַאֲמַצְיָהוּ בֶן-יוֹאָשׁ מֶלֶךְ יְהוּדָה" = "מָלַךְ יָרָבְעָם בֶּן-יוֹאָשׁ מֶלֶךְ-יִשְׂרָאֵל בְּשׁוֹמְרוֹן"...

ואמנם, "האמת ההיסטורית" = "אֲמַצְיָהוּ בֶן-יוֹאָשׁ מָלַךְ 29 שָׁנָה בִּירוּשָׁלִָם" = "14 שָׁנָה + 15 שָׁנָה"...

ואם-כן, "האמת ההיסטורית" = "בִּשְׁנַת חֲמֵשׁ-עֶשְׂרֵה שָׁנָה לְיָרָבְעָם מֶלֶךְ יִשְׂרָאֵל" = "מוֹת אֲמַצְיָהוּ בֶן-יוֹאָשׁ" = "מָלַךְ עֲזַרְיָה בֶן-אֲמַצְיָה מֶלֶךְ יְהוּדָה בֶּן-שֵׁשׁ עֶשְׂרֵה שָׁנָה הָיָה בְמָלְכוֹ וַחֲמִשִּׁים וּשְׁתַּיִם שָׁנָה מָלַךְ בִּירוּשָׁלִָם"...

4.
(מלכים ב' טו' 1) "בִּשְׁנַת עֶשְׂרִים וָשֶׁבַע שָׁנָה לְיָרָבְעָם מֶלֶךְ יִשְׂרָאֵל"...

כלומר, "האמת ההיסטורית" = "בִּשְׁנַת 27 שָׁנָה לְיָרָבְעָם מֶלֶךְ יִשְׂרָאֵל" = "דִּבְרֵי עָמוֹס אֲשֶׁר-הָיָה בַנּוֹקְדִים מִתְּקוֹעַ אֲשֶׁר חָזָה עַל-יִשְׂרָאֵל בִּימֵי עֻזִּיָּה מֶלֶךְ-יְהוּדָה וּבִימֵי יָרָבְעָם בֶּן-יוֹאָשׁ מֶלֶךְ יִשְׂרָאֵל--שְׁנָתַיִם לִפְנֵי הָרָעַשׁ" (עמוס א' 1)...

כלומר, "האמת ההיסטורית" = "שְׁנָתַיִם לִפְנֵי הָרָעַשׁ" = "דִּבְרֵי עָמוֹס" = "בִּשְׁנַת 27 לְיָרָבְעָם מֶלֶךְ יִשְׂרָאֵל"...

ואמנם, "האמת ההיסטורית" = "שְׁנָתַיִם לִפְנֵי הָרָעַשׁ" = "27 שנה - 15 שנה" = "12 שנה"...

ואכן-כן, "האמת ההיסטורית" = "בִּשְׁנַת 12 לְעֻזִּיָּה מֶלֶךְ-יְהוּדָה" = "שְׁנָתַיִם לִפְנֵי הָרָעַשׁ"...

5.
ואמנם, "האמת ההיסטורית" = "הָרָעַשׁ" = "בִּשְׁנַת 29 לְיָרָבְעָם מֶלֶךְ יִשְׂרָאֵל"...

ואכן-כן, "האמת ההיסטורית" = "הָרָעַשׁ" = "בִּשְׁנַת 14 לְעֻזִּיָּה מֶלֶךְ-יְהוּדָה" = "אֲנָךְ" = "700 שנה ליציאת מצרים"...
