(יא) וַתִּקַּח יְהוֹשַׁבְעַת בַּת הַמֶּלֶךְ אֶת יוֹאָשׁ בֶּן אֲחַזְיָהוּ וַתִּגְנֹב אֹתוֹ מִתּוֹךְ בְּנֵי הַמֶּלֶךְ הַמּוּמָתִים וַתִּתֵּן אֹתוֹ וְאֶת מֵינִקְתּוֹ בַּחֲדַר הַמִּטּוֹת וַתַּסְתִּירֵהוּ יְהוֹשַׁבְעַת בַּת הַמֶּלֶךְ יְהוֹרָם אֵשֶׁת יְהוֹיָדָע הַכֹּהֵן כִּי הִיא הָיְתָה אֲחוֹת אֲחַזְיָהוּ מִפְּנֵי עֲתַלְיָהוּ וְלֹא הֱמִיתָתְהוּ:

ספר מלכים ב פרק יא
(א) וַעֲתַלְיָה אֵם אֲחַזְיָהוּ ורָאֲתָה {רָאֲתָה} כִּי מֵת בְּנָהּ וַתָּקָם וַתְּאַבֵּד אֵת כָּל זֶרַע הַמַּמְלָכָה:
(ב) וַתִּקַּח יְהוֹשֶׁבַע בַּת הַמֶּלֶךְ יוֹרָם אֲחוֹת אֲחַזְיָהוּ אֶת יוֹאָשׁ בֶּן אֲחַזְיָה וַתִּגְנֹב אֹתוֹ מִתּוֹךְ בְּנֵי הַמֶּלֶךְ הַמֻּמָותִתים }הַמּוּמָתִים{ אֹתוֹ וְאֶת מֵנִקְתּוֹ בַּחֲדַר הַמִּטּוֹת וַיַּסְתִּרוּ אֹתוֹ מִפְּנֵי עֲתַלְיָהוּ וְלֹא הוּמָת:

וגם גזלה יכולה להיות במשמעות לטובה כאשר בספר שמואל ב פרק כג כתוב:
(כא) וְהוּא הִכָּה אֶת אִישׁ מִצְרִי אִשׁר {אִישׁ} מַרְאֶה וּבְיַד הַמִּצְרִי חֲנִית וַיֵּרֶד אֵלָיו בַּשָּׁבֶט וַיִּגְזֹל אֶת הַחֲנִית מִיַּד הַמִּצְרִי וַיַּהַרְגֵהוּ בַּחֲנִיתוֹ:
(כב) אֵלֶּה עָשָׂה בְּנָיָהוּ בֶּן יְהוֹיָדָע וְלוֹ שֵׁם בִּשְׁלשָׁה הַגִּבֹּרִים:
וגם בספר דברי הימים א פרק יא כתוב פעם נוספת:
(כג) וְהוּא הִכָּה אֶת הָאִישׁ הַמִּצְרִי אִישׁ מִדָּה חָמֵשׁ בָּאַמָּה וּבְיַד הַמִּצְרִי חֲנִית כִּמְנוֹר אֹרְגִים וַיֵּרֶד אֵלָיו בַּשָּׁבֶט וַיִּגְזֹל אֶת הַחֲנִית מִיַּד הַמִּצְרִי וַיַּהַרְגֵהוּ בַּחֲנִיתוֹ:
(כד) אֵלֶּה עָשָׂה בְּנָיָהוּ בֶּן יְהוֹיָדָע וְלוֹ שֵׁם בִּשְׁלוֹשָׁה הַגִּבֹּרִים:
