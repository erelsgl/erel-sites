וכל הלוים נחשבים לכהנים
ספר דברים פרק 10 כתוב:
(ח) בָּעֵת הַהִוא הִבְדִּיל יְהֹוָה אֶת שֵׁבֶט הַלֵּוִי לָשֵׂאת אֶת אֲרוֹן בְּרִית יְהֹוָה לַעֲמֹד לִפְנֵי יְהֹוָה לְשָׁרְתוֹ וּלְבָרֵךְ בִּשְׁמוֹ עַד הַיּוֹם הַזֶּה:
ספר דברים פרק 31 כתוב:
(ט) וַיִּכְתֹּב משֶׁה אֶת הַתּוֹרָה הַזֹּאת וַיִּתְּנָהּ אֶל הַכֹּהֲנִים בְּנֵי לֵוִי הַנֹּשְׂאִים אֶת אֲרוֹן בְּרִית יְהֹוָה וְאֶל כָּל זִקְנֵי יִשְׂרָאֵל:
(כה) וַיְצַו משֶׁה אֶת הַלְוִיִּם נֹשְׂאֵי אֲרוֹן בְּרִית יְהֹוָה לֵאמֹר :
(כו) לָקֹחַ אֵת סֵפֶר הַתּוֹרָה הַזֶּה וְשַׂמְתֶּם אֹתוֹ מִצַּד אֲרוֹן בְּרִית יְהֹוָה אֱלֹהֵיכֶם וְהָיָה שָׁם בְּךָ לְעֵד:
ובספר ישעיהו פרק [אחרון] סו כתוב:
(כ) וְהֵבִיאוּ אֶת כָּל אֲחֵיכֶם מִכָּל הַגּוֹיִם מִנְחָה לַיהֹוָה בַּסּוּסִים וּבָרֶכֶב וּבַצַּבִּים וּבַפְּרָדִים וּבַכִּרְכָּרוֹת עַל הַר קָדְשִׁי יְרוּשָׁלִַם אָמַר יְהֹוָה כַּאֲשֶׁר יָבִיאוּ בְנֵי יִשְׂרָאֵל אֶת הַמִּנְחָה בִּכְלִי טָהוֹר בֵּית יְהֹוָה:
(כא) וְגַם מֵהֶם אֶקַּח לַכֹּהֲנִים לַלְוִיִּם אָמַר יְהֹוָה :
הנה לעדות כי גם מהגוים הצדיקים שיתייהדו וילוו [יהיו לוים] יקח יהוה לכהנים הלוים כי זרע אהרן קדוש יהוה הוא ברוח הקודש כי אל אלהי הרוחות לכל בשר
ושבט הלוי אין להם נחלה בישראל וכי הם מחולקים ונפוצים בכל שבטי בני ישראל ואלה הם האביונים אשר תורת משה על פיהם ולכהנים הלוים ללמד תורה את בני ישראל שנאמר בספר דברי הימים ב פרק טו  כֹּהֵן מוֹרֶה וכי משה איש האלהים כתב בתורתו על לוי 
בספר דברים פרק לג ככתוב:
(ח) וּלְלֵוִי אָמַר תֻּמֶּיךָ וְאוּרֶיךָ לְאִישׁ חֲסִידֶךָ אֲשֶׁר נִסִּיתוֹ בְּמַסָּה תְּרִיבֵהוּ עַל מֵי מְרִיבָה:
(ט) הָאֹמֵר לְאָבִיו וּלְאִמּוֹ לֹא רְאִיתִיו וְאֶת אֶחָיו לֹא הִכִּיר וְאֶת בָּנָו לֹא יָדָע כִּי שָׁמְרוּ אִמְרָתֶךָ וּבְרִיתְךָ יִנְצֹרוּ:
(י) יוֹרוּ מִשְׁפָּטֶיךָ לְיַעֲקֹב וְתוֹרָתְךָ לְיִשְׂרָאֵל יָשִׂימוּ קְטוֹרָה בְּאַפֶּךָ וְכָלִיל עַל מִזְבְּחֶךָ:
(יא) בָּרֵךְ יְהֹוָה חֵילוֹ וּפֹעַל יָדָיו תִּרְצֶה מְחַץ מָתְנַיִם קָמָיו וּמְשַׂנְאָיו מִן יְקוּמוּן:
וכאשר קמו בבני ישראל משנאי אלהים בחטא העגל אז נקבצו כל בני לוי אלה הם האביונים הצדיקים מכל בני ישראל והרגו איש את אחיו ואיך הרגו איש את אחיו הלא כתוב כי נאספו כל בני הלוי ובני לוי לא הרגו את אחיהם הלוים וכי מה שהיה הוא שהלוים אלה הם האביונים אנשי התורה בכל שבטי בני ישראל הרגו את אחיהם בבני ישראל כל אשר חטאו בעגל הזהב
ובספר שמות פרק לב כתוב:
(כו) וַיַּעֲמֹד משֶׁה בְּשַׁעַר הַמַּחֲנֶה וַיֹּאמֶר מִי לַיהֹוָה אֵלָי וַיֵּאָסְפוּ אֵלָיו כָּל בְּנֵי לֵוִי :
(כז) וַיֹּאמֶר לָהֶם כֹּה אָמַר יְהֹוָה אֱלֹהֵי יִשְׂרָאֵל שִׂימוּ אִישׁ חַרְבּוֹ עַל יְרֵכוֹ עִבְרוּ וָשׁוּבוּ מִשַּׁעַר לָשַׁעַר בַּמַּחֲנֶה וְהִרְגוּ אִישׁ אֶת אָחִיו וְאִישׁ אֶת רֵעֵהוּ וְאִישׁ אֶת קְרֹבוֹ:
(כח) וַיַּעֲשׂוּ בְנֵי לֵוִי כִּדְבַר משֶׁה וַיִּפֹּל מִן הָעָם בַּיּוֹם הַהוּא כִּשְׁלשֶׁת אַלְפֵי אִישׁ:
(כט) וַיֹּאמֶר משֶׁה מִלְאוּ יֶדְכֶם הַיּוֹם לַיהֹוָה כִּי אִישׁ בִּבְנוֹ וּבְאָחִיו וְלָתֵת עֲלֵיכֶם הַיּוֹם בְּרָכָה:

וכי הכהונה מאב לבן עוברת ברוח הקודש על פי הסמכה רוחנית של כהן לבנו שיכול להיות גם מאומץ 
וכך גם מנביא לנביא כאשר אליהו העביר אדרתו לאלישע ואלישע צעק אבי אבי רכב ישראל ופרשיו 
ההסמכה עוברת ברוח הקודש כאשר אלהים שולח את רוח קודשו ושופך על בניו אהוביו שומרי מצוותיו

מי נחשב לוי
ובכן אדם נחשב ללוי כאשר תורת משה תהיה על פיו ולא יהיה לו חלק ונחלה בבני ישראל
אך הלוי קורח כבר לא עמד בקריטריונים וידוע כי היה עשיר כקורח לאמור היתה לו נחלה בבני ישראל וגם מרד באנשי התורה במשה ואהרן הלא כתוב מֹשֶׁה וְאַהֲרֹן בְּכֹהֲנָיו
ובכל זאת משה איש האלהים הצדיק לא התווכח עם קורח לאמור לו אתה לא לוי וכי קרא לו לקורח ולכל עדתו  בְּנֵי לֵוִי   כי היו בני לוי בעיניהם
ובספר במדבר פרק טז כתוב: (ח) וַיֹּאמֶר משֶׁה אֶל קֹרַח שִׁמְעוּ נָא בְּנֵי לֵוִי: לאמור כי משה הצדיק לא נכנס לויכוח מי לוי כי קורח היה לוי בעיניו ומה גם שמבחינת יחוס השושלת קורח הוא בן יצהר ויצהר הוא עמינדב אבי נחשון נשיא יהודה לאמור כי קורח היה יהודי במקור ולא לוי ובכל זאת בגלל שהיתה תורה בפיהם של בני יצהר נחשבו ללוים עד שקורח חמד שדות והיתה לו נחלה בבני ישראל וכבר איננו יכול להחשב לוי וכי הוא גם חמד את כספי המעשר של הכהונה 
אשר לכהנים ברוח בני אהרן הלוי ולעומת זאת שמואל הנביא מבני קורח נחשב לכהן והקריב במזבח וגם עלי הכהן קרא לו בני 
לאמור שמואל כהן מבחינה רוחנית כי בנו של עלי הכהן
ספר שמואל א פרק ג כתוב:
(טז) וַיִּקְרָא עֵלִי אֶת שְׁמוּאֵל וַיֹּאמֶר שְׁמוּאֵל בְּנִי וַיֹּאמֶר הִנֵּנִי:

 תְּחִלַּת חָכְמָה יִרְאַת יְהֹוָה וְדַעַת קְדֹשִׁים בִּינָה: וְגַם לֹא לַנְּבֹנִים עֹשֶׁר
כִּי שִׂפְתֵי כֹהֵן יִשְׁמְרוּ דַעַת וְתוֹרָה יְבַקְשׁוּ מִפִּיהוּ כִּי מַלְאַךְ יְהֹוָה צְבָאוֹת הוּא:
סֹפֵר הָאָרֹן מורה לתורה משה דַיָּן
