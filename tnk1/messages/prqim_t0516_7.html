ספר דברים פרק טז כתוב:
(א) שָׁמוֹר אֶת חֹדֶשׁ הָאָבִיב וְעָשִׂיתָ פֶּסַח לַיהֹוָה אֱלֹהֶיךָ כִּי בְּחֹדֶשׁ הָאָבִיב הוֹצִיאֲךָ יְהֹוָה אֱלֹהֶיךָ מִמִּצְרַיִם לָיְלָה:
(ב) וְזָבַחְתָּ פֶּסַח לַיהֹוָה אֱלֹהֶיךָ צֹאן וּבָקָר בַּמָּקוֹם אֲשֶׁר יִבְחַר יְהֹוָה לְשַׁכֵּן שְׁמוֹ שָׁם:
(ג) לֹא תֹאכַל עָלָיו חָמֵץ שִׁבְעַת יָמִים תֹּאכַל עָלָיו מַצּוֹת לֶחֶם עֹנִי כִּי בְחִפָּזוֹן יָצָאתָ מֵאֶרֶץ מִצְרַיִם לְמַעַן תִּזְכֹּר אֶת יוֹם צֵאתְךָ מֵאֶרֶץ מִצְרַיִם כֹּל יְמֵי חַיֶּיךָ:
(ד) וְלֹא יֵרָאֶה לְךָ שְׂאֹר בְּכָל גְּבֻלְךָ שִׁבְעַת יָמִים וְלֹא יָלִין מִן הַבָּשָׂר אֲשֶׁר תִּזְבַּח בָּעֶרֶב בַּיּוֹם הָרִאשׁוֹן לַבֹּקֶר:
(ה) לֹא תוּכַל לִזְבֹּחַ אֶת הַפָּסַח בְּאַחַד שְׁעָרֶיךָ אֲשֶׁר יְהֹוָה אֱלֹהֶיךָ נֹתֵן לָךְ:
(ו) כִּי אִם אֶל הַמָּקוֹם אֲשֶׁר יִבְחַר יְהֹוָה אֱלֹהֶיךָ לְשַׁכֵּן שְׁמוֹ שָׁם תִּזְבַּח אֶת הַפֶּסַח בָּעָרֶב כְּבוֹא הַשֶּׁמֶשׁ מוֹעֵד צֵאתְךָ מִמִּצְרָיִם:
(ז) וּבִשַּׁלְתָּ וְאָכַלְתָּ בַּמָּקוֹם אֲשֶׁר יִבְחַר יְהֹוָה אֱלֹהֶיךָ בּוֹ וּפָנִיתָ בַבֹּקֶר וְהָלַכְתָּ לְאֹהָלֶיךָ:
(ח) שֵׁשֶׁת יָמִים תֹּאכַל מַצּוֹת וּבַיּוֹם הַשְּׁבִיעִי עֲצֶרֶת לַיהֹוָה אֱלֹהֶיךָ לֹא תַעֲשֶׂה מְלָאכָה:

יש לזכור איכה יצאו בני ישראל ממצרים כאשר לקחו אזוב לחטא את דם הקורבן למען ראות המלאך כי טהורה נפשם כי עשו רצון אלהים ולא נטמאו בתועבת מצרים
וכי בני ישראל שהקריבו קורבן פסח יצאו רוחנית מגלות מצרים
הלא בתורת משה בספר דברים פרק טז כתוב:
(ו) כִּי אִם אֶל הַמָּקוֹם אֲשֶׁר יִבְחַר יְהֹוָה אֱלֹהֶיךָ לְשַׁכֵּן שְׁמוֹ שָׁם  תִּזְבַּח אֶת הַפֶּסַח בָּעָרֶב כְּבוֹא הַשֶּׁמֶשׁ מוֹעֵד צֵאתְךָ מִמִּצְרָיִם:  לאמור כי היתה זו יציאה מגלות רוחנית
וכי היציאה הרגלית [פיסית] של כל עם ישראל ממצרים היתה למחרת ב בֹּקֶר

ספר במדבר פרק לג כתוב:

(א) אֵלֶּה מַסְעֵי בְנֵי יִשְׂרָאֵל אֲשֶׁר יָצְאוּ מֵאֶרֶץ מִצְרַיִם לְצִבְאֹתָם בְּיַד משֶׁה וְאַהֲרֹן:

(ב) וַיִּכְתֹּב משֶׁה אֶת מוֹצָאֵיהֶם לְמַסְעֵיהֶם עַל פִּי יְהֹוָה וְאֵלֶּה מַסְעֵיהֶם לְמוֹצָאֵיהֶם:

(ג) וַיִּסְעוּ מֵרַעְמְסֵס בַּחֹדֶשׁ הָרִאשׁוֹן בַּחֲמִשָּׁה עָשָׂר יוֹם לַחֹדֶשׁ הָרִאשׁוֹן  מִמָּחֳרַת הַפֶּסַח יָצְאוּ בְנֵי יִשְׂרָאֵל בְּיָד רָמָה לְעֵינֵי כָּל מִצְרָיִם:

(ד)  וּמִצְרַיִם מְקַבְּרִים אֵת אֲשֶׁר הִכָּה יְהֹוָה בָּהֶם כָּל בְּכוֹר וּבֵאלֹהֵיהֶם עָשָׂה יְהֹוָה שְׁפָטִים :

ובדור הזה יש עוד רבים להציל כי נמצאים בגלות מצרים עד היום
ובספר דברים פרק כח כתוב: כי אם תעשה כמעשה ארץ מצרים אז תהיה בגלות במצרים
(סח) וֶהֱשִׁיבְךָ יְהֹוָה מִצְרַיִם [ לגלות רוחנית ברוח טומאה ] בָּאֳנִיּוֹת בַּדֶּרֶךְ אֲשֶׁר אָמַרְתִּי לְךָ לֹא תֹסִיף עוֹד לִרְאֹתָהּ וְהִתְמַכַּרְתֶּם שָׁם לְאֹיְבֶיךָ לַעֲבָדִים וְלִשְׁפָחוֹת וְאֵין קֹנֶה:
ואתם אחי בני ישראל עליכם לדעת כי הקדוש ברוך הוא צדיק והואיל בטובו להוציא אתכם בני ישראל מגלות מצרים ומכל סכסוך בין איש לאחיו כי כל ישראל אחים ובין איש לרעהו ובין איש לשכנו ולא יריבו על נחלה וכי יחלקו הנחלה בצדק בין איש לאחיו לרעהו ולשכנו
ובספר שמות פרק יב כתוב:
(ד) וְאִם יִמְעַט הַבַּיִת מִהְיוֹת מִשֶּׂה וְלָקַח הוּא וּשְׁכֵנוֹ הַקָּרֹב אֶל בֵּיתוֹ בְּמִכְסַת נְפָשֹׁת אִישׁ לְפִי אָכְלוֹ תָּכֹסּוּ עַל הַשֶּׂה:

וגם לפני הקרבת הקרבן על פי מסורת אבותינו לכבוד חג הגאולה חג הפסח מבערים את החמץ [ביעור חמץ] עם נרות
לזכור את תהליך הגאולה הגדולה בעתיד
