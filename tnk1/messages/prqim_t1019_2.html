וסכסכתי מצרים במצרים

מאת: רמי ניר

נכתב ב: 08:34:31  25.02.2011, כתוספת/תגובה ל: ישעיהו יט

נבואת אחרית מתגשמת - כדלקמן:

מַשָּׂא מִצְרָיִם
הִנֵּה יְהוָה רוֹכֵב עַל-עָב קַל
וּבָא מִצְרַיִם וְנָעוּ אֱלִילֵי מִצְרַיִם מִפָּנָיו
וּלְבַב מִצְרַיִם יִמַּס בְּקִרְבּוֹ
וְסִכְסַכְתִּי מִצְרַיִם בְּמִצְרַיִם
וְנִלְחֲמוּ אִישׁ-בְּאָחִיו וְאִישׁ בְּרֵעֵהוּ
עִיר בְּעִיר מַמְלָכָה בְּמַמְלָכָה
וְנָבְקָה רוּחַ-מִצְרַיִם בְּקִרְבּוֹ וַעֲצָתוֹ אֲבַלֵּעַ...
(ישעיה יט' 1 - 3)

כלומר... תתקיים מהפכה במצרים על כל השלכותיה...

א)
בַּיּוֹם הַהוּא
יִהְיֶה מִצְרַיִם כַּנָּשִׁים
וְחָרַד וּפָחַד מִפְּנֵי תְּנוּפַת יַד-יְהוָה צְבָאוֹת אֲשֶׁר-הוּא מֵנִיף עָלָיו...
וְהָיְתָה אַדְמַת יְהוּדָה לְמִצְרַיִם לְחָגָּא כֹּל אֲשֶׁר יַזְכִּיר אוֹתָהּ אֵלָיו יִפְחָד
מִפְּנֵי עֲצַת יְהוָה צְבָאוֹת אֲשֶׁר-הוּא יוֹעֵץ עָלָיו...
(ישעיה יט' 16 - 17)

כלומר... מתקיים ההקשר הבלתי פתור ליסודה של מדינה פלסטינית...

ב)
בַּיּוֹם הַהוּא
יִהְיוּ חָמֵשׁ עָרִים בְּאֶרֶץ מִצְרַיִם מְדַבְּרוֹת שְׂפַת כְּנַעַן
וְנִשְׁבָּעוֹת לַיהוָה צְבָאוֹת...
עִיר הַהֶרֶס יֵאָמֵר לְאֶחָת...
(ישעיה יט' 18)

כלומר... מתקיים: "שְׂפַת כְּנַעַן" = "ערבית"...

ג)
בַּיּוֹם הַהוּא
יִהְיֶה מִזְבֵּחַ לַיהוָה בְּתוֹךְ אֶרֶץ מִצְרָיִם
וּמַצֵּבָה אֵצֶל-גְּבוּלָהּ לַיהוָה
וְהָיָה לְאוֹת וּלְעֵד לַיהוָה צְבָאוֹת בְּאֶרֶץ מִצְרָיִם
כִּי-יִצְעֲקוּ אֶל-יְהוָה מִפְּנֵי לוֹחֲצִים
וְיִשְׁלַח לָהֶם מוֹשִׁיעַ וָרָב וְהִצִּילָם
וְנוֹדַע יְהוָה לְמִצְרַיִם
וְיָדְעוּ מִצְרַיִם אֶת-יְהוָה בַּיּוֹם הַהוּא
וְעָבְדוּ זֶבַח וּמִנְחָה וְנָדְרוּ-נֵדֶר לַיהוָה וְשִׁלֵּמוּ
וְנָגַף יְהוָה אֶת-מִצְרַיִם נָגוֹף וְרָפוֹא
וְשָׁבוּ עַד-יְהוָה וְנֶעְתַּר לָהֶם וּרְפָאָם...
(ישעיה יט' 19 - 22)

כלומר... את "הַהֶרֶס" במצרים תגרום איראן עד אשר "יִהְיֶה" תהליך שיקום למצרים...

ד)
בַּיּוֹם הַהוּא
תִּהְיֶה מְסִלָּה מִמִּצְרַיִם אַשּׁוּרָה
וּבָא-אַשּׁוּר בְּמִצְרַיִם וּמִצְרַיִם בְּאַשּׁוּר
וְעָבְדוּ מִצְרַיִם אֶת-אַשּׁוּר...
(ישעיה יט' 23)

כלומר... תתקיים התאוששות כלכלית בין איראן למצרים...

ה)
בַּיּוֹם הַהוּא
יִהְיֶה יִשְׂרָאֵל שְׁלִישִׁיָּה לְמִצְרַיִם וּלְאַשּׁוּר
בְּרָכָה בְּקֶרֶב הָאָרֶץ
אֲשֶׁר בֵּרְכוֹ יְהוָה צְבָאוֹת לֵאמוֹר:
בָּרוּךְ עַמִּי מִצְרַיִם
וּמַעֲשֵׂה יָדַי אַשּׁוּר
וְנַחֲלָתִי יִשְׂרָאֵל...
(ישעיה יט' 24 - 25)

כלומר... תתקיים "בְּרָכָה בְּקֶרֶב הָאָרֶץ"="בְּרָכָה בעולם כולו"...
עם התגשמות הפתרון לכל בעיות המזרח התיכון...

***

כמו-כן... באמצעות "שָׂרֵי צוֹעַן" ימצא הפתרון לחידת "וְנִשְּׁתוּ-מַיִם מֵהַיָּם"...
הלא היא חידת "פאיום" למיקומה האגדי של "אֶרֶץ גּוֹשֶׁן" - כמתועד:

וַיּוֹאמֶר פַּרְעֹה אֶל-יוֹסֵף לֵאמוֹר:
אָבִיךָ וְאַחֶיךָ בָּאוּ אֵלֶיךָ
אֶרֶץ מִצְרַיִם לְפָנֶיךָ הִוא
בְּמֵיטַב הָאָרֶץ הוֹשֵׁב אֶת-אָבִיךָ וְאֶת-אַחֶיךָ
יֵשְׁבוּ בְּאֶרֶץ גּוֹשֶׁן...
(בראשית מז' 5 - 6)
