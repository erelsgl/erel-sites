בספר יהוושע פרק כד כתוב:
(לב) וְאֶת עַצְמוֹת יוֹסֵף אֲשֶׁר הֶעֱלוּ בְנֵי יִשְׂרָאֵל מִמִּצְרַיִם קָבְרוּ בִשְׁכֶם בְּחֶלְקַת הַשָּׂדֶה אֲשֶׁר קָנָה יַעֲקֹב מֵאֵת בְּנֵי חֲמוֹר אֲבִי שְׁכֶם בְּמֵאָה קְשִׂיטָה וַיִּהְיוּ לִבְנֵי יוֹסֵף לְנַחֲלָה:
ובכן באמת מדובר על העיר שכם אשר נתן יעקב ליוסף "שכם אחד על אחיו"
אשר לקחתי בחרבי ובקשתי = אשר לקחתי עם לוי ושמעון


קשת יעקב = שמעון שהיה איש קשה [מהמילה קשת] והוא שם את יוסף בבור ועל כן יוסף אשר שמע את אחיו מדברים על כך כי היה "מליץ בינהם" גם אסר אותו לנגד עיני אחיו שידעו כי יש צדק וגם יעקב אמר לבניו בספר בראשית פרק מג :
(יד) וְאֵל שַׁדַּי יִתֵּן לָכֶם רַחֲמִים לִפְנֵי הָאִישׁ וְשִׁלַּח לָכֶם אֶת אֲחִיכֶם אַחֵר וְאֶת בִּנְיָמִין וַאֲנִי כַּאֲשֶׁר שָׁכֹלְתִּי שָׁכָלְתִּי:
ובכן יעקב קיוה כי שמעון שהיה איש קשה ישתנה בעקבות מאסרו ויהיה לאיש "אחר" כדברי יעקב
ולא במקרה הופץ שבט שמעון הקשה בתוך נחלת יהודה שהיו לוחמים בקשת ועל כן בספר שמואל ב פרק א כתוב:
(יח) וַיֹּאמֶר לְלַמֵּד בְּנֵי יְהוּדָה קָשֶׁת הִנֵּה כְתוּבָה עַל סֵפֶר הַיָּשָׁר: וגם שמעון שהיה קשה למד קשת עם יהודה וכך בספר שופטים פרק א כתוב:
(ג) וַיֹּאמֶר יְהוּדָה לְשִׁמְעוֹן אָחִיו עֲלֵה אִתִּי בְגוֹרָלִי וְנִלָּחֲמָה בַּכְּנַעֲנִי וְהָלַכְתִּי גַם אֲנִי אִתְּךָ בְּגוֹרָלֶךָ וַיֵּלֶךְ אִתּוֹ שִׁמְעוֹן:
חרב יעקב = הלוי שנאמר עליו בספר שמות פרק לב כתוב:
כז) וַיֹּאמֶר לָהֶם כֹּה אָמַר יְהֹוָה אֱלֹהֵי יִשְׂרָאֵל שִׂימוּ אִישׁ חַרְבּוֹ עַל יְרֵכוֹ עִבְרוּ וָשׁוּבוּ מִשַּׁעַר לָשַׁעַר בַּמַּחֲנֶה וְהִרְגוּ אִישׁ אֶת אָחִיו וְאִישׁ אֶת רֵעֵהוּ וְאִישׁ אֶת קְרֹבוֹ:
וגם על הכהן הלוי שנקרא יהוה כתוב בספר דברים פרק לג:(כט) אַשְׁרֶיךָ יִשְׂרָאֵל מִי כָמוֹךָ עַם נוֹשַׁע בַּיהֹוָה מָגֵן עֶזְרֶךָ וַאֲשֶׁר חֶרֶב גַּאֲוָתֶךָ וְיִכָּחֲשׁוּ אֹיְבֶיךָ לָךְ וְאַתָּה עַל בָּמוֹתֵימוֹ תִדְרֹךְ:
