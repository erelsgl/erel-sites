השקר לא ינפץ את "אבן השתייה"...
השקר לא יחזיר לידי הכהן הגדול את "לוחות הברית"...

מורה לתורה משה דין עבד יהוה משיב לרשע כרשעתו:
מלחמות חינם אתה נלחם כי ברוח זרה ולא בקודש
על כן אתה מסלף את התורה בפחזותך כי גם את הלשון היהודית עברית שפת הקודש לא ידעת
כי נפש אמת לא הומיה כמו שכתוב בשיר "התקוה" כי משם לקחת את זה כי תקותך תקות כזב על כן אתה הומה ולא מייחל לאלהים
אשר כתוב בספר תהילים:
פרק מב
(ו) מַה תִּשְׁתּוֹחֲחִי נַפְשִׁי וַתֶּהֱמִי עָלָי הוֹחִלִי לֵאלֹהִים כִּי עוֹד אוֹדֶנּוּ יְשׁוּעוֹת פָּנָיו:
(יב) מַה תִּשְׁתּוֹחֲחִי נַפְשִׁי וּמַה תֶּהֱמִי עָלָי הוֹחִילִי לֵאלֹהִים כִּי עוֹד אוֹדֶנּוּ יְשׁוּעֹת פָּנַי וֵאלֹהָי
פרק מג:
(ה) מַה תִּשְׁתּוֹחֲחִי נַפְשִׁי וּמַה תֶּהֱמִי עָלָי הוֹחִילִי לֵאלֹהִים כִּי עוֹד אוֹדֶנּוּ יְשׁוּעֹת פָּנַי וֵאלֹהָי

וכעת קרא בספר תהילים את כל פרק נט כי פסוק 7 ופסוק 15 עליך כתובים:
ספר תהילים פרק נט
(א) לַמְנַצֵּחַ אַל תַּשְׁחֵת לְדָוִד מִכְתָּם בִּשְׁלֹחַ שָׁאוּל וַיִּשְׁמְרוּ אֶת הַבַּיִת לַהֲמִיתוֹ:
(ב) הַצִּילֵנִי מֵאֹיְבַי אֱלֹהָי מִמִּתְקוֹמְמַי תְּשַׂגְּבֵנִי:
(ג) הַצִּילֵנִי מִפֹּעֲלֵי אָוֶן וּמֵאַנְשֵׁי דָמִים הוֹשִׁיעֵנִי:
(ד) כִּי הִנֵּה אָרְבוּ לְנַפְשִׁי יָגוּרוּ עָלַי עַזִּים לֹא פִשְׁעִי וְלֹא חַטָּאתִי יְהֹוָה:
(ה) בְּלִי עָוֹן יְרֻצוּן וְיִכּוֹנָנוּ עוּרָה לִקְרָאתִי וּרְאֵה:
(ו) וְאַתָּה יְהֹוָה אֱלֹהִים צְבָאוֹת אֱלֹהֵי יִשְׂרָאֵל הָקִיצָה לִפְקֹד כָּל הַגּוֹיִם אַל תָּחֹן כָּל בֹּגְדֵי אָוֶן סֶלָה:
(ז) יָשׁוּבוּ לָעֶרֶב יֶהֱמוּ כַכָּלֶב וִיסוֹבְבוּ עִיר:
(ח) הִנֵּה יַבִּיעוּן בְּפִיהֶם חֲרָבוֹת בְּשִׂפְתוֹתֵיהֶם כִּי מִי שֹׁמֵעַ:
(ט) וְאַתָּה יְהֹוָה תִּשְׂחַק לָמוֹ תִּלְעַג לְכָל גּוֹיִם:
(י) עֻזּוֹ אֵלֶיךָ אֶשְׁמֹרָה כִּי אֱלֹהִים מִשְׂגַּבִּי:
(יא) אֱלֹהֵי חסדו {חַסְדִּי} יְקַדְּמֵנִי אֱלֹהִים יַרְאֵנִי בְשֹׁרְרָי:
(יב) אַל תַּהַרְגֵם פֶּן יִשְׁכְּחוּ עַמִּי הֲנִיעֵמוֹ בְחֵילְךָ וְהוֹרִידֵמוֹ מָגִנֵּנוּ אֲדֹנָי:
(יג) חַטַּאת פִּימוֹ דְּבַר שְׂפָתֵימוֹ וְיִלָּכְדוּ בִגְאוֹנָם וּמֵאָלָה וּמִכַּחַשׁ יְסַפֵּרוּ:
(יד) כַּלֵּה בְחֵמָה כַּלֵּה וְאֵינֵמוֹ וְיֵדְעוּ כִּי אֱלֹהִים מֹשֵׁל בְּיַעֲקֹב לְאַפְסֵי הָאָרֶץ סֶלָה:
(טו) וְיָשֻׁבוּ לָעֶרֶב יֶהֱמוּ כַכָּלֶב וִיסוֹבְבוּ עִיר:
(טז) הֵמָּה ינועון {יְנִיעוּן} לֶאֱכֹל אִם לֹא יִשְׂבְּעוּ וַיָּלִינוּ:
(יז) וַאֲנִי אָשִׁיר עֻזֶּךָ וַאֲרַנֵּן לַבֹּקֶר חַסְדֶּךָ כִּי הָיִיתָ מִשְׂגָּב לִי וּמָנוֹס בְּיוֹם צַר לִי:
(יח) עֻזִּי אֵלֶיךָ אֲזַמֵּרָה כִּי אֱלֹהִים מִשְׂגַּבִּי אֱלֹהֵי חַסְדִּי:
ובספר תהילים פרק פג פסוק 3 כתוב:
(ג) כִּי הִנֵּה אוֹיְבֶיךָ יֶהֱמָיוּן וּמְשַׂנְאֶיךָ נָשְׂאוּ רֹאשׁ:
על כן ידעתי כי אויב אתה לאלהים
אך אני אֶרְדּוֹף אוֹיְבַי וְאַשִֹּיגֵם וְלֹא אָשׁוּב עַד כַּלּוֹתָם:
וכעת ראה מה כתוב בספר דברים פרק לג בפסוק יא על משנאי הנָשְׂאוּ רֹאשׁ רשעים כמוך
(ח) וּלְלֵוִי אָמַר תֻּמֶּיךָ וְאוּרֶיךָ לְאִישׁ חֲסִידֶךָ אֲשֶׁר נִסִּיתוֹ בְּמַסָּה תְּרִיבֵהוּ עַל מֵי מְרִיבָה:
(ט) הָאֹמֵר לְאָבִיו וּלְאִמּוֹ לֹא רְאִיתִיו וְאֶת אֶחָיו לֹא הִכִּיר וְאֶת בָּנָו לֹא יָדָע כִּי שָׁמְרוּ אִמְרָתֶךָ וּבְרִיתְךָ יִנְצֹרוּ:
(י) יוֹרוּ מִשְׁפָּטֶיךָ לְיַעֲקֹב וְתוֹרָתְךָ לְיִשְׂרָאֵל יָשִׂימוּ קְטוֹרָה בְּאַפֶּךָ וְכָלִיל עַל מִזְבְּחֶךָ:
(יא) בָּרֵךְ יְהֹוָה חֵילוֹ וּפֹעַל יָדָיו תִּרְצֶה מְחַץ מָתְנַיִם קָמָיו וּמְשַׂנְאָיו מִן יְקוּמוּן:

באשר לדבריך "אבן השתיה"
זה מושג שלא נמצא בכל המקרא וכי אתה ממציא ומסלף את התורה
וגם יש לי חדשות בשבילך כי "התעודה האלהית" שקראת לתעודתך היא לא "התעודה" יען כי כתבת על סמך היסטוריונים זרים ברוח זרה ולא בקודש
