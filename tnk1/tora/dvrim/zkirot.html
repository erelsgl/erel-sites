זכרון וזכירות בספר דברים

קוד: זכרון וזכירות בספר דברים בתנ"ך

סוג: לא גמור

מאת: אראל

אל: לא גמור

ספר דברים מלא במצוות על הזכרון. משה רבנו מצווה על בני-ישראל לזכור אירועים רבים בתולדותיהם; לכל ציווי כזה יש מטרה אחרת.

 

יש ציוויים, שמטרתם למנוע טעות אפשרית שבני-ישראל עלולים לעשות:

  • עליהם לזכור שהם לא ראו כל תמונה במעמד הר סיני, כדי שלא יטעו ויעשו להם פסל ותמונה כסמל לה' (דברים ד9-23) "רַק הִשָּׁמֶר לְךָ וּשְׁמֹר נַפְשְׁךָ מְאֹד, פֶּן-תִּשְׁכַּח אֶת-הַדְּבָרִים אֲשֶׁר-רָאוּ עֵינֶיךָ וּפֶן-יָסוּרוּ מִלְּבָבְךָ, כֹּל, יְמֵי חַיֶּיךָ; וְהוֹדַעְתָּם לְבָנֶיךָ, וְלִבְנֵי בָנֶיךָ. יוֹם, אֲשֶׁר עָמַדְתָּ לִפְנֵי ה' אֱלֹהֶיךָ בְּחֹרֵב, בֶּאֱמֹר ה' אֵלַי הַקְהֶל-לִי אֶת-הָעָם, וְאַשְׁמִעֵם אֶת-דְּבָרָי:  אֲשֶׁר יִלְמְדוּן לְיִרְאָה אֹתִי, כָּל-הַיָּמִים אֲשֶׁר הֵם חַיִּים עַל-הָאֲדָמָה, וְאֶת-בְּנֵיהֶם, יְלַמֵּדוּן...  וְנִשְׁמַרְתֶּם מְאֹד לְנַפְשֹׁתֵיכֶם:  כִּי לֹא רְאִיתֶם, כָּל-תְּמוּנָה, בְּיוֹם דִּבֶּר ה' אֲלֵיכֶם בְּחֹרֵב, מִתּוֹךְ הָאֵשׁ.  פֶּן-תַּשְׁחִתוּן--וַעֲשִׂיתֶם לָכֶם פֶּסֶל, תְּמוּנַת כָּל-סָמֶל... הִשָּׁמְרוּ לָכֶם, פֶּן-תִּשְׁכְּחוּ אֶת-בְּרִית ה' אֱלֹהֵיכֶם, אֲשֶׁר כָּרַת, עִמָּכֶם; וַעֲשִׂיתֶם לָכֶם פֶּסֶל תְּמוּנַת כֹּל, אֲשֶׁר צִוְּךָ ה' אֱלֹהֶיךָ."
  • עליהם לזכור שה' הוציא אותם מבית עבדים, כדי שלא יטעו ויודו לאלהי הכנענים, על כך שזכו לרשת את בתיהם של הכנענים (דברים ו10-15): "וְהָיָה כִּי יְבִיאֲךָ ה' אֱלֹהֶיךָ, אֶל-הָאָרֶץ אֲשֶׁר נִשְׁבַּע לַאֲבֹתֶיךָ לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב--לָתֶת לָךְ:  עָרִים גְּדֹלֹת וְטֹבֹת, אֲשֶׁר לֹא-בָנִיתָ.  וּבָתִּים מְלֵאִים כָּל-טוּב, אֲשֶׁר לֹא-מִלֵּאתָ, וּבֹרֹת חֲצוּבִים אֲשֶׁר לֹא-חָצַבְתָּ, כְּרָמִים וְזֵיתִים אֲשֶׁר לֹא-נָטָעְתָּ; וְאָכַלְתָּ, וְשָׂבָעְתָּ.  הִשָּׁמֶר לְךָ, פֶּן-תִּשְׁכַּח אֶת-ה', אֲשֶׁר הוֹצִיאֲךָ מֵאֶרֶץ מִצְרַיִם, מִבֵּית עֲבָדִים.  אֶת-ה' אֱלֹהֶיךָ תִּירָא, וְאֹתוֹ תַעֲבֹד; וּבִשְׁמוֹ, תִּשָּׁבֵעַ.  לֹא תֵלְכוּן, אַחֲרֵי אֱלֹהִים אֲחֵרִים--מֵאֱלֹהֵי, הָעַמִּים, אֲשֶׁר, סְבִיבוֹתֵיכֶם.  כִּי אֵל קַנָּא ה' אֱלֹהֶיךָ, בְּקִרְבֶּךָ:  פֶּן-יֶחֱרֶה אַף-ה' אֱלֹהֶיךָ, בָּךְ, וְהִשְׁמִידְךָ, מֵעַל פְּנֵי הָאֲדָמָה."
  • עליהם לזכור את הפורענות שהביא ה' על פרעה ומצרים, כדי שלא יטעו ויפחדו מהגויים הרבים בארץ כנען, דבר שעלול לגרום לכך שיכרתו איתם ברית ויעבדו את אלהיהם (דברים ז16-18): "וְאָכַלְתָּ אֶת-כָּל-הָעַמִּים, אֲשֶׁר ה' אֱלֹהֶיךָ נֹתֵן לָךְ--לֹא-תָחוֹס עֵינְךָ, עֲלֵיהֶם; וְלֹא תַעֲבֹד אֶת-אֱלֹהֵיהֶם, כִּי-מוֹקֵשׁ הוּא לָךְ.  {ס} כִּי תֹאמַר בִּלְבָבְךָ, רַבִּים הַגּוֹיִם הָאֵלֶּה מִמֶּנִּי; אֵיכָה אוּכַל, לְהוֹרִישָׁם.  לֹא תִירָא, מֵהֶם:  זָכֹר תִּזְכֹּר, אֵת אֲשֶׁר-עָשָׂה ה' אֱלֹהֶיךָ, לְפַרְעֹה, וּלְכָל-מִצְרָיִם."
  • עליהם לזכור את הדרך הקשה שבה הוביל אותם ה' במדבר, כדי שלא יטעו ויחשבו שהם עשו לעצמם את כל העושר שיש להם בארץ כנען, אלא יידעו שה' נתן להם את הכוח לעשות את העושר הזה (דברים ח1-18): "כָּל-הַמִּצְוָה, אֲשֶׁר אָנֹכִי מְצַוְּךָ הַיּוֹם--תִּשְׁמְרוּן לַעֲשׂוֹת:  לְמַעַן תִּחְיוּן וּרְבִיתֶם, וּבָאתֶם וִירִשְׁתֶּם אֶת-הָאָרֶץ, אֲשֶׁר-נִשְׁבַּע ה', לַאֲבֹתֵיכֶם.  וְזָכַרְתָּ אֶת-כָּל-הַדֶּרֶךְ, אֲשֶׁר הוֹלִיכְךָ ה' אֱלֹהֶיךָ זֶה אַרְבָּעִים שָׁנָה--בַּמִּדְבָּר:  לְמַעַן עַנֹּתְךָ לְנַסֹּתְךָ, לָדַעַת אֶת-אֲשֶׁר בִּלְבָבְךָ הֲתִשְׁמֹר מִצְו ‍ ֹתָו--אִם-לֹא.  וַיְעַנְּךָ, וַיַּרְעִבֶךָ, וַיַּאֲכִלְךָ אֶת-הַמָּן אֲשֶׁר לֹא-יָדַעְתָּ, וְלֹא יָדְעוּן אֲבֹתֶיךָ:  לְמַעַן הוֹדִיעֲךָ, כִּי לֹא עַל-הַלֶּחֶם לְבַדּוֹ יִחְיֶה הָאָדָם--כִּי עַל-כָּל-מוֹצָא פִי-ה', יִחְיֶה הָאָדָם.  שִׂמְלָתְךָ לֹא בָלְתָה, מֵעָלֶיךָ, וְרַגְלְךָ, לֹא בָצֵקָה--זֶה, אַרְבָּעִים שָׁנָה... כִּי ה' אֱלֹהֶיךָ, מְבִיאֲךָ אֶל-אֶרֶץ טוֹבָה... הִשָּׁמֶר לְךָ, פֶּן-תִּשְׁכַּח אֶת-ה' אֱלֹהֶיךָ, לְבִלְתִּי שְׁמֹר מִצְו ‍ ֹתָיו וּמִשְׁפָּטָיו וְחֻקֹּתָיו, אֲשֶׁר אָנֹכִי מְצַוְּךָ הַיּוֹם.  פֶּן-תֹּאכַל, וְשָׂבָעְתָּ; וּבָתִּים טֹבִים תִּבְנֶה, וְיָשָׁבְתָּ...  וְרָם, לְבָבֶךָ; וְשָׁכַחְתָּ אֶת-ה' אֱלֹהֶיךָ, הַמּוֹצִיאֲךָ מֵאֶרֶץ מִצְרַיִם מִבֵּית עֲבָדִים. הַמּוֹלִיכְךָ בַּמִּדְבָּר הַגָּדֹל וְהַנּוֹרָא, נָחָשׁ שָׂרָף וְעַקְרָב, וְצִמָּאוֹן, אֲשֶׁר אֵין-מָיִם; הַמּוֹצִיא לְךָ מַיִם, מִצּוּר הַחַלָּמִישׁ...  וְאָמַרְתָּ, בִּלְבָבֶךָ:  כֹּחִי וְעֹצֶם יָדִי, עָשָׂה לִי אֶת-הַחַיִל הַזֶּה.  וְזָכַרְתָּ, אֶת-ה' אֱלֹהֶיךָ--כִּי הוּא הַנֹּתֵן לְךָ כֹּחַ, לַעֲשׂוֹת חָיִל:  לְמַעַן הָקִים אֶת-בְּרִיתוֹ אֲשֶׁר-נִשְׁבַּע לַאֲבֹתֶיךָ, כַּיּוֹם הַזֶּה." (ע"ע " כוחי ועוצם ידי עשה לי את החיל הזה ")
  • עליהם לזכור את כל הפעמים שהקציפו את ה' במדבר, כדי שלא יטעו ויחשבו שהם יורשים את הארץ בזכות צדקתם, אלא יידעו שהם נמצאים שם "על תנאי" ויש להם עוד הרבה לשפר (דברים ט4-7): "אַל-תֹּאמַר בִּלְבָבְךָ, בַּהֲדֹף ה' אֱלֹהֶיךָ אֹתָם מִלְּפָנֶיךָ לֵאמֹר, בְּצִדְקָתִי הֱבִיאַנִי ה', לָרֶשֶׁת אֶת-הָאָרֶץ הַזֹּאת:  וּבְרִשְׁעַת הַגּוֹיִם הָאֵלֶּה, ה' מוֹרִישָׁם מִפָּנֶיךָ.  לֹא בְצִדְקָתְךָ, וּבְיֹשֶׁר לְבָבְךָ, אַתָּה בָא, לָרֶשֶׁת אֶת-אַרְצָם:  כִּי בְּרִשְׁעַת הַגּוֹיִם הָאֵלֶּה, ה' אֱלֹהֶיךָ מוֹרִישָׁם מִפָּנֶיךָ, וּלְמַעַן הָקִים אֶת-הַדָּבָר אֲשֶׁר נִשְׁבַּע ה' לַאֲבֹתֶיךָ, לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב.  וְיָדַעְתָּ, כִּי לֹא בְצִדְקָתְךָ ה' אֱלֹהֶיךָ נֹתֵן לְךָ אֶת-הָאָרֶץ הַטּוֹבָה הַזֹּאת--לְרִשְׁתָּהּ:  כִּי עַם-קְשֵׁה-עֹרֶף, אָתָּה.  זְכֹר, אַל-תִּשְׁכַּח, אֵת אֲשֶׁר-הִקְצַפְתָּ אֶת-ה' אֱלֹהֶיךָ, בַּמִּדְבָּר:  לְמִן-הַיּוֹם אֲשֶׁר-יָצָאתָ מֵאֶרֶץ מִצְרַיִם, עַד-בֹּאֲכֶם עַד-הַמָּקוֹם הַזֶּה, מַמְרִים הֱיִיתֶם, עִם-ה'".

 

ויש ציוויים שנותנים טעם ונימוק למצוות שונות:

 

1. ציוויים לזכור את העבדות במצרים ואת יציאת מצרים:

  • זכרון העבדות במצרים נותן טעם לשמחת החג: השמחה של האדם גדולה יותר, כאשר הוא זוכר מאיזו צרה גדולה הוא נחלץ (דברים טז9-12): "שִׁבְעָה שָׁבֻעֹת, תִּסְפָּר-לָךְ:  מֵהָחֵל חֶרְמֵשׁ, בַּקָּמָה, תָּחֵל לִסְפֹּר, שִׁבְעָה שָׁבֻעוֹת.  וְעָשִׂיתָ חַג שָׁבֻעוֹת, לה' אֱלֹהֶיךָ--מִסַּת נִדְבַת יָדְךָ, אֲשֶׁר תִּתֵּן:  כַּאֲשֶׁר יְבָרֶכְךָ, ה' אֱלֹהֶיךָ.  וְשָׂמַחְתָּ לִפְנֵי ה' אֱלֹהֶיךָ, אַתָּה וּבִנְךָ וּבִתֶּךָ וְעַבְדְּךָ וַאֲמָתֶךָ, וְהַלֵּוִי אֲשֶׁר בִּשְׁעָרֶיךָ, וְהַגֵּר וְהַיָּתוֹם וְהָאַלְמָנָה אֲשֶׁר בְּקִרְבֶּךָ--בַּמָּקוֹם, אֲשֶׁר יִבְחַר ה' אֱלֹהֶיךָ, לְשַׁכֵּן שְׁמוֹ, שָׁם.  וְזָכַרְתָּ, כִּי-עֶבֶד הָיִיתָ בְּמִצְרָיִם; וְשָׁמַרְתָּ וְעָשִׂיתָ, אֶת-הַחֻקִּים הָאֵלֶּה".
  • זכרון העבדות במצרים, והפדיון משם, נותן טעם למצוה לשפוט בצדק את החלשים - הגר, היתום והאלמנה (דברים כד17-18): "לֹא תַטֶּה, מִשְׁפַּט גֵּר יָתוֹם; וְלֹא תַחֲבֹל, בֶּגֶד אַלְמָנָה. וְזָכַרְתָּ, כִּי עֶבֶד הָיִיתָ בְּמִצְרַיִם, וַיִּפְדְּךָ ה' אֱלֹהֶיךָ מִשָּׁם; עַל-כֵּן אָנֹכִי מְצַוְּךָ, לַעֲשׂוֹת, אֶת-הַדָּבָר, הַזֶּה."

 

2. ציוויים לזכור את העבדות בארץ מצרים:

  • זכרון העבדות בארץ מצרים, והיציאה מהעבדות אל המנוחה, נותן טעם לכך שצריך לתת מנוחה לעבדים ביום השבת (דברים ה11-14): "שָׁמוֹר אֶת-יוֹם הַשַּׁבָּת, לְקַדְּשׁוֹ, כַּאֲשֶׁר צִוְּךָ, ה' אֱלֹהֶיךָ.  שֵׁשֶׁת יָמִים תַּעֲבֹד, וְעָשִׂיתָ כָּל-מְלַאכְתֶּךָ.  וְיוֹם, הַשְּׁבִיעִי--שַׁבָּת, לַיהוָה אֱלֹהֶיךָ:  לֹא תַעֲשֶׂה כָל-מְלָאכָה אַתָּה וּבִנְךָ-וּבִתֶּךָ וְעַבְדְּךָ-וַאֲמָתֶךָ וְשׁוֹרְךָ וַחֲמֹרְךָ וְכָל-בְּהֶמְתֶּךָ, וְגֵרְךָ אֲשֶׁר בִּשְׁעָרֶיךָ--לְמַעַן יָנוּחַ עַבְדְּךָ וַאֲמָתְךָ, כָּמוֹךָ.  וְזָכַרְתָּ, כִּי עֶבֶד הָיִיתָ בְּאֶרֶץ מִצְרַיִם, וַיֹּצִאֲךָ ה' אֱלֹהֶיךָ מִשָּׁם בְּיָד חֲזָקָה וּבִזְרֹעַ נְטוּיָה; עַל-כֵּן, צִוְּךָ ה' אֱלֹהֶיךָ, לַעֲשׂוֹת, אֶת-יוֹם הַשַּׁבָּת."
  • זכרון העבדות בארץ מצרים, והפדיון ברכוש גדול, נותן טעם לכך שצריך להעניק לעבדים משתחררים מתנות, מיבול הארץ שה' נתן לנו במתנה (דברים טו12-15): "כִּי-יִמָּכֵר לְךָ אָחִיךָ הָעִבְרִי, אוֹ הָעִבְרִיָּה--וַעֲבָדְךָ, שֵׁשׁ שָׁנִים; וּבַשָּׁנָה, הַשְּׁבִיעִת, תְּשַׁלְּחֶנּוּ חָפְשִׁי, מֵעִמָּךְ.  וְכִי-תְשַׁלְּחֶנּוּ חָפְשִׁי, מֵעִמָּךְ--לֹא תְשַׁלְּחֶנּוּ, רֵיקָם.  הַעֲנֵיק תַּעֲנִיק, לוֹ, מִצֹּאנְךָ, וּמִגָּרְנְךָ וּמִיִּקְבֶךָ:  אֲשֶׁר בֵּרַכְךָ ה' אֱלֹהֶיךָ, תִּתֶּן-לוֹ.  וְזָכַרְתָּ, כִּי עֶבֶד הָיִיתָ בְּאֶרֶץ מִצְרַיִם, וַיִּפְדְּךָ ה' אֱלֹהֶיךָ; עַל-כֵּן אָנֹכִי מְצַוְּךָ, אֶת-הַדָּבָר הַזֶּה—הַיּוֹם".
  • זכרון העבדות בארץ מצרים, ארץ שלא היתה שלנו, נותן טעם למצוה להשאיר בשדה חלק מהיבול של הארץ שקיבלנו מה' לאחר שהשתחררנו משם (דברים כד20-22): "כִּי תַחְבֹּט זֵיתְךָ, לֹא תְפַאֵר אַחֲרֶיךָ:  לַגֵּר לַיָּתוֹם וְלָאַלְמָנָה, יִהְיֶה.  כִּי תִבְצֹר כַּרְמְךָ, לֹא תְעוֹלֵל אַחֲרֶיךָ:  לַגֵּר לַיָּתוֹם וְלָאַלְמָנָה, יִהְיֶה. וְזָכַרְתָּ, כִּי-עֶבֶד הָיִיתָ בְּאֶרֶץ מִצְרָיִם; עַל-כֵּן אָנֹכִי מְצַוְּךָ, לַעֲשׂוֹת, אֶת-הַדָּבָר, הַזֶּה".

 

3. ציוויים לזכור אירועים שקרו בדרך ממצרים לארץ:

  • זכרון הצרעת שבאה על מרים לאחר שאמרה לשון-הרע, נותן טעם למצוה להישמר בנגע הצרעת, כי הצרעת היא אות שהאדם צריך להקפיד יותר על שמירת הלשון (דברים כד8-9): "הִשָּׁמֶר בְּנֶגַע-הַצָּרַעַת לִשְׁמֹר מְאֹד, וְלַעֲשׂוֹת:  כְּכֹל אֲשֶׁר-יוֹרוּ אֶתְכֶם הַכֹּהֲנִים הַלְוִיִּם, כַּאֲשֶׁר צִוִּיתִם--תִּשְׁמְרוּ לַעֲשׂוֹת.  זָכוֹר, אֵת אֲשֶׁר-עָשָׂה ה' אֱלֹהֶיךָ לְמִרְיָם, בַּדֶּרֶךְ, בְּצֵאתְכֶם מִמִּצְרָיִם"
  • זכרון מעשיו הרעים של עמלק, נותן טעם למחייתו (דברים כה14-16): "זָכוֹר, אֵת אֲשֶׁר-עָשָׂה לְךָ עֲמָלֵק, בַּדֶּרֶךְ, בְּצֵאתְכֶם מִמִּצְרָיִם.  אֲשֶׁר קָרְךָ בַּדֶּרֶךְ, וַיְזַנֵּב בְּךָ כָּל-הַנֶּחֱשָׁלִים אַחֲרֶיךָ--וְאַתָּה, עָיֵף וְיָגֵעַ; וְלֹא יָרֵא, אֱלֹהִים.  וְהָיָה בְּהָנִיחַ ה' אֱלֹהֶיךָ לְךָ מִכָּל-אֹיְבֶיךָ מִסָּבִיב, בָּאָרֶץ אֲשֶׁר ה'-אֱלֹהֶיךָ נֹתֵן לְךָ נַחֲלָה לְרִשְׁתָּהּ--תִּמְחֶה אֶת-זֵכֶר עֲמָלֵק, מִתַּחַת הַשָּׁמָיִם; לֹא, תִּשְׁכָּח".

 

פסוקי זיכרון נוספים בספר דברים

 

  • זכרון המצוות שקשורות למעשר (דברים כו 12-13): "כִּי תְכַלֶּה לַעְשֵׂר אֶת-כָּל-מַעְשַׂר תְּבוּאָתְךָ, בַּשָּׁנָה הַשְּׁלִישִׁת--שְׁנַת הַמַּעֲשֵׂר:  וְנָתַתָּה לַלֵּוִי, לַגֵּר לַיָּתוֹם וְלָאַלְמָנָה, וְאָכְלוּ בִשְׁעָרֶיךָ, וְשָׂבֵעוּ.  וְאָמַרְתָּ לִפְנֵי ה' אֱלֹהֶיךָ בִּעַרְתִּי הַקֹּדֶשׁ מִן-הַבַּיִת, וְגַם נְתַתִּיו לַלֵּוִי וְלַגֵּר לַיָּתוֹם וְלָאַלְמָנָה, כְּכָל-מִצְוָתְךָ, אֲשֶׁר צִוִּיתָנִי:  לֹא-עָבַרְתִּי מִמִּצְו ‍ ֹתֶיךָ, וְלֹא שָׁכָחְתִּי".
  • זכרון "שירת האזינו", שמעידה על כך שה' ידע מראש את הצרות שיבואו על ישראל (דברים לא21): "וְהָיָה כִּי-תִמְצֶאןָ אֹתוֹ רָעוֹת רַבּוֹת, וְצָרוֹת, וְעָנְתָה הַשִּׁירָה הַזֹּאת לְפָנָיו לְעֵד, כִּי לֹא תִשָּׁכַח מִפִּי זַרְעוֹ:  כִּי יָדַעְתִּי אֶת-יִצְרוֹ, אֲשֶׁר הוּא עֹשֶׂה הַיּוֹם, בְּטֶרֶם אֲבִיאֶנּוּ, אֶל-הָאָרֶץ אֲשֶׁר נִשְׁבַּעְתִּי". ובשירה עצמה (דברים לב5-9): "שִׁחֵת לוֹ לֹא, בָּנָיו מוּמָם:  {ס}  דּוֹר עִקֵּשׁ, וּפְתַלְתֹּל.  {ר} הַ  לה', תִּגְמְלוּ-זֹאת--  {ס}  עַם נָבָל, וְלֹא חָכָם:  {ר}   הֲלוֹא-הוּא אָבִיךָ קָּנֶךָ,  {ס}  הוּא עָשְׂךָ וַיְכֹנְנֶךָ.  {ר} זְכֹר יְמוֹת עוֹלָם, בִּינוּ שְׁנוֹת דֹּר-וָדֹר;  {ס}  שְׁאַל אָבִיךָ וְיַגֵּדְךָ, זְקֵנֶיךָ וְיֹאמְרוּ לָךְ.  {ר} בְּהַנְחֵל עֶלְיוֹן גּוֹיִם,  {ס}  בְּהַפְרִידוֹ בְּנֵי אָדָם;  {ר} יַצֵּב גְּבֻלֹת עַמִּים,  {ס}  לְמִסְפַּר בְּנֵי יִשְׂרָאֵל.  {ר} כִּי חֵלֶק ה', עַמּוֹ:  {ס}  יַעֲקֹב, חֶבֶל נַחֲלָתוֹ.  {ר}"
