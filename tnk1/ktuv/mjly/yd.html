ראשי>הגדרות מושגים>

יד = כוח העשיה והעבודה

קוד: יד = כוח העשיה והעבודה בתנ"ך

סוג: הגדרה

מאת: אראל

אל:

יד היא האמצעי העיקרי שבעזרתו האדם עובד ופועל, ולכן המילה יד משמשת בהשאלה לציין את כוח העשיה והעבודה של האדם, למשל:
  • דברים טו10: "נָתוֹן תִּתֵּן לוֹ, וְלֹא יֵרַע לְבָבְךָ בְּתִתְּךָ לוֹ, כִּי בִּגְלַל הַדָּבָר הַזֶּה יְבָרֶכְךָ ה' אֱלֹהֶיךָ בְּכָל מַעֲשֶׂךָ וּבְכֹל מִשְׁלַח יָדֶךָ", וכן פסוקים נוספים עם הביטויים מעשה יד / משלח יד, המציינים עיסוק בעבודה או מסחר.
  • ויקרא כה35: "וְכִי יָמוּךְ אָחִיךָ, וּמָטָה יָדוֹ עִמָּךְ, וְהֶחֱזַקְתָּ בּוֹ, גֵּר וְתוֹשָׁב וָחַי עִמָּךְ" - אדם שמטה ידו הוא אדם שאיבד את כוח העבודה שלו (פירוט).
  • במדבר יא23: "וַיֹּאמֶר ה' אֶל מֹשֶׁה 'הֲיַד ה' תִּקְצָר?! עַתָּה תִרְאֶה הֲיִקְרְךָ דְבָרִי אִם לֹא'", וכן פסוקים נוספים עם הביטוי קצר יד, המציין חוסר יכולת לעשות.
זוהי המשמעות העיקרית של המילה יד בספר משלי, המלמד אותנו להשתמש בחכמה וחריצות בכוח העבודה שלנו, ולעשות כמיטב יכולתנו בכל התחומים:
  • משלי ג27: "אַל תִּמְנַע טוֹב מִבְּעָלָיו, בִּהְיוֹת לְאֵל ידיך[יָדְךָ] לַעֲשׂוֹת" (פירוט).
  • משלי ו10: "מְעַט שֵׁנוֹת מְעַט תְּנוּמוֹת, מְעַט חִבֻּק יָדַיִם לִשְׁכָּב" (פירוט).
  • משלי ו16-17: "שֶׁשׁ הֵנָּה שָׂנֵא ה', וְשֶׁבַע [תּוֹעֲבַת] נַפְשׁוֹ: עֵינַיִם רָמוֹת, לְשׁוֹן שָׁקֶר, וְיָדַיִם שֹׁפְכוֹת דָּם נָקִי"
  • משלי י4: "רָאשׁ עֹשֶׂה כַף רְמִיָּה, וְיַד חָרוּצִים תַּעֲשִׁיר" (פירוט).
  • משלי יב14: "מִפְּרִי פִי אִישׁ יִשְׂבַּע טוֹב, וּגְמוּל יְדֵי אָדָם ישוב[יָשִׁיב] לוֹ" (פירוט).
  • משלי יב24: "יַד חָרוּצִים תִּמְשׁוֹל, וּרְמִיָּה תִּהְיֶה לָמַס" (פירוט).
  • משלי יג11: "הוֹן מֵהֶבֶל יִמְעָט, וְקֹבֵץ עַל יָד יַרְבֶּה" - מי שאוסף את הונו בהדרגה ובעבודה קשה, בסופו של דבר יהיה לו הרבה (פירוט).
  • משלי יד1: "חַכְמוֹת נָשִׁים בָּנְתָה בֵיתָהּ, וְאִוֶּלֶת בְּיָדֶיהָ תֶהֶרְסֶנּוּ" (פירוט).
  • משלי יז16: "לָמָּה זֶּה מְחִיר בְּיַד כְּסִיל לִקְנוֹת חָכְמָה, וְלֶב אָיִן?" (פירוט).
  • משלי יט24: "טָמַן עָצֵל יָדוֹ בַּצַּלָּחַת, גַּם אֶל פִּיהוּ לֹא יְשִׁיבֶנָּה" - העצל מנסה להסתיר את כוח העבודה שלו (פירוט), ולכן משלי כו15: "טָמַן עָצֵל יָדוֹ בַּצַּלָּחַת, נִלְאָה לַהֲשִׁיבָהּ אֶל פִּיו" - בסופו של דבר באמת יאבד את כוח העבודה שלו (פירוט).
  • משלי כא25: "תַּאֲוַת עָצֵל תְּמִיתֶנּוּ, כִּי מֵאֲנוּ יָדָיו לַעֲשׂוֹת"
  • משלי כד33: "מְעַט שֵׁנוֹת, מְעַט תְּנוּמוֹת, מְעַט חִבֻּק יָדַיִם לִשְׁכָּב"
  • משלי ל28: "שְׂמָמִית בְּיָדַיִם תְּתַפֵּשׂ, וְהִיא בְּהֵיכְלֵי מֶלֶךְ" - השממית (העכבישה) היא אמנם קטנה, אבל היא עובדת קשה ומצליחה לבנות לה בית מפואר כמו היכל (פירוט).
  • משלי לא19: "יָדֶיהָ שִׁלְּחָה בַכִּישׁוֹר, וְכַפֶּיהָ תָּמְכוּ פָלֶךְ. כפה פרשה לעני, וידיה שִׁלְּחָה לאביון" - אשת חיל עובדת וגם עוזרת לנזקקים (פירוט).
  • משלי לא31: "תְּנוּ לָהּ מִפְּרִי יָדֶיהָ, וִיהַלְלוּהָ בַשְּׁעָרִים מַעֲשֶׂיהָ" (פירוט).
יד מסמלת גם שלטון, כמו בפסוקים:
  • בראשית ט2: "וּמוֹרַאֲכֶם וְחִתְּכֶם יִהְיֶה עַל כָּל חַיַּת הָאָרֶץ וְעַל כָּל עוֹף הַשָּׁמָיִם בְּכֹל אֲשֶׁר תִּרְמֹשׂ הָאֲדָמָה וּבְכָל דְּגֵי הַיָּם בְּיֶדְכֶם נִתָּנוּ"
  • בראשית יד20: "וּבָרוּךְ אֵל עֶלְיוֹן אֲשֶׁר מִגֵּן צָרֶיךָ בְּיָדֶךָ וַיִּתֶּן לוֹ מַעֲשֵׂר מִכֹּל"
וכך גם בספר משלי, המלמד את האדם להשתחרר משיעבוד לאדם, ולהיות תחת שלטון ה' בלבד:
  • משלי ו5: "הִנָּצֵל כִּצְבִי מִיָּד, וּכְצִפּוֹר מִיַּד יָקוּשׁ" (פירוט)
  • משלי כא1: "פַּלְגֵי מַיִם לֶב מֶלֶךְ בְּיַד ה', עַל כָּל אֲשֶׁר יַחְפֹּץ יַטֶּנּוּ" (פירוט)
הביטוי הייחודי יד ליד מסמל את הצדק, העובר מיד ליד, מסמכות לסמכות, עד שבסופו של דבר הרע בא על עונשו:
  • משלי יא21: "יָד לְיָד לֹא יִנָּקֶה רָּע, וְזֶרַע צַדִּיקִים נִמְלָט" (פירוט).
  • משלי טז5: "תּוֹעֲבַת ה' כָּל גְּבַהּ לֵב, יָד לְיָד לֹא יִנָּקֶה" (פירוט)
היד יכולה לאטום את הפה, ולכן הביטוי יד לפה גם מסמל שתיקה:
  • משלי ל32: "אִם נָבַלְתָּ בְהִתְנַשֵּׂא, וְאִם זַמּוֹתָ - יָד לְפֶה" (פירוט)
  • ואולי גם ב משלי יח21: "מָוֶת וְחַיִּים בְּיַד לָשׁוֹן, וְאֹהֲבֶיהָ יֹאכַל פִּרְיָהּ" (פירוט)
פסוקים נוספים:
  • משלי א24: "יַעַן קָרָאתִי וַתְּמָאֵנוּ, נָטִיתִי יָדִי וְאֵין מַקְשִׁיב" (פירוט)
  • משלי ז20: "צְרוֹר הַכֶּסֶף לָקַח בְּיָדוֹ, לְיוֹם הַכֵּסֶא יָבֹא בֵיתוֹ" (פירוט)
  • משלי ח3: "לְיַד שְׁעָרִים לְפִי קָרֶת, מְבוֹא פְתָחִים תָּרֹנָּה" (פירוט)
  • משלי כו6: "מְקַצֶּה רַגְלַיִם, חָמָס שֹׁתֶה, שֹׁלֵחַ דְּבָרִים בְּיַד כְּסִיל" (פירוט)
  • משלי כו9: "חוֹחַ עָלָה בְיַד שִׁכּוֹר, וּמָשָׁל בְּפִי כְסִילִים" (פירוט)
למילה יד ישנן משמעויות רבות נוספות, ראו יד.
