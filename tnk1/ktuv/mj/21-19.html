טוב שבת בארץ מדבר, מאשת מדיינים וכעס

קוד: ביאור:משלי כא19 בתנ"ך

סוג: דיון1

מאת: אראל

אל:

→כא 18— משלי כא 19 — כא 20←

משלי כא19 : "טוֹב שֶׁבֶת בְּאֶרֶץ מִדְבָּר, מֵאֵשֶׁת מדונים[מִדְיָנִים] וָכָעַס."

 סגולות

טוב יותר לשבת (לגור) לבד, בארץ מדבר שוממה, מלגור עם אישה האוהבת מדיינים (מריבות) וכעס.

 מצודות

טוב שבת בארץ מדבר - במקום שאין אנשים, מלשבת במקום מיושב עם אשת-מדינים והתמדת כעס.


 עצות

על המשמעות המעשית של הפסוק, ראו בפירוש על הפסוק המקביל, משלי כא9: "טוֹב לָשֶׁבֶת עַל פִּנַּת גָּג, מֵאֵשֶׁת מִדְיָנִים וּבֵית חָבֶר" (פירוט). הפסוק שלנו מתאר מצב חמור יותר: "כבר אמר למעלה, שטוב לשבת על פנת גג מאשת מדינים אם אינה בעלת כעס, רק מעוררת מדנים עם שכניה ובני ביתה, צריך להתרחק מן החבורה, וידור עמה על פנת גג, שלא תמצא עם מי להתקוטט; אבל אם היא בעלת כעס, תכעיסהו גם שמה, וצריך לברוח ממנה אל המדבר" (מלבי"ם).

 הקבלות

1. בתחילת הפרק כבר נאמר, משלי כא9: "טוֹב לָשֶׁבֶת עַל פִּנַּת גָּג, מֵאֵשֶׁת מִדְיָנִים וּבֵית חָבֶר" (פירוט). הפסוק שלנו מתאר מצב חמור יותר: "כבר אמר למעלה, שטוב לשבת על פנת גג מאשת מדינים אם אינה בעלת כעס, רק מעוררת מדנים עם שכניה ובני ביתה, צריך להתרחק מן החבורה, וידור עמה על פנת גג, שלא תמצא עם מי להתקוטט; אבל אם היא בעלת כעס, תכעיסהו גם שמה, וצריך לברוח ממנה אל המדבר" (מלבי"ם).

2. אשת מדיינים - ע"פ הפשט הכוונה לאישה שאוהבת לריב, אבל ייתכן שהפסוק רומז גם לנשים המדייניות מבמדבר כה6: "וְהִנֵּה אִישׁ מִבְּנֵי יִשְׂרָאֵל בָּא, וַיַּקְרֵב אֶל אֶחָיו אֶת הַמִּדְיָנִית לְעֵינֵי מֹשֶׁה וּלְעֵינֵי כָּל עֲדַת בְּנֵי יִשְׂרָאֵל", במדבר כה3: "וַיִּצָּמֶד יִשְׂרָאֵל לְבַעַל פְּעוֹר, וַיִּחַר אַף ה' בְּיִשְׂרָאֵל".    טוב היה להם, לבני ישראל, לשבת בארץ מדבר, רחוק ממקום מושבם של עמים אחרים, מלבוא אל ארץ מדיין ולפגוש שם אשת מדיינים שגרמה להם להכעיס את ה'.

3. הרעיון שלפיו עדיף לשבת במדבר מלהיות בחברה נזכר גם בירמיהו ט1-2: "מִי יִתְּנֵנִי בַמִּדְבָּר מְלוֹן אֹרְחִים, וְאֶעֶזְבָה אֶת עַמִּי וְאֵלְכָה מֵאִתָּם, כִּי כֻלָּם מְנָאֲפִים, עֲצֶרֶת בֹּגְדִים. וַיַּדְרְכוּ אֶת לְשׁוֹנָם קַשְׁתָּם שֶׁקֶר וְלֹא לֶאֱמוּנָה גָּבְרוּ בָאָרֶץ כִּי מֵרָעָה אֶל רָעָה יָצָאוּ וְאֹתִי לֹא יָדָעוּ נְאֻם ה'". 

בנבואות הסמוכות נזכר גם כעס, כגון בירמיהו ח19: "הִנֵּה קוֹל שַׁוְעַת בַּת עַמִּי מֵאֶרֶץ מַרְחַקִּים: הה' אֵין בְּצִיּוֹן? אִם מַלְכָּהּ אֵין בָּהּ? מַדּוּעַ הִכְעִסוּנִי בִּפְסִלֵיהֶם, בְּהַבְלֵי נֵכָר?" (פירוט) ירמיהו יא17: "וה' צְבָאוֹת הַנּוֹטֵעַ אוֹתָךְ דִּבֶּר עָלַיִךְ רָעָה בִּגְלַל רָעַת בֵּית יִשְׂרָאֵל וּבֵית יְהוּדָה אֲשֶׁר עָשׂוּ לָהֶם לְהַכְעִסֵנִי לְקַטֵּר לַבָּעַל".

מכאן יש לפרש את פסוקנו: טוב יותר לאדם לשבת לבד בארץ מדבר, מלשבת עם בני ובנות ישראל המכעיסים את ה'.

4. מדבר נזכר פעמים רבות בקשר ליחסים שבין ה' לעם ישראל, בדברי הנביא ירמיהו ונביאים נוספים:

ירמיהו ב2: "הָלֹךְ וְקָרָאתָ בְאָזְנֵי יְרוּשָׁלִַם לֵאמֹר: כֹּה אָמַר ה': זָכַרְתִּי לָךְ חֶסֶד נְעוּרַיִךְ, אַהֲבַת כְּלוּלֹתָיִךְ, לֶכְתֵּךְ אַחֲרַי בַּמִּדְבָּר בְּאֶרֶץ לֹא זְרוּעָה" (פירוט).

ירמיהו ב6: "וְלֹא אָמְרוּ אַיֵּה ה', הַמַּעֲלֶה אֹתָנוּ מֵאֶרֶץ מִצְרָיִם, הַמּוֹלִיךְ אֹתָנוּ בַּמִּדְבָּר בְּאֶרֶץ עֲרָבָה וְשׁוּחָה בְּאֶרֶץ צִיָּה וְצַלְמָוֶת בְּאֶרֶץ לֹא עָבַר בָּהּ אִישׁ וְלֹא יָשַׁב אָדָם שָׁם".

ירמיהו ב24: "פֶּרֶה לִמֻּד מִדְבָּר בְּאַוַּת נפשו[נַפְשָׁהּ] שָׁאֲפָה רוּחַ תַּאֲנָתָהּ מִי יְשִׁיבֶנָּה כָּל מְבַקְשֶׁיהָ לֹא יִיעָפוּ בְּחָדְשָׁהּ יִמְצָאוּנְהָ" (פירוט).

ירמיהו ב31: "הַדּוֹר אַתֶּם רְאוּ דְבַר ה'! הֲמִדְבָּר הָיִיתִי לְיִשְׂרָאֵל, אִם אֶרֶץ מַאְפֵּלְיָה? מַדּוּעַ אָמְרוּ עַמִּי רַדְנוּ לוֹא נָבוֹא עוֹד אֵלֶיךָ"?

ירמיהו ד11: "בָּעֵת הַהִיא יֵאָמֵר לָעָם הַזֶּה וְלִירוּשָׁלִַם רוּחַ צַח שְׁפָיִים בַּמִּדְבָּר דֶּרֶךְ בַּת עַמִּי לוֹא לִזְרוֹת וְלוֹא לְהָבַר" (פירוט).

ירמיהו יב12: "עַל כָּל שְׁפָיִם בַּמִּדְבָּר בָּאוּ שֹׁדְדִים כִּי חֶרֶב לה' אֹכְלָה מִקְצֵה אֶרֶץ וְעַד קְצֵה הָאָרֶץ אֵין שָׁלוֹם לְכָל בָּשָׂר".

ירמיהו יג24: "וַאֲפִיצֵם כְּקַשׁ עוֹבֵר לְרוּחַ מִדְבָּר".

ירמיהו יז6: "וְהָיָה כְּעַרְעָר בָּעֲרָבָה וְלֹא יִרְאֶה כִּי יָבוֹא טוֹב וְשָׁכַן חֲרֵרִים בַּמִּדְבָּר אֶרֶץ מְלֵחָה וְלֹא תֵשֵׁב".

ירמיהו כג10: "כִּי מְנָאֲפִים מָלְאָה הָאָרֶץ כִּי מִפְּנֵי אָלָה אָבְלָה הָאָרֶץ יָבְשׁוּ נְאוֹת מִדְבָּר וַתְּהִי מְרוּצָתָם רָעָה וּגְבוּרָתָם לֹא כֵן".

ירמיהו לא1: "כֹּה אָמַר ה' מָצָא חֵן בַּמִּדְבָּר עַם שְׂרִידֵי חָרֶב הָלוֹךְ לְהַרְגִּיעוֹ יִשְׂרָאֵל".

יחזקאל יט13: "וְעַתָּה שְׁתוּלָה בַמִּדְבָּר בְּאֶרֶץ צִיָּה וְצָמָא".

יחזקאל כ10: "וָאוֹצִיאֵם מֵאֶרֶץ מִצְרָיִם וָאֲבִאֵם אֶל הַמִּדְבָּר".

יחזקאל כ13: "וַיַּמְרוּ בִי בֵית יִשְׂרָאֵל בַּמִּדְבָּר בְּחֻקּוֹתַי לֹא הָלָכוּ וְאֶת מִשְׁפָּטַי מָאָסוּ אֲשֶׁר יַעֲשֶׂה אֹתָם הָאָדָם וָחַי בָּהֶם וְאֶת שַׁבְּתֹתַי חִלְּלוּ מְאֹד וָאֹמַר לִשְׁפֹּךְ חֲמָתִי עֲלֵיהֶם בַּמִּדְבָּר לְכַלּוֹתָם".

יחזקאל כ15: "וְגַם אֲנִי נָשָׂאתִי יָדִי לָהֶם בַּמִּדְבָּר לְבִלְתִּי הָבִיא אוֹתָם אֶל הָאָרֶץ אֲשֶׁר נָתַתִּי זָבַת חָלָב וּדְבַשׁ צְבִי הִיא לְכָל הָאֲרָצוֹת".

יחזקאל כ17: "וַתָּחָס עֵינִי עֲלֵיהֶם מִשַּׁחֲתָם וְלֹא עָשִׂיתִי אוֹתָם כָּלָה בַּמִּדְבָּר".

יחזקאל כ18: "וָאֹמַר אַל בְּנֵיהֶם בַּמִּדְבָּר בְּחוּקֵּי אֲבוֹתֵיכֶם אַל תֵּלֵכוּ וְאֶת מִשְׁפְּטֵיהֶם אֶל תִּשְׁמֹרוּ וּבְגִלּוּלֵיהֶם אַל תִּטַּמָּאוּ".

יחזקאל כ21: "וַיַּמְרוּ בִי הַבָּנִים בְּחֻקּוֹתַי לֹא הָלָכוּ וְאֶת מִשְׁפָּטַי לֹא שָׁמְרוּ לַעֲשׂוֹת אוֹתָם אֲשֶׁר יַעֲשֶׂה אוֹתָם הָאָדָם וָחַי בָּהֶם אֶת שַׁבְּתוֹתַי חִלֵּלוּ וָאֹמַר לִשְׁפֹּךְ חֲמָתִי עֲלֵיהֶם לְכַלּוֹת אַפִּי בָּם בַּמִּדְבָּר".

יחזקאל כ23: "גַּם אֲנִי נָשָׂאתִי אֶת יָדִי לָהֶם בַּמִּדְבָּר לְהָפִיץ אֹתָם בַּגּוֹיִם וּלְזָרוֹת אוֹתָם בָּאֲרָצוֹת".

יחזקאל כ35-36: "וְהֵבֵאתִי אִתְּכֶם אֶל מִדְבַּר הָעַמִּים וְנִשְׁפַּטְתִּי אֶתְכֶם שָׁם פָּנִים אֶל פָּנִים. כַּאֲשֶׁר נִשְׁפַּטְתִּי אֶת אֲבוֹתֵיכֶם בְּמִדְבַּר אֶרֶץ מִצְרָיִם כֵּן אִשָּׁפֵט אִתְּכֶם נְאֻם אֲדֹנָי ה'".

יחזקאל כג42: "וְקוֹל הָמוֹן שָׁלֵו בָהּ וְאֶל אֲנָשִׁים מֵרֹב אָדָם מוּבָאִים סובאים[סָבָאִים] מִמִּדְבָּר וַיִּתְּנוּ צְמִידִים אֶל יְדֵיהֶן וַעֲטֶרֶת תִּפְאֶרֶת עַל רָאשֵׁיהֶן".

יחזקאל לד25: "וְכָרַתִּי לָהֶם בְּרִית שָׁלוֹם וְהִשְׁבַּתִּי חַיָּה רָעָה מִן הָאָרֶץ וְיָשְׁבוּ בַמִּדְבָּר לָבֶטַח וְיָשְׁנוּ בַּיְּעָרִים".

הושע ב5: "פֶּן אַפְשִׁיטֶנָּה עֲרֻמָּה וְהִצַּגְתִּיהָ כְּיוֹם הִוָּלְדָהּ וְשַׂמְתִּיהָ כַמִּדְבָּר וְשַׁתִּהָ כְּאֶרֶץ צִיָּה וַהֲמִתִּיהָ בַּצָּמָא".

הושע ב16: "לָכֵן הִנֵּה אָנֹכִי מְפַתֶּיהָ וְהֹלַכְתִּיהָ הַמִּדְבָּר וְדִבַּרְתִּי עַל לִבָּהּ".

הושע ט10: "כַּעֲנָבִים בַּמִּדְבָּר מָצָאתִי יִשְׂרָאֵל כְּבִכּוּרָה בִתְאֵנָה בְּרֵאשִׁיתָהּ רָאִיתִי אֲבוֹתֵיכֶם הֵמָּה בָּאוּ בַעַל פְּעוֹר וַיִּנָּזְרוּ לַבֹּשֶׁת וַיִּהְיוּ שִׁקּוּצִים כְּאָהֳבָם".

הושע יג5: "אֲנִי יְדַעְתִּיךָ בַּמִּדְבָּר בְּאֶרֶץ תַּלְאֻבוֹת".

הושע יג15: "כִּי הוּא בֵּן אַחִים יַפְרִיא יָבוֹא קָדִים רוּחַ ה' מִמִּדְבָּר עֹלֶה וְיֵבוֹשׁ מְקוֹרוֹ וְיֶחֱרַב מַעְיָנוֹ הוּא יִשְׁסֶה אוֹצַר כָּל כְּלִי חֶמְדָּה" (פירוט)

עמוס ב10: "וְאָנֹכִי הֶעֱלֵיתִי אֶתְכֶם מֵאֶרֶץ מִצְרָיִם וָאוֹלֵךְ אֶתְכֶם בַּמִּדְבָּר אַרְבָּעִים שָׁנָה לָרֶשֶׁת אֶת אֶרֶץ הָאֱמֹרִי" (פירוט).

עמוס ה25: "הַזְּבָחִים וּמִנְחָה הִגַּשְׁתֶּם לִי בַמִּדְבָּר אַרְבָּעִים שָׁנָה בֵּית יִשְׂרָאֵל".

המדבר נתפס, בחלק מהפסוקים, כמקום שבו האומה הישראלית היתה בקשר טהור וייחודי עם ה', לפני שהכעיסה אותו ובגדה בו עם "אלהי נכר הארץ".

מכאן כנראה פירשו חז"ל, שגם הפסוק הזה "על סלוק השכינה נאמר" (רש"י, ע"פ אבות דרבי נתן לד, בבלי ראש השנה לא:).
