הפך = הסתובב או שינה או הרס

קוד: הפך= בתנ"ך

סוג: הגדרה

מאת: אראל

אל:

השורש הפך מציין שינוי פתאומי וקיצוני:

שינוי תנוחה:

  • שופטים ז13: "וַיָּבֹא גִדְעוֹן וְהִנֵּה אִישׁ מְסַפֵּר לְרֵעֵהוּ חֲלוֹם וַיֹּאמֶר הִנֵּה חֲלוֹם חָלַמְתִּי וְהִנֵּה צלול[צְלִיל] לֶחֶם שְׂעֹרִים מִתְהַפֵּךְ בְּמַחֲנֵה מִדְיָן, וַיָּבֹא עַד הָאֹהֶל וַיַּכֵּהוּ וַיִּפֹּל וַיַּהַפְכֵהוּ לְמַעְלָה וְנָפַל הָאֹהֶל"
  • מלכים ב כא13: "וְנָטִיתִי עַל יְרוּשָׁלִַם אֶת קָו שֹׁמְרוֹן וְאֶת מִשְׁקֹלֶת בֵּית אַחְאָב וּמָחִיתִי אֵת יְרוּשָׁלִַם כַּאֲשֶׁר יִמְחֶה אֶת הַצַּלַּחַת מָחָה וְהָפַךְ עַל פָּנֶיהָ"
  • ירמיהו כ2-3: "וַיַּכֶּה פַשְׁחוּר אֵת יִרְמְיָהוּ הַנָּבִיא וַיִּתֵּן אֹתוֹ עַל הַמַּהְפֶּכֶת אֲשֶׁר בְּשַׁעַר בִּנְיָמִן הָעֶלְיוֹן אֲשֶׁר בְּבֵית ה'..." = מכשיר עינויים שבו תולים אדם בתנוחה הפוכה;
  • הושע ז8: "אֶפְרַיִם בָּעַמִּים הוּא יִתְבּוֹלָל, אֶפְרַיִם הָיָה עֻגָה בְּלִי הֲפוּכָה" = עוגה ששכחו להפוך אותה בתנור;

שינוי כיוון:

  • שמות י19: "וַיַּהֲפֹךְ ה' רוּחַ יָם חָזָק מְאֹד..." - מ"רוח קדים" (מזרחית) ל"רוח ים" (מערבית);
  • יהושע ז8: "בִּי אֲדֹנָי, מָה אֹמַר אַחֲרֵי אֲשֶׁר הָפַךְ יִשְׂרָאֵל עֹרֶף לִפְנֵי אֹיְבָיו" - מרדיפה לבריחה; יהושע ח20: "וַיִּפְנוּ אַנְשֵׁי הָעַי אַחֲרֵיהֶם וַיִּרְאוּ וָהֵנָּה עָלָה עֲשַׁן הָעִיר הַשָּׁמַיְמָה וְלֹא הָיָה בָהֶם יָדַיִם לָנוּס הֵנָּה וְהִנֵּה וְהָעָם הַנָּס הַמִּדְבָּר נֶהְפַּךְ אֶל הָרוֹדֵף" - מבריחה לרדיפה.
  • שופטים כ39-41: "וַיַּהֲפֹךְ אִישׁ יִשְׂרָאֵל בַּמִּלְחָמָה... וּבִנְיָמִן הֵחֵל לְהַכּוֹת חֲלָלִים בְּאִישׁ יִשְׂרָאֵל כִּשְׁלֹשִׁים אִישׁ כִּי אָמְרוּ אַךְ נִגּוֹף נִגָּף הוּא לְפָנֵינוּ כַּמִּלְחָמָה הָרִאשֹׁנָה...  וְאִישׁ יִשְׂרָאֵל הָפַךְ, וַיִּבָּהֵל אִישׁ בִּנְיָמִן כִּי רָאָה כִּי נָגְעָה עָלָיו הָרָעָה" - מרדיפה לבריחה ובחזרה;
  • שמואל א כה12: "וַיַּהַפְכוּ נַעֲרֵי דָוִד לְדַרְכָּם, וַיָּשֻׁבוּ וַיָּבֹאוּ וַיַּגִּדוּ לוֹ כְּכֹל הַדְּבָרִים הָאֵלֶּה"
  • מלכים א כב34: "וְאִישׁ מָשַׁךְ בַּקֶּשֶׁת לְתֻמּוֹ וַיַּכֶּה אֶת מֶלֶךְ יִשְׂרָאֵל בֵּין הַדְּבָקִים וּבֵין הַשִּׁרְיָן וַיֹּאמֶר לְרַכָּבוֹ הֲפֹךְ יָדְךָ וְהוֹצִיאֵנִי מִן הַמַּחֲנֶה כִּי הָחֳלֵיתִי"
  • מלכים ב ה26: "וַיֹּאמֶר אֵלָיו לֹא לִבִּי הָלַךְ כַּאֲשֶׁר הָפַךְ אִישׁ מֵעַל מֶרְכַּבְתּוֹ לִקְרָאֶתֶךָ..."
  • מלכים ב ט23: "וַיַּהֲפֹךְ יְהוֹרָם יָדָיו וַיָּנֹס..."
שינוי מראה וצבע:
  • ויקרא יג3-10-: "וְרָאָה הַכֹּהֵן אֶת הַנֶּגַע בְּעוֹר הַבָּשָׂר, וְשֵׂעָר בַּנֶּגַע הָפַךְ לָבָן... וְרָאָה הַכֹּהֵן והנה שאֶת לבנה בְּעוֹר והיא הפכה שער לָבָן ומחית בשר חי בשאֶת..."
  • ירמיהו יג23: "הֲיַהֲפֹךְ כּוּשִׁי עוֹרוֹ וְנָמֵר חֲבַרְבֻּרֹתָיו?! גַּם אַתֶּם תּוּכְלוּ לְהֵיטִיב, לִמֻּדֵי הָרֵעַ!"
הרס פתאומי, שהופך בבת אחת עיר בנויה לתל חרבות:
  • בראשית יט21-29: "וַיֹּאמֶר אֵלָיו הִנֵּה נָשָׂאֶתִי פָנֶיךָ גַּם לַדָּבָר הַזֶּה לְבִלְתִּי הָפְכִּי אֶת הָעִיר אֲשֶׁר דִּבַּרְתָּ... וַיַּהֲפֹךְ אֶת הֶעָרִים הָאֵל וְאֵת כָּל הַכִּכָּר וְאֵת כָּל יֹשְׁבֵי הֶעָרִים וְצֶמַח הָאֲדָמָה ... וַיְהִי בְּשַׁחֵת אֱלֹהִים אֶת עָרֵי הַכִּכָּר וַיִּזְכֹּר אֱלֹהִים אֶת אַבְרָהָם וַיְשַׁלַּח אֶת לוֹט מִתּוֹךְ הַהֲפֵכָה בַּהֲפֹךְ אֶת הֶעָרִים אֲשֶׁר יָשַׁב בָּהֵן לוֹט", וכן דברים כט22: "... כְּמַהְפֵּכַת סְדֹם וַעֲמֹרָה אַדְמָה וצביים[וּצְבוֹיִם] אֲשֶׁר הָפַךְ ה' בְּאַפּוֹ וּבַחֲמָתוֹ", ישעיהו יג19,  ירמיהו מט18, ירמיהו נ40,  איכה ד6: "וַיִּגְדַּל עון בַּת עַמִּי מֵחַטַּאת סְדֹם הַהֲפוּכָה כְמוֹ רָגַע וְלֹא חָלוּ בָהּ יָדָיִם".
  • שמואל ב י3: "...הֲלוֹא בַּעֲבוּר חֲקוֹר אֶת הָעִיר וּלְרַגְּלָהּ וּלְהָפְכָהּ שָׁלַח דָּוִד אֶת עֲבָדָיו אֵלֶיךָ"
  • ישעיהו א7: "אַרְצְכֶם שְׁמָמָה עָרֵיכֶם שְׂרֻפוֹת אֵשׁ אַדְמַתְכֶם לְנֶגְדְּכֶם זָרִים אֹכְלִים אֹתָהּ וּשְׁמָמָה כְּמַהְפֵּכַת זָרִים"
  • ירמיהו כ16: "וְהָיָה הָאִישׁ הַהוּא כֶּעָרִים אֲשֶׁר הָפַךְ ה' וְלֹא נִחָם, וְשָׁמַע זְעָקָה בַּבֹּקֶר וּתְרוּעָה בְּעֵת צָהֳרָיִם"
  • חגי ב22: "וְהָפַכְתִּי כִּסֵּא מַמְלְכוֹת וְהִשְׁמַדְתִּי חֹזֶק מַמְלָכוֹת הַגּוֹיִם וְהָפַכְתִּי מֶרְכָּבָה וְרֹכְבֶיהָ וְיָרְדוּ סוּסִים וְרֹכְבֵיהֶם אִישׁ בְּחֶרֶב אָחִיו"

שינוי פלאי של חפץ לחפץ אחר:

  • שמות ז15: "...וְהַמַּטֶּה אֲשֶׁר נֶהְפַּךְ לְנָחָשׁ תִּקַּח בְּיָדֶךָ"
  • שמות ז17: "... הנה אנכי מכה במטה אשר בידי על המים אשר ביאר ונהפכו לדם"
שינוי פתאומי בדעה או בדיבור:
  • שמות יד5: "וַיֻּגַּד לְמֶלֶךְ מִצְרַיִם כִּי בָרַח הָעָם, וַיֵּהָפֵךְ לְבַב פַּרְעֹה וַעֲבָדָיו אֶל הָעָם, וַיֹּאמרוּ 'מַה זֹּאֶת עָשִׂינוּ כִּי שִׁלַּחְנוּ אֶת יִשְׂרָאֵל מֵעָבְדֵנוּ'"
  • דברים כג6: "וְלֹא אָבָה ה' אֱלֹהֶיךָ לִשְׁמֹעַ אֶל בִּלְעָם, וַיַּהֲפֹךְ ה' אֱלֹהֶיךָ לְּךָ אֶת הַקְּלָלָה לִבְרָכָה, כִּי אֲהֵבְךָ ה' אֱלֹהֶיךָ", וכן נחמיה יג2
  • ישעיהו סג10: "וְהֵמָּה מָרוּ וְעִצְּבוּ אֶת רוּחַ קָדְשׁוֹ, וַיֵּהָפֵךְ לָהֶם לְאוֹיֵב הוּא נִלְחַם בָּם"
  • ראו גם: תהפוכות = זיג-זג פוליטי;
שינוי קיצוני באישיות:
  • שמואל א י6: "וְצָלְחָה עָלֶיךָ רוּחַ ה', וְהִתְנַבִּיתָ עִמָּם, וְנֶהְפַּכְתָּ לְאִישׁ אַחֵר"
  • שמואל א י9: "וְהָיָה כְּהַפְנֹתוֹ שִׁכְמוֹ לָלֶכֶת מֵעִם שְׁמוּאֵל, וַיַּהֲפָךְ לוֹ אֱלֹהִים לֵב אַחֵר, וַיָּבֹאוּ כָּל הָאֹתוֹת הָאֵלֶּה בַּיּוֹם הַהוּא"
  • ירמיהו ב21: "וְאָנֹכִי נְטַעְתִּיךְ שֹׂרֵק כֻּלֹּה זֶרַע אֱמֶת וְאֵיךְ נֶהְפַּכְתְּ לִי סוּרֵי הַגֶּפֶן נָכְרִיָּה"
מצב שונה באופן קיצוני מהמקובל:
  • יחזקאל טז34: "וַיְהִי בָךְ הֵפֶךְ מִן הַנָּשִׁים בְּתַזְנוּתַיִךְ וְאַחֲרַיִךְ לֹא זוּנָּה וּבְתִתֵּךְ אֶתְנָן וְאֶתְנַן לֹא נִתַּן לָךְ וַתְּהִי לְהֶפֶךְ"

פסוקים שניתן לפרש לפי כמה משמעויות

  • משלי יב7: "הָפוֹךְ רְשָׁעִים וְאֵינָם, וּבֵית צַדִּיקִים יַעֲמֹד"  (פירוט)

פסוקים נוספים

  • בראשית ג24: "וַיְגָרֶשׁ אֶת הָאָדָם וַיַּשְׁכֵּן מִקֶּדֶם לְגַן עֵדֶן אֶת הַכְּרֻבִים וְאֵת לַהַט הַחֶרֶב הַמִּתְהַפֶּכֶת לִשְׁמֹר אֶת דֶּרֶךְ עֵץ הַחַיִּים"
  • שמואל א ד19: "וְכַלָּתוֹ אֵשֶׁת פִּינְחָס הָרָה לָלַת וַתִּשְׁמַע אֶת הַשְּׁמֻעָה אֶל הִלָּקַח אֲרוֹן הָאֱלֹהִים וּמֵת חָמִיהָ וְאִישָׁהּ וַתִּכְרַע וַתֵּלֶד כִּי נֶהֶפְכוּ עָלֶיהָ צִרֶיהָ"
  • ישעיהו כט16: "הַפְכְּכֶם! אִם כְּחֹמֶר הַיֹּצֵר יֵחָשֵׁב כִּי יֹאמַר מַעֲשֶׂה לְעֹשֵׂהוּ לֹא עָשָׂנִי וְיֵצֶר אָמַר לְיוֹצְרוֹ לֹא הֵבִין"
  • ישעיהו לד9: "וְנֶהֶפְכוּ נְחָלֶיהָ לְזֶפֶת וַעֲפָרָהּ לְגָפְרִית, וְהָיְתָה אַרְצָהּ לְזֶפֶת בֹּעֵרָה"
  • ירמיהו כג36: "וּמַשָּׂא ידוד לֹא תִזְכְּרוּ עוֹד כִּי הַמַּשָּׂא יִהְיֶה לְאִישׁ דְּבָרוֹ וַהֲפַכְתֶּם אֶת דִּבְרֵי אֱלֹהִים חַיִּים ידוד צְבָאוֹת אֱלֹהֵינוּ"
  • ירמיהו כט26: "ידוד נְתָנְךָ כֹהֵן תַּחַת יְהוֹיָדָע הַכֹּהֵן לִהְיוֹת פְּקִדִים בֵּית ידוד לְכָל אִישׁ מְשֻׁגָּע וּמִתְנַבֵּא וְנָתַתָּה אֹתוֹ אֶל הַמַּהְפֶּכֶת וְאֶל הַצִּינֹק"
