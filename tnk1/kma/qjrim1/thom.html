ראשי>לשון המקרא>תהום>

תהום = מים רבים ועמוקים

קוד: תהום= בתנ"ך

סוג: הגדרה

מאת: אראל

אל:

בלשון ימינו תהום = עמק תלול, אך בלשון המקרא תהום מופיע בדרך-כלל יחד עם מים רבים, למשל:

  1. בראשית ז11: "...בַּיּוֹם הַזֶּה נִבְקְעוּ כָּל מעינות תְּהוֹם רַבָּה, וַאֲרֻבֹּת הַשָּׁמַיִם נִפְתָּחוּ", בראשית ח2: "וַיִּסָּכְרוּ מַעְיְנֹת תְּהוֹם וַאֲרֻבֹּת הַשָּׁמָיִם וַיִּכָּלֵא הַגֶּשֶׁם מִן הַשָּׁמָיִם"
  2. שמות טו5: "תְּהֹמֹת יְכַסְיֻמוּ, יָרְדוּ בִמְצוֹלֹת כְּמוֹ אָבֶן"

מכאן נראה, שהמילה "תהום" משמעה "מקור המים בעומק", או "ריבוי מים" (שבי"ל)- מים רבים ועמוקים.

פסוקים נוספים

  1. בראשית א2: "וְהָאָרֶץ הָיְתָה תֹהוּ וָבֹהוּ, וְחֹשֶׁךְ עַל פְּנֵי תְהוֹם, וְרוּחַ אֱלֹהִים מְרַחֶפֶת עַל פְּנֵי הַמָּיִם", תהלים קד6: "תְּהוֹם כַּלְּבוּשׁ כִּסִּיתוֹ, עַל הָרִים יַעַמְדוּ מָיִם" (ע"ע חוק הכלים השלובים והיקוות המים)
  2. בראשית מט25: "מֵאֵל אָבִיךָ וְיַעְזְרֶךָּ, וְאֵת שַׁדַּי וִיבָרְכֶךָּ; בִּרְכֹת שָׁמַיִם מֵעָל, בִּרְכֹת תְּהוֹם רֹבֶצֶת תָּחַת; בִּרְכֹת שָׁדַיִם וָרָחַם"
  3. שמות טו8: "וּבְרוּחַ אַפֶּיךָ נֶעֶרְמוּ מַיִם, נִצְּבוּ כְמוֹ נֵד נֹזְלִיָם, קָפְאוּ תְהֹמֹת בְּלֶב יָם"
  4. דברים ח7: "כִּי ידוד אֱלֹהֶיךָ מְבִיאֲךָ אֶל אֶרֶץ טוֹבָה אֶרֶץ נַחֲלֵי מָיִם עֲיָנֹת וּתְהֹמֹת יֹצְאִים בַּבִּקְעָה וּבָהָר"
  5. דברים לג13: "וּלְיוֹסֵף אָמַר: מְבֹרֶכֶת ה' אַרְצוֹ, מִמֶּגֶד שָׁמַיִם מִטָּל וּמִתְּהוֹם רֹבֶצֶת תָּחַת"
  6. ישעיהו נא10: "הֲלוֹא אַתְּ הִיא הַמַּחֲרֶבֶת יָם מֵי תְּהוֹם רַבָּה הַשָּׂמָה מַעֲמַקֵּי יָם דֶּרֶךְ לַעֲבֹר גְּאוּלִיָם"
  7. ישעיהו סג13: "מוֹלִיכָם בַּתְּהֹמוֹת כַּסּוּס בַּמִּדְבָּר לֹא יִכָּשֵׁלוּ"
  8. יחזקאל כו19: "...בְּהַעֲלוֹת עָלַיִךְ אֶת תְּהוֹם וְכִסּוּךְ הַמַּיִם הָרַבִּים"
  9. יחזקאל לא4: "מַיִם גִּדְּלוּהוּ תְּהוֹם רֹמְמָתְהוּ אֶת נַהֲרֹתֶיהָ הֹלֵךְ סְבִיבוֹת מַטָּעָהּ וְאֶת תְּעָלֹתֶיהָ שִׁלְחָה אֶל כָּל עֲצֵי הַשָּׂדֶה"
  10. יחזקאל לא15: "כֹּה אָמַר ד' ה': בְּיוֹם רִדְתּוֹ שְׁאוֹלָה הֶאֱבַלְתִּי כִּסֵּתִי עָלָיו אֶת תְּהוֹם וָאֶמְנַע נַהֲרוֹתֶיהָ וַיִּכָּלְאוּ מַיִם רַבִּים וָאַקְדִּר עָלָיו לְבָנוֹן וְכָל עֲצֵי הַשָּׂדֶה עָלָיו עֻלְפֶּה"
  11. עמוס ז4: "...וְהִנֵּה קֹרֵא לָרִב בָּאֵשׁ ד' ה', וַתֹּאכַל אֶת תְּהוֹם רַבָּה וְאָכְלָה אֶת הַחֵלֶק"
  12. יונה ב6: "אֲפָפוּנִי מַיִם עַד נֶפֶשׁ, תְּהוֹם יְסֹבְבֵנִי סוּף חָבוּשׁ לְרֹאשִׁי"
  13. חבקוק ג10: "רָאוּךָ יָחִילוּ הָרִים, זֶרֶם מַיִם עָבָר; נָתַן תְּהוֹם קוֹלוֹ, רוֹם יָדֵיהוּ נָשָׂא"
  14. תהלים לג7: "כֹּנֵס כַּנֵּד מֵי הַיָּם, נֹתֵן בְּאֹצָרוֹת תְּהוֹמוֹת"
  15. תהלים לו7: "צִדְקָתְךָ כְּהַרְרֵי אֵל, מִשְׁפָּטֶךָ תְּהוֹם רַבָּה; אָדָם וּבְהֵמָה תוֹשִׁיעַ ה'"
  16. תהלים מב8: "תְּהוֹם אֶל תְּהוֹם קוֹרֵא לְקוֹל צִנּוֹרֶיךָ, כָּל מִשְׁבָּרֶיךָ וְגַלֶּיךָ עָלַי עָבָרוּ"
  17. תהלים עא20: "אֲשֶׁר הראיתנו[הִרְאִיתַנִי] צָרוֹת רַבּוֹת וְרָעוֹת תָּשׁוּב תחיינו[תְּחַיֵּינִי] וּמִתְּהֹמוֹת הָאָרֶץ תָּשׁוּב תַּעֲלֵנִי"
  18. תהלים עז17: "רָאוּךָ מַּיִם אלהים רָאוּךָ מַּיִם יָחִילוּ אַף יִרְגְּזוּ תְהֹמוֹת"
  19. תהלים עח15: "יְבַקַּע צֻרִים בַּמִּדְבָּר, וַיַּשְׁקְ כִּתְהֹמוֹת רַבָּה"
  20. תהלים קו9: "וַיִּגְעַר בְּיַם סוּף וַיֶּחֱרָב,  וַיּוֹלִיכֵם בַּתְּהֹמוֹת כַּמִּדְבָּר"
  21. תהלים קז26: "יַעֲלוּ שָׁמַיִם יֵרְדוּ תְהוֹמוֹת, נַפְשָׁם בְּרָעָה תִתְמוֹגָג"
  22. תהלים קלה6: "כֹּל אֲשֶׁר חָפֵץ ה' עָשָׂה בַּשָּׁמַיִם וּבָאָרֶץ בַּיַּמִּים וְכָל תְּהוֹמוֹת"
  23. תהלים קמח7: "הַלְלוּ אֶת ה' מִן הָאָרֶץ, תַּנִּינִים וְכָל תְּהֹמוֹת"
  24. איוב כח14: "תְּהוֹם אָמַר לֹא בִי הִיא, וְיָם אָמַר אֵין עִמָּדִי"
  25. איוב לח16: "הֲבָאתָ עַד נִבְכֵי יָם, וּבְחֵקֶר תְּהוֹם הִתְהַלָּכְתָּ"
  26. איוב לח30: "כָּאֶבֶן מַיִם יִתְחַבָּאוּ, וּפְנֵי תְהוֹם יִתְלַכָּדוּ"
  27. איוב מא24: "אַחֲרָיו יָאִיר נָתִיב, יַחְשֹׁב תְּהוֹם לְשֵׂיבָה"
  28. משלי ג20: "בְּדַעְתּוֹ תְּהוֹמוֹת נִבְקָעוּ, וּשְׁחָקִים יִרְעֲפוּ טָל"
  29. משלי ח24-28: "בְּאֵין תְּהֹמוֹת חוֹלָלְתִּי, בְּאֵין מַעְיָנוֹת נִכְבַּדֵּי מָיִם... בַּהֲכִינוֹ שָׁמַיִם שָׁם אָנִי, בְּחוּקוֹ חוּג עַל פְּנֵי תְּהוֹם... בְּאַמְּצוֹ שְׁחָקִים מִמָּעַל, בַּעֲזוֹז עִינוֹת תְהוֹם"
