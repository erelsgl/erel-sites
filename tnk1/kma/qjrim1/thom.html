ראשי>לשון המקרא>תהום>

תהום = מים רבים ועמוקים

קוד: תהום= בתנ"ך

סוג: הגדרה

מאת: אראל

אל:

בלשון ימינו תהום הוא עמק תלול, אך בלשון המקרא תהום מופיע בדרך-כלל יחד עם מים רבים, למשל:

  • בבריאה, בראשית א2: "וְהָאָרֶץ הָיְתָה תֹהוּ וָבֹהוּ וְחֹשֶׁךְ עַל פְּנֵי תְהוֹם, וְרוּחַ אֱלֹהִים מְרַחֶפֶת עַל פְּנֵי הַמָּיִם" (פירוט), תהלים קד6: "תְּהוֹם כַּלְּבוּשׁ כִּסִּיתוֹ, עַל הָרִים יַעַמְדוּ מָיִם" (ראו חוק הכלים השלובים והיקוות המים).
  • במבול, בראשית ז11: "...בַּיּוֹם הַזֶּה נִבְקְעוּ כָּל מַעְיְנוֹת תְּהוֹם רַבָּה, וַאֲרֻבֹּת הַשָּׁמַיִם נִפְתָּחוּ", בראשית ח2: "וַיִּסָּכְרוּ מַעְיְנֹת תְּהוֹם וַאֲרֻבֹּת הַשָּׁמָיִם וַיִּכָּלֵא הַגֶּשֶׁם מִן הַשָּׁמָיִם".
  • בקריעת הים, שמות טו5: "תְּהֹמֹת יְכַסְיֻמוּ, יָרְדוּ בִמְצוֹלֹת כְּמוֹ אָבֶן",  שמות טו8: "וּבְרוּחַ אַפֶּיךָ נֶעֶרְמוּ מַיִם, נִצְּבוּ כְמוֹ נֵד נֹזְלִיָם, קָפְאוּ תְהֹמֹת בְּלֶב יָם".
  • בארץ ישראל,   דברים ח7: "כִּי ה' אֱלֹהֶיךָ מְבִיאֲךָ אֶל אֶרֶץ טוֹבָה אֶרֶץ נַחֲלֵי מָיִם עֲיָנֹת וּתְהֹמֹת יֹצְאִים בַּבִּקְעָה וּבָהָר",  דברים לג13: "וּלְיוֹסֵף אָמַר: מְבֹרֶכֶת ה' אַרְצוֹ, מִמֶּגֶד שָׁמַיִם מִטָּל וּמִתְּהוֹם רֹבֶצֶת תָּחַת".

מכא שהמילה תהום משמעה מים רבים ועמוקים - "מקור המים בעומק" או "ריבוי מים" (שבי"ל, קונקורדנציה).

פסוקים נוספים

  • בראשית מט25: "מֵאֵל אָבִיךָ וְיַעְזְרֶךָּ, וְאֵת שַׁדַּי וִיבָרְכֶךָּ; בִּרְכֹת שָׁמַיִם מֵעָל, בִּרְכֹת תְּהוֹם רֹבֶצֶת תָּחַת; בִּרְכֹת שָׁדַיִם וָרָחַם"
  • ישעיהו נא10: "הֲלוֹא אַתְּ הִיא הַמַּחֲרֶבֶת יָם מֵי תְּהוֹם רַבָּה הַשָּׂמָה מַעֲמַקֵּי יָם דֶּרֶךְ לַעֲבֹר גְּאוּלִיָם"
  • ישעיהו סג13: "מוֹלִיכָם בַּתְּהֹמוֹת כַּסּוּס בַּמִּדְבָּר לֹא יִכָּשֵׁלוּ"
  • יחזקאל כו19: "...בְּהַעֲלוֹת עָלַיִךְ אֶת תְּהוֹם וְכִסּוּךְ הַמַּיִם הָרַבִּים"
  • יחזקאל לא4: "מַיִם גִּדְּלוּהוּ תְּהוֹם רֹמְמָתְהוּ אֶת נַהֲרֹתֶיהָ הֹלֵךְ סְבִיבוֹת מַטָּעָהּ וְאֶת תְּעָלֹתֶיהָ שִׁלְחָה אֶל כָּל עֲצֵי הַשָּׂדֶה"
  • יחזקאל לא15: "כֹּה אָמַר ד' ה': בְּיוֹם רִדְתּוֹ שְׁאוֹלָה הֶאֱבַלְתִּי כִּסֵּתִי עָלָיו אֶת תְּהוֹם וָאֶמְנַע נַהֲרוֹתֶיהָ וַיִּכָּלְאוּ מַיִם רַבִּים וָאַקְדִּר עָלָיו לְבָנוֹן וְכָל עֲצֵי הַשָּׂדֶה עָלָיו עֻלְפֶּה"
  • עמוס ז4: "...וְהִנֵּה קֹרֵא לָרִב בָּאֵשׁ ד' ה', וַתֹּאכַל אֶת תְּהוֹם רַבָּה וְאָכְלָה אֶת הַחֵלֶק"
  • יונה ב6: "אֲפָפוּנִי מַיִם עַד נֶפֶשׁ, תְּהוֹם יְסֹבְבֵנִי סוּף חָבוּשׁ לְרֹאשִׁי"
  • חבקוק ג10: "רָאוּךָ יָחִילוּ הָרִים, זֶרֶם מַיִם עָבָר; נָתַן תְּהוֹם קוֹלוֹ, רוֹם יָדֵיהוּ נָשָׂא"
  • תהלים לג7: "כֹּנֵס כַּנֵּד מֵי הַיָּם, נֹתֵן בְּאֹצָרוֹת תְּהוֹמוֹת"
  • תהלים לו7: "צִדְקָתְךָ כְּהַרְרֵי אֵל, מִשְׁפָּטֶךָ תְּהוֹם רַבָּה; אָדָם וּבְהֵמָה תוֹשִׁיעַ ה'"
  • תהלים מב8: "תְּהוֹם אֶל תְּהוֹם קוֹרֵא לְקוֹל צִנּוֹרֶיךָ, כָּל מִשְׁבָּרֶיךָ וְגַלֶּיךָ עָלַי עָבָרוּ"
  • תהלים עא20: "אֲשֶׁר הראיתנו[הִרְאִיתַנִי] צָרוֹת רַבּוֹת וְרָעוֹת תָּשׁוּב תחיינו[תְּחַיֵּינִי] וּמִתְּהֹמוֹת הָאָרֶץ תָּשׁוּב תַּעֲלֵנִי"
  • תהלים עז17: "רָאוּךָ מַּיִם אלהים רָאוּךָ מַּיִם יָחִילוּ אַף יִרְגְּזוּ תְהֹמוֹת"
  • תהלים עח15: "יְבַקַּע צֻרִים בַּמִּדְבָּר, וַיַּשְׁקְ כִּתְהֹמוֹת רַבָּה"
  • תהלים קו9: "וַיִּגְעַר בְּיַם סוּף וַיֶּחֱרָב,  וַיּוֹלִיכֵם בַּתְּהֹמוֹת כַּמִּדְבָּר"
  • תהלים קז26: "יַעֲלוּ שָׁמַיִם יֵרְדוּ תְהוֹמוֹת, נַפְשָׁם בְּרָעָה תִתְמוֹגָג"
  • תהלים קלה6: "כֹּל אֲשֶׁר חָפֵץ ה' עָשָׂה בַּשָּׁמַיִם וּבָאָרֶץ בַּיַּמִּים וְכָל תְּהוֹמוֹת"
  • תהלים קמח7: "הַלְלוּ אֶת ה' מִן הָאָרֶץ, תַּנִּינִים וְכָל תְּהֹמוֹת"
  • איוב כח14: "תְּהוֹם אָמַר לֹא בִי הִיא, וְיָם אָמַר אֵין עִמָּדִי"
  • איוב לח16: "הֲבָאתָ עַד נִבְכֵי יָם, וּבְחֵקֶר תְּהוֹם הִתְהַלָּכְתָּ"
  • איוב לח30: "כָּאֶבֶן מַיִם יִתְחַבָּאוּ, וּפְנֵי תְהוֹם יִתְלַכָּדוּ"
  • איוב מא24: "אַחֲרָיו יָאִיר נָתִיב, יַחְשֹׁב תְּהוֹם לְשֵׂיבָה"
  • משלי ג20: "בְּדַעְתּוֹ תְּהוֹמוֹת נִבְקָעוּ, וּשְׁחָקִים יִרְעֲפוּ טָל"
  • משלי ח24-28: "בְּאֵין תְּהֹמוֹת חוֹלָלְתִּי, בְּאֵין מַעְיָנוֹת נִכְבַּדֵּי מָיִם... בַּהֲכִינוֹ שָׁמַיִם שָׁם אָנִי, בְּחוּקוֹ חוּג עַל פְּנֵי תְּהוֹם... בְּאַמְּצוֹ שְׁחָקִים מִמָּעַל, בַּעֲזוֹז עִינוֹת תְהוֹם"
