ראשי>גדד1>

גדוד = ארגון צבאי עויין

מאת: אראל

גדוד בלשון ימינו הוא יחידה צבאית מאורגנת. גם בלשון המקרא גדוד מציין ארגון צבאי, אולם ברוב המקומות מדובר בארגון עויין, ארגון של שודדים או צבאות זרים:

  • שמואל א ל8: " וַיִּשְׁאַל דָּוִד בה' לֵאמֹר 'אֶרְדֹּף אַחֲרֵי הַגְּדוּד הַזֶּה, הַאַשִּׂגֶנּוּ?' וַיֹּאמֶר לוֹ 'רְדֹף כִּי הַשֵּׂג תַּשִּׂיג וְהַצֵּל תַּצִּיל' " - גדוד של עמלקים שפשטו על צקלג. וכן ב שמואל א ל15: " וַיֹּאמֶר אֵלָיו דָּוִד 'הֲתוֹרִדֵנִי אֶל הַגְּדוּד הַזֶּה? וַיֹּאמֶר 'הִשָּׁבְעָה לִּי בֵאלֹהִים אִם תְּמִיתֵנִי וְאִם תַּסְגִּרֵנִי בְּיַד אֲדֹנִי וְאוֹרִדְךָ אֶל הַגְּדוּד הַזֶּה' ", שמואל א ל23: " וַיֹּאמֶר דָּוִד לֹא תַעֲשׂוּ כֵן אֶחָי אֵת אֲשֶׁר נָתַן ה' לָנוּ וַיִּשְׁמֹר אֹתָנוּ וַיִּתֵּן אֶת הַגְּדוּד הַבָּא עָלֵינוּ בְּיָדֵנוּ ".
  • שמואל ב כב30: " כִּי בְכָה אָרוּץ גְּדוּד, בֵּאלֹהַי אֲדַלֶּג שׁוּר " - דוד אומר: בעזרתך, ה', אני מצליח לרוצץ ולשבור את הגדוד התוקף אותי. וכן ב תהלים יח30: " כִּי בְךָ אָרֻץ גְּדוּד , וּבֵאלֹהַי אֲדַלֶּג שׁוּר ".
  • מלכים א יא24: " וַיִּקְבֹּץ עָלָיו אֲנָשִׁים, וַיְהִי שַׂר גְּדוּד בַּהֲרֹג דָּוִד אֹתָם, וַיֵּלְכוּ דַמֶּשֶׂק וַיֵּשְׁבוּ בָהּ וַיִּמְלְכוּ בְּדַמָּשֶׂק " - רזון בן אלידע הקים גדוד ארמי עויין לישראל.
  • מלכים ב ה2: " וַאֲרָם יָצְאוּ גְדוּדִים , וַיִּשְׁבּוּ מֵאֶרֶץ יִשְׂרָאֵל נַעֲרָה קְטַנָּה... " - כמה דורות מאוחר יותר, אותם גדודים ארמיים כבר פושטים על ישראל ולוקחים שבויים. וכן מלכים ב ו23: " וַיִּכְרֶה לָהֶם כֵּרָה גְדוֹלָה, וַיֹּאכְלוּ וַיִּשְׁתּוּ וַיְשַׁלְּחֵם וַיֵּלְכוּ אֶל אֲדֹנֵיהֶם; וְלֹא יָסְפוּ עוֹד גְּדוּדֵי אֲרָם לָבוֹא בְּאֶרֶץ יִשְׂרָאֵל "
  • מלכים ב יג20: " וַיָּמָת אֱלִישָׁע וַיִּקְבְּרֻהוּ, וּגְדוּדֵי מוֹאָב יָבֹאוּ בָאָרֶץ בָּא שָׁנָה ". מלכים ב יג21: " וַיְהִי הֵם קֹבְרִים אִישׁ, וְהִנֵּה רָאוּ אֶת הַגְּדוּד ... " - בדור הבא, גם גדודים מואביים פושטים על ישראל.
  • מלכים ב כד2: " וַיְשַׁלַּח ה' בּוֹ אֶת גְּדוּדֵי כַשְׂדִּים וְאֶת גְּדוּדֵי אֲרָם וְאֵת גְּדוּדֵי מוֹאָב וְאֵת גְּדוּדֵי בְנֵי עַמּוֹן; וַיְשַׁלְּחֵם בִּיהוּדָה לְהַאֲבִידוֹ, כִּדְבַר ה' אֲשֶׁר דִּבֶּר בְּיַד עֲבָדָיו הַנְּבִיאִים " - גדודים (מיליציות) מכל הכיוונים תוקפים את יהודה.
  • דברי הימים ב כב1: " וַיַּמְלִיכוּ יוֹשְׁבֵי יְרוּשָׁלַים אֶת אֲחַזְיָהוּ בְנוֹ הַקָּטֹן תַּחְתָּיו, כִּי כָל הָרִאשֹׁנִים הָרַג הַגְּדוּד הַבָּא בָעַרְבִים לַמַּחֲנֶה; וַיִּמְלֹךְ אֲחַזְיָהוּ בֶן יְהוֹרָם מֶלֶךְ יְהוּדָה " - מיליציות ערביות.
  • ירמיהו יח22: " תִּשָּׁמַע זְעָקָה מִבָּתֵּיהֶם, כִּי תָבִיא עֲלֵיהֶם גְּדוּד פִּתְאֹם; כִּי כָרוּ שיחה[שׁוּחָה] לְלָכְדֵנִי, וּפַחִים טָמְנוּ לְרַגְלָי " - קללה לאויבי הנביא, שיבוא עליהם גדוד של אויבים במתקפת פתע.
  • הושע ז1: " כְּרָפְאִי לְיִשְׂרָאֵל, וְנִגְלָה עון אֶפְרַיִם וְרָעוֹת שֹׁמְרוֹן; כִּי פָעֲלוּ שָׁקֶר וְגַנָּב יָבוֹא, פָּשַׁט גְּדוּד בַּחוּץ " - גדוד שודדים תוקף אותם בגלוי, לאור יום ( פירוט).
  • איוב יט12: " יַחַד יָבֹאוּ גְדוּדָיו , וַיָּסֹלּוּ עָלַי דַּרְכָּם, וַיַּחֲנוּ סָבִיב לְאָהֳלִי " - כנראה הכוונה לגדודי הכשדים שתקפו את איוב.

אמנם ישנם כמה פסוקים יוצאי-דופן, שבהם גדוד מתייחס ליחידה בצבא ישראל; רובם נמצאים בדברי הימים:

  • שמואל ב ד2: " וּשְׁנֵי אֲנָשִׁים שָׂרֵי גְדוּדִים הָיוּ בֶן שָׁאוּל, שֵׁם הָאֶחָד בַּעֲנָה וְשֵׁם הַשֵּׁנִי רֵכָב... "
  • דברי הימים א ז4: " וַעֲלֵיהֶם לְתֹלְדוֹתָם לְבֵית אֲבוֹתָם גְּדוּדֵי צְבָא מִלְחָמָה שְׁלֹשִׁים וְשִׁשָּׁה אָלֶף, כִּי הִרְבּוּ נָשִׁים וּבָנִים " (בני יששכר), דברי הימים א יב19: " וַיְקַבְּלֵם דָוִיד וַיִּתְּנֵם בְּרָאשֵׁי הַגְּדוּד ", דברי הימים א יב22: " וְהֵמָּה עָזְרוּ עִם דָּוִיד עַל הַגְּדוּד כִּי גִבּוֹרֵי חַיִל כֻּלָּם, וַיִּהְיוּ שָׂרִים בַּצָּבָא ".
  • דברי הימים ב כה9: " וַיֹּאמֶר אֲמַצְיָהוּ לְאִישׁ הָאֱלֹהִים 'וּמַה לַּעֲשׂוֹת לִמְאַת הַכִּכָּר אֲשֶׁר נָתַתִּי לִגְדוּד יִשְׂרָאֵל?' וַיֹּאמֶר אִישׁ הָאֱלֹהִים יֵשׁ לה' לָתֶת לְךָ הַרְבֵּה מִזֶּה " - כאן מדובר בגדוד מממלכת ישראל הצפונית, שאמציהו מלך יהודה שכר כדי לעזור לו להילחם באדום. וכן ב דברי הימים ב כה10: " וַיַּבְדִּילֵם אֲמַצְיָהוּ לְהַגְּדוּד אֲשֶׁר בָּא אֵלָיו מֵאֶפְרַיִם לָלֶכֶת לִמְקוֹמָם, וַיִּחַר אַפָּם מְאֹד בִּיהוּדָה וַיָּשׁוּבוּ לִמְקוֹמָם בָּחֳרִי אָף ". והם הפכו לגדוד שודדים, דברי הימים ב כה13: " וּבְנֵי הַגְּדוּד אֲשֶׁר הֵשִׁיב אֲמַצְיָהוּ מִלֶּכֶת עִמּוֹ לַמִּלְחָמָה וַיִּפְשְׁטוּ בְּעָרֵי יְהוּדָה מִשֹּׁמְרוֹן וְעַד בֵּית חוֹרוֹן וַיַּכּוּ מֵהֶם שְׁלֹשֶׁת אֲלָפִים וַיָּבֹזּוּ בִּזָּה רַבָּה ".
  • דברי הימים ב כו11: " וַיְהִי לְעֻזִּיָּהוּ חַיִל עֹשֵׂה מִלְחָמָה יוֹצְאֵי צָבָא לִגְדוּד בְּמִסְפַּר פְּקֻדָּתָם בְּיַד יעואל[יְעִיאֵל] הַסּוֹפֵר וּמַעֲשֵׂיָהוּ הַשּׁוֹטֵר עַל יַד חֲנַנְיָהוּ מִשָּׂרֵי הַמֶּלֶךְ "

פסוקים נוספים

  • בראשית מט19: " גָּד גְּדוּד יְגוּדֶנּוּ, וְהוּא יָגֻד עָקֵב " ( פירוט) - שבט גד יצטרך להתמודד עם גדודי אויבים שיתקפו אותו (כנראה כי התיישב בספר, בעבר הירדן המזרחי).
  • שמואל ב ג22: " וְהִנֵּה עַבְדֵי דָּוִד וְיוֹאָב בָּא מֵהַגְּדוּד , וְשָׁלָל רָב עִמָּם הֵבִיאוּ " - יואב נלחם בגדודי שודדים (כמו גדודי העמלקים מצקלג) והצליח להכות אותם ולקחת שלל רב.
  • הושע ו9: " וּכְחַכֵּי אִישׁ גְּדוּדִים , חֶבֶר כֹּהֲנִים; דֶּרֶךְ יְרַצְּחוּ שֶׁכְמָה, כִּי זִמָּה עָשׂוּ " - תוכחה קשה של הנביא הושע כלפי חבורת הכהנים, המתנהגת כמו גדודי שודדים.
  • מיכה ד14: " עַתָּה תִּתְגֹּדְדִי בַת גְּדוּד, מָצוֹר שָׂם עָלֵינוּ; בַּשֵּׁבֶט יַכּוּ עַל הַלְּחִי אֵת שֹׁפֵט יִשְׂרָאֵל " - צ"ע.
  • תהלים סה11: " תְּלָמֶיהָ רַוֵּה, נַחֵת גְּדוּדֶיהָ; בִּרְבִיבִים תְּמֹגְגֶנָּה, צִמְחָהּ תְּבָרֵךְ " - כאן גדוד הוא גוש-אדמה.
  • איוב כה3: " הֲיֵשׁ מִסְפָּר לִגְדוּדָיו ? וְעַל מִי לֹא יָקוּם אוֹרֵהוּ " - כאן מדובר כנראה על גדודי הכוכבים או המלאכים.
  • איוב כט25: " אֶבֲחַר דַּרְכָּם וְאֵשֵׁב רֹאשׁ, וְאֶשְׁכּוֹן כְּמֶלֶךְ בַּגְּדוּד , כַּאֲשֶׁר אֲבֵלִים יְנַחֵם ".
