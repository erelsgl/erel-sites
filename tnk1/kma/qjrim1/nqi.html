ראשי>לשון המקרא>נקי>

נקי = א. ללא חטא גלוי, ב. פטור מעונש או התחייבות

קוד: נקה= בתנ"ך

סוג: הגדרה

מאת: אראל

אל:

בלשון ימינו, נקי משמש בעיקר במשמעות מוחשית - לא מלוכלך, ללא זיהום חיצוני.

בלשון המקרא, המשמעות המוחשית מופיעה רק בחלק קטן מהפסוקים:

  • שמות כה29: "וְעָשִׂיתָ קְּעָרֹתָיו וְכַפֹּתָיו וּקְשׂוֹתָיו וּמְנַקִּיֹּתָיו אֲשֶׁר יֻסַּךְ בָּהֵן, זָהָב טָהוֹר תַּעֲשֶׂה אֹתָם" - ע"פ המפרשים, "מנקיות" הם כלים שמטרתם לשמור על הנקיון של לחם הפנים במקדש מפני לכלוך ועובש (וכן שמות לז16, במדבר ד7, ירמיהו נב19).
  • עמוס ד6: "וְגַם אֲנִי נָתַתִּי לָכֶם נִקְיוֹן שִׁנַּיִם בְּכָל עָרֵיכֶם וְחֹסֶר לֶחֶם בְּכֹל מְקוֹמֹתֵיכֶם, וְלֹא שַׁבְתֶּם עָדַי נְאֻם ה'" - כאן במשמעות צינית: השיניים שלכם אמנם לא מלוכלכות, אבל רק משום שאין לכם לחם (פירוט).
ברוב המקומות המשמעות היא מופשטת:

א. נקי = פטור ממחוייבות

  • בראשית כד8: "וְאִם לֹא תֹאבֶה הָאִשָּׁה לָלֶכֶת אַחֲרֶיךָ, וְנִקִּיתָ מִשְּׁבֻעָתִי זֹאֶת; רַק אֶת בְּנִי לֹא תָשֵׁב שָׁמָּה", בראשית כד41: "אָז תִּנָּקֶה מֵאָלָתִי כִּי תָבוֹא אֶל מִשְׁפַּחְתִּי וְאִם לֹא יִתְּנוּ לָךְ וְהָיִיתָ נָקִי מֵאָלָתִי" - אם לא תמצא אישה ממשפחתי שתסכים לעלות לארץ, תהיה פטור מההתחייבות שנשבעת לי עליה, ותהיה רשאי לקחת אישה לבני מבנות הכנעני שנמצאות כאן בארץ.
  • במדבר לב22: "וְנִכְבְּשָׁה הָאָרֶץ לִפְנֵי ה' וְאַחַר תָּשֻׁבוּ וִהְיִיתֶם נקים מה' וּמִיִּשְׂרָאֵל וְהָיְתָה הָאָרֶץ הַזֹּאת לָכֶם לַאֲחֻזָּה לִפְנֵי ה'" - אחרי שהארץ תיכבש, תהיו פטורים מההתחייבות לצאת חלוצים למלחמה לפני עם ישראל, ותוכלו לחזור לבתיכם ממזרח לירדן.
  • דברים כד5: "כִּי יִקַּח אִישׁ אִשָּׁה חֲדָשָׁה, לֹא יֵצֵא בַּצָּבָא וְלֹא יַעֲבֹר עָלָיו לְכָל דָּבָר; נָקִי יִהְיֶה לְבֵיתוֹ שָׁנָה אֶחָת וְשִׂמַּח אֶת אִשְׁתּוֹ אֲשֶׁר לָקָח" = החתן הטרי פטור משרות צבאי.
  • יהושע ב17-20: "וַיֹּאמְרוּ אֵלֶיהָ הָאֲנָשִׁים נְקִיִּם אֲנַחְנוּ מִשְּׁבֻעָתֵךְ הַזֶּה אֲשֶׁר הִשְׁבַּעְתָּנוּ... וְהָיָה כָּל אֲשֶׁר יֵצֵא מִדַּלְתֵי בֵיתֵךְ הַחוּצָה דָּמוֹ בְרֹאשׁוֹ וַאֲנַחְנוּ נְקִיִּם וְכֹל אֲשֶׁר יִהְיֶה אִתָּךְ בַּבַּיִת דָּמוֹ בְרֹאשֵׁנוּ אִם יָד תִּהְיֶה בּוֹ. וְאִם תַּגִּיָדִי אֶת דְּבָרֵנוּ זֶה וְהָיִינוּ נְקִיִּם מִשְּׁבֻעָתֵךְ אֲשֶׁר הִשְׁבַּעְתָּנוּ"  = אם תגלי את הסוד, אנחנו פטורים מההתחייבות שהתחייבנו לך בשבועה.
  • מלכים א טו22: "וְהַמֶּלֶךְ אָסָא הִשְׁמִיעַ אֶת כָּל יְהוּדָה אֵין נָקִי וַיִּשְׂאוּ אֶת אַבְנֵי הָרָמָה וְאֶת עֵצֶיהָ אֲשֶׁר בָּנָה בַּעְשָׁא וַיִּבֶן בָּם הַמֶּלֶךְ אָסָא אֶת גֶּבַע בִּנְיָמִן וְאֶת הַמִּצְפָּה" = אף אחד לא קיבל פטור.

ב. נקי = פטור מעונש

  • בראשית מד10: "וַיֹּאמֶר 'גַּם עַתָּה כְדִבְרֵיכֶם כֶּן הוּא, אֲשֶׁר יִמָּצֵא אִתּוֹ יִהְיֶה לִּי עָבֶד, וְאַתֶּם תִּהְיוּ נְקִיִּם'" - הוּא ייענש וְאַתֶּם תִּהְיוּ פטורים מעונש.
  • שמות כא19: "אם יקום והתהלך בחוץ על משענתו, ונקה המכה; רק שבתו יתן, ורפא ירפא" - יהיה פטור מעונש מוות.
  • שמות כא28: "וְכִי יִגַּח שׁוֹר אֶת אִישׁ אוֹ אֶת אִשָּׁה וָמֵת, סָקוֹל יִסָּקֵל הַשּׁוֹר וְלֹא יֵאָכֵל אֶת בְּשָׂרוֹ, וּבַעַל הַשּׁוֹר נָקִי" - השור ימות אבל בעל השור יהיה פטור מעונש מוות או מעונש אחר (פירוט)
  • שמות לד7: "נֹצֵר חֶסֶד לָאֲלָפִים, נֹשֵׂא עון וָפֶשַׁע וְחַטָּאָה וְנַקֵּה לֹא יְנַקֶּה, פֹּקֵד עון אָבוֹת עַל בָּנִים וְעַל בְּנֵי בָנִים עַל שִׁלֵּשִׁים וְעַל רִבֵּעִים", וכן ירמיהו ל11: "כִּי אִתְּךָ אֲנִי נְאֻם ה' לְהוֹשִׁיעֶךָ, כִּי אֶעֱשֶׂה כָלָה בְּכָל הַגּוֹיִם אֲשֶׁר הֲפִצוֹתִיךָ שָּׁם אַךְ אֹתְךָ לֹא אֶעֱשֶׂה כָלָה וְיִסַּרְתִּיךָ לַמִּשְׁפָּט וְנַקֵּה לֹא אֲנַקֶּךָּ" - לא מוותר לגמרי על העונש אלא רק דוחה אותו; או: לא מעניש עד הסוף, לא מביא את האדם למצב שהוא פטור מעונש. ראו מידות הרחמים שהתגלו למשה.
  • במדבר ה19: "וְהִשְׁבִּיעַ אֹתָהּ הַכֹּהֵן וְאָמַר אֶל הָאִשָּׁה 'אִם לֹא שָׁכַב אִישׁ אֹתָךְ וְאִם לֹא שָׂטִית טֻמְאָה תַּחַת אִישֵׁךְ - הִנָּקִי מִמֵּי הַמָּרִים הַמְאָרֲרִים הָאֵלֶּה'" = תהיה פטורה מהקללה של המים המרים.
  • במדבר ה31: "וְנִקָּה הָאִישׁ מעון, וְהָאִשָּׁה הַהִוא תִּשָּׂא אֶת עונה" - האיש יהיה פטור מעונש על כך שחשד באשתו או על כך ששכב עמה לאחר שנאפה.
  • שמואל א כו9: "וַיֹּאמֶר דָּוִד אַל אֲבִישַׁי 'אֶל תַּשְׁחִיתֵהוּ, כִּי מִי שָׁלַח יָדוֹ בִּמְשִׁיחַ ה' וְנִקָּה?'" - מי יהיה פטור מעונש על פגיעה במשיח ה'?!
  • שמואל ב ג28: "וַיִּשְׁמַע דָּוִד מֵאַחֲרֵי כֵן וַיֹּאמֶר ' נָקִי אָנֹכִי וּמַמְלַכְתִּי מֵעִם ה' עַד עוֹלָם מִדְּמֵי אַבְנֵר בֶּן נֵר'" - אני פטור מעונש, כל העונש יבוא על ראש יואב הרוצח.
  • מלכים א ב9: "וְעַתָּה אַל תְּנַקֵּהוּ כִּי אִישׁ חָכָם אָתָּה וְיָדַעְתָּ אֶת אֲשֶׁר תַּעֲשֶׂה לּוֹ וְהוֹרַדְתָּ אֵת שֵׂיבָתוֹ בְּדָם שְׁאוֹל" - אל תפטור אותו מעונש.
  • ירמיהו ב35: "וַתֹּאמְרִי כִּי נִקֵּיתִי אַךְ שָׁב אַפּוֹ מִמֶּנִּי הִנְנִי נִשְׁפָּט אוֹתָךְ עַל אָמְרֵךְ לֹא חָטָאתִי"
  • איוב ט23: "אִם שׁוֹט יָמִית פִּתְאֹם, לְמַסַּת נְקִיִּם יִלְעָג" - אולי הכוונה לאדם שנדון למלקות, ופתאום קיבל מכה חזקה מדי ומת, למרות שהוא נקי ופטור מעונש מוות.
בכמה נבואות ותוכחות נאמר, שהנוהג באופן מסויים "לא ינקה", כלומר לא יהיה פטור מעונש. גם אם לא ייענש מייד, העונש יגיע בסופו של דבר:
  • שמות כ6: "לֹא תִשָּׂא אֶת שֵׁם ה' אֱלֹהֶיךָ לַשָּׁוְא, כִּי לֹא יְנַקֶּה ה' אֵת אֲשֶׁר יִשָּׂא אֶת שְׁמוֹ לַשָּׁוְא", וכן דברים ה10
  • משלי ו29: "כֵּן הַבָּא אֶל אֵשֶׁת רֵעֵהוּ, לֹא יִנָּקֶה כָּל הַנֹּגֵעַ בָּהּ"
  • משלי יא21: "יָד לְיָד לֹא יִנָּקֶה רָּע, וְזֶרַע צַדִּיקִים נִמְלָט" (פירוט)
  • משלי טז5: "תּוֹעֲבַת ה' כָּל גְּבַהּ לֵב, יָד לְיָד לֹא יִנָּקֶה" (פירוט)
  • משלי יז5: "לֹעֵג לָרָשׁ חֵרֵף עֹשֵׂהוּ, שָׂמֵחַ לְאֵיד לֹא יִנָּקֶה" (פירוט)
  • משלי יט5: "עֵד שְׁקָרִים לֹא יִנָּקֶה, וְיָפִיחַ כְּזָבִים לֹא יִמָּלֵט" (פירוט)
  • משלי יט9: "עֵד שְׁקָרִים לֹא יִנָּקֶה, וְיָפִיחַ כְּזָבִים יֹאבֵד" (פירוט)
  • משלי כח20: "אִישׁ אֱמוּנוֹת רַב בְּרָכוֹת, וְאָץ לְהַעֲשִׁיר לֹא יִנָּקֶה" (פירוט)
  • ירמיהו כה29: "כִּי הִנֵּה בָעִיר אֲשֶׁר נִקְרָא שְׁמִי עָלֶיהָ אָנֹכִי מֵחֵל לְהָרַע וְאַתֶּם הִנָּקֵה תִנָּקוּ? לֹא תִנָּקוּ, כִּי חֶרֶב אֲנִי קֹרֵא עַל כָּל יֹשְׁבֵי הָאָרֶץ נְאֻם ידוד צְבָאוֹת"
בפסוקים רבים המדברים על רצח, נזכר הביטוי "דם נקי"; הכוונה היא לנרצח שהוא פטור מעונש מוות - גם אם אינו צדיק גמור בעיני הרוצח:
  • דברים יט10: "וְלֹא יִשָּׁפֵךְ דָּם נָקִי בְּקֶרֶב אַרְצְךָ אֲשֶׁר ה' אֱלֹהֶיךָ נֹתֵן לְךָ נַחֲלָה וְהָיָה עָלֶיךָ דָּמִים" = לא תתן לגואל הדם להרוג אדם שרצח בשוגג, כי הרוצח בשוגג הוא נקי ופטור מעונש מוות.
  • דברים יט13: "לֹא תָחוֹס עֵינְךָ עָלָיו, וּבִעַרְתָּ דַם הַנָּקִי מִיִּשְׂרָאֵל וְטוֹב לָךְ" - צריך להרוג את הרוצח במזיד, כי הוא לא נקי, אבל הקרבן שלו היה נקי ופטור ממוות.
  • שמואל א יט5: "וַיָּשֶׂם אֶת נַפְשׁוֹ בְכַפּוֹ וַיַּךְ אֶת הַפְּלִשְׁתִּי וַיַּעַשׂ ה' תְּשׁוּעָה גְדוֹלָה לְכָל יִשְׂרָאֵל, רָאִיתָ וַתִּשְׂמָח, וְלָמָּה תֶחֱטָא בְּדָם נָקִי לְהָמִית אֶת דָּוִד חִנָּם?" - יהונתן אומר לשאול שדוד פטור ממוות.
  • ירמיהו כו15: "אַךְ יָדֹעַ תֵּדְעוּ כִּי אִם מְמִתִים אַתֶּם אֹתִי כִּי דָם נָקִי אַתֶּם נֹתְנִים עֲלֵיכֶם וְאֶל הָעִיר הַזֹּאֵת וְאֶל יֹשְׁבֶיהָ כִּי בֶאֱמֶת שְׁלָחַנִי ידוד עֲלֵיכֶם לְדַבֵּר בְּאָזְנֵיכֶם אֵת כָּל הַדְּבָרִים הָאֵלֶּה" - ירמיהו מנסה לשכנע את שופטיו שהוא פטור מעונש מוות.

ג. נקי = ללא חטא גלוי

כמו המשמעות המוחשית של השורש "נקי", גם המשמעות המופשטת מתייחסת לנקיון גלוי של המעשים - מעשים ללא חטא; בדרך כלל משתמשים בביטוי "נקי כפיים" או "נקיון כפיים":

  • בראשית כ5: "... בְּתָם לְבָבִי וּבְנִקְיֹן כַּפַּי עָשִׂיתִי זֹאת" - תם לבב = במחשבות הנסתרות, נקיון כפיים = במעשים הגלויים.
  • שופטים טו3: "וַיֹּאמֶר לָהֶם שִׁמְשׁוֹן ' נִקֵּיתִי הַפַּעַם מִפְּלִשְׁתִּים כִּי עֹשֶׂה אֲנִי עִמָּם רָעָה'" - אחרי מה שהם עשו לי, ברור וגלוי לכולם שמותר לי להחזיר להם ואין בכך חטא.
  • ירמיהו ב34: "גַּם בִּכְנָפַיִךְ נִמְצְאוּ דַּם נַפְשׁוֹת אֶבְיוֹנִים נְקִיִּים לֹא בַמַּחְתֶּרֶת מְצָאתִים כִּי עַל כָּל אֵלֶּה" - אביונים שדמם נקי (ראו סעיף ב למעלה), או אביונים שצדקתם גלויה, ואתם הרגתם אותם בגלוי - לא במחתרת.
  • תהלים כד4: "נְקִי כַפַּיִם וּבַר לֵבָב, אֲשֶׁר לֹא נָשָׂא לַשָּׁוְא נַפְשִׁי וְלֹא נִשְׁבַּע לְמִרְמָה" - נקי כפיים = במעשים הגלויים, בר לבב = במחשבות הנסתרות.
  • תהלים כו6: "אֶרְחַץ בְּנִקָּיוֹן כַּפָּי, וַאֲסֹבְבָה אֶת מִזְבַּחֲךָ ה'"
  • תהלים עג13: "אַךְ רִיק זִכִּיתִי לְבָבִי, וָאֶרְחַץ בְּנִקָּיוֹן כַּפָּי" זכיתי לבבי = מחשבותיי הנסתרות, נקיון כפיי = מעשיי הגלויים.
פסוקים שאפשר לפרש בכמה דרכים:
  • שמות כג7: "מִדְּבַר שֶׁקֶר תִּרְחָק, וְנָקִי וְצַדִּיק אַל תַּהֲרֹג, כִּי לֹא אַצְדִּיק רָשָׁע" - "אם יש ספק קטן ביותר שמא הנאשם הוא חף מפשע, אל תהרוג אותו"; או: "אל תהרוג את מי שכבר יצא מבית הדין כשהוא פטור מעונש" (פירוט).
  • ישעיהו ג26: "וְאָנוּ וְאָבְלוּ פְּתָחֶיהָ, וְנִקָּתָה לָאָרֶץ תֵּשֵׁב" - "הצרות והאבל ינקו ויטהרו אותה מחטאיה", או "אחרי החורבן היא תהיה 'נקיה' מתושביה" (פירוט).


מושגים קרובים

נקי - צדיק

ע"פ מלבי"ם: "הנקי הוא בהשקף אל עצמו, שהוא עצמו חף מפשע. והצדיק הוא בהשקף אל אחרים, שאחרים יצדיקו אותו", "הנקי יציין שהוא בפני עצמו נקי מחטא, הגם שיוכל להיות שבני אדם יחשדו אותו... והצדיק יציין שהוא צדיק בדינו בעיני בני אדם..." (מכילתא שמות ג7 סימן ריא, איוב ד7 ועוד); אולם, כפי שהסברנו למעלה, נראה שההיפך הוא הנכון:
  • השורש "נקה" מציין במקרים רבים פטור מעונש בדיני אדם;
  • הביטוי "נקיון כפיים" מציין מעשים שהם חיצונים וגלויים (לא מצאנו ביטוי כמו "נקיון לבב");
  • ההאשמה החמורה על שפיכת "דם נקי" היא משום שחפותם של הנרצחים גלויה וידועה; אם אותם נרצחים, במעשיהם הגלויים, הם פושעים מסוכנים, ורק בסתר הם נקיים - קשה להאשים את מי ששופך את דמם.
ראו גם צדיק.

ראו תמים.
