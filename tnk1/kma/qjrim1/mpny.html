ראשי>הגדרות מושגים>

מפני - = תיאור סיבה

קוד: מפני-= בתנ"ך

סוג: הגדרה

מאת: אראל

אל:

בלשון ימינו המילה מִפְּנֵי- מציינת תיאור סיבה, כמו בִּגְלַל-.  אולם בלשון המקרא, ישנם פסוקים שבהם ניתן לפרש את המילה באופן מילולי, מהפנים של-:

  • בראשית טז8: "וַתֹּאמֶר ' מִפְּנֵי שָׂרַי גְּבִרְתִּי אָנֹכִי בֹּרַחַת'" - אני בורחת מהפנים של שרי, אני לא רוצה שהיא תראה אותי.  וכן בראשית לה1: "וַיֹּאמֶר אֱלֹהִים אֶל יַעֲקֹב... בְּבָרְחֲךָ מִפְּנֵי עֵשָׂו אָחִיךָ" (פירוט),   בראשית לו6: "וַיִּקַּח עֵשָׂו אֶת נָשָׁיו... וַיֵּלֶךְ אֶל אֶרֶץ מִפְּנֵי יַעֲקֹב אָחִיו",  שמות ב15: "וַיִּבְרַח מֹשֶׁה מִפְּנֵי פַרְעֹה".
  • בראשית ג8: "וַיִּשְׁמְעוּ אֶת קוֹל ה' אֱלֹהִים מִתְהַלֵּךְ בַּגָּן לְרוּחַ הַיּוֹם, וַיִּתְחַבֵּא הָאָדָם וְאִשְׁתּוֹ מִפְּנֵי ה' אֱלֹהִים בְּתוֹךְ עֵץ הַגָּן" - הם שמעו את קול ה', אבל התחבאו כי לא רצו שה' יראה אותם.
  • בראשית כז46: "וַתֹּאמֶר רִבְקָה אֶל יִצְחָק 'קַצְתִּי בְחַיַּי מִפְּנֵי בְּנוֹת חֵת. אִם לֹקֵחַ יַעֲקֹב אִשָּׁה מִבְּנוֹת חֵת כָּאֵלֶּה מִבְּנוֹת הָאָרֶץ לָמָּה לִּי חַיִּים?'" - נמאס לי לראות את הפנים של בנות חת. וכן  שמות א12: "וַיָּקֻצוּ מִפְּנֵי בְּנֵי יִשְׂרָאֵל".
  • ויקרא יט32: "מִפְּנֵי שֵׂיבָה תָּקוּם, וְהָדַרְתָּ פְּנֵי זָקֵן" - כשאתה רואה פנים של שיבה, תקום.
  • דברים א17: "לֹא תַכִּירוּ פָנִים בַּמִּשְׁפָּט, כַּקָּטֹן כַּגָּדֹל תִּשְׁמָעוּן, לֹא תָגוּרוּ מִפְּנֵי אִישׁ" (פירוט) - אל תפחדו מהפנים של אף אדם.

אולם ישנם פסוקים רבים שבהם אי אפשר לפרש כך כי אין "פנים", ושם מפני- מציינת תיאור סיבה כמו בלשוננו:

  • בראשית ז7: "וַיָּבֹא נֹחַ וּבָנָיו וְאִשְׁתּוֹ וּנְשֵׁי בָנָיו אִתּוֹ אֶל הַתֵּבָה מִפְּנֵי מֵי הַמַּבּוּל"
  • בראשית לו7: "כִּי הָיָה רְכוּשָׁם רָב מִשֶּׁבֶת יַחְדָּו, וְלֹא יָכְלָה אֶרֶץ מְגוּרֵיהֶם לָשֵׂאת אֹתָם מִפְּנֵי מִקְנֵיהֶם"
  • בראשית מא31: "וְלֹא יִוָּדַע הַשָּׂבָע בָּאָרֶץ מִפְּנֵי הָרָעָב הַהוּא אַחֲרֵי כֵן, כִּי כָבֵד הוּא מְאֹד" (פירוט)
  • שמות ט11: "וְלֹא יָכְלוּ הַחַרְטֻמִּים לַעֲמֹד לִפְנֵי מֹשֶׁה מִפְּנֵי הַשְּׁחִין, כִּי הָיָה הַשְּׁחִין בַּחֲרְטֻמִּם וּבְכָל מִצְרָיִם"
  • גם הצירוף "מפני ש-" קיים, שמות יט18: "וְהַר סִינַי עָשַׁן כֻּלּוֹ מִפְּנֵי אֲשֶׁר יָרַד עָלָיו ה' בָּאֵשׁ",  ירמיהו מד23: " מִפְּנֵי אֲשֶׁר קִטַּרְתֶּם וַאֲשֶׁר חֲטָאתֶם לה' וְלֹא שְׁמַעְתֶּם בְּקוֹל ה' וּבְתֹרָתוֹ וּבְחֻקֹּתָיו ובעדותיו לֹא הֲלַכְתֶּם - עַל כֵּן קָרָאת אֶתְכֶם הָרָעָה הַזֹּאת כַּיּוֹם הַזֶּה".

פסוקים נוספים

  • שמות ג7: "וַיֹּאמֶר ה' רָאֹה רָאִיתִי אֶת עֳנִי עַמִּי אֲשֶׁר בְּמִצְרָיִם וְאֶת צַעֲקָתָם שָׁמַעְתִּי מִפְּנֵי נֹגְשָׂיו כִּי יָדַעְתִּי אֶת מַכְאֹבָיו" (פירוט)
  • שמות ח20: "וַיַּעַשׂ ה' כֵּן וַיָּבֹא עָרֹב כָּבֵד בֵּיתָה פַרְעֹה וּבֵית עֲבָדָיו וּבְכָל אֶרֶץ מִצְרַיִם תִּשָּׁחֵת הָאָרֶץ מִפְּנֵי הֶעָרֹב"
  • שמות ט30: "וְאַתָּה וַעֲבָדֶיךָ יָדַעְתִּי כִּי טֶרֶם תִּירְאוּן מִפְּנֵי ה' אֱלֹהִים"
  • שמות י3: "וַיָּבֹא מֹשֶׁה וְאַהֲרֹן אֶל פַּרְעֹה וַיֹּאמְרוּ אֵלָיו כֹּה אָמַר ה' אֱלֹהֵי הָעִבְרִים עַד מָתַי מֵאַנְתָּ לֵעָנֹת מִפָּנָי שַׁלַּח עַמִּי וְיַעַבְדֻנִי"
  • שמות יד25: "וַיָּסַר אֵת אֹפַן מַרְכְּבֹתָיו וַיְנַהֲגֵהוּ בִכְבֵדֻת וַיֹּאמֶר מִצְרַיִם אָנוּסָה מִפְּנֵי יִשְׂרָאֵל כִּי ה' נִלְחָם לָהֶם בְּמִצְרָיִם"
  • ויקרא כו10: "וַאֲכַלְתֶּם יָשָׁן נוֹשָׁן, וְיָשָׁן מִפְּנֵי חָדָשׁ תּוֹצִיאוּ"
  • ויקרא כו37: "וְכָשְׁלוּ אִישׁ בְּאָחִיו כְּמִפְּנֵי חֶרֶב וְרֹדֵף אָיִן וְלֹא תִהְיֶה לָכֶם תְּקוּמָה לִפְנֵי אֹיְבֵיכֶם"
  • במדבר כ6: "וַיָּבֹא מֹשֶׁה וְאַהֲרֹן מִפְּנֵי הַקָּהָל אֶל פֶּתַח אֹהֶל מוֹעֵד וַיִּפְּלוּ עַל פְּנֵיהֶם וַיֵּרָא כְבוֹד ה' אֲלֵיהֶם"
  • במדבר כב3: "וַיָּגָר מוֹאָב מִפְּנֵי הָעָם מְאֹד כִּי רַב הוּא וַיָּקָץ מוֹאָב מִפְּנֵי בְּנֵי יִשְׂרָאֵל"
  • במדבר כב33: "וַתִּרְאַנִי הָאָתוֹן וַתֵּט לְפָנַי זֶה שָׁלֹשׁ רְגָלִים אוּלַי נָטְתָה מִפָּנַי כִּי עַתָּה גַּם אֹתְכָה הָרַגְתִּי וְאוֹתָהּ הֶחֱיֵיתִי"
  • במדבר לב17: "וַאֲנַחְנוּ נֵחָלֵץ חֻשִׁים לִפְנֵי בְּנֵי יִשְׂרָאֵל עַד אֲשֶׁר אִם הֲבִיאֹנֻם אֶל מְקוֹמָם וְיָשַׁב טַפֵּנוּ בְּעָרֵי הַמִּבְצָר מִפְּנֵי יֹשְׁבֵי הָאָרֶץ"
  • במדבר לג8: "וַיִּסְעוּ מִפְּנֵי הַחִירֹת וַיַּעַבְרוּ בְתוֹךְ הַיָּם הַמִּדְבָּרָה וַיֵּלְכוּ דֶּרֶךְ שְׁלֹשֶׁת יָמִים בְּמִדְבַּר אֵתָם וַיַּחֲנוּ בְּמָרָה"
  • דברים ה5: "אָנֹכִי עֹמֵד בֵּין ה' וּבֵינֵיכֶם בָּעֵת הַהִוא לְהַגִּיד לָכֶם אֶת דְּבַר ה' כִּי יְרֵאתֶם מִפְּנֵי הָאֵשׁ וְלֹא עֲלִיתֶם בָּהָר לֵאמֹר"
  • דברים ט19: "כִּי יָגֹרְתִּי מִפְּנֵי הָאַף וְהַחֵמָה אֲשֶׁר קָצַף ה' עֲלֵיכֶם לְהַשְׁמִיד אֶתְכֶם וַיִּשְׁמַע ה' אֵלַי גַּם בַּפַּעַם הַהִוא"
  • דברים כח20: "יְשַׁלַּח ה' בְּךָ אֶת הַמְּאֵרָה אֶת הַמְּהוּמָה וְאֶת הַמִּגְעֶרֶת בְּכָל מִשְׁלַח יָדְךָ אֲשֶׁר תַּעֲשֶׂה עַד הִשָּׁמֶדְךָ וְעַד אֲבָדְךָ מַהֵר מִפְּנֵי רֹעַ מַעֲלָלֶיךָ אֲשֶׁר עֲזַבְתָּנִי"
  • יהושע ד7: "וַאֲמַרְתֶּם לָהֶם אֲשֶׁר נִכְרְתוּ מֵימֵי הַיַּרְדֵּן מִפְּנֵי אֲרוֹן בְּרִית ה' בְּעָבְרוֹ בַּיַּרְדֵּן נִכְרְתוּ מֵי הַיַּרְדֵּן וְהָיוּ הָאֲבָנִים הָאֵלֶּה לְזִכָּרוֹן לִבְנֵי יִשְׂרָאֵל עַד עוֹלָם"
  • יהושע ה1: "וַיְהִי כִשְׁמֹעַ כָּל מַלְכֵי הָאֱמֹרִי אֲשֶׁר בְּעֵבֶר הַיַּרְדֵּן יָמָּה וְכָל מַלְכֵי הַכְּנַעֲנִי אֲשֶׁר עַל הַיָּם אֵת אֲשֶׁר הוֹבִישׁ ה' אֶת מֵי הַיַּרְדֵּן מִפְּנֵי בְּנֵי יִשְׂרָאֵל עַד עברנו[עָבְרָם] וַיִּמַּס לְבָבָם וְלֹא הָיָה בָם עוֹד רוּחַ מִפְּנֵי בְנֵי יִשְׂרָאֵל"
