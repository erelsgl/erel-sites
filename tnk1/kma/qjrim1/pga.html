ראשי>הגדרות מושגים>

פגע = פגש בכאב

קוד: פגע= בתנ"ך

סוג: הגדרה

מאת: אראל

אל:

פגיעה היא פגישה כואבת, פגישה שיש בה עימות ומאבק. פגיעה יכולה להיות בגוף:
  • שמות ה3: "וַיֹּאמְרוּ 'אֱלֹהֵי הָעִבְרִים נִקְרָא עָלֵינוּ, נֵלֲכָה נָּא דֶּרֶךְ שְׁלֹשֶׁת יָמִים בַּמִּדְבָּר וְנִזְבְּחָה לה' אֱלֹהֵינוּ פֶּן יִפְגָּעֵנוּ בַּדֶּבֶר אוֹ בֶחָרֶב'".
  • במדבר לה19: "גֹּאֵל הַדָּם הוּא יָמִית אֶת הָרֹצֵחַ בְּפִגְעוֹ בוֹ הוּא ימתנו"
  • שופטים יח25: "וַיֹּאמְרוּ אֵלָיו בְּנֵי דָן 'אַל תַּשְׁמַע קוֹלְךָ עִמָּנוּ פֶּן יִפְגְּעוּ בָכֶם אֲנָשִׁים מָרֵי נֶפֶשׁ, וְאָסַפְתָּה נַפְשְׁךָ וְנֶפֶשׁ בֵּיתֶךָ'"
  • שמואל א כב17-18: "וַיֹּאמֶר הַמֶּלֶךְ לָרָצִים הַנִּצָּבִים עָלָיו 'סֹבּוּ וְהָמִיתוּ כֹּהֲנֵי ה' כִּי גַם יָדָם עִם דָּוִד וְכִי יָדְעוּ כִּי בֹרֵחַ הוּא וְלֹא גָלוּ אֶת אזנו[אָזְנִי] ' וְלֹא אָבוּ עַבְדֵי הַמֶּלֶךְ לִשְׁלֹחַ אֶת יָדָם לִפְגֹעַ בְּכֹהֲנֵי ה'. וַיֹּאמֶר הַמֶּלֶךְ ל דויג[דּוֹאֵג] 'סֹב אַתָּה וּפְגַע בַּכֹּהֲנִים' וַיִּסֹּב דויג[דּוֹאֵג] הָאֲדֹמִי וַיִּפְגַּע הוּא בַּכֹּהֲנִים, וַיָּמֶת בַּיּוֹם הַהוּא שְׁמֹנִים וַחֲמִשָּׁה אִישׁ נֹשֵׂא אֵפוֹד בָּד".
  • מלכים א ב31-32: "וַיֹּאמֶר לוֹ הַמֶּלֶךְ 'עֲשֵׂה כַּאֲשֶׁר דִּבֶּר וּפְגַע בּוֹ וּקְבַרְתּוֹ, וַהֲסִירֹתָ דְּמֵי חִנָּם אֲשֶׁר שָׁפַךְ יוֹאָב מֵעָלַי וּמֵעַל בֵּית אָבִי. וְהֵשִׁיב ה' אֶת דָּמוֹ עַל רֹאשׁוֹ, אֲשֶׁר פָּגַע בִּשְׁנֵי אֲנָשִׁים צַדִּקִים וְטֹבִים מִמֶּנּוּ וַיַּהַרְגֵם בַּחֶרֶב וְאָבִי דָוִד לֹא יָדָע, אֶת אַבְנֵר בֶּן נֵר שַׂר צְבָא יִשְׂרָאֵל וְאֶת עֲמָשָׂא בֶן יֶתֶר שַׂר צְבָא יְהוּדָה"
פגיעה יכולה להיות גם בדברים - הפצרה עיקשת באדם או תפילה עיקשת לה':
  • בראשית כג8: "וַיְדַבֵּר אִתָּם לֵאמֹר אִם יֵשׁ אֶת נַפְשְׁכֶם לִקְבֹּר אֶת מֵתִי מִלְּפָנַי שְׁמָעוּנִי וּפִגְעוּ לִי בְּעֶפְרוֹן בֶּן צֹחַר"
  • ירמיהו ז16: "וְאַתָּה אַל תִּתְפַּלֵּל בְּעַד הָעָם הַזֶּה וְאַל תִּשָּׂא בַעֲדָם רִנָּה וּתְפִלָּה וְאַל תִּפְגַּע בִּי כִּי אֵינֶנִּי שֹׁמֵעַ אֹתָךְ"
  • ירמיהו טו11: "אָמַר ה' 'אִם לֹא שרותך[שֵׁרִיתִיךָ] לְטוֹב, אִם לוֹא הִפְגַּעְתִּי בְךָ בְּעֵת רָעָה וּבְעֵת צָרָה אֶת הָאֹיֵב'"
  • ירמיהו כז18: "וְאִם נְבִאִים הֵם וְאִם יֵשׁ דְּבַר ה' אִתָּם, יִפְגְּעוּ נָא בה' צְבָאוֹת לְבִלְתִּי בֹאוּ הַכֵּלִים הַנּוֹתָרִים בְּבֵית ה' וּבֵית מֶלֶךְ יְהוּדָה וּבִירוּשָׁלִַם בָּבֶלָה"
  • ירמיהו לו25: "וְגַם אֶלְנָתָן וּדְלָיָהוּ וּגְמַרְיָהוּ הִפְגִּעוּ בַמֶּלֶךְ לְבִלְתִּי שְׂרֹף אֶת הַמְּגִלָּה וְלֹא שָׁמַע אֲלֵיהֶם"
"ענינו פגישה במקום או בדבר מה, ויבוא גם לענין נגיעה ואחיזה במי לעשות לו רעה, ורובו מכת מות [ויש גם לטוב]. ובמוסרי ענינו הפצרה במי בתפילה ובקשה." (שבי"ל, ספר המילים).

פסוקים נוספים

  • בראשית כח11: "וַיִּפְגַּע בַּמָּקוֹם וַיָּלֶן שָׁם כִּי בָא הַשֶּׁמֶשׁ וַיִּקַּח מֵאַבְנֵי הַמָּקוֹם וַיָּשֶׂם מְרַאֲשֹׁתָיו וַיִּשְׁכַּב בַּמָּקוֹם הַהוּא" - ע"פ הפשט הכוונה לפגישה, ויש מפרשים שהכוונה לתפילה (פירוט).
  • בראשית לב2: "וְיַעֲקֹב הָלַךְ לְדַרְכּוֹ, וַיִּפְגְּעוּ בוֹ מַלְאֲכֵי אֱלֹהִים"
  • שמות ה20: "וַיִּפְגְּעוּ אֶת מֹשֶׁה וְאֶת אַהֲרֹן נִצָּבִים לִקְרָאתָם בְּצֵאתָם מֵאֵת פַּרְעֹה"
  • שמות כג4: "כִּי תִפְגַּע שׁוֹר אֹיִבְךָ אוֹ חֲמֹרוֹ תֹּעֶה, הָשֵׁב תְּשִׁיבֶנּוּ לוֹ"
  • במדבר לה21: "אוֹ בְאֵיבָה הִכָּהוּ בְיָדוֹ וַיָּמֹת מוֹת יוּמַת הַמַּכֶּה רֹצֵחַ הוּא גֹּאֵל הַדָּם יָמִית אֶת הָרֹצֵחַ בְּפִגְעוֹ בוֹ"
  • יהושע ב16: "וַתֹּאמֶר לָהֶם הָהָרָה לֵּכוּ פֶּן יִפְגְּעוּ בָכֶם הָרֹדְפִים וְנַחְבֵּתֶם שָׁמָּה שְׁלֹשֶׁת יָמִים עַד שׁוֹב הָרֹדְפִים וְאַחַר תֵּלְכוּ לְדַרְכְּכֶם"
  • יהושע טז7: "וְיָרַד מִיָּנוֹחָה עֲטָרוֹת וְנַעֲרָתָה וּפָגַע בִּירִיחוֹ וְיָצָא הַיַּרְדֵּן"
  • יהושע יז10: "נֶגְבָּה לְאֶפְרַיִם וְצָפוֹנָה לִמְנַשֶּׁה וַיְהִי הַיָּם גְּבוּלוֹ וּבְאָשֵׁר יִפְגְּעוּן מִצָּפוֹן וּבְיִשָּׂשכָר מִמִּזְרָח"
  • יהושע יט11: "וְעָלָה גְבוּלָם לַיָּמָּה וּמַרְעֲלָה וּפָגַע בְּדַבָּשֶׁת וּפָגַע אֶל הַנַּחַל אֲשֶׁר עַל פְּנֵי יָקְנְעָם"
  • יהושע יט22: "וּפָגַע הַגְּבוּל בְּתָבוֹר ושחצומה[וְשַׁחֲצִימָה] וּבֵית שֶׁמֶשׁ וְהָיוּ תֹּצְאוֹת גְּבוּלָם הַיַּרְדֵּן עָרִים שֵׁשׁ עֶשְׂרֵה וְחַצְרֵיהֶן"
  • יהושע יט26: "וְאַלַמֶּלֶךְ וְעַמְעָד וּמִשְׁאָל וּפָגַע בְּכַרְמֶל הַיָּמָּה וּבְשִׁיחוֹר לִבְנָת"
  • יהושע יט27: "וְשָׁב מִזְרַח הַשֶּׁמֶשׁ בֵּית דָּגֹן וּפָגַע בִּזְבֻלוּן וּבְגֵי יִפְתַּח אֶל צָפוֹנָה בֵּית הָעֵמֶק וּנְעִיאֵל וְיָצָא אֵל כָּבוּל מִשְּׂמֹאל"
  • יהושע יט34: "וְשָׁב הַגְּבוּל יָמָּה אַזְנוֹת תָּבוֹר וְיָצָא מִשָּׁם חוּקֹקָה וּפָגַע בִּזְבֻלוּן מִנֶּגֶב וּבְאָשֵׁר פָּגַע מִיָּם וּבִיהוּדָה הַיַּרְדֵּן מִזְרַח הַשָּׁמֶשׁ"
  • שופטים ח21: "וַיֹּאמֶר זֶבַח וְצַלְמֻנָּע קוּם אַתָּה וּפְגַע בָּנוּ כִּי כָאִישׁ גְּבוּרָתוֹ וַיָּקָם גִּדְעוֹן וַיַּהֲרֹג אֶת זֶבַח וְאֶת צַלְמֻנָּע וַיִּקַּח אֶת הַשַּׂהֲרֹנִים אֲשֶׁר בְּצַוְּארֵי גְמַלֵּיהֶם"
  • שופטים טו12: "וַיֹּאמְרוּ לוֹ לֶאֱסָרְךָ יָרַדְנוּ לְתִתְּךָ בְּיַד פְּלִשְׁתִּים וַיֹּאמֶר לָהֶם שִׁמְשׁוֹן הִשָּׁבְעוּ לִי פֶּן תִּפְגְּעוּן בִּי אַתֶּם"
  • שמואל א י5: "אַחַר כֵּן תָּבוֹא גִּבְעַת הָאֱלֹהִים אֲשֶׁר שָׁם נְצִבֵי פְלִשְׁתִּים וִיהִי כְבֹאֲךָ שָׁם הָעִיר וּפָגַעְתָּ חֶבֶל נְבִיאִים יֹרְדִים מֵהַבָּמָה וְלִפְנֵיהֶם נֵבֶל וְתֹף וְחָלִיל וְכִנּוֹר וְהֵמָּה מִתְנַבְּאִים"
  • מלכים א ה18: "וְעַתָּה הֵנִיחַ ה' אֱלֹהַי לִי מִסָּבִיב אֵין שָׂטָן וְאֵין פֶּגַע רָע"
  • ישעיהו מז3: "תִּגָּל עֶרְוָתֵךְ גַּם תֵּרָאֶה חֶרְפָּתֵךְ נָקָם אֶקָּח וְלֹא אֶפְגַּע אָדָם"
  • ישעיהו סד4: "פָּגַעְתָּ אֶת שָׂשׂ וְעֹשֵׂה צֶדֶק בִּדְרָכֶיךָ יִזְכְּרוּךָ הֵן אַתָּה קָצַפְתָּ וַנֶּחֱטָא בָּהֶם עוֹלָם וְנִוָּשֵׁעַ"
  • עמוס ה19: "כַּאֲשֶׁר יָנוּס אִישׁ מִפְּנֵי הָאֲרִי וּפְגָעוֹ הַדֹּב וּבָא הַבַּיִת וְסָמַךְ יָדוֹ עַל הַקִּיר וּנְשָׁכוֹ הַנָּחָשׁ"
  • איוב ז20: "חָטָאתִי מָה אֶפְעַל לָךְ נֹצֵר הָאָדָם לָמָה שַׂמְתַּנִי לְמִפְגָּע לָךְ וָאֶהְיֶה עָלַי לְמַשָּׂא"
  • איוב כא15: "מַה שַׁדַּי כִּי נַעַבְדֶנּוּ וּמַה נּוֹעִיל כִּי נִפְגַּע בּוֹ"
  • רות א16: "וַתֹּאמֶר רוּת אֶל תִּפְגְּעִי בִי לְעָזְבֵךְ לָשׁוּב מֵאַחֲרָיִךְ כִּי אַל אֲשֶׁר תֵּלְכִי אֵלֵךְ וּבַאֲשֶׁר תָּלִינִי אָלִין עַמֵּךְ עַמִּי וֵאלֹהַיִךְ אֱלֹהָי"
  • רות ב22: "וַתֹּאמֶר נָעֳמִי אֶל רוּת כַּלָּתָהּ טוֹב בִּתִּי כִּי תֵצְאִי עִם נַעֲרוֹתָיו וְלֹא יִפְגְּעוּ בָךְ בְּשָׂדֶה אַחֵר"
  • קהלת ט11: "שַׁבְתִּי וְרָאֹה תַחַת הַשֶּׁמֶשׁ כִּי לֹא לַקַּלִּים הַמֵּרוֹץ וְלֹא לַגִּבּוֹרִים הַמִּלְחָמָה וְגַם לֹא לַחֲכָמִים לֶחֶם וְגַם לֹא לַנְּבֹנִים עֹשֶׁר וְגַם לֹא לַיֹּדְעִים חֵן כִּי עֵת וָפֶגַע יִקְרֶה אֶת כֻּלָּם"
