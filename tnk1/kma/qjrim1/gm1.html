גם בתחילת הפסוק = בנוסף לפסוק הקודם, או "אפילו"

קוד: גם בתחילת הפסוק= בתנ"ך

סוג: הגדרה

מאת: אראל

אל:

ברוב הפסוקים שמתחילים במילה "גם", המילה באה לחבר את הפסוק לפסוקים הקודמים ולומר "בנוסף למה שנזכר בפסוקים הקודמים - יש עוד":

  • בראשית ז3: "גַּם מֵעוֹף הַשָּׁמַיִם שִׁבְעָה שִׁבְעָה זָכָר וּנְקֵבָה לְחַיּוֹת זֶרַע עַל פְּנֵי כָל הָאָרֶץ" -  נח צריך לקחת לא רק מהבהמות אלא גם מהעופות.
  • שמות יב32: "גַּם צֹאנְכֶם גַּם בְּקַרְכֶם קְחוּ כַּאֲשֶׁר דִּבַּרְתֶּם וָלֵכוּ וּבֵרַכְתֶּם גַּם אֹתִי" - לא רק את נשיכם וילדכם תקחו, אלא גם את הבהמות.
  • דברים א37: "גַּם בִּי הִתְאַנַּף ה' בִּגְלַלְכֶם לֵאמֹר גַּם אַתָּה לֹא תָבֹא שָׁם" - לא רק עליכם כעס ה' (לאחר חטא המרגלים), אלא גם עליי.
  • דברים כח61: "גַּם כָּל חֳלִי וְכָל מַכָּה אֲשֶׁר לֹא כָתוּב בְּסֵפֶר הַתּוֹרָה הַזֹּאת יַעְלֵם ה' עָלֶיךָ עַד הִשָּׁמְדָךְ" - לא רק הקללות שכתובות בפסוקים הקודמים יבואו עליכם, אלא גם כל חולי וכל מכה שאינם כתובים.
  • שופטים ב21: "גַּם אֲנִי לֹא אוֹסִיף לְהוֹרִישׁ אִישׁ מִפְּנֵיהֶם מִן הַגּוֹיִם אֲשֶׁר עָזַב יְהוֹשֻׁעַ וַיָּמֹת" - כמו שבני-ישראל הפרו את הברית ועבדו אלילים, גם אני אפר את החלק שלי בברית ולא אוריש מפניהם את הגויים.
  • שמואל א ב15: "גַּם בְּטֶרֶם יַקְטִרוּן אֶת הַחֵלֶב, וּבָא נַעַר הַכֹּהֵן וְאָמַר לָאִישׁ הַזֹּבֵחַ תְּנָה בָּשָׂר לִצְלוֹת לַכֹּהֵן וְלֹא יִקַּח מִמְּךָ בָשָׂר מְבֻשָּׁל כִּי אִם חָי" - בנוסף לחטא שנזכר בפסוקים הקודמים (שהכהנים לקחו כל אשר יעלה המזלג), היתה להם עבירה נוספת.
  • שמואל א יב16: "גַּם עַתָּה הִתְיַצְּבוּ וּרְאוּ אֶת הַדָּבָר הַגָּדוֹל הַזֶּה אֲשֶׁר ה' עֹשֶׂה לְעֵינֵיכֶם" - בנוסף להוכחות השכליות וההסטוריות שהבאתי לכם בפסוקים הקודמים, כדי להוכיח לכם שאתם חוטאים - אני מתכוון להביא לכם הוכחה ניסית ברגע זה ממש (גשם ביום קציר).
  • שמואל א יב23: "גַּם אָנֹכִי, חָלִילָה לִּי מֵחֲטֹא לה' מֵחֲדֹל לְהִתְפַּלֵּל בַּעַדְכֶם, וְהוֹרֵיתִי אֶתְכֶם בְּדֶרֶךְ הַטּוֹבָה וְהַיְשָׁרָה" - בנוסף לכך שה' לא ייטוש אתכם בעבור שמו הגדול, גם אני אתפלל בעדכם.
  • שמואל ב ה2: "גַּם אֶתְמוֹל גַּם שִׁלְשׁוֹם בִּהְיוֹת שָׁאוּל מֶלֶךְ עָלֵינוּ אַתָּה [הָיִיתָ]המוציאוהמביאֶת יִשְׂרָאֵל, וַיֹּאמֶר ה' לְךָ אַתָּה תִרְעֶה אֶת עַמִּי אֶת יִשְׂרָאֵל וְאַתָּה תִּהְיֶה לְנָגִיד עַל יִשְׂרָאֵל" - בנוסף לסיבה שנזכרה בפסוק הקודם ("עצמך ובשרך אנחנו"), יש שתי סיבות נוספות לכך שאתה ראוי למלוך עלינו.
  • שמואל ב ח11: "גַּם אֹתָם הִקְדִּישׁ הַמֶּלֶךְ דָּוִד לה', עִם הַכֶּסֶף וְהַזָּהָב אֲשֶׁר הִקְדִּישׁ מִכָּל הַגּוֹיִם אֲשֶׁר כִּבֵּשׁ" - ההסבר כתוב בסוף הפסוק
  • מלכים ב יז19: "גַּם יְהוּדָה לֹא שָׁמַר אֶת מצות ה' אֱלֹהֵיהֶם, וַיֵּלְכוּ בְּחֻקּוֹת יִשְׂרָאֵל אֲשֶׁר עָשׂוּ" - כנ"ל
  • ישעיהו יד8: "גַּם בְּרוֹשִׁים שָׂמְחוּ לְךָ אַרְזֵי לְבָנוֹן; מֵאָז שָׁכַבְתָּ לֹא יַעֲלֶה הַכֹּרֵת עָלֵינוּ" - לא רק אנשים שמחו שהלכת, אלא גם עצים
  • ישעיהו כח29: "גַּם זֹאת מֵעִם ה' צְבָאוֹת יָצָאָה, הִפְלִיא עֵצָה הִגְדִּיל תּוּשִׁיָּה" - הקטע כולו לא ברור, אבל יש בו שני חלקים; בסוף החלק הראשון נאמר "אלהיו יורנו", ובסוף החלק השני נאמר "גם זאת מעם ה' צבאות יצאה".
  • ישעיהו סו4: "גַּם אֲנִי אֶבְחַר בְּתַעֲלֻלֵיהֶם וּמְגוּרֹתָם אָבִיא לָהֶם, יַעַן קָרָאתִי וְאֵין עוֹנֶה דִּבַּרְתִּי וְלֹא שָׁמֵעוּ וַיַּעֲשׂוּ הָרַע בְּעֵינַי וּבַאֲשֶׁר לֹא חָפַצְתִּי בָּחָרוּ" - כמו שהם בחרו בדרכים רעות, גם אני אבחר במעשים רעים לעשות להם.
  • ירמיהו ב16: "גַּם בְּנֵי נֹף ותחפנסיִרְעוּךְ קָדְקֹד" - בנוסף לכפירים שבזזו את ארצך, גם המצרים מרצצים את קדקדך.
  • ירמיהו ב34: "גַּם בִּכְנָפַיִךְ נִמְצְאוּ דַּם נַפְשׁוֹת אֶבְיוֹנִים נְקִיִּים לֹא בַמַּחְתֶּרֶת מְצָאתִים כִּי עַל כָּל אֵלֶּה" - בנוסף לכל המעשים הרעים שנזכרו בפסוקים הקודמים.
  • ירמיהו ב37: "גַּם מֵאֵת זֶה תֵּצְאִי וְיָדַיִךְ עַל רֹאשֵׁךְ כִּי מָאַס ה' בְּמִבְטַחַיִךְ וְלֹא תַצְלִיחִי לָהֶם" - כמו שהתאכזבת מאשור, כך תתאכזבי גם ממצרים.
  • ירמיהו לג21-26: "גַּם בְּרִיתִי תֻפַר אֶת דָּוִד עַבְדִּי מִהְיוֹת לוֹ בֵן מֹלֵךְ עַל כִּסְאוֹ וְאֶת הַלְוִיִּם הַכֹּהֲנִים מְשָׁרְתֵי... גַּם זֶרַע יַעֲקוֹב וְדָוִד עַבְדִּי אֶמְאַס מִקַּחַת מִזַּרְעוֹ מֹשְׁלִים אֶל זֶרַע אַבְרָהָם יִשְׂחָק וְיַעֲקֹב כִּי אשוב[אָשִׁיב] אֶת שְׁבוּתָם וְרִחַמְתִּים"" - אם יתבטלו חוקי השמים והארץ, תתבטל גם הברית בין ה' לבין בית דוד.
  • ירמיהו מו21: "גַּם שְׂכִרֶיהָ בְקִרְבָּהּ כְּעֶגְלֵי מַרְבֵּק, כִּי גַם הֵמָּה הִפְנוּ נָסוּ יַחְדָּיו לֹא עָמָדוּ" - מצרים היא כמו עגלה יפהפיה וחלשה, וגם החיילים השכירים שבאו להגן עליה הם כמו עגלים שמנים ופחדנים.
  • יחזקאל טז52: "גַּם אַתְּ שְׂאִי כְלִמָּתֵךְ אֲשֶׁר פִּלַּלְתְּ לַאֲחוֹתֵךְ בְּחַטֹּאַתְַּיִךְ אֲשֶׁר הִתְעַבְתְּ מֵהֵן תִּצְדַּקְנָה מִמֵּךְ" - כמו ששומרון אחותך התביישה, גם את (יהודה) שאי כלימתך.
  • יחזקאל כ23: "גַּם אֲנִי נָשָׂאֶתִי אֶת יָדִי לָהֶם בַּמִּדְבָּר לְהָפִיץ אֹתָם בַּגּוֹיִם וּלְזָרוֹת אוֹתָם בָּאֲרָצוֹת" - כמו שהם הפרו את הברית, גם אני.
  • יחזקאל לא17: "גַּם הֵם אִתּוֹ יָרְדוּ שְׁאוֹלָה אֶל חַלְלֵי חָרֶב וּזְרֹעוֹ יָשְׁבוּ בְצִלּוֹ בְּתוֹךְ גּוֹיִם" - כמו פרעה מלך מצרים שעליו מדברים הפסוקים הקודמים, גם גויים אחרים ירדו שאולה
  • הושע ו11: "גַּם יְהוּדָה שָׁת קָצִיר לָךְ בְּשׁוּבִי שְׁבוּת עַמִּי" - כמו בית ישראל שנזכרו בפסוק הקודם, גם בית יהודה.
  • הושע י6: "גַּם אוֹתוֹ לְאַשּׁוּר יוּבָל מִנְחָה לְמֶלֶךְ יָרֵב בָּשְׁנָה אֶפְרַיִם יִקָּח וְיֵבוֹשׁ יִשְׂרָאֵל מֵעֲצָתוֹ" - גם כבודו (של העגל) גלה, וגם הוא עצמו יגלה ויובל לאשור.
  • יואל א20: "גַּם בַּהֲמוֹת שָׂדֶה תַּעֲרוֹג אֵלֶיךָ כִּי יָבְשׁוּ אֲפִיקֵי מָיִם וְאֵשׁ אָכְלָה נְאוֹת הַמִּדְבָּר" - לא רק אני קורא אליך, גם הבהמות.
  • נחום ג10-11: "גַּם הִיא לַגֹּלָה הָלְכָה בַשֶּׁבִי גַּם עֹלָלֶיהָ יְרֻטְּשׁוּ בְּרֹאשׁ כָּל חוּצוֹת וְעַל נִכְבַּדֶּיהָ יַדּוּ גוֹרָל וְכָל גְּדוֹלֶיהָ רֻתְּקוּ בַזִּקִּים. גַּם אַתְּ תִּשְׁכְּרִי תְּהִי נַעֲלָמָה גַּם אַתְּ תְּבַקְשִׁי מָעוֹז מֵאוֹיֵב" - כוש, מצרים, פוט, לוב, וגם נינוה.
  • צפניה ב12: "גַּם אַתֶּם כּוּשִׁים חַלְלֵי חַרְבִּי הֵמָּה" - כמו כל הגויים שנזכרו בפסוק הקודם.
  • דברי הימים ב כא11: "גַּם הוּא עָשָׂה בָמוֹת בְּהָרֵי יְהוּדָה וַיֶּזֶן אֶת יֹשְׁבֵי יְרוּשָׁלִַם וַיַּדַּח אֶת יְהוּדָה" - כמו שאדום פשעו (בפסוק הקודם), גם יהורם פשע. או: כמו שבית אחאב (שנזכרו 5 פסוקים קודם) עבדו אלילים, גם יהורם.
  • דברי הימים ב כב3-5: "גַּם הוּא הָלַךְ בְּדַרְכֵי בֵּית אַחְאָב כִּי אִמּוֹ הָיְתָה יוֹעַצְתּוֹ לְהַרְשִׁיעַ... גַּם בַּעֲצָתָם הָלַךְ וַיֵּלֶךְ אֶת יְהוֹרָם בֶּן אַחְאָב מֶלֶךְ יִשְׂרָאֵל לַמִּלְחָמָה עַל חֲזָאֵל מֶלֶךְ אֲרָם" - כמו אביו יהורם.
  • דברי הימים ב כט7: "גַּם סָגְרוּ דַּלְתוֹת הָאוּלָם וַיְכַבּוּ אֶת הַנֵּרוֹת וּקְטֹרֶת לֹא הִקְטִירוּ וְעֹלָה לֹא הֶעֱלוּ בַקֹּדֶשׁ לֵאלֹהֵי יִשְׂרָאֵל" - בנוסף לפשעים שנזכרו בפסוק הקודם.
  • דברי הימים ב ל12: "גַּם בִּיהוּדָה הָיְתָה יַד הָאֱלֹהִים לָתֵת לָהֶם לֵב אֶחָד לַעֲשׂוֹת מִצְוַת הַמֶּלֶךְ וְהַשָּׂרִים בִּדְבַר ה'" - גם אנשים מאשר מנשה וזבולון באו לחג, וגם מיהודה.
  • דברי הימים ב לו14: "גַּם כָּל שָׂרֵי הַכֹּהֲנִים וְהָעָם הִרְבּוּ למעולמַעַל כְּכֹל תֹּעֲבוֹת הַגּוֹיִם, וַיְטַמְּאוּ אֶת בֵּית ה' אֲשֶׁר הִקְדִּישׁ בִּירוּשָׁלִָם" - בנוסף למלך צדקיהו, גם השרים והכהנים.
  • תהלים יט12: "גַּם עַבְדְּךָ נִזְהָר בָּהֶם בְּשָׁמְרָם עֵקֶב רָב" - כמו כל האנשים שחומדים את דברי ה' ("הנחמדים מזהב"), גם אני (ראו מצודת דוד).
  • תהלים יט14: "גַּם מִזֵּדִים חֲשֹׂךְ עַבְדֶּךָ אַל יִמְשְׁלוּ בִי אָז אֵיתָם וְנִקֵּיתִי מִפֶּשַׁע רָב" - הצל אותי משגיאות ומנסתרות, וגם מעבירות בזדון.
  • תהלים כה3: "גַּם כָּל קֹוֶיךָ לֹא יֵבֹשׁוּ יֵבֹשׁוּ הַבּוֹגְדִים רֵיקָם" - בנוסף לכך שאני לא אבוש - אני מבקש שגם כל שאר קוויך לא יבושו.

  • תהלים פג9: "גַּם אַשּׁוּר נִלְוָה עִמָּם הָיוּ זְרוֹעַ לִבְנֵי לוֹט סֶלָה" - בנוסף לשאר העמים שנזכרו בפסוקים הקודמים, גם אשור.
  • תהלים פה13: "גַּם ה' יִתֵּן הַטּוֹב וְאַרְצֵנוּ תִּתֵּן יְבוּלָהּ" - "מלבד גמול המעשה, יתן גם את הטוב = מה שהבטיח בחסדו" (מצודת דוד)
  • תהלים קיט23: "גַּם יָשְׁבוּ שָׂרִים בִּי נִדְבָּרוּ עַבְדְּךָ יָשִׂיחַ בְּחֻקֶּיךָ"  - לא רק שחירפו וביזו אותי באקראי (בפסוק הקודם), אלא גם ישבו דרך קבע לדבר עליי דברים רעים (מצודת דוד).
  • תהלים קיט24: "גַּם עֵדֹתֶיךָ שַׁעֲשֻׁעָי אַנְשֵׁי עֲצָתִי" - לא רק שאשיח בחוקיך ברצינות, אלא גם אשתעשע בהם בשעות הפנאי שלי, וגם אתייעץ בהם.

  • נחמיה ה13: "גַּם חָצְנִי נָעַרְתִּי וָאֹמְרָה 'כָּכָה יְנַעֵר הָאֱלֹהִים אֶת כָּל הָאִישׁ אֲשֶׁר לֹא יָקִים אֶת הַדָּבָר הַזֶּה מִבֵּיתוֹ וּמִיגִיעוֹ וְכָכָה יִהְיֶה נָעוּר וָרֵק'" - בנוסף לדברי התוכחה שאמרתי לנושים, גם עשיתי מעשה שממחיש אותם.
  • נחמיה ה14: "גַּם מִיּוֹם אֲשֶׁר צִוָּה אֹתִי לִהְיוֹת פֶּחָם בְּאֶרֶץ יְהוּדָה... לֶחֶם הַפֶּחָה לֹא אָכַלְתִּי" - בנוסף למאבקי למען הלווים העניים.
  • נחמיה ו17: "גַּם בַּיָּמִים הָהֵם מַרְבִּים חֹרֵי יְהוּדָה אִגְּרֹתֵיהֶם הוֹלְכוֹת עַל טוֹבִיָּה וַאֲשֶׁר לְטוֹבִיָּה בָּאוֹת אֲלֵיהֶם...  גַּם טובתיו היו אמרים לפני ודברי היו מוציאים לו אגרות שלח טוֹבִיָּה ליראני" - בנוסף לצרות האחרות שעשו האויבים לנחמיה (בתחילת הפרק).
  • נחמיה יג23: "גַּם בַּיָּמִים הָהֵם רָאִיתִי אֶת הַיְּהוּדִים הֹשִׁיבוּ נָשִׁים אשדודיותעמוניותמוֹאֲבִיּוֹת" - בנוסף לחילול-השבת שנזכר לפני כן.
  • משלי א26: "גַּם אֲנִי בְּאֵידְכֶם אֶשְׂחָק אֶלְעַג בְּבֹא פַחְדְּכֶם" - כמו שאתם מתלוצצים, גם אני אצחק כשיבוא עליכם אסון.
  • משלי יח9: "גם מתרפה במלאכתו, אח הוא לבעל משחית" - בנוסף לנרגן (שנזכר בפסוק הקודם), גם אדם שמתבטל ממלאכתו עוזר להשחתת היחסים בין הבריות (פירוט)
  • משלי כד23: "גַּם אֵלֶּה לַחֲכָמִים: הַכֵּר פָּנִים בְּמִשְׁפָּט בַּל טוֹב" - בנוסף למשלים שנזכרו עד עכשיו, גם המשלים שבאים בהמשך הם של חכמים אחרים (ראו משלי שלמה ומשלי חכמים אחרים).
  • משלי כה1: "גַּם אֵלֶּה מִשְׁלֵי שְׁלֹמֹה אֲשֶׁר הֶעְתִּיקוּ אַנְשֵׁי חִזְקִיָּה מֶלֶךְ יְהוּדָה" - בנוסף למשלים שנזכרו קודם, גם המשלים שבאים בהמשך הם של שלמה.

ויש פסוקים שיכולים להתאים לשני הפירושים:

  • תהלים פד4: "גַּם צִפּוֹר מָצְאָה בַיִת וּדְרוֹר קֵן לָהּ אֲשֶׁר שָׁתָה אֶפְרֹחֶיהָ אֶת מִזְבְּחוֹתֶיךָ ה' צְבָאוֹת מַלְכִּי וֵאלֹהָי" - 1 לא רק אני מתאווה להיות במקדש, גם הציפורים;  2 אפילו הציפור האוהבת לנדוד והדרור האוהב חופש החליטו להשתכן ליד המזבח (פירוט).
  • משלי יז26: "גם ענוש לצדיק לא טוב, להכות נדיבים על ישר" - 1 לא רק שהוא מכעיס את הוריו כמו שנזכר בפסוק הקודם, אלא גם מעניש את הצדיקים;  2 גם לוותר לרשע בגלל שהוא במעמד גבוה זה לא טוב, וגם להחמיר יותר מדי עם צדיק שנמצא במעמד גבוה זה לא טוב (פירוט).
  • משלי יז28: "גם אויל מחריש חכם יחשב, אטם שפתיו נבון" - 1 בנוסף לשתקנים שנזכרו בפסוק הקודם, גם אויל יכול להרויח משתיקה;  2 אפילו אויל, שבדרך-כלל מזלזלים בו, יכול להיחשב לחכם אם ישתוק. (פירוט)
  • משלי יט2: "גם בלא דעת נפש לא טוב, ואץ ברגלים חוטא" - 1 בנוסף ל"עיקש שפתיו" שנזכר בפסוק הקודם, גם אדם שאינו נמצא בקשר קרוב עם אף אחד גם הוא עלול להיות לא טוב, כי אין מי שימתח עליו ביקורת אם הוא טועה;  2 אפילו אדם שחוטא בבלי דעת וכוונה, גם הוא לא טוב (פירוט).
  • משלי יד13: "גַּם בִּשְׂחוֹק יִכְאַב לֵב, וְאַחֲרִיתָהּ שִׂמְחָה תוּגָה" - 1 בנוסף לפסוק הקודם, שאמר "יש דרך ישר לפני איש ואחריתה דרכי מוות", גם כשהאדם צוחק ונהנה מדרכו - הוא עלול להגיע לכאב לב ולמוות;   2. אפילו כשאדם נראה צוחק, ייתכן שליבו כואב ועצוב (פירוט).
