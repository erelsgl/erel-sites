ראשי>הגדרות מושגים>
ראשי>לשון המקרא>עזז>

עוז = ביצור והגנה, או מרץ להתקפה

קוד: עזז= בתנ"ך

סוג: הגדרה

מאת: אראל

אל:

השורש עזז, והמילה עוֹז, מציינים, בלשון המקרא, כוח, ובמיוחד כוח הגנתי - כוח של מבצר או כוח שקול לזה:
  • במדבר יג28: "אֶפֶס כִּי עַז הָעָם הַיֹּשֵׁב בָּאָרֶץ, וְהֶעָרִים בְּצֻרוֹת גְּדֹלֹת מְאֹד; וְגַם יְלִדֵי הָעֲנָק רָאִינוּ שָׁם"
  • במדבר כא24: "ויכהו ישראל לפי חרב ויירש את ארצו מארנן עד יבק עד בני עמון, כי עז גבול בני עמון" - מבוצר (כנראה היתה להם חומת הפרדה..).
  • שופטים ט51: "וּמִגְדַּל עֹז הָיָה בְתוֹךְ הָעִיר, וַיָּנֻסוּ שָׁמָּה כָּל הָאֲנָשִׁים וְהַנָּשִׁים וְכֹל בַּעֲלֵי הָעִיר וַיִּסְגְּרוּ בַּעֲדָם וַיַּעֲלוּ עַל גַּג הַמִּגְדָּל"
  • שמואל א ב10: "...ה' יָדִין אַפְסֵי אָרֶץ, וְיִתֶּן עֹז לְמַלְכּוֹ, וְיָרֵם קֶרֶן מְשִׁיחוֹ"
  • ישעיהו כו1: "... עיר עז לנו, ישועה ישית חומות וחל"
  • עמוס ה9: "הַמַּבְלִיג שֹׁד עַל עָז וְשֹׁד עַל מִבְצָר יָבוֹא"
  • תהלים כח8: "ה' עז למו ומעוז ישועות משיחו הוא" - מגן על משיחו.
  • תהלים כט11: "ה' עֹז לְעַמּוֹ יִתֵּן, ה' יְבָרֵךְ אֶת עַמּוֹ בַשָּׁלוֹם" - ה' יתן לעם ישראל בטחון ושלום.
  • תהלים נט10: "עזו אליך אשמרה, כי אלהים משגבי"
  • תהלים עא7: "כְּמוֹפֵת הָיִיתִי לְרַבִּים, וְאַתָּה מַחֲסִי עֹז"
  • תהלים עח61: "וַיִּתֵּן לַשְּׁבִי עֻזּוֹ, וְתִפְאַרְתּוֹ בְיַד צָר" - ירושלים והמקדש, שהם כביכול המבצר של ה'.
  • תהלים פד6: "אַשְׁרֵי אָדָם עוֹז לוֹ בָךְ, מְסִלּוֹת בִּלְבָבָם" - אשרי אדם שהבטחון שלו הוא בך.
  • תהלים קמ8: "ה' ד' עֹז יְשׁוּעָתִי, סַכֹּתָה לְרֹאשִׁי בְּיוֹם נָשֶׁק" - הגנת על ראשי כמו סוכך בלתי חדיר לנשק.
  • משלי י15: "הוֹן עָשִׁיר קִרְיַת עֻזּוֹ, מְחִתַּת דַּלִּים רֵישָׁם" (פירוט)
  • משלי יד26: "בְּיִרְאַת ה' מִבְטַח עֹז, וּלְבָנָיו יִהְיֶה מַחְסֶה" (פירוט)
  • משלי יח10: "מִגְדַּל עֹז שֵׁם ה', בּוֹ יָרוּץ צַדִּיק וְנִשְׂגָּב"
  • משלי יח11: "הוֹן עָשִׁיר קִרְיַת עֻזּוֹ, וּכְחוֹמָה נִשְׂגָּבָה בְּמַשְׂכִּיתוֹ" - רכושו של העשיר מגן עליו, לפחות במחשבתו (פירוט).
  • משלי יח19: "אָח נִפְשָׁע מִקִּרְיַת עֹז, ומדונים[וּמִדְיָנִים] כִּבְרִיחַ אַרְמוֹן" - כאשר פושעים באחד האחים ומגרשים אותו מהעיר המבוצרת, הדבר גורם למריבות שנועלות את המשפחה (פירוט).
  • משלי כא22: "עִיר גִּבֹּרִים עָלָה חָכָם, וַיֹּרֶד עֹז מִבְטֶחָה" - החכם מצליח להוריד ולבטל את ההגנות של העיר המבוצרת (פירוט).
  • משלי לא25: "עז והדר לבושה, ותשחק ליום אחרון" - אשת חיל תופרת בגדים שמגנים עליה ועל בני ביתה.
  • ואולי גם משלי י29: "מָעוֹז לַתֹּם דֶּרֶךְ ה', וּמְחִתָּה לְפֹעֲלֵי אָוֶן" - דרכו של ה' היא להגן על התמימים (פירוט).

אבל גם כוח התקפתי, "ענינו תוקף כוח וגבורה" (שבי"ל)
  • בראשית מט7: "אָרוּר אַפָּם כִּי עָז וְעֶבְרָתָם כִּי קָשָׁתָה..."
  • שופטים ה21: "נַחַל קִישׁוֹן גְּרָפָם... תִּדְרְכִי נַפְשִׁי עֹז"
  • שופטים יד18: "מַהאכל יצא מאכל ומעַז יצא מָּתוֹק... מַה מָּתוֹק מִדְּבַשׁ וּמֶה עַז מֵאֲרִי?"
  • שמואל ב כב18: "יַצִּילֵנִי מֵאֹיְבִי עָז, מִשֹּׂנְאַי כִּי אָמְצוּ מִמֶּנִּי"
  • ישעיהו נא9: "עוּרִי עוּרִי לִבְשִׁי עֹז זְרוֹעַ ה'... הֲלוֹא אַתְּ הִיא הַמַּחְצֶבֶת רַהַב מְחוֹלֶלֶת תַּנִּין"
  • ישעיהו סב8: "נִשְׁבַּע ידוד בִּימִינוֹ וּבִזְרוֹעַ עֻזּוֹ אִם אֶתֵּן אֶת דְּגָנֵךְ עוֹד מַאֲכָל לְאֹיְבַיִךְ וְאִם יִשְׁתּוּ בְנֵי נֵכָר תִּירוֹשֵׁךְ אֲשֶׁר יָגַעַתְּ בּוֹ"
  • תהלים צ11: "מִי יוֹדֵעַ עֹז אַפֶּךָ, וּכְיִרְאָתְךָ עֶבְרָתֶךָ"
  • משלי יח23: "תַּחֲנוּנִים יְדַבֶּר רָשׁ, וְעָשִׁיר יַעֲנֶה עַזּוֹת" (פירוט).
  • משלי כא14: "מַתָּן בַּסֵּתֶר יִכְפֶּה אָף, וְשֹׁחַד בַּחֵק חֵמָה עַזָּה" (פירוט).
וגם מרץ:
  • שמואל ב ו14: "וְדָוִד מְכַרְכֵּר בְּכָל עֹז לִפְנֵי ה' וְדָוִד חָגוּר אֵפוֹד בָּד", דברי הימים א יג8
ויש פסוקים שאפשר לפרש בכמה דרכים:
  • משלי כד5: "גֶּבֶר חָכָם בַּעוֹז, וְאִישׁ דַּעַת מְאַמֶּץ כֹּחַ" - גבר חכם שומר על כוחו ההגנתי, ומצליח גם לתקוף (פירוט) 
  • משלי ל25: "הַנְּמָלִים עַם לֹא עָז, וַיָּכִינוּ בַקַּיִץ לַחְמָם"

מילים וביטויים

  • עזאזל - יש אומרים שהכוונה להר עז, גבוה כמבצר, שממנו היו זורקים את ה"שעיר לעזאזל".
  • עזניה - יש אומרים שהכוונה לציפור טרף עזה ותקיפה.
  • העז את פניו = נתן עוז וחוזק לפניו - שיקר או הזיק מבלי שפניו הביעו בושה או צער.

פסוקים נוספים

  • בראשית מט3-4: "רְאוּבֵן בְּכֹרִי אַתָּה, כֹּחִי וְרֵאשִׁית אוֹנִי, יֶתֶר שְׂאֵת וְיֶתֶר עָז"
  • ישעיהו יט4: "וְסִכַּרְתִּי אֶת מִצְרַיִם בְּיַד אֲדֹנִים קָשֶׁה וּמֶלֶךְ עַז יִמְשָׁל בָּם נְאֻם הָאָדוֹן ידוד צְבָאוֹת"
  • ישעיהו כה3: "עַל כֵּן יְכַבְּדוּךָ עַם עָז, קִרְיַת גּוֹיִם עָרִיצִים יִירָאוּךָ"
  • ירמיהו מח17: "נֻדוּ לוֹ כָּל סְבִיבָיו וְכֹל יֹדְעֵי שְׁמוֹ אִמְרוּ אֵיכָה נִשְׁבַּר מַטֵּה עֹז מַקֵּל תִּפְאָרָה"
  • יחזקאל יט11: "וַיִּהְיוּ לָהּ מַטּוֹת עֹז אֶל שִׁבְטֵי מֹשְׁלִים, וַתִּגְבַּהּ קוֹמָתוֹ עַל בֵּין עֲבֹתִים וַיֵּרָא בְגָבְהוֹ בְּרֹב דָּלִיֹּתָיו... ותצא אש ממטה בדיה פריה אכלָהּ ולא היה בה מטה עֹז שבט למשול קינה היא ותהי לקינה"
  • דברי הימים א טז27: "הוֹד וְהָדָר לְפָנָיו, עֹז וְחֶדְוָה בִּמְקֹמוֹ"
  • דברי הימים ב ל21: "וַיַּעֲשׂוּ בְנֵי יִשְׂרָאֵל הַנִּמְצְאִים בִּירוּשָׁלִַם אֶת חַג הַמַּצּוֹת שִׁבְעַת יָמִים בְּשִׂמְחָה גְדוֹלָה וּמְהַלְלִים לידוד יוֹם בְּיוֹם הַלְוִיִּם וְהַכֹּהֲנִים בִּכְלֵי עֹז לידוד"
  • תהלים ח3: "מִפִּי עוֹלְלִים וְיֹנְקִים יִסַּדְתָּ עֹז לְמַעַן צוֹרְרֶיךָ לְהַשְׁבִּית אוֹיֵב וּמִתְנַקֵּם"
  • תהלים יח18: "יַצִּילֵנִי מֵאֹיְבִי עָז, וּמִשֹּׂנְאַי כִּי אָמְצוּ מִמֶּנִּי"
  • תהלים ל8: "ידוד בִּרְצוֹנְךָ הֶעֱמַדְתָּה לְהַרְרִי עֹז, הִסְתַּרְתָּ פָנֶיךָ הָיִיתִי נִבְהָל"
  • תהלים סא4: "כִּי הָיִיתָ מַחְסֶה לִי מִגְדַּל עֹז מִפְּנֵי אוֹיֵב"
  • תהלים סב12: "אַחַת דִּבֶּר אֱלֹהִים שְׁתַּיִם זוּּ שָׁמָעְתִּי כִּי עֹז לֵאלֹהִים"
  • תהלים סח35: "תְּנוּ עֹז לֵאלֹהִים עַל יִשְׂרָאֵל גַּאֲוָתוֹ וְעֻזּוֹ בַּשְּׁחָקִים. נורא אלהים ממקדשיך אל יִשְׂרָאֵל הוא נתן עֹז ותעצמות לעם ברוך אלהים"
  • תהלים קנ1: "הַלְלוּ יָהּ הַלְלוּ אֵל בְּקָדְשׁוֹ הַלְלוּהוּ בִּרְקִיעַ עֻזּוֹ"
  • תהלים צג1: "ה' מָלָךְ גֵּאוּת לָבֵשׁ, לָבֵשׁ ה' עֹז הִתְאַזָּר, אַף תִּכּוֹן תֵּבֵל בַּל תִּמּוֹט"
  • תהלים צו6: "הוֹד וְהָדָר לְפָנָיו , עֹז וְתִפְאֶרֶת בְּמִקְדָּשׁוֹ"
  • תהלים קלח3: "בְּיוֹם קָרָאתִי וַתַּעֲנֵנִי תַּרְהִבֵנִי בְנַפְשִׁי עֹז"
  • איוב יב16: "עִמּוֹ עֹז וְתוּשִׁיָּה, לוֹ שֹׁגֵג וּמַשְׁגֶּה"
  • איוב כו2: "מֶה עָזַרְתָּ לְלֹא כֹחַ הוֹשַׁעְתָּ זְרוֹעַ לֹא עֹז?"
  • איוב לז6-7: "כִּי לַשֶּׁלַג יֹאמַר הֱוֵא אָרֶץ וְגֶשֶׁם מָטָר וְגֶשֶׁם מִטְרוֹת עֻזּוֹ בְּיַד כָּל אָדָם יַחְתּוֹם לָדַעַת כָּל אַנְשֵׁי מַעֲשֵׂהוּ"
  • איוב מא14: "בְּצַוָּארוֹ יָלִין עֹז, וּלְפָנָיו תָּדוּץ דְּאָבָה"
  • שיר השירים ח6: "שימני כחותם על לבך כחותם על זרועך, כי עזה כמות אהבה קשה כשאול קנאה רשפיה רשפי אש שלהבתיה"
