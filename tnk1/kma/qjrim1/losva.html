ראשי>הגדרות מושגים>

לא שבע = א. לא היה לו מספיק; ב. רצה יותר מדי

קוד: לא שבע= בתנ"ך

סוג: הגדרה

מאת: אראל

אל:

בלשון המקרא, כשאומרים שמישהו לא שבע, יש לכך שתי משמעויות שונות:

א. עונש וקללה: לא היה לו מספיק אוכל כדי לשבוע, או שהאוכל התקלל בתוך מעיו והוא לא הצליח לשבוע ממנו:

  • ויקרא כו26: "בְּשִׁבְרִי לָכֶם מַטֵּה לֶחֶם, וְאָפוּ עֶשֶׂר נָשִׁים לַחְמְכֶם בְּתַנּוּר אֶחָד; וְהֵשִׁיבוּ לַחְמְכֶם בַּמִּשְׁקָל, וַאֲכַלְתֶּם וְלֹא תִשְׂבָּעוּ"
  • יחזקאל ז19: "כַּסְפָּם בַּחוּצוֹת יַשְׁלִיכוּ, וּזְהָבָם לְנִדָּה יִהְיֶה; כַּסְפָּם וּזְהָבָם לֹא יוּכַל לְהַצִּילָם בְּיוֹם עֶבְרַת ה'; נַפְשָׁם לֹא יְשַׂבֵּעוּ וּמֵעֵיהֶם לֹא יְמַלֵּאוּ, כִּי מִכְשׁוֹל עונם הָיָה"
  • עמוס ד8: "וְנָעוּ שְׁתַּיִם שָׁלֹשׁ עָרִים אֶל עִיר אַחַת לִשְׁתּוֹת מַיִם וְלֹא יִשְׂבָּעוּ וְלֹא שַׁבְתֶּם עָדַי נְאֻם ה'"
  • מיכה ו14: "אַתָּה תֹאכַל וְלֹא תִשְׂבָּע וְיֶשְׁחֲךָ בְּקִרְבֶּךָ וְתַסֵּג וְלֹא תַפְלִיט וַאֲשֶׁר תְּפַלֵּט לַחֶרֶב אֶתֵּן"
  • איוב כז14: "אִם יִרְבּוּ בָנָיו לְמוֹ חָרֶב וְצֶאֱצָאָיו לֹא יִשְׂבְּעוּ לָחֶם"
  • בראשית מא31: "וְלֹא יִוָּדַע הַשָּׂבָע בָּאָרֶץ מִפְּנֵי הָרָעָב הַהוּא אַחֲרֵי כֵן כִּי כָבֵד הוּא מְאֹד"
ב. תאווה שאינה נגמרת: למרות שהיה לו יותר מדי אוכל, הוא בכל-זאת לא הסתפק בכך ורצה לאכול עוד:
  • ישעיהו ט19: "וַיִּגְזֹר עַל יָמִין וְרָעֵב וַיֹּאכַל עַל שְׂמֹאול וְלֹא שָׂבֵעוּ אִישׁ בְּשַׂר זְרֹעוֹ יֹאכֵלוּ"
  • ישעיהו נו11: "וְהַכְּלָבִים עַזֵּי נֶפֶשׁ לֹא יָדְעוּ שָׂבְעָה וְהֵמָּה רֹעִים לֹא יָדְעוּ הָבִין כֻּלָּם לְדַרְכָּם פָּנוּ אִישׁ לְבִצְעוֹ מִקָּצֵהוּ"
  • יחזקאל טז28-29: "וַתִּזְנִי אֶל בְּנֵי אַשּׁוּר מִבִּלְתִּי שָׂבְעָתֵךְ וַתִּזְנִים וְגַם לֹא שָׂבָעַתְּ. וַתַּרְבִּי אֶת תַּזְנוּתֵךְ אֶל אֶרֶץ כְּנַעַן כַּשְׂדִּימָה וְגַם בְּזֹאֶת לֹא שָׂבָעַתְּ"
  • חבקוק ב5: "...אֲשֶׁר הִרְחִיב כִּשְׁאוֹל נַפְשׁוֹ, וְהוּא כַמָּוֶת וְלֹא יִשְׂבָּע; וַיֶּאֱסֹף אֵלָיו כָּל הַגּוֹיִם, וַיִּקְבֹּץ אֵלָיו כָּל הָעַמִּים"
  • איוב יט22: "לָמָּה תִּרְדְּפֻנִי כְמוֹ אֵל וּמִבְּשָׂרִי לֹא תִשְׂבָּעוּ"
  • איוב לא31: "אִם לֹא אָמְרוּ מְתֵי אָהֳלִי מִי יִתֵּן מִבְּשָׂרוֹ לֹא נִשְׂבָּע"
  • משלי כז20: "שְׁאוֹל ואבדה[וַאֲבַדּוֹ] לֹא תִשְׂבַּעְנָה, וְעֵינֵי הָאָדָם לֹא תִשְׂבַּעְנָה" (פירוט).
  • משלי ל15-16: "לַעֲלוּקָה שְׁתֵּי בָנוֹת הַב הַב שָׁלוֹשׁ הֵנָּה לֹא תִשְׂבַּעְנָה אַרְבַּע לֹא אָמְרוּ הוֹן.  שְׁאוֹל וְעֹצֶר רָחַם אֶרֶץ לֹא שָׂבְעָה מַּיִם וְאֵשׁ לֹא אָמְרָה הוֹן"
יש פסוקים שבהם שני הפירושים אפשריים:
  • הושע ד10: "וְאָכְלוּ וְלֹא יִשְׂבָּעוּ, הִזְנוּ וְלֹא יִפְרֹצוּ; כִּי אֶת ה' עָזְבוּ לִשְׁמֹר" - מקובל לפרש כפירוש א - עונש וקללה לכהנים; אך אפשר גם לפרש כפירוש ב - תיאור פשעי הכהנים, שאוכלים ואוכלים ועדיין רוצים עוד...
  • חגי א6: "זְרַעְתֶּם הַרְבֵּה וְהָבֵא מְעָט, אָכוֹל וְאֵין לְשָׂבְעָה, שָׁתוֹ וְאֵין לְשָׁכְרָה, לָבוֹשׁ וְאֵין לְחֹם לוֹ, וְהַמִּשְׂתַּכֵּר מִשְׂתַּכֵּר אֶל צְרוֹר נָקוּב" - מקובל לפרש כפירוש א - עונש וקללה לישראל על כך שאינם בונים את המקדש; אך אפשר גם לפרש כפירוש ב - תוכחה לישראל על כך שהם חיים בצורה בזבזנית ואינם משקיעים בבניין המקדש (לפירוט).
פסוקים נוספים:
  • תהלים נט16: "הֵמָּה ינועון[יְנִיעוּן] לֶאֱכֹל אִם לֹא יִשְׂבְּעוּ וַיָּלִינוּ"
  • קהלת א8: "כָּל הַדְּבָרִים יְגֵעִים לֹא יוּכַל אִישׁ לְדַבֵּר לֹא תִשְׂבַּע עַיִן לִרְאוֹת וְלֹא תִמָּלֵא אֹזֶן מִשְּׁמֹעַ"
  • קהלת ד8: "יֵשׁ אֶחָד וְאֵין שֵׁנִי גַּם בֵּן וָאָח אֵין לוֹ וְאֵין קֵץ לְכָל עֲמָלוֹ גַּם עיניו[עֵינוֹ] לֹא תִשְׂבַּע עֹשֶׁר וּלְמִי אֲנִי עָמֵל וּמְחַסֵּר אֶת נַפְשִׁי מִטּוֹבָה גַּם זֶה הֶבֶל וְעִנְיַן רָע הוּא"
  • קהלת ה9: "אֹהֵב כֶּסֶף לֹא יִשְׂבַּע כֶּסֶף; וּמִי אֹהֵב בֶּהָמוֹן לֹא תְבוּאָה גַּם זֶה הָבֶל"
  • קהלת ו3: "אִם יוֹלִיד אִישׁ מֵאָה, וְשָׁנִים רַבּוֹת יִחְיֶה, וְרַב שֶׁיִּהְיוּ יְמֵי שָׁנָיו, וְנַפְשׁוֹ לֹא תִשְׂבַּע מִן הַטּוֹבָה, וְגַם קְבוּרָה לֹא הָיְתָה לּוֹ - אָמַרְתִּי טוֹב מִמֶּנּוּ הַנָּפֶל"
