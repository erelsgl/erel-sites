ראשי>הגדרות מושגים>

חפץ = רצון מתוך מחשבה תועלתית

קוד: חפץ= בתנ"ך

סוג: הגדרה

מאת: אראל

אל:

חפץ הוא רצון הנובע מתוך מחשבה תועלתית, אינטרס בלע"ז:
  • בראשית לד19: "וְלֹא אֵחַר הַנַּעַר לַעֲשׂוֹת הַדָּבָר, כִּי חָפֵץ בְּבַת יַעֲקֹב, וְהוּא נִכְבָּד מִכֹּל בֵּית אָבִיו" = שכם רצה את בת יעקב גם מתוך תשוקה (כמו שנזכר בפסוקים הקודמים), וגם משיקולים של תועלת, מפני שהיא "בת יעקב" העשיר והנכבד: "אחר שראה מעלת יעקב ובניו, התעורר אצלו חפץ מצד התועלת, שהיא בת יעקב שהוא עשיר גדול, חוץ מהחשק והאהבה בהנערה עצמה שכבר היתה ברשותו..." (מלבי"ם).
  • במדבר יד8: "אִם חָפֵץ בָּנוּ ה', וְהֵבִיא אֹתָנוּ אֶל הָאָרֶץ הַזֹּאת וּנְתָנָהּ לָנוּ, אֶרֶץ אֲשֶׁר הִוא זָבַת חָלָב וּדְבָשׁ" = "אם ה' ירצה בנו כי יראה שאנחנו מועילים לו ומקיימים את מצוותיו, הוא יתן לנו את ארץ כנען".
  • שמואל א יח22: "וַיְצַו שָׁאוּל אֶת עֲבָדָו: דַּבְּרוּ אֶל דָּוִד בַּלָּט לֵאמֹר: 'הִנֵּה חָפֵץ בְּךָ הַמֶּלֶךְ וְכָל עֲבָדָיו אֲהֵבוּךָ, וְעַתָּה הִתְחַתֵּן בַּמֶּלֶךְ'" - כולם אוהבים אותך, והמלך גם חפץ בך כי אתה מביא לו תועלת במלחמותיך בפלשתים.
  • מלכים א ט1: "וַיְהִי כְּכַלּוֹת שְׁלֹמֹה לִבְנוֹת אֶת בֵּית ה' וְאֶת בֵּית הַמֶּלֶךְ וְאֵת כָּל חֵשֶׁק שְׁלֹמֹה אֲשֶׁר חָפֵץ לַעֲשׂוֹת" - שלמה בנה בניינים גם מתוך חשק רגשי וגם מתוך חפץ תועלתי; ייתכן שהוא בנה את בית ה' מתוך חשק ואהבה לה', ואת בית המלך - מתוך חפץ וצורך תועלתי.
  • מלכים א יג33: "אַחַר הַדָּבָר הַזֶּה לֹא שָׁב יָרָבְעָם מִדַּרְכּוֹ הָרָעָה, וַיָּשָׁב וַיַּעַשׂ מִקְצוֹת הָעָם כֹּהֲנֵי בָמוֹת; הֶחָפֵץ יְמַלֵּא אֶת יָדוֹ וִיהִי כֹּהֲנֵי בָמוֹת" - ירבעם הפך את עבודת ה' ממטרה לאמצעי - כל מי שחשב שהוא יכול להפיק תועלת מהקרבת קרבנות, היה יכול להתמנות ל"כהן" בבמות.
  • שיר השירים ב7: "הִשְׁבַּעְתִּי אֶתְכֶם, בְּנוֹת יְרוּשָׁלִַם, בִּצְבָאוֹת אוֹ בְּאַיְלוֹת הַשָּׂדֶה, אִם תָּעִירוּ וְאִם תְּעוֹרְרוּ אֶת הָאַהֲבָה עַד שֶׁתֶּחְפָּץ" (וכן שיר השירים ג5,  שיר השירים ח4) - אל תבזבזו את רגשות האהבה שלכן, חכו עד שתמצאו מישהו שירצה אתכן ואתן תרצו אותו, רצון שמבוסס על מחשבה הגיונית (פירוט).
  • קהלת ג1: "לַכֹּל זְמָן, וְעֵת לְכָל חֵפֶץ תַּחַת הַשָּׁמָיִם" - כל דבר שהאדם רוצה לעשות, ויש בו גם תועלת, יוכל לעשותו בסופו של דבר, צריך רק לחכות בסבלנות לזמן המתאים  (ייתכן שהמילה " חפץ " רומזת גם לאהבה שנזכרה בשיר השירים).
  • קהלת יב1: "וּזְכֹר אֶת בּוֹרְאֶיךָ בִּימֵי בְּחוּרֹתֶיךָ, עַד אֲשֶׁר לֹא יָבֹאוּ יְמֵי הָרָעָה וְהִגִּיעוּ שָׁנִים אֲשֶׁר תֹּאמַר 'אֵין לִי בָהֶם חֵפֶץ'" - האדם צריך לזכור את ה' כל עוד הוא צעיר ויש לו כוח לעשות מעשים מועילים, לא לחכות לימי הזקנה שבהם לאדם אין כוח לממש את רצונותיו; מעין האזהרה ב משלי ה11-12: "וְנָהַמְתָּ בְאַחֲרִיתֶךָ, בִּכְלוֹת בְּשָׂרְךָ וּשְׁאֵרֶךָ. וְאָמַרְתָּ 'אֵיךְ שָׂנֵאתִי מוּסָר, וְתוֹכַחַת נָאַץ לִבִּי...'". (פירוט).
  • משלי לא13: "דָּרְשָׁה צֶמֶר וּפִשְׁתִּים, וַתַּעַשׂ בְּחֵפֶץ כַּפֶּיהָ" - אשת חיל מעסיקה את ידיה בכל דבר המביא תועלת, ועושה זאת מתוך רצון (פירוט).

אבני חפץ הן אבנים יקרות, שאנשים רוצים בהן בגלל ערכן הגבוה:
  • ישעיהו נד12: "וְשַׂמְתִּי כַּדְכֹד שִׁמְשֹׁתַיִךְ, וּשְׁעָרַיִךְ לְאַבְנֵי אֶקְדָּח, וְכָל גְּבוּלֵךְ לְאַבְנֵי חֵפֶץ" (פירוט)
ואולי מכאן באה המשמעות החדשה של המילה חפץ - אובייקט, עצם כלשהו שבעליו חפץ ורוצה בו כי הוא מביא לו תועלת (ואפשר לפרש כך גם את  משלי לא13: "דָּרְשָׁה צֶמֶר וּפִשְׁתִּים, וַתַּעַשׂ בְּחֵפֶץ כַּפֶּיהָ": אשת חיל עושה חפצים במלאכת כפיה).

הנביאים הדגישו בכמה מקומות, שה' אינו חפץ בזבחים - הוא אינו מפיק מהם כל תועלת. הוא חפץ רק בכך שאנשים יקיימו את מצוותיו ויעשו צדקה ומשפט (כי הדבר מועיל לקיום הבריאה):

  • שמואל א טו22: "וַיֹּאמֶר שְׁמוּאֵל: 'הַחֵפֶץ לה' בְּעֹלוֹת וּזְבָחִים, כִּשְׁמֹעַ בְּקוֹל ה'?! הִנֵּה שְׁמֹעַ מִזֶּבַח טוֹב, לְהַקְשִׁיב מֵחֵלֶב אֵילִים!'"
  • ישעיהו א11: "לָמָּה לִּי רֹב זִבְחֵיכֶם, יֹאמַר ה'? שָׂבַעְתִּי עֹלוֹת אֵילִים וְחֵלֶב מְרִיאִים, וְדַם פָּרִים וּכְבָשִׂים וְעַתּוּדִים לֹא חָפָצְתִּי"
  • ירמיהו ט23: "כִּי אִם בְּזֹאת יִתְהַלֵּל הַמִּתְהַלֵּל הַשְׂכֵּל וְיָדֹעַ אוֹתִי, כִּי אֲנִי ה' עֹשֶׂה חֶסֶד מִשְׁפָּט וּצְדָקָה בָּאָרֶץ כִּי בְאֵלֶּה חָפַצְתִּי נְאֻם ה'"
  • הושע ו6: "כִּי חֶסֶד חָפַצְתִּי וְלֹא זָבַח, וְדַעַת אֱלֹהִים מֵעֹלוֹת"
  • יחזקאל יח23: "הֶחָפֹץ אֶחְפֹּץ מוֹת רָשָׁע, נְאֻם ד' ה'?! הֲלוֹא בְּשׁוּבוֹ מִדְּרָכָיו וְחָיָה!": ה' אינו חפץ גם ב"קרבנות אדם", אלא בתשובה (וכן יחזקאל יח32,  יחזקאל לג11).
  • מיכה ז18: "מִי אֵל כָּמוֹךָ נֹשֵׂא עון וְעֹבֵר עַל פֶּשַׁע לִשְׁאֵרִית נַחֲלָתוֹ, לֹא הֶחֱזִיק לָעַד אַפּוֹ כִּי חָפֵץ חֶסֶד הוּא"
  • תהלים מ7: "זֶבַח וּמִנְחָה לֹא חָפַצְתָּ,אָזְנַיִם כָּרִיתָ לִּי, עוֹלָה וַחֲטָאָה לֹא שָׁאָלְתָּ"
  • תהלים קמז10: "לֹא בִגְבוּרַת הַסּוּס יֶחְפָּץ, לֹא בְשׁוֹקֵי הָאִישׁ יִרְצֶה. רוצה ה' את יראיו, את המיחלים לחסדו."
ה' מצפה שגם בני ישראל יחפצו באותם דברים, ומוכיח אותם על כך שהם עוסקים בחפצים אחרים, יש להם רצונות שאינם משתלבים עם הרצונות של ה':
  • ישעיהו נו4: "כִּי כֹה אָמַר ה' לַסָּרִיסִים אֲשֶׁר יִשְׁמְרוּ אֶת שַׁבְּתוֹתַי, וּבָחֲרוּ בַּאֲשֶׁר חָפָצְתִּי, וּמַחֲזִיקִים בִּבְרִיתִי" - ה' משבח את הסריסים הבוחרים לרצות את אותם דברים שה' חפץ בהם.
  • ישעיהו נח2: "וְאוֹתִי יוֹם יוֹם יִדְרֹשׁוּן וְדַעַת דְּרָכַי יֶחְפָּצוּן, כְּגוֹי אֲשֶׁר צְדָקָה עָשָׂה וּמִשְׁפַּט אֱלֹהָיו לֹא עָזָב; יִשְׁאָלוּנִי מִשְׁפְּטֵי צֶדֶק, קִרְבַת אֱלֹהִים יֶחְפָּצוּן" - ה' מגנה את בני ישראל, הרוצים רק לימוד וידיעה, התעלות רוחנית וקרבת ה', אך לא מעשים של צדקה וחסד, שהם הדברים שה' חפץ באמת (פירוט).
  • ישעיהו נח3: "'לָמָּה צַּמְנוּ וְלֹא רָאִיתָ, עִנִּינוּ נַפְשֵׁנוּ וְלֹא תֵדָע?' הֵן בְּיוֹם צֹמְכֶם תִּמְצְאוּ חֵפֶץ, וְכָל עַצְּבֵיכֶם תִּנְגֹּשׂוּ!" - ביום צומכם, במקום לצמצם את עצמכם ולהשתדל לעסוק בחפצי ה',  אתם עוסקים כרגיל בחפצים האישיים שלכם  (פירוט). חפצי ה' ביום צום הם לעשות צדקה ומשפט, כפי שהנביא מסביר בהמשך (פסוקים 6-7).
  • ישעיהו נח13: "אִם תָּשִׁיב מִשַּׁבָּת רַגְלֶךָ, עֲשׂוֹת חֲפָצֶיךָ בְּיוֹם קָדְשִׁי; וְקָרָאתָ לַשַּׁבָּת עֹנֶג, לִקְדוֹשׁ ה' מְכֻבָּד; וְכִבַּדְתּוֹ מֵעֲשׂוֹת דְּרָכֶיךָ, מִמְּצוֹא חֶפְצְךָ וְדַבֵּר דָּבָר" - בשבת האדם צריך להתנתק מחפציו ועסקיו, המביאים לו תועלת; וכן להימנע מללכת ברגליו למקום שבו הוא מתכוון לעשות את חפציו לאחר השבת.
  • ישעיהו סו3: "שׁוֹחֵט הַשּׁוֹר מַכֵּה אִישׁ, זוֹבֵחַ הַשֶּׂה עֹרֵף כֶּלֶב, מַעֲלֵה מִנְחָה דַּם חֲזִיר, מַזְכִּיר לְבֹנָה מְבָרֵךְ אָוֶן, גַּם הֵמָּה בָּחֲרוּ בְּדַרְכֵיהֶם וּבְשִׁקּוּצֵיהֶם נַפְשָׁם חָפֵצָה. גַּם אני אבחר בתעלליהם ומגורתם אביא להם, יען קראתי ואין עונה, דברתי ולא שמעו, ויעשו הרע בעיני, ובאשר לא חפצתי בָּחֲרוּ" (פירוט)
  • ירמיהו ו10: "עַל מִי אֲדַבְּרָה וְאָעִידָה וְיִשְׁמָעוּ? הִנֵּה עֲרֵלָה אָזְנָם וְלֹא יוּכְלוּ לְהַקְשִׁיב, הִנֵּה דְבַר ה' הָיָה לָהֶם לְחֶרְפָּה לֹא יַחְפְּצוּ בוֹ"
ועל כך אמרו חז"ל: "עשה רצונך כרצונו" (משנה אבות ב ד).

החכמה מועילה יותר מכל חפץ, כלומר מכל דבר שהאדם רוצה לעשות, מכל עסק שהאדם חושב שמביא לו תועלת:

  • משלי ג15: "יְקָרָה הִיא מפניים[מִפְּנִינִים], וְכָל חֲפָצֶיךָ לֹא יִשְׁווּ בָהּ" - אי אפשר להשוות בין החכמה לבין כל חפץ אחר, כי כל חפץ הוא סופי ומוגבל, והחכמה היא אינסופית ובאמצעותה אפשר לשנות את גבולות המציאות (ע"פ גליה).
  • משלי ח11: "כִּי טוֹבָה חָכְמָה מִפְּנִינִים, וְכָל חֲפָצִים לֹא יִשְׁווּ בָהּ" (פירוט)
אולם הכסיל אינו מודע לתועלת שבתבונה, ולכן:
  • משלי יח2: "לֹא יַחְפֹּץ כְּסִיל בִּתְבוּנָה, כִּי אִם בְּהִתְגַּלּוֹת לִבּוֹ" (פירוט)

פירושים נוספים

בלשון המשנה ובלשון ימינו, המילה חפץ מציינת דבר כלשהו. יש שפירשו שהמשמעות הזו קיימת גם בלשון המקרא: "לכל חפץ - לכל דבר; כל הדברים קרוים חפצים בלשון משנה" (רש"י על קהלת ג1).

פסוקים נוספים

חפצי אדם

  • דברים כא14: "וְהָיָה אִם לֹא חָפַצְתָּ בָּהּ, וְשִׁלַּחְתָּהּ לְנַפְשָׁהּ, וּמָכֹר לֹא תִמְכְּרֶנָּה בַּכָּסֶף; לֹא תִתְעַמֵּר בָּהּ תַּחַת אֲשֶׁר עִנִּיתָהּ"
  • דברים כה7: "וְאִם לֹא יַחְפֹּץ הָאִישׁ לָקַחַת אֶת יְבִמְתּוֹ וְעָלְתָה יְבִמְתּוֹ הַשַּׁעְרָה אֶל הַזְּקֵנִים וְאָמְרָה מאן יַבְּמִי לְהָקִים לְאָחִיו שֵׁם בְּיִשְׂרָאֵל לֹא אָבָה יְבָמִי. וקראו לו זקני עירו ודברו אֶליו ועמד ואמר 'לֹא חפצתי לָקַחַתה'".
  • שמואֶל א יח25: "ויאמר שאול כה תאמרו לדוד 'אין חפץ למלך במהר כי במאה ערלות פלשתים להנקם באיבי המלך' ושאול חשב להפיל אֶת דוד ביד פלשתים" (פירוט)
  • שמואל א יט1: "וַיְדַבֵּר שָׁאוּל אֶל יוֹנָתָן בְּנוֹ וְאֶל כָּל עֲבָדָיו לְהָמִית אֶת דָּוִד, וִיהוֹנָתָן בֶּן שָׁאוּל חָפֵץ בְּדָוִד מְאֹד"
  • שמואל ב כ11: "וְאִישׁ עָמַד עָלָיו מִנַּעֲרֵי יוֹאָב, וַיֹּאמֶר: 'מִי אֲשֶׁר חָפֵץ בְּיוֹאָב וּמִי אֲשֶׁר לְדָוִד - אַחֲרֵי יוֹאָב'"
  • שמואל ב כד3: "וַיֹּאמֶר יוֹאָב אֶל הַמֶּלֶךְ 'וְיוֹסֵף ה' אֱלֹהֶיךָ אֶל הָעָם כָּהֵם וְכָהֵם מֵאָה פְעָמִים וְעֵינֵי אֲדֹנִי הַמֶּלֶךְ רֹאוֹת וַאדֹנִי הַמֶּלֶךְ לָמָּה חָפֵץ בַּדָּבָר הַזֶּה?'"
  • מלכים א ה22: "וַיִּשְׁלַח חִירָם אֶל שְׁלֹמֹה לֵאמֹר 'שָׁמַעְתִּי אֶת אֲשֶׁר שָׁלַחְתָּ אֵלָי, אֲנִי אֶעֱשֶׂה אֵת כָּל חֶפְצְךָ בַּעֲצֵי אֲרָזִים וּבַעֲצֵי בְרוֹשִׁים. עבדי ירדו מן הלבנון ימה ואֲנִי אשימם דברות בים עד המקום אֲשֶׁר תשלח אֵלָי ונפצתים שם ואֶתה תשא ואֵתה תעשה אֶת חפצי לתת לחם ביתי'. ויהי חירום נתן לשְׁלֹמֹה עצי אֲרָזִים ועצי בְרוֹשִׁים כָּל חפצו".
  • מלכים א ט11: "חִירָם מֶלֶךְ צֹר נִשָּׂא אֶת שְׁלֹמֹה בַּעֲצֵי אֲרָזִים וּבַעֲצֵי בְרוֹשִׁים וּבַזָּהָב לְכָל חֶפְצוֹ, אָז יִתֵּן הַמֶּלֶךְ שְׁלֹמֹה לְחִירָם עֶשְׂרִים עִיר בְּאֶרֶץ הַגָּלִיל"
  • מלכים א כא6: "וַיְדַבֵּר אֵלֶיהָ 'כִּי אֲדַבֵּר אֶל נָבוֹת הַיִּזְרְעֵאֶלִי וָאֹמַר לוֹ תְּנָה לִּי אֶת כַּרְמְךָ בְּכֶסֶף אוֹ אִם חָפֵץ אַתָּה אֶתְּנָה לְךָ כֶרֶם תַּחְתָּיו וַיֹּאמֶר לֹא אֶתֵּן לְךָ אֶת כַּרְמִי'"
  • ישעיהו יג17: "הִנְנִי מֵעִיר עֲלֵיהֶם אֶת מָדָי, אֲשֶׁר כֶּסֶף לֹא יַחְשֹׁבוֹּ וְזָהָב לֹא יַחְפְּצוּ בוֹ"
  • ירמיהו מב22: "וְעַתָּה יָדֹעַ תֵּדְעוּ כִּי בַּחֶרֶב בָּרָעָב וּבַדֶּבֶר תָּמוּתוּ בַּמָּקוֹם אֲשֶׁר חֲפַצְתֶּם לָבוֹא לָגוּר שָׁם"
  • מלאכי ג1: "הִנְנִי שֹׁלֵחַ מַלְאָכִי וּפִנָּה דֶרֶךְ לְפָנָי וּפִתְאֹם יָבוֹא אֶל הֵיכָלוֹ הָאָדוֹן אֲשֶׁר אַתֶּם מְבַקְשִׁים וּמַלְאַךְ הַבְּרִית אֲשֶׁר אַתֶּם חֲפֵצִים הִנֵּה בָא אָמַר ה' צְבָאוֹת"
  • דברי הימים א כח9: "וְאַתָּה שְׁלֹמֹה בְנִי דַּע אֶת אֱלֹהֵי אָבִיךָ וְעָבְדֵהוּ בְּלֵב שָׁלֵם וּבְנֶפֶשׁ חֲפֵצָה כִּי כָל לְבָבוֹת דּוֹרֵשׁ ידוד וְכָל יֵצֶר מַחֲשָׁבוֹת מֵבִין אִם תִּדְרְשֶׁנּוּ יִמָּצֵא לָךְ וְאִם תַּעַזְבֶנּוּ יַזְנִיחֲךָ לָעַד"
  • דברי הימים ב ט12: "וְהַמֶּלֶךְ שְׁלֹמֹה נָתַן לְמַלְכַּת שְׁבָא אֶת כָּל חֶפְצָהּ אֲשֶׁר שָׁאָלָה מִלְּבַד אֲשֶׁר הֵבִיאָה אֶל הַמֶּלֶךְ וַתַּהֲפֹךְ וַתֵּלֶךְ לְאַרְצָהּ הִיא וַעֲבָדֶיהָ"
  • תהלים א2: "כִּי אִם בְּתוֹרַת ידוד חֶפְצוֹ וּבְתוֹרָתוֹ יֶהְגֶּה יוֹמָם וָלָיְלָה"
  • תהלים לד13: "מִי הָאִישׁ הֶחָפֵץ חַיִּים, אֹהֵב יָמִים לִרְאוֹת טוֹב" (פירוט)
  • קהלת ה7: "אִם עֹשֶׁק רָשׁ וְגֵזֶל מִשְׁפָּט וָצֶדֶק תִּרְאֶה בַמְּדִינָה אַל תִּתְמַהּ עַל הַחֵפֶץ כִּי גָּבֹהַּ מֵעַל גָבֹהַּ שֹׁמֵר וּגְבֹהִים עֲלֵיהֶם" (פירוט)
  • קהלת ח3: "אַל תִּבָּהֵל מִפָּנָיו תֵּלֵךְ, אַל תַּעֲמֹד בְּדָבָר רָע, כִּי כָּל אֲשֶׁר יַחְפֹּץ יַעֲשֶׂה"
  • קהלת ח6: "כִּי לְכָל חֵפֶץ יֵשׁ עֵת וּמִשְׁפָּט כִּי רָעַת הָאָדָם רַבָּה עָלָיו"
  • קהלת יב10: "בִּקֵּשׁ קֹהֶלֶת לִמְצֹא דִּבְרֵי חֵפֶץ וְכָתוּב יֹשֶׁר דִּבְרֵי אֱמֶת"
  • אסתר ב14: "בָּעֶרֶב הִיא בָאָה וּבַבֹּקֶר הִיא שָׁבָה אֶל בֵּית הַנָּשִׁים שֵׁנִי אֶל יַד שַׁעֲשְׁגַז סְרִיס הַמֶּלֶךְ שֹׁמֵר הַפִּילַגְשִׁים לֹא תָבוֹא עוֹד אֶל הַמֶּלֶךְ כִּי אִם חָפֵץ בָּהּ הַמֶּלֶךְ וְנִקְרְאָה בְשֵׁם"
  • אסתר ו6: "וַיָּבוֹא הָמָן וַיֹּאמֶר לוֹ הַמֶּלֶךְ מַה לַעֲשׂוֹת בָּאִישׁ אֲשֶׁר הַמֶּלֶךְ חָפֵץ בִּיקָרוֹ? וַיֹּאמֶר הָמָן בְּלִבּוֹ לְמִי יַחְפֹּץ הַמֶּלֶךְ לַעֲשׂוֹת יְקָר יוֹתֵר מִמֶּנִּי. וַיֹּאמֶר הָמָן אל הַמֶּלֶךְ איש אֲשֶׁר הַמֶּלֶךְ חָפֵץ בִּיקָרוֹ... ונתון הלבוש והסוס על יד איש משרי הַמֶּלֶךְ הפרתמים והלבשו את האיש אֲשֶׁר הַמֶּלֶךְ חָפֵץ ביְקָרו והרכיבהו על הסוס ברחוב העיר וקראו לפניו ככה יעשה לאיש אשר הַמֶּלֶךְ חפץ בִּיקָרוֹ... ויקח הָמָן את הלבוש ואת הסוס וילבש את מרדכי וירכיבהו ברחוב העיר ויקרא לפניו ככה יעשה לאיש אשר הַמֶּלֶךְ חפץ בִּיקָרוֹ"
  • נחמיה א11: "אָנָּא אֲדֹנָי תְּהִי נָּא אָזְנְךָ קַשֶּׁבֶת אֶל תְּפִלַּת עַבְדְּךָ וְאֶל תְּפִלַּת עֲבָדֶיךָ הַחֲפֵצִים לְיִרְאָה אֶת שְׁמֶךָ וְהַצְלִיחָה נָא לְעַבְדְּךָ הַיּוֹם וּתְנֵהוּ לְרַחֲמִים לִפְנֵי הָאִישׁ הַזֶּה וַאֲנִי הָיִיתִי מַשְׁקֶה לַמֶּלֶךְ"

חפצי ה'

  • שופטים יג23: "וַתֹּאמֶר לוֹֹּּ אִשְׁתּוֹ 'לו חָפֵץ ה' לַהֲמִיתֵנוּ לֹא לָקַח מִיָּדֵנוּ עֹלָה וּמִנְחָה וְלֹא הֶרְאָנוּ אֶת כָּל אֵלֶּה וְכָעֵת לֹא הִשְׁמִיעָנוּ כָּזֹאֶת'"
  • שמואל א ב25: "'אִם יֶחֱטָא אִישׁ לְאִישׁ וּפִלְלוֹ אֱלֹהִים, וְאִם לה' יֶחֱטָא אִישׁ מִי יִתְפַּלֶּל לו?' וְלֹא יִשְׁמְעוּ לְקוֹל אֲבִיהֶם כִּי חָפֵץ ה' לַהֲמִיתָם" (פירוט)
  • שמואל ב טו26: "וְאִם כֹּה יֹאמַר 'לֹא חָפַצְתִּי בָּךְ' - הִנְנִי, יַעֲשֶׂה לִּי כַּאֲשֶׁר טוֹב בְּעֵינָיו"
  • שמואל ב כב20: "וַיֹּצֵא לַמֶּרְחָב אֹתִי, יְחַלְּצֵנִי כִּי חָפֵץ בִּי"
  • שמואל ב כג5: "כִּי לֹא כֵן בֵּיתִי עִם אֵל, כִּי בְרִית עוֹלָם שָׂם לִי עֲרוּכָה בַכֹּל וּשְׁמֻרָה; כִּי כָל יִשְׁעִי וְכָל חֵפֶץ כִּי לֹא יַצְמִיחַ"
  • מלכים א י9: "'יְהִי ה' אֱלֹהֶיךָ בָּרוּךְ אֲשֶׁר חָפֵץ בְּךָ לְתִתְּךָ עַל כִּסֵּא יִשְׂרָאֵל, בְּאַהֲבַת ה' אֶת יִשְׂרָאֵל לְעֹלָם וַיְשִׂימְךָ לְמֶלֶךְ לַעֲשׂוֹת מִשְׁפָּט וּצְדָקָה'... והמלך שלמה נתן למלכת שבא אֶת כל חפצה אֲשֶׁר שאלה, מלבד אשר נתן לה כיד המלך שלמה ותפן ותלך לארצה היא ועבדיה"
  • ישעיהו מב21: "ה' חָפֵץ לְמַעַן צִדְקוֹ, יַגְדִּיל תּוֹרָה וְיַאְדִּיר": לה' יש רצון, וכדי שהאדם יצדיק את קיומו ויקיים את רצון ה' בצדק, הוא צריך להגדיל תורה ולהאדיר, להגדיל את ידיעותיו בתורת ה', שהיא המגדירה במדוייק במה ה' חפץ.
  • ישעיהו מד28: "הָאֹמֵר לְכוֹרֶשׁ רֹעִי וְכָל חֶפְצִי יַשְׁלִם, וְלֵאמֹר לִירוּשָׁלִַם תִּבָּנֶה וְהֵיכָל תִּוָּסֵד"
  • ישעיהו מו10: "מַגִּיד מֵרֵאשִׁית אַחֲרִית וּמִקֶּדֶם אֲשֶׁר לֹא נַעֲשׂוּ, אֹמֵר עֲצָתִי תָקוּם וְכָל חֶפְצִי אֶעֱשֶׂה"
  • ישעיהו מח14: "הִקָּבְצוּ כֻלְּכֶם וּשֲׁמָעוּ, מִי בָהֶם הִגִּיד אֶת אֵלֶּה? ה' אֲהֵבוֹ, יַעֲשֶׂה חֶפְצוֹ בְּבָבֶל וּזְרֹעוֹ כַּשְׂדִּים"
  • ישעיהו נג10: "וה' חפץ דכאו החלי, אם תשים אשם נפשו יראה זרע יאריך ימים וחפץ ה' בידו יצלח"
  • ישעיהו נה11: "כֵּן יִהְיֶה דְבָרִי אֲשֶׁר יֵצֵא מִפִּי, לֹא יָשׁוּב אֵלַי רֵיקָם כִּי אִם עָשָׂה אֶת אֲשֶׁר חָפַצְתִּי וְהִצְלִיחַ אֲשֶׁר שְׁלַחְתִּיו"
