חיל = צבא של אנשים, בהמות או רכוש

קוד: חיל= בתנ"ך

סוג: הגדרה

מאת: אראל

אל:

חַיִל הוא, כמו בלשון ימינו, צבא של לוחמים:
  • שמות יד4: "וְחִזַּקְתִּי אֶת לֵב פַּרְעֹה, וְרָדַף אַחֲרֵיהֶם, וְאִכָּבְדָה בְּפַרְעֹה וּבְכָל חֵילוֹ..."
  • שמואל א יד48: "וַיַּעַשׂ חַיִל וַיַּךְ אֶת עֲמָלֵק וַיַּצֵּל אֶת יִשְׂרָאֵל מִיַּד שֹׁסֵהוּ" - שאול אסף צבא והיכה את עמלק.
  • שמואל ב ח9: "וַיִּשְׁמַע תֹּעִי מֶלֶךְ חֲמָת כִּי הִכָּה דָוִד אֵת כָּל חֵיל הֲדַדְעָזֶר"
  • מלכים ב כה5: "וַיִּרְדְּפוּ חֵיל כַּשְׂדִּים אַחַר הַמֶּלֶךְ וַיַּשִּׂגוּ אֹתוֹ בְּעַרְבוֹת יְרֵחוֹ וְכָל חֵילוֹ נָפֹצוּ מֵעָלָיו"
  • תהלים לג16: "אֵין הַמֶּלֶךְ נוֹשָׁע בְּרָב חָיִל, גִּבּוֹר לֹא יִנָּצֵל בְּרָב כֹּחַ"
גם "צבא" של בני אדם:
  • יחזקאל לז10: "וְהִנַּבֵּאתִי כַּאֲשֶׁר צִוָּנִי וַתָּבוֹא בָהֶם הָרוּחַ וַיִּחְיוּ וַיַּעַמְדוּ עַל רַגְלֵיהֶם חַיִל גָּדוֹל מְאֹד מְאֹד"
גם "צבא" של חיות:
  • יואל ב25: "וְשִׁלַּמְתִּי לָכֶם אֶת הַשָּׁנִים אֲשֶׁר אָכַל הָאַרְבֶּה הַיֶּלֶק וְהֶחָסִיל וְהַגָּזָם, חֵילִי הַגָּדוֹל אֲשֶׁר שִׁלַּחְתִּי בָּכֶם"
וגם של צמחים:
  • יואל ב22: "אַל תִּירְאוּ בַּהֲמוֹת שָׂדַי כִּי דָשְׁאוּ נְאוֹת מִדְבָּר, כִּי עֵץ נָשָׂא פִרְיוֹ תְּאֵנָה וָגֶפֶן נָתְנוּ חֵילָם"

  • דברים ח17: "וְאָמַרְתָּ בִּלְבָבֶךָ 'כֹּחִי וְעֹצֶם יָדִי עָשָׂה לִי אֶת הַחַיִל הַזֶּה'"
  • מלכים א י2: "וַתָּבֹא יְרוּשָׁלְַמָה בְּחַיִל כָּבֵד מְאֹד גְּמַלִּים נֹשְׂאִים בְּשָׂמִים וְזָהָב רַב מְאֹד וְאֶבֶן יְקָרָה וַתָּבֹא אֶל שְׁלֹמֹה וַתְּדַבֵּר אֵלָיו אֵת כָּל אֲשֶׁר הָיָה עִם לְבָבָהּ"
  • ישעיהו י14: "וַתִּמְצָא כַקֵּן יָדִי לְחֵיל הָעַמִּים וְכֶאֱסֹף בֵּיצִים עֲזֻבוֹת כָּל הָאָרֶץ אֲנִי אָסָפְתִּי וְלֹא הָיָה נֹדֵד כָּנָף וּפֹצֶה פֶה וּמְצַפְצֵף"
  • ישעיהו ח4: "כִּי בְּטֶרֶם יֵדַע הַנַּעַר קְרֹא אָבִי וְאִמִּי יִשָּׂא אֶת חֵיל דַּמֶּשֶׂק וְאֵת שְׁלַל שֹׁמְרוֹן לִפְנֵי מֶלֶךְ אַשּׁוּר"
  • ישעיהו ל6: "מַשָּׂא בַּהֲמוֹת נֶגֶב בְּאֶרֶץ צָרָה וְצוּקָה לָבִיא וָלַיִשׁ מֵהֶם אֶפְעֶה וְשָׂרָף מְעוֹפֵף יִשְׂאוּ עַל כֶּתֶף עֲיָרִים חֵילֵהֶם וְעַל דַּבֶּשֶׁת גְּמַלִּים אוֹצְרֹתָם עַל עַם לֹא יוֹעִילוּ"
  • ישעיהו ס5: "אָז תִּרְאִי וְנָהַרְתְּ וּפָחַד וְרָחַב לְבָבֵךְ, כִּי יֵהָפֵךְ עָלַיִךְ הֲמוֹן יָם חֵיל גּוֹיִם יָבֹאוּ לָךְ"
  • ירמיהו טו13: "חֵילְךָ וְאוֹצְרוֹתֶיךָ לָבַז אֶתֵּן לֹא בִמְחִיר וּבְכָל חַטֹּאותֶיךָ וּבְכָל גְּבוּלֶיךָ"
  • יחזקאל כח5: "בְּרֹב חָכְמָתְךָ בִּרְכֻלָּתְךָ הִרְבִּיתָ חֵילֶךָ וַיִּגְבַּהּ לְבָבְךָ בְּחֵילֶךָ"
  • עובדיה א13: "אַל תָּבוֹא בְשַׁעַר עַמִּי בְּיוֹם אֵידָם אַל תֵּרֶא גַם אַתָּה בְּרָעָתוֹ בְּיוֹם אֵידוֹ וְאַל תִּשְׁלַחְנָה בְחֵילוֹ בְּיוֹם אֵידוֹ"
  • מיכה ד13: "קוּמִי וָדוֹשִׁי בַת צִיּוֹן כִּי קַרְנֵךְ אָשִׂים בַּרְזֶל וּפַרְסֹתַיִךְ אָשִׂים נְחוּשָׁה וַהֲדִקּוֹת עַמִּים רַבִּים וְהַחֲרַמְתִּי לידוד בִּצְעָם וְחֵילָם לַאֲדוֹן כָּל הָאָרֶץ"
  • צפניה א13: "וְהָיָה חֵילָם לִמְשִׁסָּה וּבָתֵּיהֶם לִשְׁמָמָה וּבָנוּ בָתִּים וְלֹא יֵשֵׁבוּ וְנָטְעוּ כְרָמִים וְלֹא יִשְׁתּוּ אֶת יֵינָם"
  • זכריה ט4: "הִנֵּה אֲדֹנָי יוֹרִשֶׁנָּה וְהִכָּה בַיָּם חֵילָהּ וְהִיא בָּאֵשׁ תֵּאָכֵל"
  • רות ד11: "... ועשה חיל באפרתה וקרא שם בבית לחם"
  • איוב טו29: "לֹא יֶעְשַׁר וְלֹא יָקוּם חֵילוֹ וְלֹא יִטֶּה לָאָרֶץ מִנְלָם"
  • בראשית לד29: "וְאֵת כָּל חֵילָם וְאֶת כָּל טַפָּם וְאֶת נְשֵׁיהֶם שָׁבוּ וַיָּבֹזּוּ וְאֶת כָּל אֲשֶׁר בַּבָּיִת", במדבר לא9: "וַיִּשְׁבּוּ בְנֵי יִשְׂרָאֵל אֶת נְשֵׁי מִדְיָן וְאֶת טַפָּם וְאֶת כָּל בְּהֶמְתָּם וְאֵת כָּל מִקְנֵהֶם וְאֶת כָּל חֵילָם בָּזָזוּ" - הבהמות או הרכוש (כי את הגברים, אנשי הצבא, הרגו)
  • תהלים מט7: "הַבֹּטְחִים עַל חֵילָם וּבְרֹב עָשְׁרָם יִתְהַלָּלוּ"

אנשי חיל ונשות חיל

איש חיל היא איש המשרת בצבא (חַיָּל), או איש הראוי לשרת בצבא, כלומר לוחם טוב, חזק ואמיץ:
  • שופטים ג29: "וַיַּכּוּ אֶת מוֹאָב בָּעֵת הַהִיא כַּעֲשֶׂרֶת אֲלָפִים אִישׁ כָּל שָׁמֵן וְכָל אִישׁ חָיִל וְלֹא נִמְלַט אִישׁ"
  • שמואל א לא12: "וַיָּקוּמוּ כָּל אִישׁ חַיִל וַיֵּלְכוּ כָל הַלַּיְלָה וַיִּקְחוּ אֶת גְּוִיַּת שָׁאוּל וְאֵת גְּוִיֹּת בָּנָיו מֵחוֹמַת בֵּית שָׁן וַיָּבֹאוּ יָבֵשָׁה וַיִּשְׂרְפוּ אֹתָם שָׁם"
  • ירמיהו מח14: "אֵיךְ תֹּאמְרוּ גִּבּוֹרִים אֲנָחְנוּ וְאַנְשֵׁי חַיִל לַמִּלְחָמָה"
  • שופטים כ44: "וַיִּפְּלוּ מִבִּנְיָמִן שְׁמֹנָה עָשָׂר אֶלֶף אִישׁ אֶת כָּל אֵלֶּה אַנְשֵׁי חָיִל"
  • שמות יח21: "וְאַתָּה תֶחֱזֶה מִכָּל הָעָם אַנְשֵׁי חַיִל יִרְאֵי אֱלֹהִים אַנְשֵׁי אֱמֶת שֹׂנְאֵי בָצַע וְשַׂמְתָּ עֲלֵהֶם שָׂרֵי אֲלָפִים שָׂרֵי מֵאוֹת שָׂרֵי חֲמִשִּׁים וְשָׂרֵי עֲשָׂרֹת" (פירוט)
  • ישעיהו ה22: "הוֹי גִּבּוֹרִים לִשְׁתּוֹת יָיִן וְאַנְשֵׁי חַיִל לִמְסֹךְ שֵׁכָר" (פירוט)
  • שמואל ב יא16: "וַיְהִי בִּשְׁמוֹר יוֹאָב אֶל הָעִיר וַיִּתֵּן אֶת אוּרִיָּה אֶל הַמָּקוֹם אֲשֶׁר יָדַע כִּי אַנְשֵׁי חַיִל שָׁם"
  • שמואל ב כד9: "וַיִּתֵּן יוֹאָב אֶת מִסְפַּר מִפְקַד הָעָם אֶל הַמֶּלֶךְ וַתְּהִי יִשְׂרָאֵל שְׁמֹנֶה מֵאוֹת אֶלֶף אִישׁ חַיִל שֹׁלֵף חֶרֶב וְאִישׁ יְהוּדָה חֲמֵשׁ מֵאוֹת אֶלֶף אִישׁ"
  • מלכים א א42: "עוֹדֶנּוּ מְדַבֵּר וְהִנֵּה יוֹנָתָן בֶּן אֶבְיָתָר הַכֹּהֵן בֹּא וַיֹּאמֶר אֲדֹנִיָּהוּ בָּא כִּי אִישׁ חַיִל אַתָּה וְטוֹב תְּבַשֵּׂר"
  • מלכים ב כד16: "וְאֵת כָּל אַנְשֵׁי הַחַיִל שִׁבְעַת אֲלָפִים וְהֶחָרָשׁ וְהַמַּסְגֵּר אֶלֶף הַכֹּל גִּבּוֹרִים עֹשֵׂי מִלְחָמָה וַיְבִיאֵם מֶלֶךְ בָּבֶל גּוֹלָה בָּבֶלָה"
  • נחום ב4: "מָגֵן גִּבֹּרֵיהוּ מְאָדָּם אַנְשֵׁי חַיִל מְתֻלָּעִים בְּאֵשׁ פְּלָדוֹת הָרֶכֶב בְּיוֹם הֲכִינוֹ וְהַבְּרֹשִׁים הָרְעָלוּ"
  • גם אנשים היודעים לנהל חיל של בהמות נקראים כך: בראשית מז6: "...וְאִם יָדַעְתָּ וְיֶשׁ בָּם אַנְשֵׁי חַיִל, וְשַׂמְתָּם שָׂרֵי מִקְנֶה עַל אֲשֶׁר לִי"
  • וגם אנשים היודעים לעבוד עם חיל של רכוש: דברי הימים א כו8: "כָּל אֵלֶּה מִבְּנֵי עֹבֵד אֱדֹם הֵמָּה וּבְנֵיהֶם וַאֲחֵיהֶם אִישׁ חַיִל בַּכֹּחַ לַעֲבֹדָה שִׁשִּׁים וּשְׁנַיִם לְעֹבֵד אֱדֹם"
נשים בזמן התנ"ך לא נהגו לשרת בצבא, ולכן אשת חיל היא, כנראה, אישה אמיצה ומוכשרת, בעלת כישרון לניהול בית ורכוש:
  • משלי יב4: "אֵשֶׁת חַיִל עֲטֶרֶת בַּעְלָהּ, וּכְרָקָב בְּעַצְמוֹתָיו מְבִישָׁה" (פירוט)
  • משלי לא10: "אֵשֶׁת חַיִל מִי יִמְצָא וְרָחֹק מִפְּנִינִים מִכְרָהּ... רבות בנות עשו חַיִל ואת עלית על כלנה" - יודעת לנהל את ביתה ולצבור רכוש רב.
  • רות ג11: "וְעַתָּה בִּתִּי אַל תִּירְאִי כָּל אֲשֶׁר תֹּאמְרִי אֶעֱשֶׂה לָּךְ כִּי יוֹדֵעַ כֹּל שַׁעַר עַמִּי כִּי אֵשֶׁת חַיִל אָתְּ" - אמיצה ובעלת כישרון לפרנס את עצמה ואת חמותה.

גם בן חיל:
  • דברים ג18: "וָאֲצַו אֶתְכֶם בָּעֵת הַהִוא לֵאמֹר ידוד אֱלֹהֵיכֶם נָתַן לָכֶם אֶת הָאָרֶץ הַזֹּאֶת לְרִשְׁתָּהּ חֲלוּצִים תַּעַבְרוּ לִפְנֵי אֲחֵיכֶם בְּנֵי יִשְׂרָאֵל כָּל בְּנֵי חָיִל"

  • דניאל ה7: "קָרֵא מַלְכָּא בְּחַיִל לְהֶעָלָה לְאָשְׁפַיָּא כשדִּיאוְגָזְרַיָּא עָנֵה מַלְכָּא וְאָמַר לְחַכִּימֵי בָבֶל דִּי כָל אֱנָשׁ דִי יִקְרֵה כְּתָבָה דְנָה וּפִשְׁרֵהּ יְחַוִּנַּנִי אַרְגְּוָנָא יִלְבַּשׁ והמונכאדִּי דַהֲבָא עַל צַוְּארֵהּ וְתַלְתִּי בְמַלְכוּתָא יִשְׁלַט"

פירושים נוספים

יש מפרשים, שלמילה חיל יש גם משמעות מופשטת - כוח: "ויש מפרשים אנשי חיל - אנשי כח וזריזות" (רמב"ן על שמות כא18); אבל פירוש זה אינו הכרחי:
  • שמואל א ב4: "קֶשֶׁת גִּבֹּרִים חַתִּים, וְנִכְשָׁלִים אָזְרוּ חָיִל" - אפשר לפרש "נכשלים קיבלו כוח", אבל אפשר גם לפרש "נכשלים הצליחו לפקד על צבא" או "נכשלים הצליחו להשיג רכוש".
  • שמואל ב כב40: "וַתַּזְרֵנִי חַיִל לַמִּלְחָמָה, תַּכְרִיעַ קָמַי תַּחְתֵּנִי"
  • שמואל ב כב33: "הָאֵל מָעוּזִּי חָיִל, וַיַּתֵּר תָּמִים דרכו"  - מעוז = מבצר ומגן; אפשר לפרש "ה' היה מבצר לכוח שלי", ואפשר לפרש "ה' היה מבצר לצבא שלי".
  • ירמיהו מו22: "קוֹלָהּ כַּנָּחָשׁ יֵלֵךְ, כִּי בְחַיִל יֵלֵכוּ וּבְקַרְדֻּמּוֹת בָּאוּ לָהּ כְּחֹטְבֵי עֵצִים" - אפשר לפרש "בכוח" ואפשר לפרש "עם צבא".
  • חבקוק ג19: "ה' ד' חֵילִי, וַיָּשֶׂם רַגְלַי כָּאַיָּלוֹת, וְעַל בָּמוֹתַי יַדְרִכֵנִי, לַמְנַצֵּחַ בִּנְגִינוֹתָי" - ה' הוא הכוח שלי, או: ה' מגן עלי כמו צבא.
  • זכריה ד6: "וַיַּעַן וַיֹּאמֶר אֵלַי לֵאמֹר זֶה דְּבַר ה' אֶל זְרֻבָּבֶל לֵאמֹר לֹא בְחַיִל וְלֹא בְכֹחַ כִּי אִם בְּרוּחִי אָמַר ידוד צְבָאוֹת" - אפשר לפרש ש"חיל" נרדף ל"כוח", ואפשר לפרש שאלו שני דברים שונים.
  • תהלים לג17: "שֶׁקֶר הַסּוּס לִתְשׁוּעָה וּבְרֹב חֵילוֹ לֹא יְמַלֵּט" - ברוב כוחו, או ברוב צבאו.
  • תהלים מח14: "שִׁיתוּ לִבְּכֶם לְחֵילָה פַּסְּגוּ אַרְמְנוֹתֶיהָ לְמַעַן תְּסַפְּרוּ לְדוֹר אַחֲרוֹן" - לכוחה (של ירושלים), או לצבא השומר עליה.
  • משלי לא3: "אַל תִּתֵּן לַנָּשִׁים חֵילֶךָ, וּדְרָכֶיךָ לַמְחוֹת מְלָכִין" - כוחך או רכושך (פירוט).
  • עזרא ד23: "...אֲזַלוּ בִבְהִילוּ לִירוּשְׁלֶם עַל יְהוּדָיֵא, וּבַטִּלוּ הִמּוֹ בְּאֶדְרָע וְחָיִל" - אפשר לפרש "ביטלו אותם מבניית המקדש בזרוע ובכוח", אבל אפשר גם לפרש "בזרוע ובצבא".

פסוקים שבהם כמה משמעויות אפשריות

  • במדבר כד18: "וְהָיָה אֱדוֹם יְרֵשָׁה וְהָיָה יְרֵשָׁה שֵׂעִיר אֹיְבָיו וְיִשְׂרָאֵל עֹשֶׂה חָיִל"
  • דברים לג11: "בָּרֵךְ ה' חֵילוֹ וּפֹעַל יָדָיו תִּרְצֶה, מְחַץ מָתְנַיִם קָמָיו וּמְשַׂנְאָיו מִן יְקוּמוּן" - שבט לוי לא יצא למלחמה, אבל בני לוי עבדו כשופטים ושוטרים והיו צריכים להילחם בפושעים; ואפשר גם לפרש: רכושו.
  • עובדיה א11: "בְּיוֹם עֲמָדְךָ מִנֶּגֶד בְּיוֹם שְׁבוֹת זָרִים חֵילוֹ וְנָכְרִים בָּאוּ שערווְעַל יְרוּשָׁלִַם יַדּוּ גוֹרָל גַּם אַתָּה כְּאַחַד מֵהֶם"
  • ישעיהו מג17: "הַמּוֹצִיא רֶכֶב וָסוּס חַיִל וְעִזּוּז יַחְדָּו יִשְׁכְּבוּ בַּל יָקוּמוּ דָּעֲכוּ כַּפִּשְׁתָּה כָבוּ"
  • נחום ג8: "הֲתֵיטְבִי מִנֹּא אָמוֹן הַיֹּשְׁבָה בַּיְאֹרִיָם, מִיָּם סָבִיב לָהּ, אֲשֶׁר חֵיל ים מַיִם חוֹמָתָהּ"
  • דניאל יא13: "וְשָׁב מֶלֶךְ הַצָּפוֹן וְהֶעֱמִיד הָמוֹן רַב מִן הָרִאשׁוֹן וּלְקֵץ הָעִתִּים שָׁנִים יָבוֹא בוֹא בְּחַיִל גָּדוֹל וּבִרְכוּשׁ רָב"
