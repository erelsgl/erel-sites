עצב = א. צער מהול בכעס, ב. מתן צורה לחומר, ג. מאמץ ויגיעה

קוד: עצב= בתנ"ך

סוג: הגדרה

מאת: אראל

אל: סגלות משלי

לשורש עצב יש בלשון המקרא כמה משמעויות שונות.

1. צער, כעס, כאב

עֶצֶב בלשון ימינו הוא צער, וגם בלשון המקרא השורש עצב מציין צער - היפך שמחה:

  • נחמיה ח10-11: "וַיֹּאמֶר לָהֶם 'לְכוּ אִכְלוֹּ מַשְׁמַנִּים וּשְׁתוּ מַמְתַקִּים וְשִׁלְחוּ מָנוֹת לְאֵין נָכוֹן לוֹ כִּי קָדוֹשׁ הַיּוֹם לַאֲדֹנֵינוּ וְאַל תֵּעָצֵבוּ כִּי חֶדְוַת ה' הִיא מָעֻזְּכֶם'. וְהַלְוִיִּם מַחְשִׁים לְכָל הָעָם לֵאמֹר 'הַסּוּ כִּי הַיּוֹם קָדֹשׁ וְאַל תֵּעָצֵבוּ'"
  • משלי טו13: "לֵב שָׂמֵחַ יֵיטִב פָּנִים, וּבְעַצְּבַת לֵב רוּחַ נְכֵאָה" (פירוט)
  • תהלים קמז3: "הָרֹפֵא לִשְׁבוּרֵי לֵב, וּמְחַבֵּשׁ לְעַצְּבוֹתָם"
  • ישעיהו נד6: "כִּי כְאִשָּׁה עֲזוּבָה וַעֲצוּבַת רוּחַ קְרָאָךְ ה', וְאֵשֶׁת נְעוּרִים כִּי תִמָּאֵס אָמַר אֱלֹהָיִךְ"

ברוב הפסוקים, השורש עצב מציין צער שנובע ממעשה רע של אדם אחר, צער מהול בכעס, כמו בלשון ימינו במילה עַצְבָּנוּת:

  • בראשית ו6: "וַיִּנָּחֶם ה' כִּי עָשָׂה אֶת הָאָדָם בָּאָרֶץ, וַיִּתְעַצֵּב אֶל לִבּוֹ" - הצטער וגם כעס.
  • בראשית לד7: "וּבְנֵי יַעֲקֹב בָּאוּ מִן הַשָּׂדֶה כְּשָׁמְעָם, וַיִּתְעַצְּבוּ הָאֲנָשִׁים, וַיִּחַר לָהֶם מְאֹד כִּי נְבָלָה עָשָׂה בְיִשְׂרָאֵל לִשְׁכַּב אֶת בַּת יַעֲקֹב וְכֵן לֹא יֵעָשֶׂה" - הצטערו וגם כעסו.
  • בראשית מה5: "וְעַתָּה אַל תֵּעָצְבוּ וְאַל יִחַר בְּעֵינֵיכֶם כִּי מְכַרְתֶּם אֹתִי הֵנָּה כִּי לְמִחְיָה שְׁלָחַנִי אֱלֹהִים לִפְנֵיכֶם" - אל תצטערו ואל תכעסו זה על זה.
  • שמואל א כ3: "יָדֹעַ יֵדַע אָבִיךָ כִּי מָצָאתִי חֵן בְּעֵינֶיךָ וַיֹּאמֶר 'אַל יָדַע זֹאת יְהוֹנָתָן פֶּן יֵעָצֵב...'" - יצטער ויכעס.
  • משלי י10: "קֹרֵץ עַיִן יִתֵּן עַצָּבֶת, וֶאֱוִיל שְׂפָתַיִם יִלָּבֵט" (פירוט).

פסוקים נוספים

  • שמואל א כ34: "... ולא אכל ביום החדש השני לחם, כי נעצב אל דוד כי הכלמו אביו" - הצטער וכעס;
  • שמואל ב יט3: "וַתְּהִי הַתְּשֻׁעָה בַּיּוֹם הַהוּא לְאֵבֶל לְכָל הָעָם כִּי שָׁמַע הָעָם בַּיּוֹם הַהוּא לֵאמֹר נֶעֱצַב הַמֶּלֶךְ עַל בְּנוֹ" - הצטער וכעס;
  • ישעיהו יד3: "וְהָיָה בְּיוֹם הָנִיחַ ה' לְךָ מֵעָצְבְּךָ וּמֵרָגְזֶךָ וּמִן הָעֲבֹדָה הַקָּשָׁה אֲשֶׁר עֻבַּד בָּךְ..."
  • ישעיהו נ11: "הֵן כֻּלְּכֶם קֹדְחֵי אֵשׁ, מְאַזְּרֵי זִיקוֹת; לְכוּ בְּאוּר אֶשְׁכֶם, וּבְזִיקוֹת בִּעַרְתֶּם; מִיָּדִי הָיְתָה זֹּאת לָכֶם, לְמַעֲצֵבָה תִּשְׁכָּבוּן"
  • ישעיהו סג10: "וְהֵמָּה מָרוּ וְעִצְּבוּ אֶת רוּחַ קָדְשׁוֹ, וַיֵּהָפֵךְ לָהֶם לְאוֹיֵב הוּא נִלְחַם בָּם"
  • דניאל ו21: "וּכְמִקְרְבֵהּ לְגֻבָּא לְדָנִיֵּאל, בְּקָל עֲצִיב זְעִק, עָנֵה מַלְכָּא וְאָמַר לְדָנִיֵּאל: 'דָּנִיֵּאל עֲבֵד אֱלָהָא חַיָּא, אֱלָהָךְ דִּי [אַנְתְּ]ה פָּלַח לֵהּ בִּתְדִירָא, הַיְכִל לְשֵׁיזָבוּתָךְ מִן אַרְיָוָתָא?'" - בקול שיש בו צער ומתח;
  • תהלים עח40: "כַּמָּה יַמְרוּהוּ בַמִּדְבָּר, יַעֲצִיבוּהוּ בִּישִׁימוֹן" 

2. נתינת צורה, פיסול

הפועל עִצֵּב (עיצב) בלשון ימינו משמעה "נתן צורה לחומר", וגם בלשון המקרא:

  • איוב י8: "יָדֶיךָ עִצְּבוּנִי וַיַּעֲשׂוּנִי, יַחַד סָבִיב וַתְּבַלְּעֵנִי" = ידיך עיצבו את הצורה שלי ועשו את גופי.
  • קהלת י9: "מַסִּיעַ אֲבָנִים יֵעָצֵב בָּהֶם, בּוֹקֵעַ עֵצִים יִסָּכֶן בָּם" = מי שטרח והסיע את האבנים ממקורן, יש לו זכות לעצב אותן ולקבוע את צורתן (פירוט).

לכן המילה עֶצֶב בלשון המקרא משמעה גם פסל או כלי כלשהו -

  • ירמיהו כב28: "הַעֶצֶב נִבְזֶה נָפוּץ הָאִישׁ הַזֶּה כָּנְיָהוּ, אִם כְּלִי אֵין חֵפֶץ בּוֹ? מַדּוּעַ הוּטֲלוּ הוּא וְזַרְעוֹ וְהֻשְׁלְכוּ עַל הָאָרֶץ אֲשֶׁר לֹא יָדָעוּ"

בדרך כלל (לא תמיד), הכוונה לפסל של עבודת אלילים, כי באלילות הדגש הוא על הצורה החיצונית - על העיצוב (ראו גם דברים ד16: "פֶּן תַּשְׁחִתוּן וַעֲשִׂיתֶם לָכֶם פֶּסֶל תְּמוּנַת כָּל סָמֶל,תַּבְנִית זָכָר אוֹ נְקֵבָה..." - עבודת אלילים מוגדרת ע"י מילים רבות הקשורות לצורה חיצונית):

  • שמואל א לא9: "וַיִּכְרְתוּ אֶת רֹאשׁוֹ וַיַּפְשִׁיטוּ אֶת כֵּלָיו וַיְשַׁלְּחוּ בְאֶרֶץ פְּלִשְׁתִּים סָבִיב לְבַשֵּׂר בֵּית עֲצַבֵּיהֶם וְאֶת הָעָם", וכן דברי הימים א י9
  • שמואל ב ה21: "וַיַּעַזְבוּ שָׁם אֶת עֲצַבֵּיהֶם, וַיִּשָּׂאֵם דָּוִד וַאֲנָשָׁיו"
  • ישעיהו י11: "הֲלֹא כַּאֲשֶׁר עָשִׂיתִי לְשֹׁמְרוֹן וְלֶאֱלִילֶיהָ, כֵּן אֶעֱשֶׂה לִירוּשָׁלִַם וְלַעֲצַבֶּיהָ"
  • ישעיהו מו1: "כָּרַע בֵּל, קֹרֵס נְבוֹ; הָיוּ עֲצַבֵּיהֶם לַחַיָּה וְלַבְּהֵמָה, נְשֻׂאֹתֵיכֶם עֲמוּסוֹת מַשָּׂא לַעֲיֵפָה"
  • ואולי גם משלי י22: "בִּרְכַּת ה' הִיא תַעֲשִׁיר, וְלֹא יוֹסִף עֶצֶב עִמָּהּ" (פירוט).

פסוקים נוספים 

  • ישעיהו מח5: "וָאַגִּיד לְךָ מֵאָז, בְּטֶרֶם תָּבוֹא הִשְׁמַעְתִּיךָ; פֶּן תֹּאמַר עָצְבִּי עָשָׂם וּפִסְלִי וְנִסְכִּי צִוָּם"
  • ירמיהו נ2: "... נלכדה בבל, הביש בל, חת מרדך, הבישו עצביה, חתו גלוליה"
  • הושע ד17: "חֲבוּר עֲצַבִּים אֶפְרָיִם, הַנַּח לוֹ"
  • הושע ח4: "הֵם הִמְלִיכוּ וְלֹא מִמֶּנִּי, הֵשִׂירוּ וְלֹא יָדָעְתִּי; כַּסְפָּם וּזְהָבָם עָשׂוּ לָהֶם עֲצַבִּים לְמַעַן יִכָּרֵת"
  • הושע יג2: "וְעַתָּה יוֹסִפוּ לַחֲטֹא וַיַּעְשׂוּ לָהֶם מַסֵּכָה מִכַּסְפָּם כִּתְבוּנָם עֲצַבִּים מַעֲשֵׂה חָרָשִׁים כֻּלֹּה לָהֶם הֵם אֹמְרִים זֹבְחֵי אָדָם עֲגָלִים יִשָּׁקוּן"
  • הושע יד9: "אֶפְרַיִם מַה לִּי עוֹד לָעֲצַבִּים אֲנִי עָנִיתִי וַאֲשׁוּרֶנּוּ אֲנִי כִּבְרוֹשׁ רַעֲנָן מִמֶּנִּי פֶּרְיְךָ נִמְצָא"
  • מיכה א7: "וְכָל פְּסִילֶיהָ יֻכַּתּוּ, וְכָל אֶתְנַנֶּיהָ יִשָּׂרְפוּ בָאֵשׁ, וְכָל עֲצַבֶּיהָ אָשִׂים שְׁמָמָה, כִּי מֵאֶתְנַן זוֹנָה קִבָּצָה וְעַד אֶתְנַן זוֹנָה יָשׁוּבוּ"
  • זכריה יג2: "וְהָיָה בַיּוֹם הַהוּא, נְאֻם ה' צְבָאוֹת, אַכְרִית אֶת שְׁמוֹת הָעֲצַבִּים מִן הָאָרֶץ, וְלֹא יִזָּכְרוּ עוֹד; וְגַם אֶת הַנְּבִיאִים וְאֶת רוּחַ הַטֻּמְאָה אַעֲבִיר מִן הָאָרֶץ"
  • דברי הימים ב כד18: "וַיַּעַזְבוּ אֶת בֵּית ה' אֱלֹהֵי אֲבוֹתֵיהֶם, וַיַּעַבְדוּ אֶת הָאֲשֵׁרִים וְאֶת הָעֲצַבִּים; וַיְהִי קֶצֶף עַל יְהוּדָה וִירוּשָׁלִַם בְּאַשְׁמָתָם זֹאֶת"
  • תהלים קו36-38: "וַיַּעַבְדוּ אֶת עֲצַבֵּיהֶם וַיִּהְיוּ לָהֶם לְמוֹקֵשׁ... וַיִּשְׁפְּכוּ דַּם נָקִי דָם בְּנֵיהֶם וּבְנוֹתֵיהֶם אֲשֶׁר זִבְּחוּ לַעֲצַבֵּי כְנָעַן וַתֶּחֱנַף הָאָרֶץ בַּדָּמִים"
  • תהלים קטו4: "עצביהם כֶּסֶף וְזָהָב, מַעֲשֵׂה יְדֵי אָדָם"
  • תהלים קלה15: "עֲצַבֵּי הַגּוֹיִם כֶּסֶף וְזָהָב מַעֲשֵׂה יְדֵי אָדָם"

3. עבודה מאומצת

לפי חלק מהמפרשים (ראו רש"י ודעת-מקרא על משלי יד23, וגם בתגובת אביתר כהן), לשורש עצב ישנה משמעות נוספת - עבודה מאומצת:

  • משלי טו1: "מַעֲנֶה רַּךְ יָשִׁיב חֵמָה, וּדְבַר עֶצֶב יַעֲלֶה אָף" - "דבר עצב" מנוגד ל"מענה רך", מכאן - עצב = קשה, מתאמץ להתנגד לדברי הזולת (פירוט).

פסוקים שבהם כמה משמעויות אפשריות

  • בראשית ג16: "אֶל הָאִשָּׁה אָמַר 'הַרְבָּה אַרְבֶּה עִצְּבוֹנֵךְ וְהֵרֹנֵךְ, בְּעֶצֶב תֵּלְדִי בָנִים...'" = 1 בצער;   2 3 במאמץ שיש בו יצירה.
  • בראשית ג17: "וּלְאָדָם אָמַר '...אֲרוּרָה הָאֲדָמָה בַּעֲבוּרֶךָ, בְּעִצָּבוֹן תֹּאכֲלֶנָּה כֹּל יְמֵי חַיֶּיךָ'" - 1 עבודת האדמה כרוכה בצער לפחות חלק מהזמן, כמו שנאמר ( תהלים קכו5): "הַזֹּרְעִים בְּדִמְעָה - בְּרִנָּה יִקְצֹרוּ";   3 עבודת האדמה כרוכה במאמץ, כמו שנאמר "בזיעת אפיך תאכל לחם". 
  • בראשית ה29: "וַיִּקְרָא אֶת שְׁמוֹ נֹחַ לֵאמֹר 'זֶה יְנַחֲמֵנוּ מִמַּעֲשֵׂנוּ וּמֵעִצְּבוֹן יָדֵינוּ, מִן הָאֲדָמָה אֲשֶׁר אֵרְרָהּ ה''" - נוח ייצר יין, שמשמח ומשכיח את צער העבודה הקשה.
