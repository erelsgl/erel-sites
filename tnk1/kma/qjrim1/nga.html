נגע = א. יצר קשר גופני; ב. פגע והזיק

קוד: נגע= בתנ"ך

סוג: הגדרה

מאת: אראל

אל:

נגיעה בלשון ימינו, וגם בלשון המקרא, מציינת קרבה פיסית, כמו למשל בפסוקים:

  • שמוֹת יט12: "וְהִגְבַּלְתָּ אֶת הָעָם סָבִיב לֵאמֹר 'הִשָּׁמְרוּ לָכֶם עֲלוֹת בָּהָר וּנְגֹעַ בְּקָצֵהוּ, כָּל הַנֹּגֵעַ בָּהָר מוֹת יוּמָת'"
  • שמות כט37: "שִׁבְעַת יָמִים תְּכַפֵּר עַל הַמִּזְבֵּחַ וְקִדַּשְׁתָּ אֹתוֹ, וְהָיָה הַמִּזְבֵּחַ קֹדֶשׁ קָדָשִׁים; 'כָּל הַנֹּגֵעַ בַּמִּזְבֵּחַ יִקְדָּשׁ'"
  • שמות ל29: "וְקִדַּשְׁתָּ אֹתָם, וְהָיוּ קֹדֶשׁ קָדָשִׁים; כָּל הַנֹּגֵעַ בָּהֶם יִקְדָּשׁ"
  • ויקרא יא24: "וּלְאֵלֶּה תִּטַּמָּאוּ: כָּל הַנֹּגֵעַ בְּנִבְלָתָם יִטְמָא עַד הָעָרֶב"
  • מלכים א ו27: "וַיִּתֵּן אֶת הַכְּרוּבִים בְּתוֹךְ הַבַּיִת הַפְּנִימִי וַיִּפְרְשׂוּ אֶת כַּנְפֵי הַכְּרֻבִים וַתִּגַּע כְּנַף הָאֶחָד בַּקִּיר וּכְנַף הַכְּרוּב הַשֵּׁנִי נֹגְעֹת בַּקִּיר הַשֵּׁנִי וְכַנְפֵיהֶם אֶל תּוֹךְ הַבַּיִת נֹגַעַת כָּנָף אֶל כָּנָף"
  • מלכים א יט5: "וַיִּשְׁכַּב וַיִּישַׁן תַּחַת רֹתֶם אֶחָד וְהִנֵּה זֶה מַלְאָךְ נֹגֵעַ בּוֹ וַיֹּאמֶר לוֹ קוּם אֱכוֹל"
  • ישעיהו ו7: "וַיַּגַּע עַל פִּי וַיֹּאמֶר הִנֵּה נָגַע זֶה עַל שְׂפָתֶיךָ וְסָר עונך וְחַטָּאתְךָ תְּכֻפָּר"
  • עמוס ט5: "וַאדֹנָי ידוד הַצְּבָאוֹת הַנּוֹגֵעַ בָּאָרֶץ וַתָּמוֹג וְאָבְלוּ כָּל יוֹשְׁבֵי בָהּ וְעָלְתָה כַיְאֹר כֻּלָּהּ וְשָׁקְעָה כִּיאֹר מִצְרָיִם"
  • דניאל ח5: "וַאֲנִי הָיִיתִי מֵבִין וְהִנֵּה צְפִיר הָעִזִּים בָּא מִן הַמַּעֲרָב עַל פְּנֵי כָל הָאָרֶץ וְאֵין נוֹגֵעַ בָּאָרֶץ וְהַצָּפִיר קֶרֶן חָזוּת בֵּין עֵינָיו"

אולם, בפסוקים שמדברים על נגיעה בבני-אדם, הכוונה היא בדרך כלל לקרבה שלילית - נגיעה שיש בה פגיעה ונזק:

  • בראשית יב17: "וַיְנַגַּע ה' אֶת פַּרְעֹה נְגָעִים גְּדֹלִים וְאֶת בֵּיתוֹ עַל דְּבַר שָׂרַי אֵשֶׁת אַבְרָם"
  • בראשית כ6: "וַיֹּאמֶר אֵלָיו הָאֱלֹהִים בַּחֲלֹם 'גַּם אָנֹכִי יָדַעְתִּי כִּי בְתָם לְבָבְךָ עָשִׂיתָ זֹּאת וָאֶחְשֹׂךְ גַּם אָנֹכִי אוֹתְךָ מֵחֲטוֹ לִי עַל כֵּן לֹא נְתַתִּיךָ לִנְגֹּעַ אֵלֶיהָ'"
  • בראשית כו11: "וַיְצַו אֲבִימֶלֶךְ אֶת כָּל הָעָם לֵאמֹר 'הַנֹּגֵעַ בָּאִישׁ הַזֶּה וּבְאִשְׁתּוֹ מוֹת יוּמָת'"
  • בראשית כו29: "אִם תַּעֲשֵׂה עִמָּנוּ רָעָה כַּאֲשֶׁר לֹא נְגַעֲנוּךָ וְכַאֲשֶׁר עָשִׂינוּ עִמְּךָ רַק טוֹב וַנְּשַׁלֵּחֲךָ בְּשָׁלוֹם; אַתָּה עַתָּה בְּרוּךְ ה'"
  • בראשית לב33: "עַל כֵּן לֹא יֹאכְלוּ בְנֵי יִשְׂרָאֵל אֶת גִּיד הַנָּשֶׁה אֲשֶׁר עַל כַּף הַיָּרֵךְ עַד הַיּוֹם הַזֶּה כִּי נָגַע בְּכַף יֶרֶךְ יַעֲקֹב בְּגִיד הַנָּשֶׁה"
  • שמות יא1: "וַיֹּאמֶר ה' אֶל מֹשֶׁה 'עוֹד נֶגַע אֶחָד אָבִיא עַל פַּרְעֹה וְעַל מִצְרַיִם אַחֲרֵי כֵן יְשַׁלַּח אֶתְכֶם מִזֶּה כְּשַׁלְּחוֹ כָּלָה גָּרֵשׁ יְגָרֵשׁ אֶתְכֶם מִזֶּה'"
  • יהושע ח15: "וַיִּנָּגְעוּ יְהוֹשֻׁעַ וְכָל יִשְׂרָאֵל לִפְנֵיהֶם וַיָּנֻסוּ דֶּרֶךְ הַמִּדְבָּר"
  • יהושע ט19: "וַיֹּאמְרוּ כָּל הַנְּשִׂיאִים אֶל כָל הָעֵדָה 'אֲנַחְנוּ נִשְׁבַּעְנוּ לָהֶם בה' אֱלֹהֵי יִשְׂרָאֵל, וְעַתָּה לֹא נוּכַל לִנְגֹּעַ בָּהֶם'"
  • שופטים כ34: "וַיָּבֹאוּ מִנֶּגֶד לַגִּבְעָה עֲשֶׂרֶת אֲלָפִים אִישׁ בָּחוּר מִכָּל יִשְׂרָאֵל וְהַמִּלְחָמָה כָּבֵדָה וְהֵם לֹא יָדְעוּ כִּי נֹגַעַת עֲלֵיהֶם הָרָעָה"
  • שמואל א ו9: "וּרְאִיתֶם אִם דֶּרֶךְ גְּבוּלוֹ יַעֲלֶה בֵּית שֶׁמֶשׁ הוּא עָשָׂה לָנוּ אֶת הָרָעָה הַגְּדוֹלָה הַזֹּאֶת וְאִם לֹא וְיָדַעְנוּ כִּי לֹא יָדוֹ נָגְעָה בָּנוּ מִקְרֶה הוּא הָיָה לָנוּ"
  • ירמיהו ד10: "וָאֹמַר 'אֲהָהּ ד' ה' אָכֵן הַשֵּׁא הִשֵּׁאתָ לָעָם הַזֶּה וְלִירוּשָׁלִַם לֵאמֹר 'שָׁלוֹם יִהְיֶה לָכֶם' וְנָגְעָה חֶרֶב עַד הַנָּפֶשׁ'"
  • ירמיהו ד18: "דַּרְכֵּךְ וּמַעֲלָלַיִךְ עָשׂוֹ אֵלֶּה לָךְ, זֹאת רָעָתֵךְ כִּי מָר כִּי נָגַע עַד לִבֵּךְ"
  • ירמיהו יב14: "כה אמר ה' על כל שכני הרעים הנגעים בנחלה אשר הנחלתי את עמי את ישראל הנני נתשם מעל אדמתם ואת בית יהודה אתוש מתוכם" (פירוט)
  • ירמיהו מח32: "מִבְּכִי יַעְזֵר אֶבְכֶּה לָּךְ הַגֶּפֶן שִׂבְמָה נְטִישֹׁתַיִךְ עָבְרוּ יָם עַד יָם יַעְזֵר נָגָעוּ עַל קֵיצֵךְ וְעַל בְּצִירֵךְ שֹׁדֵד נָפָל"

ואולי מכאן באה המילה "נֶגַע", שמשמעה "מחלה מזיקה":

  • ויקרא יג2: "אָדָם כִּי יִהְיֶה בְעוֹר בְּשָׂרוֹ שְׂאֵת אוֹ סַפַּחַת אוֹ בַהֶרֶת, וְהָיָה בְעוֹר בְּשָׂרוֹ לְנֶגַע צָרָעַת, וְהוּבָא אֶל אַהֲרֹן הַכֹּהֵן אוֹ אֶל אַחַד מִבָּנָיו הַכֹּהֲנִים..." (וכן בפרקים יד, טו)
  • שמואל ב ז14: "אֲנִי אֶהְיֶה לּוֹ לְאָב וְהוּא יִהְיֶה לִּי לְבֵן, אֲשֶׁר בהעותו וְהֹכַחְתִּיו בְּשֵׁבֶט אֲנָשִׁים וּבְנִגְעֵי בְּנֵי אָדָם"
  • מלכים א ח37-38: "רָעָב כִּי יִהְיֶה בְּאֶרֶץ דֶּבֶר כִּי יִהְיֶה שִׁדָּפוֹן יֵרָקוֹן אַרְבֶּה חָסִיל כִּי יִהְיֶה כִּי יָצַר לוֹ אֹיְבוֹ בָאָרֶץ שְׁעָרָיו כָּל נֶגַע כָּל מַחֲלָה. כָּל תְּפִלָּה כָל תְּחִנָּה אֲשֶׁר תִהְיֶה לְכָל הָאָדָם לְכֹל עַמְּךָ יִשְׂרָאֵל אֲשֶׁר יֵדְעוּן אִישׁ נֶגַע לְבָבוֹ וּפָרַשׂ כַּפָּיו אֶל הַבַּיִת הַזֶּה"
  • מלכים ב טו5: "וַיְנַגַּע ידוד אֶת הַמֶּלֶךְ וַיְהִי מְצֹרָע עַד יוֹם מֹתוֹ וַיֵּשֶׁב בְּבֵית הַחָפְשִׁית וְיוֹתָם בֶּן הַמֶּלֶךְ עַל הַבַּיִת שֹׁפֵט אֶת עַם הָאָרֶץ"

פסוק אחד שונה מכל הפסוקים שלמעלה - הוא מציין נגיעה רוחנית:

  • שמואל א י26: "וְגַם שָׁאוּל הָלַךְ לְבֵיתוֹ גִּבְעָתָה, וַיֵּלְכוּ עִמּוֹ הַחַיִל אֲשֶׁר נָגַע אֱלֹהִים בְּלִבָּם"

פירשנו שהמשמעויות של השורש נגע קשורות זו לזו;

אך ע"פ שבי"ל בקונקורדנציה, השורש נגע כולל שתי משמעויות שונות שאין קשר ביניהן:
1. קריבות ופגיעות דבר בדבר,
2. כאב ומדוה.
