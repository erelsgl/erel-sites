בטן = מקום ההולדה ומקום ההתחלה, מקום העיכול ומקום הרגשות המיידיים

קוד: בטן = כרס בתנ"ך

סוג: הגדרה

מאת: אראל

אל:

לבטן ישנם שני תפקידים עיקריים בעולם המוחשי, וכנגדם שני תפקידים עיקריים בעולם המשל והנמשל.

א. בטן = מקום ההולדה, וגם משל להתחלה

בטן היא המקום שממנו יוצאים הילדים, למשל:
  • בראשית לח27: "וַיְהִי בְּעֵת לִדְתָּהּ, וְהִנֵּה תְאוֹמִים בְּבִטְנָהּ"
  • דברים ז13: "וַאֲהֵבְךָ וּבֵרַכְךָ וְהִרְבֶּךָ, וּבֵרַךְ פְּרִי בִטְנְךָ וּפְרִי אַדְמָתֶךָ דְּגָנְךָ וְתִירֹשְׁךָ וְיִצְהָרֶךָ שְׁגַר אֲלָפֶיךָ וְעַשְׁתְּרֹת צֹאנֶךָ עַל הָאֲדָמָה אֲשֶׁר נִשְׁבַּע לַאֲבֹתֶיךָ לָתֶת לָךְ"
בטן האם היא תחילת דרכו של האדם בעולם, ולכן משתמשים בביטוי "מהבטן" כדי לציין שליחות המחייבת את האדם מתחילת הדרך:
  • תחילת דרכו של שמשון: שופטים יג5: "כִּי הִנָּךְ הָרָה וְיֹלַדְתְּ בֵּן וּמוֹרָה לֹא יַעֲלֶה עַל רֹאשׁוֹ כִּי נְזִיר אֱלֹהִים יִהְיֶה הַנַּעַר מִן הַבָּטֶן וְהוּא יָחֵל לְהוֹשִׁיעַ אֶת יִשְׂרָאֵל מִיַּד פְּלִשְׁתִּים... ויאמר לי הִנָּךְ הָרָה וְיֹלַדְתְּ בֵּן ועתה אל תשתי יין ושכר ואל תאכלי כל טמאה כִּי נְזִיר אֱלֹהִים יִהְיֶה הַנַּעַר מִן הַבָּטֶן עד יום מותו"
  • תחילת דרכו של עם ישראל: ישעיהו מד2: "כֹּה אָמַר ה' עֹשֶׂךָ וְיֹצֶרְךָ מִבֶּטֶן יַעְזְרֶךָּ אַל תִּירָא עַבְדִּי יַעֲקֹב וִישֻׁרוּן בָּחַרְתִּי בוֹ"
  • תחילת דרכו של ישעיהו הנביא: ישעיהו מט1: "שִׁמְעוּ אִיִּים אֵלַי וְהַקְשִׁיבוּ לְאֻמִּים מֵרָחוֹק: ה' מִבֶּטֶן קְרָאָנִי, מִמְּעֵי אִמִּי הִזְכִּיר שְׁמִי"
  • תחילת דרכו של ירמיהו הנביא: ירמיהו א5: "בְּטֶרֶם אצורך[אֶצָּרְךָ] בַבֶּטֶן יְדַעְתִּיךָ וּבְטֶרֶם תֵּצֵא מֵרֶחֶם הִקְדַּשְׁתִּיךָ, נָבִיא לַגּוֹיִם נְתַתִּיךָ"
  • תחילת דרכו של כל אדם בעולם: קהלת ה14: "כַּאֲשֶׁר יָצָא מִבֶּטֶן אִמּוֹ עָרוֹם יָשׁוּב לָלֶכֶת כְּשֶׁבָּא, וּמְאוּמָה לֹא יִשָּׂא בַעֲמָלוֹ שֶׁיֹּלֵךְ בְּיָדוֹ"

פסוקים נוספים

  • ישעיהו מט15: "הֲתִשְׁכַּח אִשָּׁה עוּלָהּ מֵרַחֵם בֶּן בִּטְנָהּ? גַּם אֵלֶּה תִשְׁכַּחְנָה וְאָנֹכִי לֹא אֶשְׁכָּחֵךְ"
  • הושע ט11: "אֶפְרַיִם כָּעוֹף יִתְעוֹפֵף כְּבוֹדָם' מִלֵּדָה וּמִבֶּטֶן וּמֵהֵרָיוֹן"
  • הושע ט16: "הֻכָּה אֶפְרַיִם שָׁרְשָׁם יָבֵשׁ פְּרִי [בַל]י יַעֲשׂוּן; גַּם כִּי יֵלֵדוּן וְהֵמַתִּי מַחֲמַדֵּי בִטְנָם"
  • הושע יב4: "בַּבֶּטֶן עָקַב אֶת אָחִיו וּבְאוֹנוֹ שָׂרָה אֶת אֱלֹהִים"
  • מיכה ו7: "הֲיִרְצֶה ה' בְּאַלְפֵי אֵילִים בְּרִבְבוֹת נַחֲלֵי שָׁמֶן הַאֶתֵּן בְּכוֹרִי פִּשְׁעִי פְּרִי בִטְנִי חַטַּאת נַפְשִׁי"
  • תהלים כב10: "כִּי אַתָּה גֹחִי מִבָּטֶן מַבְטִיחִי עַל שְׁדֵי אִמִּי. עַליך השלכתי מרחם מִבָּטֶן אִמִּי אלי אַתָּה"
  • תהלים נח4: "זֹרוּ רְשָׁעִים מֵרָחֶם, תָּעוּ מִבֶּטֶן דֹּבְרֵי כָזָב"
  • תהלים עא6: "עָלֶיךָ נִסְמַכְתִּי מִבֶּטֶן מִמְּעֵי אִמִּי אַתָּה גוֹזִי בְּךָ תְהִלָּתִי תָמִיד"
  • תהלים קכז3: "הִנֵּה נַחֲלַת ה' בָּנִים, שָׂכָר פְּרִי הַבָּטֶן"
  • תהלים קלט13: "כִּי אַתָּה קָנִיתָ כִלְיֹתָי, תְּסֻכֵּנִי בְּבֶטֶן אִמִּי" (פירוט)
  • איוב א21: "וַיֹּאמֶר עָרֹם יצתי[יָצָאתִי] מִבֶּטֶן אִמִּי וְעָרֹם אָשׁוּב שָׁמָה..."
  • איוב ג10: "כִּי לֹא סָגַר דַּלְתֵי בִטְנִי, וַיַּסְתֵּר עָמָל מֵעֵינָי"
  • איוב ג11: "לָמָּה לֹּא מֵרֶחֶם אָמוּת מִבֶּטֶן יָצָאתִי וְאֶגְוָע"
  • איוב י19: "כַּאֲשֶׁר לֹא הָיִיתִי אֶהְיֶה מִבֶּטֶן לַקֶּבֶר אוּבָל"
  • איוב יט17: "רוּחִי זָרָה לְאִשְׁתִּי, וְחַנֹּתִי לִבְנֵי בִטְנִי"
  • איוב לא15: "הלא בַבֶּטֶן עֹשֵׂנִי עָשָׂהוּ, וַיְכֻנֶנּוּ בָּרֶחֶם אֶחָד"
  • איוב לא18: "כִּי מִנְּעוּרַי גְּדֵלַנִי כְאָב, וּמִבֶּטֶן אִמִּי אַנְחֶנָּה"
  • איוב לב19: "הִנֵּה בִטְנִי כְּיַיִן לֹא יִפָּתֵחַ, כְּאֹבוֹת חֲדָשִׁים יִבָּקֵעַ"
  • איוב לח29: "מִבֶּטֶן מִי יָצָא הַקָּרַח, וּכְפֹר שָׁמַיִם מִי יְלָדוֹ"
  • משלי לא2: "מַה בְּרִי וּמַה בַּר בִּטְנִי וּמֶה בַּר נְדָרָי"

ב. בטן = מקום עיכול המזון, וגם משל לרגשות מיידיים ותת-הכרתיים

הבטן היא גם המקום שמתמלא באוכל מייד אחרי האכילה:
  • תהלים יז14: "מִמְתִים יָדְךָ ה' מִמְתִים מֵחֶלֶד, חֶלְקָם בַּחַיִּים וצפינך[וּצְפוּנְךָ] תְּמַלֵּא בִטְנָם, יִשְׂבְּעוּ בָנִים וְהִנִּיחוּ יִתְרָם לְעוֹלְלֵיהֶם"
  • משלי יג25: "צַדִּיק אֹכֵל לְשֹׂבַע נַפְשׁוֹ, וּבֶטֶן רְשָׁעִים תֶּחְסָר"
  • איוב כ15: "חַיִל בָּלַע וַיְקִאֶנּוּ, מִבִּטְנוֹ יוֹרִשֶׁנּוּ אֵל... כי לא ידע שלו בבטנו, בחמודו לא ימלט... יהי למלא בטנו, ישלח בו חרון אפו וימטר עלימו בלחומו" - הרשע ממלא את בטנו באוכל וברכוש גזול, אך ה' ידאג שהוא יקיא את הכל וייענש על הגזלה.
ולכן משתמשים במושג "בטן" כדי לציין רגשות שממלאים את האדם כתגובה מיידית לאירועים חיצוניים:
  • חבקוק ג16: "שָׁמַעְתִּי וַתִּרְגַּז בִּטְנִי, לְקוֹל צָלֲלוּ שְׂפָתַי, יָבוֹא רָקָב בַּעֲצָמַי וְתַחְתַּי אֶרְגָּז, אֲשֶׁר אָנוּחַ לְיוֹם צָרָה לַעֲלוֹת לְעַם יְגוּדֶנּוּ" = שמעתי על הצרות העומדות לבוא עליי, ומייד הרגשתי רוגז בבטני.
  • תהלים לא10: "חָנֵּנִי ה' כִּי צַר לִי, עָשְׁשָׁה בְכַעַס עֵינִי נַפְשִׁי וּבִטְנִי" = ראיתי מראות מכעיסים, ומייד התמלאה בטני בכעס.
  • משלי יח8: "דִּבְרֵי נִרְגָּן כְּמִתְלַהֲמִים, וְהֵם יָרְדוּ חַדְרֵי בָטֶן" = דברים של אדם המאשים את רעהו בכוונות רעות גורמים לתחושה רעה בבטן (פירוט).
  • משלי יח20: "מִפְּרִי פִי אִישׁ תִּשְׂבַּע בִּטְנוֹ, תְּבוּאַת שְׂפָתָיו יִשְׂבָּע" (פירוט)
בניגוד ללב, המציין את הרגשות המודעים, הנובעים ממחשבה - הבטן מציינת רגשות מיידיים שאינם נובעים ממחשבה, כמו הביטוי בלשון ימינו " מגיב מהבטן " שמשמעו "מגיב מיידית, בלי לחשוב". לפעמים הבטן מציינת רגשות שאף אינם מודעים:
  • משלי כ27: "נֵר ה' נִשְׁמַת אָדָם, חֹפֵשׂ כָּל חַדְרֵי בָטֶן" (פירוט)
  • משלי כ30: "חַבֻּרוֹת פֶּצַע תמריק[תַּמְרוּק] בְּרָע וּמַכּוֹת חַדְרֵי בָטֶן" (פירוט)
  • משלי כב18: "כִּי נָעִים כִּי תִשְׁמְרֵם בְּבִטְנֶךָ, יִכֹּנוּ יַחְדָּו עַל שְׂפָתֶיךָ" - את החכמות החיצוניות יש לשמור בבטן, בניגוד לחכמה העיקרית שנשמרת בלב (פירוט).
  • איוב טו2: "הֶחָכָם יַעֲנֶה דַעַת רוּחַ, וִימַלֵּא קָדִים בִּטְנוֹ?!" - אליפז מאשים את איוב שהוא "מדבר מהבטן", מדבר בלי לחשוב, ואומר דברי רוח והבל; וכן בהמשך הפרק: איוב טו35: "הָרֹה עָמָל וְיָלֹד אָוֶן, וּבִטְנָם תָּכִין מִרְמָה". ואיוב 'מודה באשמה': איוב לב18: "כִּי מָלֵתִי מִלִּים, הֱצִיקַתְנִי רוּחַ בִּטְנִי".
מקור המשל הוא בדברי ה' ליחזקאל:
  • יחזקאל ג3: "וַיֹּאמֶר אֵלַי 'בֶּן אָדָם! בִּטְנְךָ תַאֲכֵל וּמֵעֶיךָ תְמַלֵּא אֵת הַמְּגִלָּה הַזֹּאֵת אֲשֶׁר אֲנִי נֹתֵן אֵלֶיךָ!' וָאֹכְלָה, וַתְּהִי בְּפִי כִּדְבַשׁ לְמָתוֹק" - ה' נתן ליחזקאל מגילה ובה נבואות, וציווה עליו לאכול את הנבואה ולמלא בה את בטנו; הפעולה הזאת היא משל לכך שהנבואות אמורות להתמזג בנפשו של הנביא גם מבחינה רגשית, הוא אמור להרגיש הזדהות מלאה עם הנבואות.

ג. פסוקים נוספים

  • במדבר ה21: "וְהִשְׁבִּיעַ הַכֹּהֵן אֶת הָאִשָּׁה בִּשְׁבֻעַת הָאָלָה וְאָמַר הַכֹּהֵן לָאִשָּׁה יִתֵּן ה' אוֹתָךְ לְאָלָה וְלִשְׁבֻעָה בְּתוֹךְ עַמֵּךְ בְּתֵת ידוד אֶת יְרֵכֵךְ נֹפֶלֶת וְאֶת בִּטְנֵךְ צָבָה. ובאו המים המאררים הָאָלָה במעיך לצבות בטן ולנפל ירך וְאָמַרה הָאִשָּׁה אמן אמן... והשקה אֶת המים והיתה אם נטמאה ותמעל מעל באישה ובאו בה המים המאררים למרים וצבתה בטנה ונפלה ירכה והיתה הָאִשָּׁה לְאָלָה בקרב עמה"
  • שופטים ג21: "וַיִּשְׁלַח אֵהוּד אֶת יַד שְׂמֹאלוֹ וַיִּקַּח אֶת הַחֶרֶב מֵעַל יֶרֶךְ יְמִינוֹ וַיִּתְקָעֶהָ בְּבִטְנוֹ"
