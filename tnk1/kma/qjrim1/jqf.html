ראשי>לשון המקרא>השקיף>

השקיף, נשקף = הביט מלמעלה למטה

קוד: שקף= בתנ"ך

סוג: הגדרה

מאת: אראל

אל:

השורש "שקף" מציין הבטה ממקום גבוה למקום נמוך:

א. הבטה מתוך חלון:

  • בראשית כו8: "... וַיַּשְׁקֵף אֲבִימֶלֶךְ מֶלֶךְ פְּלִשְׁתִּים בְּעַד החלוֹן, וַיַּרְא וְהִנֵּה יִצְחָק מְצַחֵק אֵת רִבְקָה אִשְׁתּוֹ"
  • שופטים ה28: "בְּעַד הַחַלּוֹן נִשְׁקְפָה וַתְּיַבֵּב אֵם סִיסְרָא בְּעַד הָאֶשְׁנָב: מַדּוּעַ בֹּשֵׁשׁ רִכְבּוֹ לָבוֹא מַדּוּעַ אֶחֱרוּ פַּעֲמֵי מַרְכְּבוֹתָיו"
  • שמואל ב ו16: "...וּמִיכַל בַּת שָׁאוּל נִשְׁקְפָה בְּעַד הַחַלּוֹן, וַתֵּרֶא אֶת הַמֶּלֶךְ דָּוִד מְפַזֵּז וּמְכַרְכֵּר לִפְנֵי ידָּוִד וַתִּבֶז לוֹ בְּלִבָּהּ", דברי הימים א טו29
  • מלכים ב ט30-32: "...וְאִיזֶבֶל שָׁמְעָה וַתָּשֶׂם בַּפּוּךְ עֵינֶיהָ וַתֵּיטֶב אֶת רֹאשָׁהּ, וַתַּשְׁקֵף בְּעַד הַחַלּוֹן...  וַיִּשָּׂא פָנָיו אֶל הַחַלּוֹן וַיֹּאמֶר 'מִי אִתִּי מִי?', וַיַּשְׁקִיפוּ אֵלָיו שְׁנַיִם שְׁלֹשָׁה סָרִיסִים"
  • משלי ז6-7: "כִּי בְּחַלּוֹן בֵּיתִי בְּעַד אֶשְׁנַבִּי נִשְׁקָפְתִּי, וָאֵרֶא בַפְּתָאיִם אָבִינָה בַבָּנִים נַעַר חֲסַר לֵב" - הבטה מלמעלה שהיא גם משל לראיית התמונה הכללית.
  • וכנראה גם כאן: שמואל ב כד20: "וַיַּשְׁקֵף אֲרַוְנָה, וַיַּרְא אֶת הַמֶּלֶךְ וְאֶת עֲבָדָיו עֹבְרִים עָלָיו, וַיֵּצֵא אֲרַוְנָה וַיִּשְׁתַּחוּ לַמֶּלֶךְ אַפָּיו אָרְצָה" - ארונה היה בביתו, השקיף מהחלון ואז יצא לקראת המלך.

ב. הבטה מאיזור הררי לאיזור נמוך יותר:

  • בראשית יח16: "וַיָּקֻמוּ מִשָּׁם הָאֲנָשִׁים, וַיַּשְׁקִפוּ עַל פְּנֵי סְדֹם, וְאַבְרָהָם הֹלֵךְ עִמָּם לְשַׁלְּחָם", וכן בהמשך בראשית יט28: "וַיַּשְׁקֵף עַל פְּנֵי סְדֹם וַעֲמֹרָה וְעַל כָּל פְּנֵי אֶרֶץ הַכִּכָּר וַיַּרְא וְהִנֵּה עָלָה קִיטֹר הָאָרֶץ כְּקִיטֹר הַכִּבְשָׁן" - אברהם גר בחברון שבהרי יהודה, והשקיף על-פני סדום שנמצאת למטה, בבקעת הירדן.
  • במדבר כא20: "וּמִבָּמוֹת, הַגַּיְא אֲשֶׁר בִּשְׂדֵה מוֹאָב רֹאשׁ הַפִּסְגָּה,וְנִשְׁקָפָה עַל פְּנֵי הַיְשִׁימֹן"
  • במדבר כג28: "וַיִּקַּח בָּלָק אֶת בִּלְעָם רֹאשׁ הַפְּעוֹר, הַנִּשְׁקָף עַל פְּנֵי הַיְשִׁימֹן"
  • וכנראה גם כאן (אני לא מכיר את הטופוגרפיה): שמואל א יג18: "וְהָרֹאשׁ אֶחָד יִפְנֶה דֶּרֶךְ בֵּית חֹרוֹן, וְהָרֹאשׁ אֶחָד יִפְנֶה דֶּרֶךְ הַגְּבוּל הַנִּשְׁקָף עַל גֵּי הַצְּבֹעִים הַמִּדְבָּרָה"
  • ואולי גם כאן: ירמיהו ו1: "הָעִזוּ בְּנֵי בִניָמִן מִקֶּרֶב יְרוּשָׁלִַם וּבִתְקוֹעַ תִּקְעוּ שׁוֹפָר וְעַל בֵּית הַכֶּרֶם שְׂאוּ מַשְׂאֵת כִּי רָעָה נִשְׁקְפָה מִצָּפוֹן וְשֶׁבֶר גָּדוֹל" וצריך עיון.

ג. הבטה מהשמיים לארץ:

  • דברים כו15: "הַשְׁקִיפָה מִמְּעוֹן קָדְשְׁךָ מִן הַשָּׁמַיִם, וּבָרֵךְ אֶת עַמְּךָ אֶת יִשְׂרָאֵל וְאֵת הָאֲדָמָה אֲשֶׁר נָתַתָּה לָנוּ כַּאֲשֶׁר נִשְׁבַּעְתָּ לַאֲבֹתֵינוּ אֶרֶץ זָבַת חָלָב וּדְבָשׁ"
  • תהלים יד2: "ידוד מִשָּׁמַיִם הִשְׁקִיף עַל בְּנֵי אָדָם לִרְאוֹת הֲיֵשׁ מַשְׂכִּיל דֹּרֵשׁ אֶת אֱלֹהִים", וכן תהלים נג3
  • תהלים פה12: "אֱמֶת מֵאֶרֶץ תִּצְמָח וְצֶדֶק מִשָּׁמַיִם נִשְׁקָף"
  • תהלים קב20: "כִּי הִשְׁקִיף מִמְּרוֹם קָדְשׁוֹ ה' מִשָּׁמַיִם אֶל אֶרֶץ הִבִּיט"
  • איכה ג50: "עַד יַשְׁקִיף וְיֵרֶא ה' מִשָּׁמָיִם"
  • וכנראה זו הכוונה גם כאן, שהרי ה' תמיד מסתכל, כביכול, מהשמים לארץ: שמות יד24: "וַיְהִי בְּאַשְׁמֹרֶת הַבֹּקֶר, וַיַּשְׁקֵף ה' אֶל מַחֲנֵה מִצְרַיִם בְּעַמּוּד אֵשׁ וְעָנָן, וַיָּהָם אֵת מַחֲנֵה מִצְרָיִם"
  • וכ: שיר השירים ו10: "מִי זֹאת הַנִּשְׁקָפָה כְּמוֹ שָׁחַר, יָפָה כַלְּבָנָה בָּרָה כַּחַמָּה אֲיֻמָּה כַּנִּדְגָּלוֹת" - היא נראית כמו אור השחר, הלבנה והחמה, שמאירים מהשמיים על הארץ.

ד. משקוף = החלק העליון של הפתח - מעל שתי המזוזות:

  • שמות יב7: "וְלָקְחוּ מִן הַדָּם, וְנָתְנוּ עַל שְׁתֵּי הַמְּזוּזֹת וְעַל הַמַּשְׁקוֹף, עַל הַבָּתִּים אֲשֶׁר יֹאכְלוּ אֹתוֹ בָּהֶם" (וכן בהמשך פסוקים 22, 23; ראו פירושי רש"י וראב"ע).

ה. שקופים = חלונות גבוהים (בעיקר בבית המקדש):

  • מלכים א ו4: "וַיַּעַשׂ לַבָּיִת חַלּוֹנֵי שְׁקֻפִים אֲטֻמִים"
  • מלכים א ז4: "וּשְׁקֻפִים שְׁלֹשָׁה טוּרִים וּמֶחֱזָה אֶל מֶחֱזָה שָׁלֹשׁ פְּעָמִים"
  • מלכים א ז5: "וְכָל הַפְּתָחִים וְהַמְּזוּזוֹת רְבֻעִים שָׁקֶף וּמוּל מֶחֱזָה אֶל מֶחֱזָה שָׁלֹשׁ פְּעָמִים"
