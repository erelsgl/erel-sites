כלב - חיה בזויה

קוד: כלב = חיה בתנ"ך

סוג: הגדרה

מאת: אראל

אל:

בימינו הכלב נחשב לידידו הטוב ביותר של האדם, אבל במקרא דווקא החמור נחשב לידידו הטוב ביותר של אדם, הכלב נחשב לחיה בזויה ומתועבת:
  • שמות כב30: "וְאַנְשֵׁי קֹדֶשׁ תִּהְיוּן לִי, וּבָשָׂר בַּשָּׂדֶה טְרֵפָה לֹא תֹאכֵלוּ, לַכֶּלֶב תַּשְׁלִכוּן אֹתוֹ" - הכלב הוא ההיפך מקודש.
  • דברים כג19: "לֹא תָבִיא אֶתְנַן זוֹנָה וּמְחִיר כֶּלֶב בֵּית ידוד אֱלֹהֶיךָ לְכָל נֶדֶר כִּי תוֹעֲבַת ידוד אֱלֹהֶיךָ גַּם שְׁנֵיהֶם" - הכלב נחשב לתועבה  (הכלבים נזכרו יחד עם הזונות גם ב מלכים א כב38: "וַיִּשְׁטֹף אֶת הָרֶכֶב עַל בְּרֵכַת שֹׁמְרוֹן וַיָּלֹקּוּ הַכְּלָבִים אֶת דָּמוֹ וְהַזֹּנוֹת רָחָצוּ כִּדְבַר ה' אֲשֶׁר דִּבֵּר").

כלב ככינוי גנאי ועלבון

בנביאים וכתובים נזכר הכלב כמה פעמים ככינוי גנאי ועלבון:
  • שמואל א יז43: "וַיֹּאמֶר הַפְּלִשְׁתִּי אֶל דָּוִד ' הֲכֶלֶב אָנֹכִי כִּי אַתָּה בָא אֵלַי בַּמַּקְלוֹת?'..."
  • שמואל א כד14: "אַחֲרֵי מִי יָצָא מֶלֶךְ יִשְׂרָאֵל? אַחֲרֵי מִי אַתָּה רֹדֵף? אַחֲרֵי כֶּלֶב מֵת? אַחֲרֵי פַּרְעֹשׁ אֶחָד?"
  • שמואל ב ג8: "וַיִּחַר לְאַבְנֵר מְאֹד עַל דִּבְרֵי אִישׁ בֹּשֶׁת וַיֹּאמֶר 'הֲרֹאשׁ כֶּלֶב אָנֹכִי אֲשֶׁר לִיהוּדָה?'..."
  • שמואל ב ט8: "וַיִּשְׁתַּחוּ וַיֹּאמֶר 'מֶה עַבְדֶּךָ כִּי פָנִיתָ אֶל הַכֶּלֶב הַמֵּת אֲשֶׁר כָּמוֹנִי?'"
  • שמואל ב טז9: "וַיֹּאמֶר אֲבִישַׁי בֶּן צְרוּיָה אֶל הַמֶּלֶךְ 'לָמָּה יְקַלֵּל הַכֶּלֶב הַמֵּת הַזֶּה אֶת אֲדֹנִי הַמֶּלֶךְ? אֶעְבְּרָה נָּא וְאָסִירָה אֶת רֹאשׁוֹ!'"
  • מלכים ב ח13: "וַיֹּאמֶר חֲזָהאֵל 'כִּי מָה עַבְדְּךָ הַכֶּלֶב כִּי יַעֲשֶׂה הַדָּבָר הַגָּדוֹל הַזֶּה?' וַיֹּאמֶר אֱלִישָׁע הִרְאַנִי ה' אֹתְךָ מֶלֶךְ עַל אֲרָם"
  • ישעיהו נו10: "צפו[צֹפָיו] עִוְרִים כֻּלָּם לֹא יָדָעוּ כֻּלָּם כְּלָבִים אִלְּמִים לֹא יוּכְלוּ לִנְבֹּחַ הֹזִים שֹׁכְבִים אֹהֲבֵי לָנוּם.  והכְּלָבִים עזי נפש לֹא יָדָעוּ שבעה והמה רעים לֹא יָדָעוּ הבין כֻּלָּם לדרכם פנו איש לבצעו מקצהו"
  • משלי כו11: "כְּכֶלֶב שָׁב עַל קֵאוֹ כְּסִיל שׁוֹנֶה בְאִוַּלְתּוֹ"
  • איוב ל1: "וְעַתָּה שָׂחֲקוּ עָלַי צְעִירִים מִמֶּנִּי לְיָמִים אֲשֶׁר מָאַסְתִּי אֲבוֹתָם לָשִׁית עִם כַּלְבֵי צֹאנִי"

אכילה ע"י כלבים


אכילה ע"י כלבים נחשבה לעונש חמור ומבזה במיוחד:

  • מלכים א יד11: "הַמֵּת לְיָרָבְעָם בָּעִיר יֹאכְלוּ הַכְּלָבִים וְהַמֵּת בַּשָּׂדֶה יֹאכְלוּ עוֹף הַשָּׁמָיִם כִּי ה' דִּבֵּר"
  • מלכים א טז4: "הַמֵּת לְבַעְשָׁא בָּעִיר יֹאכְלוֹּ הַכְּלָבִים וְהַמֵּת לוֹ בַּשָּׂדֶה יֹאכְלוּ עוֹף הַשָּׁמָיִם"
  • מלכים א כא19-24: "וְדִבַּרְתָּ אֵלָיו לֵאמֹר כֹּה אָמַר ה' הֲרָצַחְתָּ וְגַם יָרָשְׁתָּ וְדִבַּרְתָּ אֵלָיו לֵאמֹר כֹּה אָמַר ה' בִּמְקוֹם אֲשֶׁר לָקְקוּ הַכְּלָבִים אֶת דַּם נָבוֹת יָלֹקּוּ הַכְּלָבִים אֶת דָּמְךָ גַּם אָתָּה...  וְגַם לְאִיזֶבֶל דִּבֶּר ה' לֵאמֹר הַכְּלָבִים יֹאכְלוּ אֶת אִיזֶבֶל בְּחֵל יִזְרְעֶאֶל.  הַמֵּת לְאַחְאָב בָּעִיר יֹאכְלוּ הַכְּלָבִים וְהַמֵּת בַּשָּׂדֶה יֹאכְלוּ עוֹף הַשָּׁמָיִם"
  • מלכים א כב38: "וַיִּשְׁטֹף אֶת הָרֶכֶב עַל בְּרֵכַת שֹׁמְרוֹן וַיָּלֹקּוּ הַכְּלָבִים אֶת דָּמוֹ וְהַזֹּנוֹת רָחָצוּ כִּדְבַר ה' אֲשֶׁר דִּבֵּר"
  • מלכים ב ט10-36: "וְאֶת אִיזֶבֶל יֹאכְלוֹּ הַכְּלָבִים בְּחֵלֶק יִזְרְעֶאֶל וְאֵין קֹבֵר וַיִּפְתַּח הַדֶּלֶת וַיָּנֹס...  וַיָּשֻׁבוּ וַיַּגִּיָדוֹּ לוֹ וַיֹּאמֶר דִּבֶּר ידָּוִד הוּא אֲשֶׁר דְּבַר בְּיַד עַבְדּוֹ אֵלִיָּהוּ הַתִּשְׁבִּי לֵאמֹר בְּחֵלֶק יִזְרְעֶאֶל יֹאכְלוּ הַכְּלָבִים אֶת בְּשַׂר אִיזֶבֶל"
  • ירמיהו טו3: "וּפָקַדְתִּי עֲלֵיהֶם אַרְבַּע מִשְׁפָּחוֹת נְאֻם ידוד אֶת הַחֶרֶב לַהֲרֹג וְאֶת הַכְּלָבִים לִסְחֹב וְאֶת עוֹף הַשָּׁמַיִם וְאֶת בֶּהֱמַת הָאָרֶץ לֶאֱכֹל וּלְהַשְׁחִית"


פסוקים נוספים

  • שמות יא7: "וּלְכֹל בְּנֵי יִשְׂרָאֵל לֹא יֶחֱרַץ כֶּלֶב לְשֹׁנוֹ לְמֵאִישׁ וְעַד בְּהֵמָה לְמַעַן תֵּדְעוּן אֲשֶׁר יַפְלֶה ידוד בֵּין מִצְרַיִם וּבֵין יִשְׂרָאֵל"
  • שופטים ז5: "וַיּוֹרֶד אֶת הָעָם אֶל הַמַּיִם וַיֹּאמֶר ידוד אֶל גִּדְעוֹן כֹּל אֲשֶׁר יָלֹק בִּלְשׁוֹנוֹ מִן הַמָּיִם כַּאֲשֶׁר יָלֹק הַכֶּלֶב תַּצִּיג אוֹתוֹ לְבָד וְכֹל אֲשֶׁר יִכְרַע עַל בִּרְכָּיו לִשְׁתּוֹת"
  • ישעיהו סו3: "שׁוֹחֵט הַשּׁוֹר מַכֵּה אִישׁ זוֹבֵחַ הַשֶּׂה עֹרֵף כֶּלֶב מַעֲלֵה מִנְחָה דַּם חֲזִיר מַזְכִּיר לְבֹנָה מְבָרֵךְ אָוֶן גַּם הֵמָּה בָּחֲרוּ בְּדַרְכֵיהֶם וּבְשִׁקּוּצֵיהֶם נַפְשָׁם חָפֵצָה"
  • תהלים כב17: "כִּי סְבָבוּנִי כְּלָבִים עֲדַת מְרֵעִים הִקִּיפוּנִי כָּאֲרִי יָדַי וְרַגְלָי"
  • תהלים כב21: "הַצִּילָה מֵחֶרֶב נַפְשִׁי מִיַּד כֶּלֶב יְחִידָתִי"
  • תהלים נט7-15: "יָשׁוּבוּ לָעֶרֶב יֶהֱמוּ כַכָּלֶב וִיסוֹבְבוּ עִיר... וְיָשׁוּבוּ לָעֶרֶב יֶהֱמוּ כַכָּלֶב וִיסוֹבְבוּ עִיר"
  • תהלים סח24: "לְמַעַן תִּמְחַץ רַגְלְךָ בְּדָם לְשׁוֹן כְּלָבֶיךָ מֵאֹיְבִים מִנֵּהוּ"
  • משלי כו17: "מַחֲזִיק בְּאָזְנֵי כָלֶב עֹבֵר מִתְעַבֵּר עַל רִיב לֹּא לוֹ"
  • קהלת ט4: "כִּי מִי אֲשֶׁר יבחר[יְחֻבַּר] אֶל כָּל הַחַיִּים יֵשׁ בִּטָּחוֹן כִּי לְכֶלֶב חַי הוּא טוֹב מִן הָאַרְיֵה הַמֵּת"
