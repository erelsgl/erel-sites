ראשי>הקבלות>קשרי מושגים>

תעה = טעה = שגה או איבד את דרכו

קוד: תעה= בתנ"ך

סוג: הגדרה

מאת: אראל

אל:

בלשון ימינו מבחינים בין טעה = עשה שגיאה, לבין תעה = איבד את דרכו. בלשון המקרא, השורש טעה מופיע רק פעם אחת, ומשמעותו, לפי המפרשים, כמו תעה:

  • יחזקאל יג10: "יַעַן וּבְיַעַן הִטְעוּ אֶת עַמִּי לֵאמֹר 'שָׁלוֹם וְאֵין שָׁלוֹם', וְהוּא בֹּנֶה חַיִץ, וְהִנָּם טָחִים אֹתוֹ תָּפֵל" (מצודת ציון שם, שבי"ל בקונקורדנציה, דעת מקרא על ישעיהו כח).

אם כך, מה משמעות השורש תעה = טעה?

1. בחלק מהפסוקים הוא נרדף לשגיאה ולאיוולת, כמו טעה בלשוננו:

  • ישעיהו יט13-14: "נוֹאֲלוּ שָׂרֵי צֹעַן, נִשְּׁאוּ שָׂרֵי נֹף, הִתְעוּ אֶת מִצְרַיִם פִּנַּת שְׁבָטֶיהָ... ה' מָסַךְ בְּקִרְבָּהּ רוּחַ עִוְעִים וְהִתְעוּ אֶת מִצְרַיִם בְּכָל מַעֲשֵׂהוּ כְּהִתָּעוֹת שִׁכּוֹר בְּקִיאוֹ" (פירוט)
  • ישעיהו כח7: "וְגַם אֵלֶּה בַּיַּיִן שָׁגוּ וּבַשֵּׁכָר תָּעוּ, כֹּהֵן וְנָבִיא שָׁגוּ בַשֵּׁכָר נִבְלְעוּ מִן הַיַּיִן, תָּעוּ מִן הַשֵּׁכָר שָׁגוּ בָּרֹאֶה פָּקוּ פְּלִילִיָּה" (פירוט);

2. אולם בחלק מהפסוקים הוא מציין אבדן דרך, כמו תעה בלשוננו:

  • בראשית לז15: "וַיִּמְצָאֵהוּ אִישׁ וְהִנֵּה תֹעֶה בַּשָּׂדֶה וַיִּשְׁאָלֵהוּ הָאִישׁ לֵאמֹר מַה תְּבַקֵּשׁ"
  • שמות כג4: "כִּי תִפְגַּע שׁוֹר אֹיִבְךָ אוֹ חֲמֹרוֹ תֹּעֶה - הָשֵׁב תְּשִׁיבֶנּוּ לוֹ"
  • ישעיהו לה8: "וְהָיָה שָׁם מַסְלוּל וָדֶרֶךְ וְדֶרֶךְ הַקֹּדֶשׁ יִקָּרֵא לָהּ לֹא יַעַבְרֶנּוּ טָמֵא וְהוּא לָמוֹ הֹלֵךְ דֶּרֶךְ וֶאֱוִילִים לֹא יִתְעוּ"
  • תהלים קז4: "תָּעוּ בַמִּדְבָּר בִּישִׁימוֹן דָּרֶךְ, עִיר מוֹשָׁב לֹא מָצָאוּ"
  • תהלים קיט176: "תָּעִיתִי כְּשֶׂה אֹבֵד, בַּקֵּשׁ עַבְדֶּךָ כִּי מצותיך לֹא שָׁכָחְתִּי" (פירוט).

3. ובחלק מהפסוקים הוא מציין התפזרות לכל עבר, לא דווקא מתוך טעות או תעיה:

  • ישעיהו מז15: "כֵּן הָיוּ לָךְ אֲשֶׁר יָגָעַתְּ סֹחֲרַיִךְ מִנְּעוּרַיִךְ, אִישׁ לְעֶבְרוֹ תָּעוּ, אֵין מוֹשִׁיעֵךְ"

מובן שיש קשר בין הפירושים: "עניינו טעות ושגיאה, ורובו בדרך" (שבי"ל, קונקורדנציה).

ישנם פסוקים רבים, שבהם כל הפירושים אפשריים, בפרט בפסוקים המדברים על מנהיגים המתעים את בני ישראל:

  • ירמיהו נ6: "צֹאן אֹבְדוֹת היה[הָיוּ] עַמִּי, רֹעֵיהֶם הִתְעוּם; הָרִים שובבים[שׁוֹבְבוּם], מֵהַר אֶל גִּבְעָה הָלָכוּ, שָׁכְחוּ רִבְצָם"  - בני ישראל נמשלים לצאן, ומנהיגיהם נמשלים לרועים: כאשר הרועים תועים (מאבדים את הדרך), הם גורמים לצאן לתעות (לאבד את הדרך ולהתפזר לכל עבר); כאשר המנהיגים תועים (שוגים), הם גורמים גם לבני ישראל לתעות (לשגות ולהתפצל לפלגים רבים):
  • ירמיהו כג13: "וּבִנְבִיאֵי שֹׁמְרוֹן רָאִיתִי תִפְלָה: הִנַּבְּאוּ בַבַּעַל וַיַּתְעוּ אֶת עַמִּי אֶת יִשְׂרָאֵל"
  • ירמיהו כג32: "הִנְנִי עַל נִבְּאֵי חֲלֹמוֹת שֶׁקֶר נְאֻם ה', וַיְסַפְּרוּם וַיַּתְעוּ אֶת עַמִּי בְּשִׁקְרֵיהֶם וּבְפַחֲזוּתָם, וְאָנֹכִי לֹא שְׁלַחְתִּים וְלֹא צִוִּיתִים, וְהוֹעֵיל לֹא יוֹעִילוּ לָעָם הַזֶּה נְאֻם ה'"
  • ישעיהו ג12: "עַמִּי מְאַשְּׁרֶיךָ מַתְעִים, וְדֶרֶךְ אֹרְחֹתֶיךָ בִּלֵּעוּ"
  • ישעיהו ט15: "וַיִּהְיוּ מְאַשְּׁרֵי הָעָם הַזֶּה מַתְעִים, וּמְאֻשָּׁרָיו מְבֻלָּעִים"
  • מיכה ג5: "כֹּה אָמַר ה' עַל הַנְּבִיאִים הַמַּתְעִים אֶת עַמִּי, הַנֹּשְׁכִים בְּשִׁנֵּיהֶם וְקָרְאוּ שָׁלוֹם וַאֲשֶׁר לֹא יִתֵּן עַל פִּיהֶם וְקִדְּשׁוּ עָלָיו מִלְחָמָה"

כך גם בפסוקים שבהם בני ישראל תועים מאחרי ה':

  • ישעיהו נג6: "כֻּלָּנוּ כַּצֹּאן תָּעִינוּ, אִישׁ לְדַרְכּוֹ פָּנִינוּ; וה' הִפְגִּיעַ בּוֹ אֵת עוון כֻּלָּנוּ" - גם ה' נמשל לרועה של בני ישראל, ובני ישראל תועים (שוגים) כאשר הם תועים (מאבדים את דרכם, מתפזרים לכל עבר ועובדים כל אליל).
  • יחזקאל יד11: "לְמַעַן לֹא יִתְעוּ עוֹד בֵּית יִשְׂרָאֵל מֵאַחֲרַי, וְלֹא יִטַּמְּאוּ עוֹד בְּכָל פִּשְׁעֵיהֶם; וְהָיוּ לִי לְעָם וַאֲנִי אֶהְיֶה לָהֶם לֵאלֹהִים נְאֻם אֲדֹנָי ה'"
  • יחזקאל מד10-15: "כִּי אִם הַלְוִיִּם אֲשֶׁר רָחֲקוּ מֵעָלַי בִּתְעוֹת יִשְׂרָאֵל אֲשֶׁר תָּעוּ מֵעָלַי אַחֲרֵי גִּלּוּלֵיהֶם וְנָשְׂאוּ עונם... וְהַכֹּהֲנִים הַלְוִיִּם בְּנֵי צָדוֹק אֲשֶׁר שָׁמְרוּ אֶת מִשְׁמֶרֶת מִקְדָּשִׁי בִּתְעוֹת בְּנֵי יִשְׂרָאֵל מֵעָלַי הֵמָּה יִקְרְבוּ אֵלַי לְשָׁרְתֵנִי וְעָמְדוּ לְפָנַי לְהַקְרִיב לִי חֵלֶב וָדָם נְאֻם אֲדֹנָי ה'"
  • יחזקאל מח11: "לַכֹּהֲנִים הַמְקֻדָּשׁ מִבְּנֵי צָדוֹק אֲשֶׁר שָׁמְרוּ מִשְׁמַרְתִּי אֲשֶׁר לֹא תָּעוּ בִּתְעוֹת בְּנֵי יִשְׂרָאֵל כַּאֲשֶׁר תָעוּ הַלְוִיִּם"
  • עמוס ב4: "כֹּה אָמַר ה' עַל שְׁלֹשָׁה פִּשְׁעֵי יְהוּדָה וְעַל אַרְבָּעָה לֹא אֲשִׁיבֶנּוּ עַל מָאֳסָם אֶת תּוֹרַת ה' וְחֻקָּיו לֹא שָׁמָרוּ וַיַּתְעוּם כִּזְבֵיהֶם אֲשֶׁר הָלְכוּ אֲבוֹתָם אַחֲרֵיהֶם"

פסוקים לעיון נוסף

  • בראשית כ13: "וַיְהִי כַּאֲשֶׁר הִתְעוּ אֹתִי אֱלֹהִים מִבֵּית אָבִי, וָאֹמַר לָהּ זֶה חַסְדֵּךְ אֲשֶׁר תַּעֲשִׂי עִמָּדִי אֶל כָּל הַמָּקוֹם אֲשֶׁר נָבוֹא שָׁמָּה אִמְרִי לִי אָחִי הוּא" (פירוט).
  • ישעיהו כט24: "וְיָדְעוּ תֹעֵי רוּחַ בִּינָה, וְרוֹגְנִים יִלְמְדוּ לֶקַח"
  • ישעיהו כא4: "תָּעָה לְבָבִי, פַּלָּצוּת בִּעֲתָתְנִי; אֵת נֶשֶׁף חִשְׁקִי שָׂם לִי לַחֲרָדָה"
  • ישעיהו ל28: "וְרוּחוֹ כְּנַחַל שׁוֹטֵף עַד צַוָּאר יֶחֱצֶה, לַהֲנָפָה גוֹיִם בְּנָפַת שָׁוְא וְרֶסֶן מַתְעֶה עַל לְחָיֵי עַמִּים"
  • ישעיהו טז8: "כִּי שַׁדְמוֹת חֶשְׁבּוֹן אֻמְלָל, גֶּפֶן שִׂבְמָה; בַּעֲלֵי גוֹיִם הָלְמוּ שְׂרוּקֶּיהָ, עַד יַעְזֵר נָגָעוּ; תָּעוּ מִדְבָּר שְׁלֻחוֹתֶיהָ, נִטְּשׁוּ עָבְרוּ יָם"
  • ישעיהו לב6: "כִּי נָבָל נְבָלָה יְדַבֵּר וְלִבּוֹ יַעֲשֶׂה אָוֶן לַעֲשׂוֹת חֹנֶף וּלְדַבֵּר אֶל ה' תּוֹעָה לְהָרִיק נֶפֶשׁ רָעֵב וּמַשְׁקֶה צָמֵא יַחְסִיר" (פירוט)
  • ירמיהו י15: "הֶבֶל הֵמָּה, מַעֲשֵׂה תַּעְתֻּעִים, בְּעֵת פְּקֻדָּתָם יֹאבֵדוּ"
  • ירמיהו נא18: "הֶבֶל הֵמָּה, מַעֲשֵׂה תַּעְתֻּעִים, בְּעֵת פְּקֻדָּתָם יֹאבֵדוּ"
  • הושע ד12: "עַמִּי בְּעֵצוֹ יִשְׁאָל וּמַקְלוֹ יַגִּיד לוֹ' כִּי רוּחַ זְנוּנִים הִתְעָה וַיִּזְנוּ מִתַּחַת אֱלֹהֵיהֶם"
  • תהלים נח4: "זֹרוּ רְשָׁעִים מֵרָחֶם, תָּעוּ מִבֶּטֶן דֹּבְרֵי כָזָב"
  • תהלים צה10: "אַרְבָּעִים שָׁנָה אָקוּט בְּדוֹר, וָאֹמַר עַם תֹּעֵי לֵבָב הֵם, וְהֵם לֹא יָדְעוּ דְרָכָי" (פירוט)
  • תהלים קיט110: "נָתְנוּ רְשָׁעִים פַּח לִי, וּמִפִּקּוּדֶיךָ לֹא תָעִיתִי" (פירוט)
  • משלי י17: "אֹרַח לְחַיִּים שׁוֹמֵר מוּסָר, וְעוֹזֵב תּוֹכַחַת מַתְעֶה" (פירוט)
  • משלי יד22: "הלא יִתְעוּ חֹרְשֵׁי רָע, וְחֶסֶד וֶאֱמֶת חֹרְשֵׁי טוֹב" (פירוט)
  • משלי כא16: "אָדָם תּוֹעֶה מִדֶּרֶךְ הַשְׂכֵּל, בִּקְהַל רְפָאִים יָנוּחַ" (פירוט)
  • איוב לח41: "מִי יָכִין לָעֹרֵב צֵידוֹ כִּי ילדו[יְלָדָיו] אֵל אֶל יְשַׁוֵּעוּ, יִתְעוּ לִבְלִי אֹכֶל"
  • איוב טו31: "אַל יַאֲמֵן בשו[בַּשָּׁיו] נִתְעָה, כִּי שָׁוְא תִּהְיֶה תְמוּרָתוֹ"
  • דברי הימים ב לו16: "וַיִּהְיוּ מַלְעִבִים בְּמַלְאֲכֵי הָאֱלֹהִים וּבוֹזִים דְּבָרָיו וּמִתַּעְתְּעִים בִּנְבִאָיו עַד עֲלוֹת חֲמַת ה' בְּעַמּוֹ עַד לְאֵין מַרְפֵּא"
  • נחמיה ד2: "וַיִּקְשְׁרוּ כֻלָּם יַחְדָּו לָבוֹא לְהִלָּחֵם בִּירוּשָׁלִָם וְלַעֲשׂוֹת לוֹ תּוֹעָה"
