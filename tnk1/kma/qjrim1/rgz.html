ראשי>הגדרות מושגים>
ראשי>לשון המקרא>רגז>

רגז = לא שקט

קוד: רגז= בתנ"ך

סוג: הגדרה

מאת: אראל

אל:

רוגז הוא אי-שקט. הוא יכול לתאר אי-שקט של הארץ, רעידת אדמה:
  • איוב ט6: "הַמַּרְגִּיז אֶרֶץ מִמְּקוֹמָהּ, וְעַמּוּדֶיהָ יִתְפַלָּצוּן"
אי שקט של אנשים בארץ, מהומה גדולה או מלחמה או גירוש:
  • שמואל א יד15: "וַתְּהִי חֲרָדָה בַמַּחֲנֶה בַשָּׂדֶה וּבְכָל הָעָם הַמַּצָּב וְהַמַּשְׁחִית חָרְדוּ גַּם הֵמָּה, וַתִּרְגַּז הָאָרֶץ וַתְּהִי לְחֶרְדַּת אֱלֹהִים"
  • שמואל ב ז10: "וְשַׂמְתִּי מָקוֹם לְעַמִּי לְיִשְׂרָאֵל וּנְטַעְתִּיו וְשָׁכַן תַּחְתָּיו וְלֹא יִרְגַּז עוֹד, וְלֹא יֹסִיפוּ בְנֵי עַוְלָה לְעַנּוֹתוֹ כַּאֲשֶׁר בָּרִאשׁוֹנָה" (פירוט)
אך בדרך-כלל הוא מתאר אי-שקט של הנפש:
  • איוב ג17: "שָׁם רְשָׁעִים חָדְלוּ רֹגֶז, וְשָׁם יָנוּחוּ יְגִיעֵי כֹחַ" - רוגז מנוגד למנוחה.
  • משלי כט9: "אִישׁ חָכָם נִשְׁפָּט אֶת אִישׁ אֱוִיל וְרָגַז וְשָׂחַק וְאֵין נָחַת"  - רוגז מנוגד לנחת (פירוט).
רוגז הוא מצב נפשי המשותף לרגשות לא נעימים שונים - חרדה, כעס, צער וחרטה - בכולם הנפש נמצאת בחוסר-מנוחה:
  • שמות טו14: "שָׁמְעוּ עַמִּים יִרְגָּזוּן, חִיל אָחַז יֹשְׁבֵי פְּלָשֶׁת" - העמים ששמעו על קריאת ים סוף לא יכלו לישון בשקט בלילה (חרדה),
  • איוב יב6: "יִשְׁלָיוּ אֹהָלִים לְשֹׁדְדִים, וּבַטֻּחוֹת לְמַרְגִּיזֵי אֵל, לַאֲשֶׁר הֵבִיא אֱלוֹהַּ בְּיָדוֹ" - אנשים שמכעיסים את ה'.
  • שמואל ב יט1: "וַיִּרְגַּז הַמֶּלֶךְ, וַיַּעַל עַל עֲלִיַּת הַשַּׁעַר וַיֵּבְךְּ, וְכֹה אָמַר בְּלֶכְתּוֹ 'בְּנִי אַבְשָׁלוֹם בְּנִי בְּנִי אַבְשָׁלוֹם מִי יִתֵּן מוּתִי אֲנִי תַחְתֶּיךָ אַבְשָׁלוֹם בְנִי בְנִי...'" - דוד לא יכל להפסיק לבכות (צער)
ויש פסוקים שבהם כמה משמעויות אפשריות:
  • בראשית מה24: "וַיְשַׁלַּח אֶת אֶחָיו וַיֵּלֵכוּ, וַיֹּאמֶר אֲלֵהֶם 'אַל תִּרְגְּזוּ בַּדָּרֶךְ'" - יוסף אומר לאחיו 'לכו בשלום ושלוה, אל תכעסו זה על זה, אל תפחדו ואל תהיו מלאים ברגשות צער וחרטה בגלל מה שעשיתם לי'.
  • דברים כח65: "וּבַגּוֹיִם הָהֵם לֹא תַרְגִּיעַ וְלֹא יִהְיֶה מָנוֹחַ לְכַף רַגְלֶךָ, וְנָתַן ה' לְךָ שָׁם לֵב רַגָּז וְכִלְיוֹן עֵינַיִם וְדַאֲבוֹן נָפֶשׁ" - בגלות לא תהיה לכם מנוחה, יהיה לכם רק צער ופחד.
  • חבקוק ג2: "ה'... בְּרֹגֶז רַחֵם תִּזְכּוֹר" - כשהעולם לא שקט, כשיש כעס או פחד או צער בעולם, זכור לרחם עלינו.
  • תהלים ד5: "רִגְזוּ וְאַל תֶּחֱטָאוּ, אִמְרוּ בִלְבַבְכֶם עַל מִשְׁכַּבְכֶם וְדֹמּוּ סֶלָה" - כדי שלא תחטאו, כדאי לכם להיות תמיד באי-שקט, "בתנועה מתמדת" (פירוט).
  • איוב יד1: "אָדָם יְלוּד אִשָּׁה, קְצַר יָמִים וּשְׂבַע רֹגֶז" - החיים קצרים ואין בהם מנוחה.

מקורות ופירושים נוספים

"לשון רגז בלשון הקודש ענינו ניענוע ורעדה (רשב"ם)... והושאל על שאון והמית הנפש לכל סיבה שתהיה... חרדת הפחד... חרדת העצבון... החימה והחרון. ובלשון ארמית נאמר תמיד על החימה והחרון" (שד"ל על בראשית מה24).

רוגז - רעש

ע"פ מלבי"ם, רעש גדול מרוגז; רוגז הוא רעידת אדמה קלה, ורעש הוא רעידת אדמה חמורה.
  • שמואל ב כב8: "ותגעשוַתִּרְעַשׁ הָאָרֶץ, מוֹסְדוֹת הַשָּׁמַיִם יִרְגָּזוּ, וַיִּתְגָּעֲשׁוּ כִּי חָרָה לוֹ"
  • ישעיהו יג13: "עַל כֵּן שָׁמַיִם אַרְגִּיז וְתִרְעַשׁ הָאָרֶץ מִמְּקוֹמָהּ, בְּעֶבְרַת ה' צְבָאוֹת וּבְיוֹם חֲרוֹן אַפּוֹ"
  • יואל ב10: "לְפָנָיו רָגְזָה אֶרֶץ, רָעֲשׁוּ שָׁמָיִם; שֶׁמֶשׁ וְיָרֵחַ קָדָרוּ וְכוֹכָבִים אָסְפוּ נָגְהָם"
  • תהלים יח8: "וַתִּגְעַשׁ וַתִּרְעַשׁ הָאָרֶץ, וּמוֹסְדֵי הָרִים יִרְגָּזוּ, וַיִּתְגָּעֲשׁוּ כִּי חָרָה לוֹ"
  • תהלים עז19: "קוֹל רַעַמְךָ בַּגַּלְגַּל הֵאִירוּ בְרָקִים תֵּבֵל רָגְזָה וַתִּרְעַשׁ הָאָרֶץ"
  • איוב לט24: "בְּרַעַשׁ וְרֹגֶז יְגַמֶּא אָרֶץ, וְלֹא יַאֲמִין כִּי קוֹל שׁוֹפָר"


פסוקים נוספים

רוגז = חרדה

  • דברים ב25: "הַיּוֹם הַזֶּה אָחֵל תֵּת פַּחְדְּךָ וְיִרְאָתְךָ עַל פְּנֵי הָעַמִּים תַּחַת כָּל הַשָּׁמָיִם, אֲשֶׁר יִשְׁמְעוּן שִׁמְעֲךָ וְרָגְזוּ וְחָלוּ מִפָּנֶיךָ"
  • ירמיהו לג9: "וְהָיְתָה לִּי לְשֵׁם שָׂשׂוֹן לִתְהִלָּה וּלְתִפְאֶרֶת לְכֹל גּוֹיֵי הָאָרֶץ אֲשֶׁר יִשְׁמְעוּ אֶת כָּל הַטּוֹבָה אֲשֶׁר אָנֹכִי עֹשֶׂה אֹתָם וּפָחֲדוּ וְרָגְזוּ עַל כָּל הַטּוֹבָה וְעַל כָּל הַשָּׁלוֹם אֲשֶׁר אָנֹכִי עֹשֶׂה לָּהּ"

רוגז = צער

  • ישעיהו יד3: "וְהָיָה בְּיוֹם הָנִיחַ ה' לְךָ מֵעָצְבְּךָ וּמֵרָגְזֶךָ וּמִן הָעֲבֹדָה הַקָּשָׁה אֲשֶׁר עֻבַּד בָּךְ"

רוגז = כעס

  • דניאל ג13: "בֵּאדַיִן נְבוּכַדְנֶצַּר בִּרְגַז וַחֲמָה אֲמַר לְהַיְתָיָה לְשַׁדְרַךְ מֵישַׁךְ וַעֲבֵד נְגוֹ בֵּאדַיִן גֻּבְרַיָּא אִלֵּךְ הֵיתָיוּ קֳדָם מַלְכָּא"
  • עזרא ה12: "לָהֵן מִן דִּי הַרְגִּזוּ אֲבָהֳתַנָא לֶאֱלָהּ שְׁמַיָּא יְהַב הִמּוֹ בְּיַד נְבוּכַדְנֶצַּר מֶלֶךְ בָּבֶל כסדִּיאוּבַיְתָה דְנָה סַתְרֵהּ וְעַמָּה הַגְלִי לְבָבֶל"

  • שמואל א כח15: "וַיֹּאמֶר שְׁמוּאֵל אֶל שָׁאוּל 'לָמָּה הִרְגַּזְתַּנִי לְהַעֲלוֹת אֹתִי?' ..." - למה לא נתת לי לנוח בשלוה בקבר?
  • מלכים ב יט27: "וְשִׁבְתְּךָ וְצֵאתְךָ וּבֹאֲךָ יָדָעְתִּי, וְאֵת הִתְרַגֶּזְךָ אֵלָי. יען הִתְרַגֶּזְךָ אֵלָי ושאננך עלה באזני, ושמתי חחי באפך ומתגי בשפתיך והשבתיך בדרך אשר באת בה"
  • ישעיהו ה25: "עַל כֵּן חָרָה אַף ה' בְּעַמּוֹ, וַיֵּט יָדוֹ עָלָיו וַיַּכֵּהוּ, וַיִּרְגְּזוּ הֶהָרִים, וַתְּהִי נִבְלָתָם כַּסּוּחָה בְּקֶרֶב חוּצוֹת; בְּכָל זֹאת לֹא שָׁב אַפּוֹ וְעוֹד יָדוֹ נְטוּיָה"
  • ישעיהו יד9: "שְׁאוֹל מִתַּחַת רָגְזָה לְךָ לִקְרַאת בּוֹאֶךָ עוֹרֵר לְךָ רְפָאִים כָּל עַתּוּדֵי אָרֶץ הֵקִים מִכִּסְאוֹתָם כֹּל מַלְכֵי גוֹיִם"
  • ישעיהו יד16: "רֹאֶיךָ אֵלֶיךָ יַשְׁגִּיחוּ אֵלֶיךָ יִתְבּוֹנָנוּ הֲזֶה הָאִישׁ מַרְגִּיז הָאָרֶץ מַרְעִישׁ מַמְלָכוֹת"
