המילה רפאים נזכרת בלשון המקרא בכמה משמעויות:

א. שבט של ענקים שחי בעבר הירדן המזרחי:

  • בראשית יד5: "וּבְאַרְבַּע עֶשְׂרֵה שָׁנָה בָּא כְדָרְלָעֹמֶר וְהַמְּלָכִים אֲשֶׁר אִתּוֹ וַיַּכּוּ אֶת רְפָאִים בְּעַשְׁתְּרֹת קַרְנַיִם וְאֶת הַזּוּזִים בְּהָם וְאֵת הָאֵימִים בְּשָׁוֵה קִרְיָתָיִם"
  • בראשית טו20: "וְאֶת הַחִתִּי וְאֶת הַפְּרִזִּי וְאֶת הָרְפָאִים"
  • דברים ב11: "רְפָאִים יֵחָשְׁבוּ אַף הֵם כָּעֲנָקִים וְהַמֹּאָבִים יִקְרְאוּ לָהֶם אֵמִים"
  • דברים ב20: "אֶרֶץ רְפָאִים תֵּחָשֵׁב אַף הִוא רְפָאִים יָשְׁבוּ בָהּ לְפָנִים וְהָעַמֹּנִים יִקְרְאוּ לָהֶם זַמְזֻמִּים"
  • דברים ג11: "כִּי רַק עוֹג מֶלֶךְ הַבָּשָׁן נִשְׁאַר מִיֶּתֶר הָרְפָאִים הִנֵּה עַרְשׂוֹ עֶרֶשׂ בַּרְזֶל הֲלֹה הִוא בְּרַבַּת בְּנֵי עַמּוֹן תֵּשַׁע אַמּוֹת אָרְכָּהּ וְאַרְבַּע אַמּוֹת רָחְבָּהּ בְּאַמַּת אִישׁ"
  • דברים ג13: "וְיֶתֶר הַגִּלְעָד וְכָל הַבָּשָׁן מַמְלֶכֶת עוֹג נָתַתִּי לַחֲצִי שֵׁבֶט הַמְנַשֶּׁה כֹּל חֶבֶל הָאַרְגֹּב לְכָל הַבָּשָׁן הַהוּא יִקָּרֵא אֶרֶץ רְפָאִים"
  • יהושע יב4: "וּגְבוּל עוֹג מֶלֶךְ הַבָּשָׁן מִיֶּתֶר הָרְפָאִים הַיּוֹשֵׁב בְּעַשְׁתָּרוֹת וּבְאֶדְרֶעִי"
  • יהושע יג12: "כָּל מַמְלְכוּת עוֹג בַּבָּשָׁן אֲשֶׁר מָלַךְ בְּעַשְׁתָּרוֹת וּבְאֶדְרֶעִי הוּא נִשְׁאַר מִיֶּתֶר הָרְפָאִים וַיַּכֵּם מֹשֶׁה וַיֹּרִשֵׁם"
  • יהושע יז15: "וַיֹּאמֶר אֲלֵיהֶם יְהוֹשֻׁעַ אִם עַם רַב אַתָּה עֲלֵה לְךָ הַיַּעְרָה וּבֵרֵאתָ לְךָ שָׁם בְּאֶרֶץ הַפְּרִזִּי וְהָרְפָאִים כִּי אָץ לְךָ הַר אֶפְרָיִם"
  • דברי הימים א כ4: "וַיְהִי אַחֲרֵיכֵן וַתַּעֲמֹד מִלְחָמָה בְּגֶזֶר עִם פְּלִשְׁתִּים אָז הִכָּה סִבְּכַי הַחֻשָׁתִי אֶת סִפַּי מִילִדֵי הָרְפָאִים וַיִּכָּנֵעוּ"

ב. שם עמק באיזור ירושלים - בין יהודה לבנימין

(כיום יש בירושלים רחוב בשם זה):

  • יהושע טו8: "וְעָלָה הַגְּבוּל גֵּי בֶן הִנֹּם אֶל כֶּתֶף הַיְבוּסִי מִנֶּגֶב הִיא יְרוּשָׁלִָם וְעָלָה הַגְּבוּל אֶל רֹאשׁ הָהָר אֲשֶׁר עַל פְּנֵי גֵי הִנֹּם יָמָּה אֲשֶׁר בִּקְצֵה עֵמֶק רְפָאִים צָפֹנָה"
  • יהושע יח16: "וְיָרַד הַגְּבוּל אֶל קְצֵה הָהָר אֲשֶׁר עַל פְּנֵי גֵּי בֶן הִנֹּם אֲשֶׁר בְּעֵמֶק רְפָאִים צָפוֹנָה וְיָרַד גֵּי הִנֹּם אֶל כֶּתֶף הַיְבוּסִי נֶגְבָּה וְיָרַד עֵין רֹגֵל"
  • שמואל ב ה18-22: "וּפְלִשְׁתִּים בָּאוּ וַיִּנָּטְשׁוּ בְּעֵמֶק רְפָאִים... וַיֹּסִפוּ עוֹד פְּלִשְׁתִּים לַעֲלוֹת וַיִּנָּטְשׁוּ בְּעֵמֶק רְפָאִים", וכן דברי הימים א יד9
  • שמואל ב כג13, דברי הימים א יא15: "וַיֵּרְדוּ שלשים מהשלשים רֹאשׁ ויבאו אֶל קציר אֶל דוד אֶל מְעָרַת עֲדֻלָּם וחית פְלִשְׁתִּים חֹנָה בְּעֵמֶק רְפָאִים"
  • ישעיהו יז5: "וְהָיָה כֶּאֱסֹף קָצִיר קָמָה וּזְרֹעוֹ שִׁבֳּלִים יִקְצוֹר וְהָיָה כִּמְלַקֵּט שִׁבֳּלִים בְּעֵמֶק רְפָאִים"

ג. כינוי למתים, שוכני שאול:

  • ישעיהו יד9: "שְׁאוֹל מִתַּחַת רָגְזָה לְךָ לִקְרַאת בּוֹאֶךָ עוֹרֵר לְךָ רְפָאִים כָּל עַתּוּדֵי אָרֶץ הֵקִים מִכִּסְאוֹתָם כֹּל מַלְכֵי גוֹיִם"
  • ישעיהו כו14: "מֵתִים בַּל יִחְיוּ רְפָאִים בַּל יָקֻמוּ לָכֵן פָּקַדְתָּ וַתַּשְׁמִידֵם וַתְּאַבֵּד כָּל זֵכֶר לָמוֹ"
  • ישעיהו כו19: "יִחְיוּ מֵתֶיךָ נְבֵלָתִי יְקוּמוּן הָקִיצוּ וְרַנְּנוּ שֹׁכְנֵי עָפָר כִּי טַל אוֹרֹת טַלֶּךָ וָאָרֶץ רְפָאִים תַּפִּיל"
  • תהלים פח11: "הֲלַמֵּתִים תַּעֲשֶׂה פֶּלֶא אִם רְפָאִים יָקוּמוּ יוֹדוּךָ סֶּלָה"
  • איוב כו5: "הָרְפָאִים יְחוֹלָלוּ מִתַּחַת מַיִם וְשֹׁכְנֵיהֶם"
  • משלי ב18: "כִּי שָׁחָה אֶל מָוֶת בֵּיתָהּ וְאֶל רְפָאִים מַעְגְּלֹתֶיהָ"
  • משלי ט18: "וְלֹא יָדַע כִּי רְפָאִים שָׁם בְּעִמְקֵי שְׁאוֹל קְרֻאֶיהָ"
  • משלי כא16: "אָדָם תּוֹעֶה מִדֶּרֶךְ הַשְׂכֵּל - בִּקְהַל רְפָאִים יָנוּחַ" (ראו מנוחה - רק במוות)

מה הקשר בין המשמעויות?

ייתכן שהמשותף לכל המשמעויות הוא הפחד והשיתוק (= רפיון לב וידיים): הענקים מפחידים את רואיהם ומרפים את ידיהם, גם המתים מפחידים את רואיהם, ובעמק רפאים - ייתכן שהיו בעבר ענקים שגרמו לעוברים בו לפחד.

ראו גם רפא = הרפה.
