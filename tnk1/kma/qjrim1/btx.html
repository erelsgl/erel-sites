בטח = היה רגוע או נשען על-, האמין שיעזור או שלא יזיק

קוד: בטח= בתנ"ך

סוג: הגדרה

מאת: אראל

אל:

השורש בטח מופיע בלשון המקרא בכמה הקשרים:

1. שלוה ללא פחד:

  • בראשית לד25: "וַיְהִי בַיּוֹם הַשְּׁלִישִׁי, בִּהְיוֹתָם כֹּאֲבִים, וַיִּקְחוּ שְׁנֵי בְנֵי יַעֲקֹב שִׁמְעוֹן וְלֵוִי אֲחֵי דִינָה אִישׁ חַרְבּוֹ, וַיָּבֹאוּ עַל הָעִיר בֶּטַח, וַיַּהַרְגוּ כָּל זָכָר" - שמעון ולוי לא חששו שאנשי שכם יפגעו בהם; או - אנשי שכם לא חששו ששמעון ולוי יפגעו בהם.
  • ויקרא כה18: "וַעֲשִׂיתֶם אֶת חֻקֹּתַי וְאֶת מִשְׁפָּטַי תִּשְׁמְרוּ וַעֲשִׂיתֶם אֹתָם' וִישַׁבְתֶּם עַל הָאָרֶץ לָבֶטַח, ונתנה הָאָרֶץ פריה ואכלתם לשבע וִישַׁבְתֶּם לָבֶטַח עַליה", ויקרא כו5: "וְהִשִּׂיג לָכֶם דַּיִשׁ אֶת בָּצִיר וּבָצִיר יַשִּׂיג אֶת זָרַע וַאֲכַלְתֶּם לַחְמְכֶם לָשֹׂבַע וִישַׁבְתֶּם לָבֶטַח בְּאַרְצְכֶם" - בלי לחשוש מפני רעב;
  • דברים יב10: "וַעֲבַרְתֶּם אֶת הַיַּרְדֵּן וִישַׁבְתֶּם בָּאָרֶץ אֲשֶׁר ידוד אֱלֹהֵיכֶם מַנְחִיל אֶתְכֶם וְהֵנִיחַ לָכֶם מִכָּל אֹיְבֵיכֶם מִסָּבִיב וִישַׁבְתֶּם בֶּטַח" - בלי לחשוש מאויבים;
  • דברים לג12: "לְבִנְיָמִן אָמַר: יְדִיד ה' יִשְׁכֹּן לָבֶטַח עָלָיו חֹפֵף עָלָיו כָּל הַיּוֹם וּבֵין כתפיו שָׁכֵן"
  • דברים לג28: "וַיִּשְׁכֹּן יִשְׂרָאֵל בֶּטַח, בָּדָד עֵין יַעֲקֹב, אֶל אֶרֶץ דָּגָן וְתִירוֹשׁ, אַף שָׁמָיו יַעַרְפוּ טָל"
  • שופטים ח11: "וַיַּעַל גִּדְעוֹן... וַיַּךְ אֶת הַמַּחֲנֶה, וְהַמַּחֲנֶה הָיָה בֶטַח" - מחנה המדיינים היה רגוע ושאנן, ולא חשש מפני גדעון.
  • שופטים יח7: "וַיֵּלְכוּ חֲמֵשֶׁת הָאֲנָשִׁים וַיָּבֹאוּ לָיְשָׁה וַיִּרְאוּ אֶת הָעָם אֲשֶׁר בְּקִרְבָּהּ יוֹשֶׁבֶת לָבֶטַח כְּמִשְׁפַּט צִדֹנִים שֹׁקֵט וּבֹטֵחַ וְאֵין מַכְלִים דָּבָר בָּאָרֶץ יוֹרֵשׁ עֶצֶר וּרְחֹקִים הֵמָּה מִצִּדֹנִים וְדָבָר אֵין לָהֶם עִם אָדָם"
  • שמואל א יב11: "... ויצל אתכם מיד איביכם מסביב ותשבו בטח"
  • מלכים א ה5: "וַיֵּשֶׁב יְהוּדָה וְיִשְׂרָאֵל ל בטח אִישׁ תַּחַת גַּפְנוֹ וְתַחַת תְּאֵנָתוֹ מִדָּן וְעַד בְּאֵר שָׁבַע כֹּל יְמֵי שְׁלֹמֹה"
  • מלכים ב יח19: "וַיֹּאמֶר אֲלֵהֶם רַב שָׁקֵה: אִמְרוּ נָא אֶל חִזְקִיָּהוּ כֹּה אָמַר הַמֶּלֶךְ הַגָּדוֹל מֶלֶךְ אַשּׁוּר: מָה הַבִּטָּחוֹן הַזֶּה אֲשֶׁר בָּטָחְתָּ?" - מדוע אתה שאנן וחושב שלא נצליח לפגוע בך?
  • ישעיהו יד30: "וְרָעוּ בְּכוֹרֵי דַלִּים וְאֶבְיוֹנִים לָבֶטַח יִרְבָּצוּ וְהֵמַתִּי בָרָעָב שָׁרְשֵׁךְ וּשְׁאֵרִיתֵךְ יַהֲרֹג"
  • משלי א33: "וְשֹׁמֵעַ לִי יִשְׁכָּן בֶּטַח, וְשַׁאֲנַן מִפַּחַד רָעָה"
  • משלי ג23: "אָז תֵּלֵךְ לָבֶטַח דַּרְכֶּךָ, וְרַגְלְךָ לֹא תִגּוֹף"
  • משלי ג29: "אַל תַּחֲרֹשׁ עַל רֵעֲךָ רָעָה, וְהוּא יוֹשֵׁב לָבֶטַח אִתָּךְ" - לא חושד בך, לא חושש שתפגע בו.
  • משלי י9: "הוֹלֵךְ בַּתֹּם יֵלֶךְ בֶּטַח, וּמְעַקֵּשׁ דְּרָכָיו יִוָּדֵעַ"
  • משלי יא15: "רַע יֵרוֹעַ כִּי עָרַב זָר, וְשֹׂנֵא תֹקְעִים בּוֹטֵחַ" - השונא לתקוע כף ולהיות ערֵב על הלוואה, יחיה בשלוה וללא פחד מהפסד פתאומי.
  • משלי כח1: "נָסוּ וְאֵין רֹדֵף רָשָׁע, וְצַדִּיקִים כִּכְפִיר יִבְטָח" - הצדיקים לא יפחדו.
2. אמונה בכוחו של מישהו או משהו לעזור ולהגן:
  • דברים כח52: "וְהֵצַר לְךָ בְּכָל שְׁעָרֶיךָ עַד רֶדֶת חֹמֹתֶיךָ הגבהת וְהַבְּצֻרוֹת אֲשֶׁר אַתָּה בֹּטֵחַ בָּהֵן בְּכָל אַרְצֶךָ..."
  • שופטים ט26: "וַיָּבֹא גַּעַל בֶּן עֶבֶד וְאֶחָיו וַיַּעַבְרוּ בִּשְׁכֶם וַיִּבְטְחוּ בוֹ בַּעֲלֵי שְׁכֶם" - האמינו שהוא יצליח להגן עליהם מפני אבימלך;
  • שופטים כ36: "וַיִּרְאוּ בְנֵי בִנְיָמִן כִּי נִגָּפוּ, וַיִּתְּנוּ אִישׁ יִשְׂרָאֵל מָקוֹם לְבִנְיָמִן, כִּי בָטְחוּ אֶל הָאֹרֵב אֲשֶׁר שָׂמוּ אֶל הַגִּבְעָה"
  • מלכים ב יח21: "עַתָּה הִנֵּה בָטַחְתָּ לְּךָ עַל מִשְׁעֶנֶת הַקָּנֶה הָרָצוּץ הַזֶּה עַל מִצְרַיִם אֲשֶׁר יִסָּמֵךְ אִישׁ עָלָיו וּבָא בְכַפּוֹ וּנְקָבָהּ כֵּן פַּרְעֹה מֶלֶךְ מִצְרַיִם לְכָל הַבֹּטְחִים עָלָיו"
  • משלי יד16: "חָכָם יָרֵא וְסָר מֵרָע, וּכְסִיל מִתְעַבֵּר וּבוֹטֵחַ", משלי כח26: "בּוֹטֵחַ בְּלִבּוֹ הוּא כְסִיל, וְהוֹלֵךְ בְּחָכְמָה הוּא יִמָּלֵט" - הכסיל מאמין בכוחה של תבונתו, ואינו נזהר;
  • משלי לא11: "בָּטַח בָּהּ לֵב בַּעְלָהּ, וְשָׁלָל לֹא יֶחְסָר" - האמין שהיא תצליח לפרנס אותו, בזמן שהוא יושב בשער העיר כשופט (פירוט).
  • ירמיהו יז7: "בָּרוּךְ הַגֶּבֶר אֲשֶׁר יִבְטַח בה', וְהָיָה ה' מִבְטַחוֹ" (פירוט).


כך גם יש לפרש את הפסוקים המדברים על בטחון בה' - אמונה שה' יעזור ויגן:
  • מלכים ב יח5: "בה' אֱלֹהֵי יִשְׂרָאֵל בָּטָח וְאַחֲרָיו לֹא הָיָה כָמֹהוּ בְּכֹל מַלְכֵי יְהוּדָה וַאֲשֶׁר הָיוּ לְפָנָיו" - האמין שה' יעזור לו במרד נגד אשור;
  • מלכים ב יח22: "וְכִי תֹאמְרוּן אֵלַי 'אֶל ה' אֱלֹהֵינוּ בָּטָחְנוּ'..." - אנחנו מאמינים שה' יגן עלינו מפני אשור; וכן
  • מלכים ב יט10: "כֹּה תֹאמְרוּן אַל חִזְקִיָּהוּ מֶלֶךְ יְהוּדָה לֵאמֹר אֶל יַשִּׁאֲךָ אֱלֹהֶיךָ אֲשֶׁר אַתָּה בֹּטֵחַ בּוֹ לֵאמֹר לֹא תִנָּתֵן יְרוּשָׁלִַם בְּיַד מֶלֶךְ אַשּׁוּר"
  • משלי ג5: "בטח אל ה' בכל לבך ואל בינתך אל תשען" (פירוט)
  • משלי טז20: "מַשְׂכִּיל עַל דָּבָר יִמְצָא טוֹב, וּבוֹטֵחַ בה' אַשְׁרָיו"
  • משלי כח25: "רְחַב נֶפֶשׁ יְגָרֶה מָדוֹן, וּבוֹטֵחַ עַל ה' יְדֻשָּׁן"
  • משלי כט25: "חֶרְדַּת אָדָם יִתֵּן מוֹקֵשׁ, וּבוֹטֵחַ בה' יְשֻׂגָּב"
ראו גם: מבטח = דבר שבוטחים בו = דבר שמאמינים בו שיעזור ויגן.

משמעות מוחשית - הישענות

לפי חלק מהפירושים, המשמעות המקורית של השורש "בטח" היא מוחשית - "נשען", ומשמעות זו קרובה למשמעות "נפל", כי הנשען על דבר אחר - יכול ליפול בקלות כאשר הדבר שהוא נשען עליו נופל; ראו:
  • משלי יא28: "בּוֹטֵחַ בְּעָשְׁרוֹ הוּא יִפֹּל, וְכֶעָלֶה צַדִּיקִים יִפְרָחוּ": ניגוד בין מי שנשען על עושרו כעל גזע, לבין מי שמתייחס אל עושרו כאל עָלֶה (פירוט)
  • ירמיהו יב5: "... ובארץ שלום אתה בוטח, ואיך תעשה בגאון הירדן?" - אם בארץ שקטה ושלוה נפלת כשהתברר לך שמי שנשענת עליו בגד בך, מה תעשה כשתגיע למקום סוער ומסוכן?

ראו גם: השורש שען - משענת חומרית ומשענת רוחנית.
