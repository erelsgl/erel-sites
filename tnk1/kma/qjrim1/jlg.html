ראשי>הקשר לפרשה>תוכן>משל ונמשל>

שלג כמשל

קוד: שלג= בתנ"ך

סוג: משל

מאת: אראל

אל:

[מאמר לימי החורף] שלג כפשוטו נזכר כמה פעמים בתנ"ך:
  • שמואל ב כג20: "וּבְנָיָהוּ בֶּן יְהוֹיָדָע בֶן אִישׁ חירַב פְּעָלִים מִקַּבְצְאֵל הוּא הִכָּה אֶת שְׁנֵי אֲרִאֵל מוֹאָב וְהוּא יָרַד וְהִכָּה אֵת [הָאֲרִי]הבְּתוֹךְ הַבֹּאר בְּיוֹם הַשָּׁלֶג"
אולם בפסוקים רבים, שלג הוא משל:

שלג = לבן

שלג מסמל צבע לבן - לבן חולה כמו של צרעת:
  • שמות ד6: "וַיֹּאמֶר ה' לוֹ עוֹד 'הָבֵא נָא יָדְךָ בְּחֵיקֶךָ', וַיָּבֵא יָדוֹ בְּחֵיקוֹ וַיּוֹצִאָהּ וְהִנֵּה יָדוֹ מְצֹרַעַת כַּשָּׁלֶג"
  • במדבר יב10: "וְהֶעָנָן סָר מֵעַל הָאֹהֶל, וְהִנֵּה מִרְיָם מְצֹרַעַת כַּשָּׁלֶג, וַיִּפֶן אַהֲרֹן אֶל מִרְיָם וְהִנֵּה מְצֹרָעַת"
  • מלכים ב ה27: "וְצָרַעַת נַעֲמָן תִּדְבַּק בְּךָ וּבְזַרְעֲךָ לְעוֹלָם; וַיֵּצֵא מִלְּפָנָיו מְצֹרָע כַּשָּׁלֶג"
או לבן בריא כמו של נזיר:
  • איכה ד7: "זַכּוּ נְזִירֶיהָ מִשֶּׁלֶג, צַחוּ מֵחָלָב, אָדְמוּ עֶצֶם מִפְּנִינִים, סַפִּיר גִּזְרָתָם" (פירוט)
וכן את לבושו של ה' בחזון דניאל:
  • דניאל ז9: "חָזֵה הֲוֵית עַד דִּי כָרְסָוָן רְמִיו וְעַתִּיק יוֹמִין יְתִב, לְבוּשֵׁהּ כִּתְלַג חִוָּר וּשְׂעַר רֵאשֵׁהּ כַּעֲמַר נְקֵא כָּרְסְיֵהּ שְׁבִיבִין דִּי נוּר גַּלְגִּלּוֹהִי נוּר דָּלִק" (פירוט)

שלג = טהור

הלובן של השלג הוא גם משל לטהרה ונקיון מעבירות:
  • ישעיהו א18: "לְכוּ נָא וְנִוָּכְחָה יֹאמַר ה': אִם יִהְיוּ חֲטָאֵיכֶם כַּשָּׁנִים - כַּשֶּׁלֶג יַלְבִּינוּ, אִם יַאְדִּימוּ כַתּוֹלָע - כַּצֶּמֶר יִהְיוּ" (פירוט).
  • תהלים נא9: "תְּחַטְּאֵנִי בְאֵזוֹב וְאֶטְהָר, תְּכַבְּסֵנִי וּמִשֶּׁלֶג אַלְבִּין"
המשל הוא כפול, כי השלג גם עשוי ממים, וניתן לטבול ולהיטהר במי שלג:
  • איוב ט30: "אִם הִתְרָחַצְתִּי במושָׁלֶג, וַהֲזִכּוֹתִי בְּבֹר כַּפָּי"

שלג = קר

שלג מסמל גם קור, וזה יכול להיות לטובה:
  • משלי כה13: "כְּצִנַּת שֶׁלֶג בְּיוֹם קָצִיר - צִיר נֶאֱמָן לְשֹׁלְחָיו, וְנֶפֶשׁ אֲדֹנָיו יָשִׁיב" (פירוט)
או לרעה:
  • משלי לא21: "לֹא תִירָא לְבֵיתָהּ מִשָּׁלֶג, כִּי כָל בֵּיתָהּ לָבֻשׁ שָׁנִים" (פירוט)
  • משלי כו1: "כַּשֶּׁלֶג בַּקַּיִץ וְכַמָּטָר בַּקָּצִיר - כֵּן לֹא נָאוֶה לִכְסִיל כָּבוֹד" (פירוט)

שלג = מעשה ה'

בכמה פסוקים, שלג מסמל את עוצמתו וחכמתו של הבורא:
  • ישעיהו נה10-11: "כִּי כַּאֲשֶׁר יֵרֵד הַגֶּשֶׁם וְהַשֶּׁלֶג מִן הַשָּׁמַיִם וְשָׁמָּה לֹא יָשׁוּב כִּי אִם הִרְוָה אֶת הָאָרֶץ וְהוֹלִידָהּ וְהִצְמִיחָהּ וְנָתַן זֶרַע לַזֹּרֵעַ וְלֶחֶם לָאֹכֵל - כֵּן יִהְיֶה דְבָרִי אֲשֶׁר יֵצֵא מִפִּי, לֹא יָשׁוּב אֵלַי רֵיקָם כִּי אִם עָשָׂה אֶת אֲשֶׁר חָפַצְתִּי וְהִצְלִיחַ אֲשֶׁר שְׁלַחְתִּיו" (פירוט).
  • תהלים קמח8: "אֵשׁ וּבָרָד, שֶׁלֶג וְקִיטוֹר, רוּחַ סְעָרָה עֹשָׂה דְבָרוֹ"
  • תהלים קמז16-17: "הַנֹּתֵן שֶׁלֶג כַּצָּמֶר, כְּפוֹר כָּאֵפֶר יְפַזֵּר. מַשְׁלִיךְ קַרְחוֹ כְפִתִּים, לִפְנֵי קָרָתוֹ מִי יַעֲמֹד"
  • איוב לז6: "כִּי לַשֶּׁלַג יֹאמַר 'הֱוֵא אָרֶץ', וְגֶשֶׁם מָטָר, וְגֶשֶׁם מִטְרוֹת עֻזּוֹ"
  • איוב לח22: "הֲבָאתָ אֶל אֹצְרוֹת שָׁלֶג, וְאֹצְרוֹת בָּרָד תִּרְאֶה?"

פסוקים שעדיין לא הבנתי

  • ירמיהו יח14: "הֲיַעֲזֹב מִצּוּר שָׂדַי שֶׁלֶג לְבָנוֹן? אִם יִנָּתְשׁוּ מַיִם זָרִים קָרִים נוֹזְלִים?"
  • תהלים סח15: "בְּפָרֵשׂ שַׁדַּי מְלָכִים בָּהּ תַּשְׁלֵג בְּצַלְמוֹן"
  • איוב ו16: "הַקֹּדְרִים מִנִּי קָרַח, עָלֵימוֹ יִתְעַלֶּם שָׁלֶג"
  • איוב כד19: "צִיָּה גַם חֹם יִגְזְלוּ מֵימֵי שֶׁלֶג שְׁאוֹל חָטָאוּ"
לעיון נוסף:
  • שלג = אשלגן, שימש להכנת סבון / אביתר כהן.
  • על משקעים ואנשים בספר משלי / אראל סגל.
