ראשי>לשון המקרא>א?ד>מילים שמשמעותן השתנתה>

קרקע = תחתית הבית

קוד: קרקע= בתנ"ך

סוג: הגדרה

מאת: אראל

אל:

בלשון התלמוד ובלשון ימינו, קרקע היא חלקת-אדמה, נחלה, שדה. אולם בלשון המקרא משמעות המילה שונה במקצת - היא מציינת תמיד את התחתית, שאנחנו קוראים לה "רִצְפָּה":
  • רצפת המשכן, במדבר ה17: "וְלָקַח הַכֹּהֵן מַיִם קְדֹשִׁים בִּכְלִי חָרֶשׂ, וּמִן הֶעָפָר אֲשֶׁר יִהְיֶה בְּקַרְקַע הַמִּשְׁכָּן יִקַּח הַכֹּהֵן וְנָתַן אֶל הַמָּיִם"
  • רצפת המקדש, מלכים א ו15-16: "וַיִּבֶן אֶת קִירוֹת הַבַּיִת מִבַּיְתָה בְּצַלְעוֹת אֲרָזִים, מִקַּרְקַע הַבַּיִת עַד קִירוֹת הַסִּפֻּן צִפָּה עֵץ מִבַּיִת, וַיְצַף אֶת קַרְקַע הַבַּיִת בְּצַלְעוֹת בְּרוֹשִׁים. וַיִּבֶן אֶת עֶשְׂרִים אַמָּה מירכותי[מִיַּרְכְּתֵי] הַבַּיִת בְּצַלְעוֹת אֲרָזִים מִן הַקַּרְקַע עַד הַקִּירוֹת וַיִּבֶן לוֹ מִבָּיִת לִדְבִיר לְקֹדֶשׁ הַקֳּדָשִׁים", מלכים א ו30: "וְאֶת קַרְקַע הַבַּיִת צִפָּה זָהָב לִפְנִימָה וְלַחִיצוֹן"
  • רצפת בית המשפט העליון, מלכים א ז7: "וְאוּלָם הַכִּסֵּא אֲשֶׁר יִשְׁפָּט שָׁם אֻלָם הַמִּשְׁפָּט עָשָׂה וְסָפוּן בָּאֶרֶז מֵהַקַּרְקַע עַד הַקַּרְקָע"
  • ורצפת הים, עמוס ט3: "וְאִם יֵחָבְאוּ בְּרֹאשׁ הַכַּרְמֶל - מִשָּׁם אֲחַפֵּשׂ וּלְקַחְתִּים, וְאִם יִסָּתְרוּ מִנֶּגֶד עֵינַי בְּקַרְקַע הַיָּם מִשָּׁם אֲצַוֶּה אֶת הַנָּחָשׁ וּנְשָׁכָם"
  • (יש גם מקום בארץ ישראל, שנקרא "קרקע", יהושע טו3: "וְיָצָא אֶל מִנֶּגֶב לְמַעֲלֵה עַקְרַבִּים וְעָבַר צִנָה וְעָלָה מִנֶּגֶב לְקָדֵשׁ בַּרְנֵעַ וְעָבַר חֶצְרוֹן וְעָלָה אַדָּרָה וְנָסַב הַקַּרְקָעָה").

לעומת זאת, המילה רצפה בלשון המקרא מציינת את הציפוי לתחתית - הריצוף, יחזקאל מ17: "וַיְבִיאֵנִי אֶל הֶחָצֵר הַחִיצוֹנָה, וְהִנֵּה לְשָׁכוֹת, וְרִצְפָה עָשׂוּי לֶחָצֵר סָבִיב סָבִיב, שְׁלֹשִׁים לְשָׁכוֹת אֶל הָרִצְפָה".

יש כאן תופעה מעניינת - שינוי-משמעות אחד גורר שינוי-משמעות אחר: שדה הפך לקרקע, קרקע הפכה לרצפה, רצפה הפכה לריצוף.
