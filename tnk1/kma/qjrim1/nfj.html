נפש = הצד החומרי של החיים

קוד: נפש= בתנ"ך

סוג: הגדרה

מאת: אראל

אל: סגלות משלי

המושג נפש מציין את החיים, כמו למשל בפסוק:
  • בראשית יב 13: "אִמְרִי נָא אֲחֹתִי אָתְּ, לְמַעַן יִיטַב לִי בַעֲבוּרֵךְ וְחָיְתָה נַפְשִׁי בִּגְלָלֵךְ".

אולם, בניגוד למושגים אחרים שמציינים חיים, המושג נפש מציין את הצד החומרי והגופני יותר של החיים - הדבר שמת כאשר הורגים את הגוף (ראו גם נפש לעומת רוח). המושג נפש מופיע בביטויים רבים, למשל:

  • יציאת הנפש = מוות, בראשית לה 18: "וַיְהִי בְּצֵאת נַפְשָׁהּ כִּי מֵתָה".
  • בקשת נפש הזולת = רצון להרגו; וכן משלי א 18: "וְהֵם לְדָמָם יֶאֱרֹבוּ, יִצְפְּנוּ לְנַפְשֹׁתָם".
  • לקיחת נפש הזולת = הריגתו.
  • נפש = גוף חי, בראשית יד 21: "וַיֹּאמֶר מֶלֶךְ סְדֹם אֶל אַבְרָם 'תֶּן לִי הַנֶּפֶשׁ והרכוש קַח לָךְ'".
  • בכל הנפש = עד מוות, דברים ו5: "וְאָהַבְתָּ אֵת ה' אֱלֹהֶיךָ בְּכָל לְבָבְךָ וּבְכָל נַפְשְׁךָ וּבְכָל מְאֹדֶךָ", "אפילו הוא נוטל את חייך".
  • כופר נפש = כסף שהאדם נותן כדי לכפר על חייו ולהינצל ממוות, שמות ל12: "כִּי תִשָּׂא אֶת רֹאשׁ בְּנֵי יִשְׂרָאֵל לִפְקֻדֵיהֶם וְנָתְנוּ אִישׁ כֹּפֶר נַפְשׁוֹ לה' בִּפְקֹד אֹתָם וְלֹא יִהְיֶה בָהֶם נֶגֶף בִּפְקֹד אֹתָם" (פירוט), משלי יג8: "כֹּפֶר נֶפֶשׁ אִישׁ עָשְׁרוֹ וְרָשׁ לֹא שָׁמַע גְּעָרָה", (פירוט), במדבר לה31: "וְלֹא תִקְחוּ כֹפֶר לְנֶפֶשׁ רֹצֵחַ אֲשֶׁר הוּא רָשָׁע לָמוּת כִּי מוֹת יוּמָת", שמות כא30: "אִם כֹּפֶר יוּשַׁת עָלָיו וְנָתַן פִּדְיֹן נַפְשׁוֹ כְּכֹל אֲשֶׁר יוּשַׁת עָלָיו".
  • שומר נפשו = מגן על חייו, משלי יג3: "נֹצֵר פִּיו שֹׁמֵר נַפְשׁוֹ", משלי טז17: "שֹׁמֵר נַפְשׁוֹ נֹצֵר דַּרְכּוֹ" (פירוט), משלי יט16: "שֹׁמֵר מִצְוָה שֹׁמֵר נַפְשׁוֹ" (פירוט), משלי כא23: "שֹׁמֵר פִּיו וּלְשׁוֹנוֹ שֹׁמֵר מִצָּרוֹת נַפְשׁוֹ", משלי כב5: "צִנִּים פַּחִים בְּדֶרֶךְ עִקֵּשׁ שׁוֹמֵר נַפְשׁוֹ יִרְחַק מֵהֶם" (פירוט).
  • תאוות נפש = רצון לאכול ולשתות, דברים יב 20: "כִּי תְאַוֶּה נַפְשְׁךָ לֶאֱכֹל בָּשָׂר בְּכָל אַוַּת נַפְשְׁךָ תֹּאכַל בָּשָׂר"; או רצון חומרי-בהמי כלשהו, משלי כא10: "נֶפֶשׁ רָשָׁע אִוְּתָה רָע, לֹא יֻחַן בְּעֵינָיו רֵעֵהוּ" (פירוט), משלי יג19: "תַּאֲוָה נִהְיָה תֶּעֱרַב לְנָפֶשׁ וְתוֹעֲבַת כְּסִילִים סוּר מֵרָע" (פירוט).
  • בעל נפש = בעל תאווה לאכול ולשתות, משלי כג 2: "ושמת שכין בלעך, אם בעל נפש אתה" - אם יש לך תאווה, שים גדר בתוך פיך כדי שלא תאכל יותר מדי (פירוט).
  • נפש עמלה = תאוה ש"עובדת" על האדם, מרמה אותו וגורמת לו לקנות מעבר לצרכיו, משלי טז 26: " נֶפֶשׁ עָמֵל עָמְלָה לּוֹ, כִּי אָכַף עָלָיו פִּיהוּ" (פירוט).
  • רחב נפש = בעל תאוה חומרית גדולה, משלי כח 25: "רְחַב נֶפֶשׁ יְגָרֶה מָדוֹן, וּבוֹטֵחַ עַל ה' יְדֻשָּׁן" - מי שיש לו רצון גדול ורחב לתענוגות חומריים, מסתכסך הרבה במריבות כאשר אינו מקבל את מה שהוא רוצה (פירוט).
  • עינוי נפש = צום וסיגופים גופניים, שלילת הצרכים החומריים של האדם.
  • גומל נפשו = מספק את הצרכים החומריים של עצמו, משלי יא17: "גֹּמֵל נַפְשׁוֹ אִישׁ חָסֶד, וְעֹכֵר שְׁאֵרוֹ אַכְזָרִי" (פירוט).
  • נפש בהמה = החיים או הצרכים החומריים של הבהמה, משלי יב10: "יוֹדֵעַ צַדִּיק נֶפֶשׁ בְּהֶמְתּוֹ, וְרַחֲמֵי רְשָׁעִים אַכְזָרִי" (פירוט), ויש אומרים שהכוונה לצרכים הבהמיים של האדם - "הנפש הבהמית".

החיים כמשל

ישנם ביטויים שבהם המושג נפש מציין דבר שהוא חשוב כמו החיים, או אפילו גדול מהחיים:
  • אהבת נפש = אהבה גדולה כמו שאדם אוהב את חייו, שמואל א כ17: "וַיּוֹסֶף יְהוֹנָתָן לְהַשְׁבִּיעַ אֶת דָּוִד בְּאַהֲבָתוֹ אֹתוֹ כִּי אַהֲבַת נַפְשׁוֹ אֲהֵבוֹ", שיר השירים ג1: "עַל מִשְׁכָּבִי בַּלֵּילוֹת בִּקַּשְׁתִּי אֵת שֶׁאָהֲבָה נַפְשִׁי בִּקַּשְׁתִּיו וְלֹא מְצָאֵתִיו".
  • נשיאת נפש = שאיפה גדולה מהחיים, שאיפה שמוכנים לסכן את החיים עבורה, דברים כד15: "כי עני הוא ואליו הוא נושא את נפשו".
  • קיצור נפש = צער שמקצר את החיים,  במדבר כא4: "וַיִּסְעוּ מֵהֹר הָהָר דֶּרֶךְ יַם סוּף לִסְבֹב אֶת אֶרֶץ אֱדוֹם, וַתִּקְצַר נֶפֶשׁ הָעָם בַּדָּרֶךְ"
  • מרירות נפש  = צער שגורם לחיים עצמם להיראות מרים ורעים, משלי לא6-7: "תְּנוּ שֵׁכָר לְאוֹבֵד, וְיַיִן לְמָרֵי נָפֶשׁ. יִשְׁתֶּה וְיִשְׁכַּח רִישׁוֹ, וַעֲמָלוֹ לֹא יִזְכָּר עוֹד" - יש לתת משקאות משכרים רק לאבלים, או לאנשים שצערם גדול כל-כך עד שהם עומדים להתאבד (פירוט).
  • חי נפשך = הדבר נכון ואמיתי כמו שחייך אמיתיים, שמואל א א26: "וַתֹּאמֶר 'בִּי אֲדֹנִי, חֵי נַפְשְׁךָ אֲדֹנִי, אֲנִי הָאִשָּׁה הַנִּצֶּבֶת עִמְּכָה בָּזֶה לְהִתְפַּלֵּל אֶל ה'".
  • אם יש את נפשך = רק אם אתה באמת רוצה, "רק אם בא לך".
  • ראו גם ביטויים שקשורים למסירות נפש.

פסוקים נוספים

א. נפש = החיים; פגיעה בנפש = הריגה

בראשית ט 4-5: "אַךְ בָּשָׂר בְּנַפְשׁוֹ דָמוֹ לֹא תֹאכֵלוּ. וְאַךְ אֶת דִּמְכֶם לְנַפְשֹׁתֵיכֶם אֶדְרֹשׁ מִיַּד כָּל חַיָּה אֶדְרְשֶׁנּוּ וּמִיַּד הָאָדָם מִיַּד אִישׁ אָחִיו אֶדְרֹשׁ אֶת נֶפֶשׁ הָאָדָם"

בראשית יט 17: "וַיְהִי כְהוֹצִיאָם אֹתָם הַחוּצָה וַיֹּאמֶר הִמָּלֵט עַל נַפְשֶׁךָ אַל תַּבִּיט אַחֲרֶיךָ... פֶּן תִּסָּפֶה... ותגדל חסדך אשר עשית עמדי להחיות את נפשי... אמלטה נא שמה הלא מצער הוא ותחי נפשי"

בראשית לב 31: "וַיִּקְרָא יַעֲקֹב שֵׁם הַמָּקוֹם פְּנִיאֵל כִּי רָאִיתִי אֱלֹהִים פָּנִים אֶל פָּנִים וַתִּנָּצֵל נַפְשִׁי"

בראשית לז 21: "וַיִּשְׁמַע רְאוּבֵן וַיַּצִּלֵהוּ מִיָּדָם וַיֹּאמֶר לֹא נַכֶּנּוּ נָפֶשׁ"

שמות כא 23: "וְאִם אָסוֹן יִהְיֶה וְנָתַתָּה נֶפֶשׁ תַּחַת נָפֶשׁ", וכן ויקרא כד, דברים יט 21

ויקרא כד 17: "וְאִישׁ כִּי יַכֶּה כָּל נֶפֶשׁ אָדָם מוֹת יוּמָת. ומכה נֶפֶשׁ בהמה ישלמנה נֶפֶשׁ תחת נֶפֶשׁ",

ויקרא יט 28: "וְשֶׂרֶט לָנֶפֶשׁ לֹא תִתְּנוּ בִּבְשַׂרְכֶם" - סריטה על אבדן החיים, לאות אבל.

ויקרא כא - כב: "ויאמר ה'אל משה אמר אל הכהנים בני אהרן ואמרת אלהם לנפש לא יטמא בעמיו... ועל כל נפשת מת לא יבא לאביו ולאמו לא יטמא... איש איש מזרע אהרן והוא צרוע או זב בקדשים לא יאכל עד אשר יטהר והנגע בכל טמא נפש או איש אשר תצא ממנו שכבת זרע... נפש אשר תגע בו וטמאה עד הערב ולא יאכל מן הקדשים כי אם רחץ בשרו במים... וכהן כי יקנה נפש קנין כספו הוא יאכל בו ויליד ביתו הם יאכלו בלחמו": טומאה לנפש = טומאה שנובעת מאבדן החיים.

במדבר יז 3: "אֵת מַחְתּוֹת הַחַטָּאִים הָאֵלֶּה בְּנַפְשֹׁתָם וְעָשׂוּ אֹתָם רִקֻּעֵי פַחִים צִפּוּי לַמִּזְבֵּחַ כִּי הִקְרִיבֻם לִפְנֵי ה'וַיִּקְדָּשׁוּ וְיִהְיוּ לְאוֹת לִבְנֵי יִשְׂרָאֵל" - הנשיאים, מקטירי הקטורת, חטאו גם בכך שגרמו לעצמם למות.

במדבר יט 11: "הַנֹּגֵעַ בְּמֵת לְכָל נֶפֶשׁ אָדָם וְטָמֵא שִׁבְעַת יָמִים"

במדבר לה 11: "וְהִקְרִיתֶם לָכֶם עָרִים עָרֵי מִקְלָט תִּהְיֶינָה לָכֶם וְנָס שָׁמָּה רֹצֵחַ מַכֵּה נֶפֶשׁ בִּשְׁגָגָה"

דברים ד 15: "וְנִשְׁמַרְתֶּם מְאֹד לְנַפְשֹׁתֵיכֶם (כִּי לֹא רְאִיתֶם כָּל תְּמוּנָה בְּיוֹם דִּבֶּר ה'אֲלֵיכֶם בְּחֹרֵב מִתּוֹךְ הָאֵשׁ), פן תשחיתון ועשיתם לכם פסל" (וכן יהושע כג 11) - פירוש הפסוק הוא " שמרו על חייכם ואל תעשו פסל "; וחז " ל למדו מהביטוי " ונשמרתם מאד לנפשותיכם ", שהאדם צריך בדרך - כלל לשמור על חייו.

דברים כב 26: "ולנערלֹא תַעֲשֶׂה דָבָר אֵין לנער חֵטְא מָוֶת כִּי כַּאֲשֶׁר יָקוּם אִישׁ עַל רֵעֵהוּ וּרְצָחוֹ נֶפֶשׁ כֵּן הַדָּבָר הַזֶּה"

יהושע ב 13: "וְהַחֲיִתֶם אֶת אָבִי וְאֶת אִמִּי וְאֵת אַחַי וְאֶת אחותי[אַחְיוֹתַי] וְאֶת כָּל אֲשֶׁר לָהֶם וְהִצַּלְתֶּם אֶת נַפְשֹׁתֵינוּ מִמָּוֶת. ויאמרו לה האנשים נפשנו תחתיכם למות אם לא תגידו אֶת דברנו זה והיה בתת ה'לנו אֶת הארץ ועשינו עמך חסד ואמת"

שמות יב 16: "אַךְ אֲשֶׁר יֵאָכֵל לְכָל נֶפֶשׁ הוּא לְבַדּוֹ יֵעָשֶׂה לָכֶם", ויקרא ז 18: "וְאִם הֵאָכֹל יֵאָכֵל מִבְּשַׂר זֶבַח שְׁלָמָיו בַּיּוֹם הַשְּׁלִישִׁי לֹא יֵרָצֶה הַמַּקְרִיב אֹתוֹ לֹא יֵחָשֵׁב לוֹ פִּגּוּל יִהְיֶה וְהַנֶּפֶשׁ הָאֹכֶלֶת מִמֶּנּוּ עונה תִּשָּׂא... וְהַנֶּפֶשׁ אשר תאכל בשר מזֶבַח השלמים אשר לה'וטמאֹתוֹ עליו ונכרתה הנפש ההוא מעמיה... ונפש כי תגע בכל טמא בטמאת אדם או בבהמה טמאה או בכל שקץ טמא ואכל מִבְּשַׂר זֶבַח השלמים אשר לה'ונכרתה הנפש ההוא מעמיה" - הנפש היא האוכלת.

ויקרא יא 43: "אַל תְּשַׁקְּצוּ אֶת נַפְשֹׁתֵיכֶם בְּכָל הַשֹּׁרֵץ הַשֶּׁרֶץ וְלֹא תִטַּמְּאוּ בָּהֶם וְנִטְמֵתֶם בָּם. כי אני ה'אַלהיכם והתקדשתם והייתם קדשים כי קדוש אני וְלֹא תִטַּמְּאוּ אֶת נַפְשֹׁתֵיכֶם בְּכָל הַשֹּׁרֵץ הרמש על הארץ" - הנפש היא הרצון לאכול, וה ' מזהיר אותנו שלא לשקץ ולטמא רצון זה באכילת שרצים.

ו. פסוקים שיכולים להתאים לכמה תת - משמעויות שונות

עונש הכרתת הנפש (“ ונכרתה הנפש ההיא מעמיה / מלפניי”) - לפי חלק מהדעות, נפש = חיים, והכוונה לקטיעת החיים החומריים; לפי דעות אחרות, נפש = יצור חי, והכוונה לקטיעת האדם ממשפחתו בעולם הזה או בעולם הבא. לפירוט ע ' כאן.

בראשית מד 30: "וְעַתָּה כְּבֹאִי אֶל עַבְדְּךָ אָבִי וְהַנַּעַר אֵינֶנּוּ אִתָּנוּ - וְנַפְשׁוֹ קְשׁוּרָה בְנַפְשׁוֹ" - לפי הפירוש נפש = חיים: החיים של יעקב קשורים בחייו של בנימין: אם בנימין ימות, גם יעקב ימות (כמו שכתוב בהמשך " והיה בראותו כי אין הנער - ומת "). ולפי הפירוש נפש = רגש חזק: ליעקב יש רגשות אהבה חזקים מאד הקשורים לבנימין, ולכן לא יוכל לחיות אם יקרה לו דבר רע.
