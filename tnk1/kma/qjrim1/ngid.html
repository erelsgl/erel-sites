ראשי>הגדרות מושגים>

נגיד = מנהיג צבאי או רועה?

קוד: נגיד= בתנ"ך

סוג: הגדרה

מאת: שאלה: אראל; תשובה: אביתר כהן

אל:

נגיד הוא אחד הכינויים של מנהיג (מהשורש נגד, שקשור גם לדיבור - ע"ע קשר בין דיבור להנהגה). אבל מה המשמעות המיוחדת של כינוי זה? מה ההבדל בין נגיד לבין מלך, מושל, שר, וכו'?

מחלק מהפסוקים נראה ש"נגיד" הוא מנהיג צבאי:

  1. שמואל א ט16: "כָּעֵת מָחָר אֶשְׁלַח אֵלֶיךָ אִישׁ מֵאֶרֶץ בִּנְיָמִן וּמְשַׁחְתּוֹ לְנָגִיד עַל עַמִּי יִשְׂרָאֵל וְהוֹשִׁיעַ אֶת עַמִּי מִיַּד פְּלִשְׁתִּים כִּי רָאִיתִי אֶת עַמִּי כִּי בָּאָה צַעֲקָתוֹ אֵלָי... שמואל א י1: "ויקח שמואל אֶת פך השמן ויצק עַל ראשו וישקהו ויאמר הלוא כִּי משחך ה' עַל נחלתו לְנָגִיד" - שאול נמשח לנגיד כדי להילחם בפלשתים;
  2. שמואל א יג14: "וְעַתָּה מַמְלַכְתְּךָ לֹא תָקוּם; בִּקֵּשׁ ה' לוֹ אִישׁ כִּלְבָבוֹ וַיְצַוֵּהוּ ה' לְנָגִיד עַל עַמּוֹ כִּי לֹא שָׁמַרְתָּ אֵת אֲשֶׁר צִוְּךָ ה'" - גם זה נאמר בקשר למלחמה בפלשתים (מלחמת מכמש);
  3. דברי הימים א ה2: "כִּי יְהוּדָה גָּבַר בְּאֶחָיו וּלְנָגִיד מִמֶּנּוּ וְהַבְּכֹרָה לְיוֹסֵף" - יהודה היה גיבור יותר מאחיו, ולכן הנגיד יצא ממנו.
  4. שמואל א כה30: "וְהָיָה כִּי יַעֲשֶׂה ה' לַאדֹנִי כְּכֹל אֲשֶׁר דִּבֶּר אֶת הַטּוֹבָה עָלֶיךָ וְצִוְּךָ לְנָגִיד עַל יִשְׂרָאֵל" - בהמשך היא אומרת "כי מלחמות ה' אדוני נלחם".
  5. שמואל ב ה2: "גַּם אֶתְמוֹל גַּם שִׁלְשׁוֹם בִּהְיוֹת שָׁאוּל מֶלֶךְ עָלֵינוּ אַתָּה [הָיִיתָ] ה מוציא[הַמּוֹצִיא] והמבי[וְהַמֵּבִיא] א אֶת יִשְׂרָאֵל; וַיֹּאמֶר ה' לְךָ אַתָּה תִרְעֶה אֶת עַמִּי אֶת יִשְׂרָאֵל וְאַתָּה תִּהְיֶה לְנָגִיד עַל יִשְׂרָאֵל"
  6. דברי הימים ב לב21: "וַיִּשְׁלַח ידוד מַלְאָךְ וַיַּכְחֵד כָּל גִּבּוֹר חַיִל וְנָגִיד וְשָׂר בְּמַחֲנֵה מֶלֶךְ אַשּׁוּר וַיָּשָׁב בְּבֹשֶׁת פָּנִים לְאַרְצוֹ וַיָּבֹא בֵּית אֱלֹהָיו ומיציאו[וּמִיצִיאֵי] מֵעָיו שָׁם הִפִּילֻהוּ בֶחָרֶב"
מחלק מהפסוקים נראה ש"נגיד" מקביל לרועה:
  1. שמואל ב ה2, דברי הימים א יא2: "גַּם אֶתְּמוֹל גַּם שִׁלְשׁוֹם בִּהְיוֹת שָׁאוּל מֶלֶךְ עַלינו אַתָּה היית הַמּוֹצִיא וְהַמֵּבִיא אֶת יִשְׂרָאֵל; וַיֹּאמֶר ה' לְךָ אַתָּה תִרְעֶה אֶת עַמִּי אֶת יִשְׂרָאֵל וְאַתָּה תִּהְיֶה לנָגִיד עַל יִשְׂרָאֵל", 
  2. שמואל ב ז8, דברי הימים א יז7: "וְעַתָּה כֹּה תֹאמַר לְעַבְדִּי לדוד כֹּה אָמַר ידוד צְבָאוֹת אֲנִי לְקַחְתִּיךָ מִן הַנָּוֶה מאחר הַצֹּאן לִהְיוֹת נָגִיד עַל עַמִּי עַל יִשְׂרָאֵל" - במקום להיות רועה צאן, תהיה נגיד על כל ישראל.
מחלק מהפסוקים נראה שהתואר "נגיד" מדגיש את מעמדו של המנהיג, הנעלה מעל העם:
  1. שמואל ב ו21: "וַיֹּאמֶר דָּוִד אֶל מִיכַל: לִפְנֵי ה' אֲשֶׁר בָּחַר בִּי מֵאָבִיךְ וּמִכָּל בֵּיתוֹ לְצַוֹּת אֹתִי נָגִיד עַל עַם ה' עַל יִשְׂרָאֵל וְשִׂחַקְתִּי לִפְנֵי ה'" - ה' ציוה אותי לנגיד, ולכן כשאני לפני ה' אני לא מתנהג כמו נגיד אלא כמו אחד העם.
  2. מלכים א יד7: "לְכִי אִמְרִי לְיָרָבְעָם כֹּה אָמַר ידוד אֱלֹהֵי יִשְׂרָאֵל יַעַן אֲשֶׁר הֲרִימֹתִיךָ מִתּוֹךְ הָעָם וָאֶתֶּנְךָ נָגִיד עַל עַמִּי יִשְׂרָאֵל"
  3. מלכים א טז2: "יַעַן אֲשֶׁר הֲרִימֹתִיךָ מִן הֶעָפָר וָאֶתֶּנְךָ נָגִיד עַל עַמִּי יִשְׂרָאֵל וַתֵּלֶךְ בְּדֶרֶךְ יָרָבְעָם וַתַּחֲטִא אֶת עַמִּי יִשְׂרָאֵל לְהַכְעִיסֵנִי בְּחַטֹּאֶתָם"
  4. ישעיהו נה4: "הֵן עֵד לְאוּמִּים נְתַתִּיו נָגִיד וּמְצַוֵּה לְאֻמִּים"
  5. יחזקאל כח2: "בֶּן אָדָם אָמַר לִנְגִיד צֹר כֹּה אֱמֹר אֲדֹנָי ידוד יַעַן גָּבַהּ לִבְּךָ וַתֹּאמֶר אֵל אָנִי מוֹשַׁב אֱלֹהִים יָשַׁבְתִּי בְּלֵב יַמִּים וְאַתָּה אָדָם וְלֹא אֵל וַתִּתֵּן לִבְּךָ כְּלֵב אֱלֹהִים"
  6. דברי הימים א ה2: "כִּי יְהוּדָה גָּבַר בְּאֶחָיו וּלְנָגִיד מִמֶּנּוּ וְהַבְּכֹרָה לְיוֹסֵף" - יהודה היה גיבור יותר מאחיו, ולכן הנגיד יצא ממנו.
  7. דברי הימים ב יא22: "וַיַּעֲמֵד לָרֹאשׁ רְחַבְעָם אֶת אֲבִיָּה בֶן מַעֲכָה לְנָגִיד בְּאֶחָיו כִּי לְהַמְלִיכוֹ" - נעלה מעל אחיו
בחלק מהפסוקים, "נגיד" הוא אדם שממונה לתפקיד מסויים, שר או פקיד:
  1. ירמיהו כ1: "וַיִּשְׁמַע פַּשְׁחוּר בֶּן אִמֵּר הַכֹּהֵן וְהוּא פָקִיד נָגִיד בְּבֵית ה' אֶת יִרְמְיָהוּ נִבָּא אֶת הַדְּבָרִים הָאֵלֶּה"
  2. דברי הימים א ט11: "וַעֲזַרְיָה בֶּן חִלְקִיָּה בֶּן מְשֻׁלָּם בֶּן צָדוֹק בֶּן מְרָיוֹת בֶן אֲחִיטוּב נְגִיד בֵּית הָאֱלֹהִים"
  3. דברי הימים א ט20: "וּפִינְחָס בֶּן אֶלְעָזָר נָגִיד הָיָה עֲלֵיהֶם לְפָנִים ה' עִמּוֹ"
  4. דברי הימים א יב28: "וִיהוֹיָדָע הַנָּגִיד לְאַהֲרֹן וְעִמּוֹ שְׁלֹשֶׁת אֲלָפִים וּשְׁבַע מֵאוֹת"
  5. דברי הימים א יג1: "וַיִּוָּעַץ דָּוִיד עִם שָׂרֵי הָאֲלָפִים וְהַמֵּאוֹת לְכָל נָגִיד"
  6. דברי הימים א כו24: "וּשְׁבֻאֵל בֶּן גֵּרְשׁוֹם בֶּן מֹשֶׁה נָגִיד עַל הָאֹצָרוֹת"
  7. דברי הימים א כז4: "וְעַל מַחֲלֹקֶת הַחֹדֶשׁ הַשֵּׁנִי דּוֹדַי הָאֲחוֹחִי וּמַחֲלֻקְתּוֹ וּמִקְלוֹת הַנָּגִיד וְעַל מַחֲלֻקְתּוֹ עֶשְׂרִים וְאַרְבָּעָה אָלֶף"
  8. דברי הימים א כז16: "וְעַל שִׁבְטֵי יִשְׂרָאֵל לָראוּבֵנִי נָגִיד אֱלִיעֶזֶר בֶּן זִכְרִי לַשִּׁמְעוֹנִי שְׁפַטְיָהוּ בֶּן מַעֲכָה"
  9. דברי הימים ב יא11: "וַיְחַזֵּק אֶת הַמְּצֻרוֹת וַיִּתֵּן בָּהֶם נְגִידִים וְאֹצְרוֹת מַאֲכָל וְשֶׁמֶן וָיָיִן"
  10. דברי הימים ב יט11: "וְהִנֵּה אֲמַרְיָהוּ כֹהֵן הָרֹאשׁ עֲלֵיכֶם לְכֹל דְּבַר ידוד וּזְבַדְיָהוּ בֶן יִשְׁמָעֵאל הַנָּגִיד לְבֵית יְהוּדָה לְכֹל דְּבַר הַמֶּלֶךְ וְשֹׁטְרִים הַלְוִיִּם לִפְנֵיכֶם חִזְקוּ וַעֲשׂוּ וִיהִי ידוד עִם הַטּוֹב"
  11. דברי הימים ב כח7: "וַיַּהֲרֹג זִכְרִי גִּבּוֹר אֶפְרַיִם אֶת מַעֲשֵׂיָהוּ בֶּן הַמֶּלֶךְ וְאֶת עַזְרִיקָם נְגִיד הַבָּיִת וְאֶת אֶלְקָנָה מִשְׁנֵה הַמֶּלֶךְ"
  12. דברי הימים ב לא12: "וַיָּבִיאוּ אֶת הַתְּרוּמָה וְהַמַּעֲשֵׂר וְהַקֳּדָשִׁים בֶּאֱמוּנָה וַעֲלֵיהֶם נָגִיד כונניהוהַלֵּוִי וְשִׁמְעִי אָחִיהוּ מִשְׁנֶה"
  13. דברי הימים ב לא13: "וִיחִיאֵל וַעֲזַזְיָהוּ וְנַחַת וַעֲשָׂהאֵל וִירִימוֹת וְיוֹזָבָד וֶאֱלִיאֵל וְיִסְמַכְיָהוּ וּמַחַת וּבְנָיָהוּ פְּקִידִים מִיַּד כונניהווְשִׁמְעִי אָחִיו בְּמִפְקַד יְחִזְקִיָּהוּ הַמֶּלֶךְ וַעֲזַרְיָהוּ נְגִיד בֵּית הָאֱלֹהִים"
  14. דברי הימים ב לה8: "וְשָׂרָיו לִנְדָבָה לָעָם לַכֹּהֲנִים וְלַלְוִיִּם הֵרִימוּ חִלְקִיָּה וּזְכַרְיָהוּ וִיחִיאֵל נְגִידֵי בֵּית הָאֱלֹהִים לַכֹּהֲנִים נָתְנוּ לַפְּסָחִים אַלְפַּיִם וְשֵׁשׁ מֵאוֹת וּבָקָר שְׁלֹשׁ מֵאוֹת"
