ראשי>הגדרות מושגים>

בוגד = צבוע, המכסה את רשעתו כמו בבגד

קוד: בוגד= בתנ"ך

סוג: הגדרה

מאת: אראל

אל:

התואר בוגד נרדף לתואר רשע:

  • משלי כא18: "כֹּפֶר לַצַּדִּיק רָשָׁע, וְתַחַת יְשָׁרִים בּוֹגֵד" (פירוט),
  • ירמיהו יב1: "מַדּוּעַ דֶּרֶךְ רְשָׁעִים צָלֵחָה, שָׁלוּ כָּל בֹּגְדֵי בָגֶד?" (פירוט),
  • חבקוק א13: "לָמָּה תַבִּיט בּוֹגְדִים תַּחֲרִישׁ בְּבַלַּע רָשָׁע צַדִּיק מִמֶּנּוּ?"

מה המשמעות הייחודית שלו?

1. כמו בלשון ימינו, בוגד הוא המפר ברית עם אדם קרוב, כגון:

  • אח, מלאכי ב10: "הֲלוֹא אָב אֶחָד לְכֻלָּנוּ, הֲלוֹא אֵל אֶחָד בְּרָאָנוּ, מַדּוּעַ נִבְגַּד אִישׁ בְּאָחִיו, לְחַלֵּל בְּרִית אֲבֹתֵינוּ?" (פירוט),  ירמיהו יב6: "כִּי גַּם אַחֶיךָ וּבֵית אָבִיךָ גַּם הֵמָּה בָּגְדוּ בָךְ, גַם הֵמָּה קָרְאוּ אַחֲרֶיךָ מָלֵא, אַל תַּאֲמֵן בָּם כִּי יְדַבְּרוּ אֵלֶיךָ טוֹבוֹת",  איוב ו15: "אַחַי בָּגְדוּ כְמוֹ נָחַל, כַּאֲפִיק נְחָלִים יַעֲבֹרוּ"
  • בת-זוג, מלאכי ב14: "כִּי ה' הֵעִיד בֵּינְךָ וּבֵין אֵשֶׁת נְעוּרֶיךָ, אֲשֶׁר אַתָּה בָּגַדְתָּה בָּהּ, וְהִיא חֲבֶרְתְּךָ וְאֵשֶׁת בְּרִיתֶךָ",  ירמיהו ט1: "מִי יִתְּנֵנִי בַמִּדְבָּר מְלוֹן אֹרְחִים וְאֶעֶזְבָה אֶת עַמִּי וְאֵלְכָה מֵאִתָּם, כִּי כֻלָּם מְנָאֲפִים עֲצֶרֶת בֹּגְדִים",   משלי כג27-28: "כִּי שׁוּחָה עֲמֻקָּה זוֹנָה וּבְאֵר צָרָה נָכְרִיָּה , אַף הִיא כְּחֶתֶף תֶּאֱרֹב וּבוֹגְדִים בְּאָדָם תּוֹסִף" (פירוט).
  • פילגש, שמות כא8: "אִם רָעָה בְּעֵינֵי אֲדֹנֶיהָ אֲשֶׁר לא[לוֹ] יְעָדָהּ - וְהֶפְדָּהּ, לְעַם נָכְרִי לֹא יִמְשֹׁל לְמָכְרָהּ בְּבִגְדוֹ בָהּ" (פירוט).
  • בן-זוג, ירמיהו ג20: "אָכֵן בָּגְדָה אִשָּׁה מֵרֵעָהּ, כֵּן בְּגַדְתֶּם בִּי בֵּית יִשְׂרָאֵל, נְאֻם ה'".
  • חבר, איכה א2: "כָּל רֵעֶיהָ בָּגְדוּ בָהּ, הָיוּ לָהּ לְאֹיְבִים".
  • מנהיג, שופטים ט23: "וַיִּשְׁלַח אֱלֹהִים רוּחַ רָעָה בֵּין אֲבִימֶלֶךְ וּבֵין בַּעֲלֵי שְׁכֶם, וַיִּבְגְּדוּ בַעֲלֵי שְׁכֶם בַּאֲבִימֶלֶךְ" (פירוט).
  • ה', הושע ו7: "וְהֵמָּה כְּאָדָם עָבְרוּ בְרִית, שָׁם בָּגְדוּ בִי",  ירמיהו ה11: "כִּי בָגוֹד בָּגְדוּ בִּי בֵּית יִשְׂרָאֵל וּבֵית יְהוּדָה, נְאֻם ה'", הושע ה7: "בה' בָּגָדוּ כִּי בָנִים זָרִים יָלָדוּ, עַתָּה יֹאכְלֵם חֹדֶשׁ אֶת חֶלְקֵיהֶם",  מלאכי ב11: "בָּגְדָה יְהוּדָה וְתוֹעֵבָה נֶעֶשְׂתָה בְיִשְׂרָאֵל וּבִירוּשָׁלָים, כִּי חִלֵּל יְהוּדָה קֹדֶשׁ ה' אֲשֶׁר אָהֵב וּבָעַל בַּת אֵל נֵכָר",  שמואל א יד33: "וַיַּגִּידוּ לְשָׁאוּל לֵאמֹר 'הִנֵּה הָעָם חֹטִאים לה' לֶאֱכֹל עַל הַדָּם' וַיֹּאמֶר ' בְּגַדְתֶּם גֹּלּוּ אֵלַי הַיּוֹם אֶבֶן גְּדוֹלָה'",  תהלים קיט158: "רָאִיתִי בֹגְדִים וָאֶתְקוֹטָטָה, אֲשֶׁר אִמְרָתְךָ לֹא שָׁמָרוּ".

2. בוגד הוא גם צבוּע, המכסה את מעשי-הרשע שלו, כמו רוצח המסתיר סכין מתחת לבגדו במטרה לתקוע אותו בגבו של הקרבן. כך אפשר לפרש בפסוקים:

  • ירמיהו יב6: "כִּי גַּם אַחֶיךָ וּבֵית אָבִיךָ גַּם הֵמָּה בָּגְדוּ בָךְ, גַם הֵמָּה קָרְאוּ אַחֲרֶיךָ מָלֵא, אַל תַּאֲמֵן בָּם כִּי יְדַבְּרוּ אֵלֶיךָ טוֹבוֹת"
  • צפניה ג4: "נְבִיאֶיהָ פֹּחֲזִים, אַנְשֵׁי בֹּגְדוֹת" - מציגים את עצמם כנביאים ואף לובשים בגדים אפייניים לנביאים, אך למעשה הם רשעים.
  • משלי ב22: "וּרְשָׁעִים מֵאֶרֶץ יִכָּרֵתוּ, וּבוֹגְדִים יִסְּחוּ מִמֶּנָּה" - הרשעים יגורשו מייד, והבוגדים יגורשו לאחר שרשעתם תתגלה (פירוט).
  • משלי כב12: "עֵינֵי ה' נָצְרוּ דָעַת, וַיְסַלֵּף דִּבְרֵי בֹגֵד" (פירוט) - הכוונה לעד-שקר המנסה להטעות את השופט; אולם ה' יסלף ויגרום לו להיכשל ב דבריו כך שהשקר ייחשף.

ויש פסוקים שאפשר לפרש בשתי דרכים:

  • משלי כה19: "שֵׁן רֹעָה וְרֶגֶל מוּעָדֶת - מִבְטָח בּוֹגֵד בְּיוֹם צָרָה" (פירוט):   1. אדם הבוגד בעמו בוטח ביום צרה, מקווה שירויח כשתבוא צרה על עמו, אך לא מבין שהוא יהיה הראשון שייפול - כמו העומד על שן -סלע רעועה ורגלו מועדת.   2. לבטוח בבוגד, המסתיר את רשעו, זה כמו לסמוך על שן או רגל רעועה - היא נראית תקינה עד שמנסים להשתמש בה ואז היא מועדת ונופלת.
  • ירמיהו ג11: "וַיֹּאמֶר ה' אֵלַי: 'צִדְּקָה נַפְשָׁהּ מְשֻׁבָה יִשְׂרָאֵל מִבֹּגֵדָה יְהוּדָה'":   מלכות שומרון נקראת "משובה" ומלכות יהודה נקראת "בוגדה", כי:    1. במלכות יהודה היה בית המקדש ובו ארון הברית, והם הכניסו את פסלי אליליהם אפילו לשם;    2. כי מלכות שומרון "היו דומים כאשה שובבה שיצאה מתחת רשות בעלה לזנות בפרהסיא", ומלכות יהודה "דומה כזונה הבוגדת בבעלה בצנעה בלי ידיעתו, אכלה ומחתה פיה ואמרה לא פעלתי און" (מלבי"ם על ירמיהו ג6).
  • תהלים עח57: "וַיִּסֹּגוּ וַיִּבְגְּדוּ כַּאֲבוֹתָם, נֶהְפְּכוּ כְּקֶשֶׁת רְמִיָּה":   1. בני-ישראל בגדו בה' והפרו את הברית ביניהם;   2. בני-ישראל התחזו לצדיקים, כמו קשת היורה לכיוון ההפוך ופוגעת בבעליה.

בוגדים לעומת ישרים

שלושה פסוקים בספר משלי מציגים את הבוגדים כניגוד ל ישרים:

  • משלי יא3: "תֻּמַּת יְשָׁרִים תַּנְחֵם, וְסֶלֶף בּוֹגְדִים ושדם[יְשָׁדֵּם]"
  • משלי יא6: "צִדְקַת יְשָׁרִים תַּצִּילֵם, וּבְהַוַּת בֹּגְדִים יִלָּכֵדוּ"
  • משלי כא18: "כֹּפֶר לַצַּדִּיק רָשָׁע, וְתַחַת יְשָׁרִים בּוֹגֵד"

ניתן לפרש ניגוד זה לפי שני הפירושים למעלה:   1. ישר הוא הנוהג בשלום ובמישור עם כולם, בניגוד לבוגד שאינו נוהג בשלום אפילו עם קרוביו;    2. ישר הוא הנוהג בדרך מישורית שקופה וגלויה לכל, בניגוד לבוגד הפועל בסתר.

פסוקים נוספים

  • ישעיהו כד16: "מִכְּנַף הָאָרֶץ זְמִרֹת שָׁמַעְנוּ צְבִי לַצַּדִּיק וָאֹמַר רָזִי לִי רָזִי לִי אוֹי לִי בֹּגְדִים בָּגָדוּ וּבֶגֶד בּוֹגְדִים בָּגָדוּ"
  • ישעיהו כא2: "חָזוּת קָשָׁה הֻגַּד לִי הַבּוֹגֵד בּוֹגֵד וְהַשּׁוֹדֵד שׁוֹדֵד עֲלִי עֵילָם צוּרִי מָדַי כָּל אַנְחָתָה הִשְׁבַּתִּי"
  • ישעיהו לג1: "הוֹי שׁוֹדֵד וְאַתָּה לֹא שָׁדוּד וּבוֹגֵד וְלֹא בָגְדוּ בוֹ! כַּהֲתִמְךָ שׁוֹדֵד תּוּשַּׁד, כַּנְּלֹתְךָ לִבְגֹּד יִבְגְּדוּ בָךְ"
  • חבקוק ב5: "וְאַף כִּי הַיַּיִן בּוֹגֵד גֶּבֶר יָהִיר וְלֹא יִנְוֶה אֲשֶׁר הִרְחִיב כִּשְׁאוֹל נַפְשׁוֹ וְהוּא כַמָּוֶת וְלֹא יִשְׂבָּע וַיֶּאֱסֹף אֵלָיו כָּל הַגּוֹיִם וַיִּקְבֹּץ אֵלָיו כָּל הָעַמִּים"
  • תהלים כה3: "גַּם כָּל קֹוֶיךָ לֹא יֵבֹשׁוּ, יֵבֹשׁוּ הַבּוֹגְדִים רֵיקָם"
  • תהלים נט6: "וְאַתָּה ה' אֱלֹהִים צְבָאוֹת אֱלֹהֵי יִשְׂרָאֵל הָקִיצָה לִפְקֹד כָּל הַגּוֹיִם אַל תָּחֹן כָּל בֹּגְדֵי אָוֶן סֶלָה"
  • תהלים עג15: "אִם אָמַרְתִּי אֲסַפְּרָה כְמוֹ, הִנֵּה דוֹר בָּנֶיךָ בָגָדְתִּי"
  • משלי יא3: "תֻּמַּת יְשָׁרִים תַּנְחֵם, וְסֶלֶף בּוֹגְדִים ושדם[יְשָׁדֵּם]" (פירוט)
  • משלי יא6: "צִדְקַת יְשָׁרִים תַּצִּילֵם, וּבְהַוַּת בֹּגְדִים יִלָּכֵדוּ" (פירוט)
  • משלי יג2: "מִפְּרִי פִי אִישׁ יֹאכַל טוֹב, וְנֶפֶשׁ בֹּגְדִים חָמָס" (פירוט)
  • משלי יג15: "שֵׂכֶל טוֹב יִתֶּן חֵן, וְדֶרֶךְ בֹּגְדִים אֵיתָן" (פירוט)
