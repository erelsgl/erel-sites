ביקש את נפשו = רצה להרגו

קוד: ביקש את נפשו= בתנ"ך

סוג: הגדרה

מאת: אראל

אל:

המושג "נפש" מציין את החיים, במיוחד את הצד הגופני שלהם. לכן, כשרוצים להגיד, בלשון המקרא, שמישהו רצה להרוג את רעהו - משתמשים בביטוי "ביקש את נפשו", כלומר - חיפש את המקום שבו הוא חי, כדי לקחת את חייו ממנו:
  • שמות ד19: "וַיֹּאמֶר ה' אֶל מֹשֶׁה בְּמִדְיָן 'לֵךְ שֻׁב מִצְרָיִם כִּי מֵתוּ כָּל הָאֲנָשִׁים הַמְבַקְשִׁים אֶת נַפְשֶׁךָ'"
  • שמואֶל א כ1: "ויברח דוד מנוות ברמה ויבא וַיֹּאמֶר לפני יהונתן 'מה עשיתי מה עוני ומה חטאֶתי לפני אביך כִּי מבקש אֶת נפשי?'"
  • שמואֶל א כב23: "שֻׁבה אֶתי אֶל תירא, כִּי אשר יבקש אֶת נפשי - יבקש אֶת נַפְשֶׁךָ, כִּי משמרת אֶתה עמדי" כלומר - אותו אדם שרוצה להרוג אותי (שאול), רוצה להרוג גם אותך, ולכן כדאי שנישאר ביחד.
  • שמואל א כג15: "וַיַּרְא דָוִד כִּי יָצָא שָׁאוּל לְבַקֵּשׁ אֶת נַפְשׁוֹ, וְדָוִד בְּמִדְבַּר זִיף בַּחֹרְשָׁה"
  • שמואל א כה29: "וַיָּקָם אָדָם לִרְדָפְךָ וּלְבַקֵּשׁ אֵת נַפְשֶׁךָ, וְהָיְתָה נֶפֶשׁ אֲדֹנִי צְרוּרָה בִּצְרוֹר הַחַיִּים אֶת ה' אֱלֹהֶיךָ וְאֵת נֶפֶשׁ אֹיְבֶיךָ יְקַלְּעֶנָּה בְּתוֹךְ כַּף הַקָּלַע" - המקובלים פירשו שהכוונה לנשמה הנצחית , אך ע " פ הפשט הכוונה לחיים החומריים : אביגיל מאחלת לדוד , שבכל פעם שמישהו יבקש את נפשו כדי להרגו , תהיה נפשו שמורה במקום של חיים כך שלא יצליחו להרגו , ונפשם של אויביו תהיה במקום מסוכן כמו כף של מקלע - אבנים .
  • שמואל ב ד8: "וַיָּבִאוּ אֶת רֹאשׁ אִישׁ בֹּשֶׁת אֶל דָּוִד חֶבְרוֹן וַיֹּאמְרוּ אֶל הַמֶּלֶךְ 'הִנֵּה רֹאשׁ אִישׁ בֹּשֶׁת בֶּן שָׁאוּל אֹיִבְךָ אֲשֶׁר בִּקֵּשׁ אֶת נַפְשֶׁךָ, וַיִּתֵּן ה' לַאדֹנִי הַמֶּלֶךְ נְקָמוֹת הַיּוֹם הַזֶּה מִשָּׁאוּל וּמִזַּרְעוֹ'"
  • שמואֶל ב טז11: "ויאמר דָּוִד אֶל אבישי ואֶל כל עבדיו 'הִנֵּה בני אֲשֶׁר יצא ממעי מבִּקֵּשׁ אֶת נפשי ואף כי עתה בֶּן הימיני...'"
  • מלכים א יט10, 14: "וַיֹּאמֶר 'קַנֹּא קִנֵּאֶתִי לה' אֱלֹהֵי צְבָאוֹת כִּי עָזְבוּ בְרִיתְךָ בְּנֵי יִשְׂרָאֵל אֶת מִזְבְּחֹתֶיךָ הָרָסוּ וְאֶת נְבִיאֶיךָ הָרְגוּ בֶחָרֶב; וָאִוָּתֵר אֲנִי לְבַדִּי, וַיְבַקְשׁוּ אֶת נַפְשִׁי לְקַחְתָּהּ'"
  • ירמיהו ד30: "[וְאַתְּ]י שָׁדוּד מַה תַּעֲשִׂי כִּי תִלְבְּשִׁי שָׁנִי כִּי תַעְדִּי עֲדִי זָהָב כִּי תִקְרְעִי בַפּוּךְ עֵינַיִךְ לַשָּׁוְא תִּתְיַפִּי מָאֲסוּ בָךְ עֹגְבִים נַפְשֵׁךְ יְבַקֵּשׁוּ"
  • ירמיהו יא21: "לָכֵן כֹּה אָמַר ה' עַל אַנְשֵׁי עֲנָתוֹת הַמְבַקְשִׁים אֶת נַפְשְׁךָ לֵאמֹר לֹא תִנָּבֵא בְּשֵׁם ה' וְלֹא תָמוּת בְּיָדֵנוּ"
  • ירמיהו יט7: "וּבַקֹּתִי אֶת עֲצַת יְהוּדָה וִירוּשָׁלִַם בַּמָּקוֹם הַזֶּה וְהִפַּלְתִּים בַּחֶרֶב לִפְנֵי אֹיְבֵיהֶם וּבְיַד מְבַקְשֵׁי נַפְשָׁם וְנָתַתִּי אֶת נִבְלָתָם לְמַאֲכָל לְעוֹף הַשָּׁמַיִם וּלְבֶהֱמַת הָאָרֶץ"
  • ירמיהו יט9: "וְהַאֲכַלְתִּים אֶת בְּשַׂר בְּנֵיהֶם וְאֵת בְּשַׂר בְּנֹתֵיהֶם וְאִישׁ בְּשַׂר רֵעֵהוּ יֹאכֵלוּ בְּמָצוֹר וּבְמָצוֹק אֲשֶׁר יָצִיקוּ לָהֶם אֹיְבֵיהֶם וּמְבַקְשֵׁי נַפְשָׁם"
  • ירמיהו כא7: "וְאַחֲרֵי כֵן נְאֻם ה' אֶתֵּן אֶת צִדְקִיָּהוּ מֶלֶךְ יְהוּדָה וְאֶת עֲבָדָיו וְאֶת הָעָם וְאֶת הַנִּשְׁאָרִים בָּעִיר הַזֹּאֶת מִן הַדֶּבֶר מִן הַחֶרֶב וּמִן הָרָעָב בְּיַד נְבוּכַדְרֶאצַּר מֶלֶךְ בָּבֶל וּבְיַד אֹיְבֵיהֶם וּבְיַד מְבַקְשֵׁי נַפְשָׁם וְהִכָּם לְפִי חֶרֶב לֹא יָחוּס עֲלֵיהֶם וְלֹא יַחְמֹל וְלֹא יְרַחֵם"
  • ירמיהו כב25: "וּנְתַתִּיךָ בְּיַד מְבַקְשֵׁי נַפְשֶׁךָ וּבְיַד אֲשֶׁר אַתָּה יָגוֹר מִפְּנֵיהֶם וּבְיַד נְבוּכַדְרֶאצַּר מֶלֶךְ בָּבֶל וּבְיַד הַכַּשְׂדִּים"
  • ירמיהו לד20: "וְנָתַתִּי אוֹתָם בְּיַד אֹיְבֵיהֶם וּבְיַד מְבַקְשֵׁי נַפְשָׁם וְהָיְתָה נִבְלָתָם לְמַאֲכָל לְעוֹף הַשָּׁמַיִם וּלְבֶהֱמַת הָאָרֶץ"
  • ירמיהו לד21: "וְאֶת צִדְקִיָּהוּ מֶלֶךְ יְהוּדָה וְאֶת שָׂרָיו אֶתֵּן בְּיַד אֹיְבֵיהֶם וּבְיַד מְבַקְשֵׁי נַפְשָׁם וּבְיַד חֵיל מֶלֶךְ בָּבֶל הָעֹלִים מֵעֲלֵיכֶם"
  • ירמיהו לח16: "וַיִּשָּׁבַע הַמֶּלֶךְ צִדְקִיָּהוּ אֶל יִרְמְיָהוּ בַּסֵּתֶר לֵאמֹר 'חַי ה' אֶת אֲשֶׁר עָשָׂה לָנוּ אֶת הַנֶּפֶשׁ הַזֹּ אֶת אִם אֲמִיתֶךָ וְאִם אֶתֶּנְךָ בְּיַד הָאֲנָשִׁים הָאֵלֶּה אֲשֶׁר מְבַקְשִׁים אֶת נַפְשֶׁךָ'"
  • ירמיהו מד30: "כֹּה אָמַר ה' הִנְנִי נֹתֵן אֶת פַּרְעֹה חָפְרַע מֶלֶךְ מִצְרַיִם בְּיַד אֹיְבָיו וּבְיַד מְבַקְשֵׁי נַפְשׁוֹ כַּאֲשֶׁר נָתַתִּי אֶת צִדְקִיָּהוּ מֶלֶךְ יְהוּדָה בְּיַד נְבוּכַדְרֶאצַּר מֶלֶךְ בָּבֶל אֹיְבוֹ וּמְבַקֵּשׁ נַפְשׁוֹ"
  • ירמיהו מו26: "וּנְתַתִּים בְּיַד מְבַקְשֵׁי נַפְשָׁם וּבְיַד נְבוּכַדְרֶאצַּר מֶלֶךְ בָּבֶל וּבְיַד עֲבָדָיו וְאַחֲרֵי כֵן תִּשְׁכֹּן כִּימֵי קֶדֶם נְאֻם ה'"
  • ירמיהו מט37: "וְהַחְתַּתִּי אֶת עֵילָם לִפְנֵי אֹיְבֵיהֶם וְלִפְנֵי מְבַקְשֵׁי נַפְשָׁם וְהֵבֵאֶתִי עֲלֵיהֶם רָעָה אֶת חֲרוֹן אַפִּי נְאֻם ה' וְשִׁלַּחְתִּי אַחֲרֵיהֶם אֶת הַחֶרֶב עַד כַּלּוֹתִי אוֹתָם"
  • תהלים לה4: "יֵבֹשׁוּ וְיִכָּלְמוּ מְבַקְשֵׁי נַפְשִׁי יִסֹּגוּ אָחוֹר וְיַחְפְּרוּ חֹשְׁבֵי רָעָתִי"
  • תהלים לח13: "וַיְנַקְשׁוּ מְבַקְשֵׁי נַפְשִׁי וְדֹרְשֵׁי רָעָתִי דִּבְּרוּ הַוּוֹת וּמִרְמוֹת כָּל הַיּוֹם יֶהְגּוּ"
  • תהלים מ15: "יֵבֹשׁוּ וְיַחְפְּרוּ יַחַד מְבַקְשֵׁי נַפְשִׁי לִסְפּוֹתָהּ יִסֹּגוּ אָחוֹר וְיִכָּלְמוּ חֲפֵצֵי רָעָתִי"
  • תהלים נד5: "כִּי זָרִים קָמוּ עָלַי וְעָרִיצִים בִּקְשׁוּ נַפְשִׁי לֹא שָׂמוּ אֱלֹהִים לְנֶגְדָּם סֶלָה"
  • תהלים סג10: "וְהֵמָּה לְשׁוֹאָה יְבַקְשׁוּ נַפְשִׁי יָבֹאוּ בְּתַחְתִּיּוֹת הָאָרֶץ"
  • תהלים ע3: "יֵבֹשׁוּ וְיַחְפְּרוּ מְבַקְשֵׁי נַפְשִׁי יִסֹּגוּ אָחוֹר וְיִכָּלְמוּ חֲפֵצֵי רָעָתִי"
  • תהלים פו14: "אֱלֹהִים זֵדִים קָמוּ עָלַי וַעֲדַת עָרִיצִים בִּקְשׁוּ נַפְשִׁי וְלֹא שָׂמוּךָ לְנֶגְדָּם"
  • וכן משלי כט10: "אנשי דמים ישנאו תם וישרים יבקשו נפשו" (פירוט).
