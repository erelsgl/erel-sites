ראשי>היחסים בין עם ישראל לעמים השכנים>

לבנון ואיראן סייעו לבניית בתי המקדש

מאת: אראל

בית המקדש הראשון נבנה בעזרת הלבנונים (מלכים א ה16): " וַיִּשְׁלַח שְׁלֹמֹה אֶל חִירָם לֵאמֹר... הִנְנִי אֹמֵר לִבְנוֹת בַּיִת לְשֵׁם ה' אֱלֹהָי... וְעַתָּה צַוֵּה וְיִכְרְתוּ לִי אֲרָזִים מִן הַלְּבָנוֹן. וַעֲבָדַי יִהְיוּ עִם עֲבָדֶיךָ, וּשְׂכַר עֲבָדֶיךָ אֶתֵּן לְךָ כְּכֹל אֲשֶׁר תֹּאמֵר, כִּי אַתָּה יָדַעְתָּ כִּי אֵין בָּנוּ אִישׁ יֹדֵעַ לִכְרָת עֵצִים כַּצִּדֹנִים. וַיְהִי כִּשְׁמֹעַ חִירָם אֶת דִּבְרֵי שְׁלֹמֹה וַיִּשְׂמַח מְאֹד, וַיֹּאמֶר 'בָּרוּךְ ה' הַיּוֹם אֲשֶׁר נָתַן לְדָוִד בֵּן חָכָם עַל הָעָם הָרָב הַזֶּה... אֲנִי אֶעֱשֶׂה אֶת כָּל חֶפְצְךָ בַּעֲצֵי אֲרָזִים וּבַעֲצֵי בְרוֹשִׁים. עֲבָדַי יֹרִדוּ מִן הַלְּבָנוֹן יָמָּה, וַאֲנִי אֲשִׂימֵם דֹּבְרוֹת בַּיָּם עַד הַמָּקוֹם אֲשֶׁר תִּשְׁלַח אֵלַי, וְנִפַּצְתִּים שָׁם וְאַתָּה תִשָּׂא; וְאַתָּה תַּעֲשֶׂה אֶת חֶפְצִי לָתֵת לֶחֶם בֵּיתִי...". שלמה מלך ישראל כרת ברית עם חירם מלך צור, וזה מכר לו עצי ארזים מהלבנון לצורך בניית המקדש.

בית המקדש השני נבנה בעזרת האיראנים - הפרסים (עזרא א1): " וּבִשְׁנַת אַחַת לְכוֹרֶשׁ מֶלֶךְ פָּרַס, לִכְלוֹת דְּבַר ה' מִפִּי יִרְמְיָה, הֵעִיר ה' אֶת רוּחַ כֹּרֶשׁ מֶלֶךְ פָּרַס , וַיַּעֲבֶר קוֹל בְּכָל מַלְכוּתוֹ וְגַם בְּמִכְתָּב לֵאמֹר. כֹּה אָמַר כֹּרֶשׁ מֶלֶךְ פָּרַס: כֹּל מַמְלְכוֹת הָאָרֶץ נָתַן לִי ה' אֱלֹהֵי הַשָּׁמָיִם, וְהוּא פָקַד עָלַי לִבְנוֹת לוֹ בַיִת בִּירוּשָׁלַ ͏ִ ם אֲשֶׁר בִּיהוּדָה. מִי בָכֶם מִכָּל עַמּוֹ יְהִי אֱלֹהָיו עִמּוֹ, וְיַעַל לִירוּשָׁלַ ͏ִ ם אֲשֶׁר בִּיהוּדָה וְיִבֶן אֶת בֵּית ה' אֱלֹהֵי יִשְׂרָאֵל, הוּא הָאֱלֹהִים אֲשֶׁר בִּירוּשָׁלָ ͏ִ ם. וְכָל הַנִּשְׁאָר מִכָּל הַמְּקֹמוֹת אֲשֶׁר הוּא גָר שָׁם יְנַשְּׂאוּהוּ אַנְשֵׁי מְקֹמוֹ בְּכֶסֶף וּבְזָהָב וּבִרְכוּשׁ וּבִבְהֵמָה, עִם הַנְּדָבָה לְבֵית הָאֱלֹהִים אֲשֶׁר בִּירוּשָׁלָ ͏ִ ם ", וגם בעזרת הלבנונים, עזרא ג7: " וַיִּתְּנוּ כֶסֶף לַחֹצְבִים וְלֶחָרָשִׁים, וּמַאֲכָל וּמִשְׁתֶּה וָשֶׁמֶן לַצִּדֹנִים וְלַצֹּרִים; לְהָבִיא עֲצֵי אֲרָזִים מִן הַלְּבָנוֹן אֶל יָם יָפוֹא , כְּרִשְׁיוֹן כּוֹרֶשׁ מֶלֶךְ פָּרַס עֲלֵיהֶם " ( פירוט).

היום, תשעה באב ה'תשפ"ד, הלבנונים והאיראנים יחד זוממים להשמיד את הבית הלאומי השלישי של העם היהודי. נתפלל שהלבנונים והאיראנים ייזכרו בתפקיד ההסטורי שמילאו בבניית שני בתי המקדש הקודמים, ובמקום להשמיד את ישראל ישתפו פעולה כדי לבנות את בית המקדש השלישי.
