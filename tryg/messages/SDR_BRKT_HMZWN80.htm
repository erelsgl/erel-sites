שלשה או יותר שאכלו חייבים לזמן

נְבָרֵךְ ( בעשרה אֱלֹהֵינוּ) שֶׁאָכַלְנוּ מִשֶּׁלּוֹ.

בָּרוּךְ ( בעשרה אֱלֹהֵינוּ) שֶׁאָכַלְנוּ מִשֶׁלּוֹ וּבְטוּבוֹ חָיִינוּ.

 

ברכה ראשונה - ברכת הזן

בָּרוּךְ אַתָּה יהוה אֱלֹהֵינוּ מֶלֶךְ הָעוֹלָם, הַזָּן אוֹתָנוּ וְאֶת הָעוֹלָם כֻּלּוֹ בְּטוֹב, בְּחֵן בְּחֶסֶד וּבְרַחֲמִים . אֱלֹהֵינוּ אָבִינוּ, רְעֵנוּ, פַּרְנְסֵנוּ, וְאַל תַּצְרִיכֵנוּ לִידֵי מַתְּנַת בָּשָׂר וָדָם. כִּי אִם לְיָדְךָ הַמְּלֵאָה וְהַפְּתוּחָה. שֶׁלֹּא נֵבוֹשׁ וְלֹא נִכָּלֵם לְעוֹלָם וָעֶד.

כָּאָמוּר: פּוֹתֵ֥חַ אֶת־יָדֶ֑ךָ וּמַשְׂבִּ֖יעַ לְכָל־חַ֣י רָצֽוֹן.           תהלים קמה, טז

בָּרוּךְ אַתָּה יהוה, הַזָּן אֶת הַכֹּל:

 

 

ברכה שנייה - ברכת הארץ

נוֹדֶה לְּךָ יהוה אֱלֹהֵינוּ כִּי הִנְחַלְתָּנוּ אֶרֶץ חֶמְדָּה טוֹבָה וּרְחָבָה, וְעַל בְּרִיתְךָ שֶׁחָתַמְתָּ בִּבְשָׂרֵנוּ, וְעַל תּוֹרָתְךָ שֶׁלִּמַּדְתָּנוּ, וְעַל הַחַיִּים וְהַמָּזוֹן שָׁאַתָּה נוֹתֵן לָנוּ בְּכָל יוֹם וָיוֹם.

 

בימי ההודיות מוסיפים כאן על הניסים (עמוד 84)

לחנוכה:

וְעַל הַנִּסִּים, הַגְּבוּרוֹת, הַתְּשׁוּעוֹת, הַמִּלְחָמוֹת וְהַפְּדוּת שֶׁעָשִׂיתָ עִמָּנוּ, וְעִם אֲבוֹתֵינוּ בַּיָּמִים הָהֵם בַּזְּמַן הַזֶּה.

בִּימֵי מַתִּתְיָהוּ בֶּן יוֹחָנָן הַכֹּהֵן חַשְׁמוֹנַאי וּבָנָיו, כְּשֶׁעָמְדָה מַלְכוּת יָוָן הָרְשָׁעָה עַל עַמְּךָ יִשְׂרָאֵל לְשַׁכְּחָם אֶת תּוֹרָתְךָ וּלְהַעֲבִירָם מֵחֻקֵּי רְצוֹנֶךָ. וְאַתָּה בְּרַחֲמֶיךָ הָרַבִּים עָמַדְתָּ לָהֶם בְּעֵת צָרָתָם. רַבְתָּ אֶת רִיבָם, דַּנְתָּ אֶת דִּינָם, נָקַמְתָּ אֶת נִקְמָתָם, מָסַרְתָּ גִּבּוֹרִים בְּיַד חַלָּשִׁים, וְרַבִּים בְּיַד מְעַטִּים, וּטְמֵאִים בְּיַד טְהוֹרִים, וּרְשָׁעִים בְּיַד צַדִּיקִים, וְזֵדִים בְּיַד עוֹשֵׂי תוֹרָתֶךָ. וּלְךָ עָשִׂיתָ שֵׁם גָּדוֹל בָּעוֹלָם, וּלְעַמְּךָ יִשְׂרָאֵל עָשִׂיתָ תְּשׁוּעָה גְדוֹלָה. וְאַחַר כָּךְ בָּאוּ בָנֶיךָ לִדְבִיר בֵּיתֶךָ, וְטִהֲרוּ אֶת מִקְדָּשֶׁךָ, וְהִדְלִיקוּ נֵרוֹת בְּחַצְרוֹת קָדְשֶׁךָ. כְּשֵׁם שֶׁעָשִֹיתָ תְּשׁוּעָה לַאֲבוֹתֵינוּ, כָּךְ עֲשֵׂה עִמָּנוּ בָּעֵת הַזּאֹת, וְנוֹדֶה לְשִׁמְךָ לָנֶצַח.

וממשיכים ועל כולם...

 

וְעַל כֻּלָּם אָנוּ מוֹדִים לְךָ וּמְבָרְכִים אֶת שִׁמְךָ לְעוֹלָם וָעֶד. כַּכָּתוּב:

וְאָֽכַלְתָּ֖ וְשָׂבָ֑עְתָּ וּבֵֽרַכְתָּ֙ אֶת־יְהוָ֣ה אֱלֹהֶ֔יךָ עַל־הָאָ֥רֶץ הַטֹּבָ֖ה                         דברים ח, י

אֲשֶׁ֥ר נָֽתַן־לָֽךְ. בָּרוּךְ אַתָּה יהוה, עַל הָאָרֶץ וְעַל הַמָּזוֹן:

ברכה שלישית - בונה ירושלים

 

 

בשבת   יש אומרים ברכה זו בנוסח זה:

נַחֲמֵנוּ יהוה אֱלֹהֵינוּ בְּבִנְיַן יְרוּשָׁלַיִם עִירֶךָ, וּבְצִיּוֹן מִשְׁכַּן כְּבוֹדֶךָ, וּבַבַּיִת הַגָּדוֹל וְהַקָּדוֹשׁ שֶׁנִּקְרָא שִׁמְךָ עָלָיו, וּבְמַלְכוּת בֵּית דָּוִד מְשִׁיחֶךָ.

 

 

רַחֵם יהוה אֱלֹהֵינוּ בְּרַחֲמֶיךָ הָרַבִּים עַל יִשְׂרָאֵל עַמֶּךָ, וְעַל יְרוּשָׁלַיִם עִירֶךָ, וְעַל צִיּוֹן מִשְׁכַּן כְּבוֹדֶךָ, וְהָאֵר פָּנֶיךָ עַל מִקְדָּשְׁךָ הַשָּׁמֵם. בְּנֵה בֵיתְךָ וְשַׁכְלֵל הֵיכָלֶךָ. וּמַלְכוּת בֵּית דָּוִד הַחֲזֵר לִמְקוֹמָהּ בְּיָמֵינוּ.


בשבת מוסיפים כאן

אֱלֹהֵינוּ וֵאלֹהֵי אֲבוֹתֵינוּ רְצֵה בִּמְנוּחָתֵנוּ, וְהַחֲלִיצֵנוּ בְּמִצְוֹתֶיךָ וּבְמִצְוַת יוֹם הַשְּׁבִיעִי הַגָּדוֹל וְהַקָּדוֹשׁ הַזֶּה. כִּי יוֹם גָּדוֹל וְקָדוֹשׁ הוּא מִלְּפָנֶיךָ, נִשְׁבּוֹת בּוֹ כְּמִצְוַת רְצוֹנֶךָ. הָנַח לָנוּ וְאַל תְּהִי צָרָה בִּמְנוּחָתֵנוּ. וּתְהִי מְנוּחָתֵנוּ מְנוּחַת אַהֲבָה וּנְדָבָה, מְנוּחַת אֱמֶת וֶאֱמוּנָה, מְנוּחַת שָׁלוֹם וְשַׁלְוָה, מְנוּחָה שְׁלֵמָה שָׁאַתָּה רוֹצֶה בָּהּ .

 

בראש חודש, ביום טוב ובחול המועד מוסיפים כאן:

 

אָנָּא אֱלֹהֵינוּ וֵאלֹהֵי אֲבוֹתֵינוּ יַעֲלֶה וְיָבוֹא, יַגִּיעַ וְיֵרָאֶה, יִזָּכֵר וְיִפָּקֵד לְפָנֶיךָ זִכְרוֹנֵנוּ וְזִכְרוֹן אֲבוֹתֵינוּ, וְזִכְרוֹן כָּל עַמְּךָ בֵּית יִשְׂרָאֵל, וְזִכְרוֹן עִירְךָ וּמִקְדָּשְךָ לְטוֹבָה בְּיוֹם:

  אָ֤ז תַּחְפֹּ֣ץ זִבְחֵי־צֶ֭דֶק עוֹלָ֣ה וְכָלִ֑יל    אָ֤ז יַֽעֲל֖וּ עַל־מִזְבַּֽחֲךָ֣ פָרִֽים.                    תהילים נא, כא

 


וְשַׂמְּחֵנוּּ וְנַחֲמֵנוּ בְּבִנְיַן עִירֶךָ וּבְתִקוּּן הֵיכָלֶךָ. בָּרוּךְ אַתָּה יהוה, מְנַחֵם עַמּוֹ בְּבִנְיַן יְרוּשָׁלָיִם. אָמֵן:

בָּרוּךְ אַתָּה יהוה, בּוֹנֵה יְרוּשָׁלָיִם. אָמֵן:


 

ברכה רביעית - הטוב והמטיב

 

בָּרוּךְ אַתָּה יהוה אֱלֹהֵינוּ מֶלֶךְ הָעוֹלָם, הָאֵל. אָבִינוּ. מַלְכֵּנוּ. בּוֹרְאֵנוּ. גּוֹאֲלֵנוּ. הַמֶּלֶךְ הַטּוֹב וְהַמֵּטִיב שֶׁבְּכָל יוֹם וָיוֹם הוּא מַרְבֶּה לְהֵיטִיב עִמָּנוּ , וְהוּא יִגְמְלֵנוּ לָעַד חֵן וָחֶסֶד, רֶוַח וְרַחֲמִים, וְכָל טוֹב. אָמֵן:  

 

ברכת האורח

יְהִי רָצוֹן שֶׁלֹא יְבוֹשׁ בַּעַל הַבַּיִת בָּעוֹלָם הַזֶּה וְלֹא יִכָּלֵם לְעוֹלָם הַבָּא וְיַצְלִיח מְאֹד בְּכֹל נְכָסָיו וְיִהְיוּ נְכָסָיו וּנְכָסֵנוּ מֻצְלָחִים וּקְרוֹבִים לָעִיר בָּרֵךְ יהוה חֵילוֹ וּפֹעַל יָדָיו תִּרְצֶה וְאַל יִזְדַּקֵּק לֹא לְפָנָיו וְלֹא לְפָנֵינוּ שׁוּם דְּבַר הִרְהוּר חֵטְא וְעָוֹן מֵעַתָּה וְעַד עוֹלָם.

 

נוהגים להוסיף בקשות מעין אלו (ובשבתות ובימים טובים אין להוסיף תחינות ובקשות):

הָרַחֲמָן יָאִיר עֵינֵינוּ בִּמְאוֹר תּוֹרָה. הָרַחֲמָן יַצְלִיחַ לָנוּ בְּכָל דְּרָכֵינוּ. הָרַחֲמָן יָשִׂים יִרְאָתוֹ לְנֶגֶד עֵינֵינוּ. הָרַחֲמָן יָסִיר מַחֲלָה מִקִּרְבֵּנוּ. הָרַחֲמָן יִגְעַר בְּחוֹשְׁבֵי רָעָתֵנוּ. הָרַחֲמָן יַנְחִילֵנוּ נִצָּחוֹן עַל כָּל אוֹיְבֵינוּ וְיִירַשׁ זַרְעֵנוּ אֵת שַׁעַר שֹׂנְׁאֵינוּ. הָרַחֲמָן יִהְיֶה לָנוּ עוֹזֵר וְסוֹמֵךְ. עוֹשֶׂה שָׁלוֹם בִּמְרוֹמָיו הוּא יַעֲשֶׂה שָׁלוֹם עָלֵינוּ וְעַל כָּל יִשְׂרָאֵל.

יש נוהגים להוסיף   מַה שֶּׁאָכַלְנוּ יִהְיֶה לְשָׂבְעָה. וּמַה שֶּׁשָּׁתִינוּ יִהְיֶה לִרְפוּאָה. וּמַה שֶּׁהוֹתַרְנוּ יִהְיֶה לִבְרָכָה כְּדִכְתִיב וַיִּתֵּן לִפְנֵיהֶם וַיּאכְלוּ וַיּוֹתִירוּ כִּדְבַר יהוה: בְּרוּכִים אַתֶּם לַיהוה. עוֹשֵׂה שָׁמַיִם וָאָרֶץ: בָּרוּךְ הַגֶּבֶר אֲשֶׁר יִבְטַח בַּיהוה. וְהָיָה יהוה מִבְטָחוֹ:    יְֽהוָ֗ה עֹ֖ז לְעַמּ֣וֹ יִתֵּ֑ן   יְהוָ֓ה ׀ יְבָרֵ֖ךְ אֶת־עַמּ֣וֹ בַשָּׁלֽוֹם:    תהלים קמה, טז

וְעַל תְּנוּבַת הַשָּדֶה, וְעַל אֶרֶץ חֶמְדָּה טוֹבָה וּרְחָבָה, שֶרָצִיתָ וְהִנְחַלְתָּנוּ. רַחֵם יהוה אֱלֹהֵינוּ עַל יִשְׂרָאֵל עַמֶּךָ, וְעַל יְרוּשָׁלַיִם עִירֶךָ, וְעַל צִיּוֹן מִשְׁכַּן כְּבוֹדֶךָ, וְהָאֵר פָּנֶיךָ עַל מִקְדָּשְׁךָ הַשָּׁמֵם .

ורצה והחליצנו ביום השבת הזה

וזכרנו לטובה ביום ראש החדש הזה

וזכרנו ביום הזכרון הזה

ושמחנו ביום חג (פלוני) הזה

וְנֹאכַל מִפִּרְיָהּ וְנִשְבַּע מִטּוּבָהּ וּנְבָרֶכְךָ עָלֶיהָ בִּקְדֻשָּה וּבְטָהֳרָה, כִּי אֵל טוֹב וּמֵטִיב אָתָּה בָּרוּךְ אַתָּה אַתָּה יהוה עַל הָאָרֶץ וְעַל

                 פְּרוֹתֶיהָ / פְּרִי גַפְנָהּ / הַמִּחְיָה עַל הַכַּלְכָּלָה
